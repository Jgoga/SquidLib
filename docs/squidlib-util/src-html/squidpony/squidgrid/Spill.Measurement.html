008/**
009 * A randomized flood-fill implementation that can be used for level generation (e.g. filling ponds and lakes), for
010 * gas propagation, or for all sorts of fluid-dynamics-on-the-cheap.
011 * Created by Tommy Ettinger on 4/7/2015.
012 * 
013 * @see Splash An alternative implementation with a lighter API
014 */
015public class Spill implements Serializable {
016    private static final long serialVersionUID = 1L;

018    /**
019     * The type of heuristic to use.
020     */
021    public enum Measurement {
022
023        /**
024         * The distance it takes when only the four primary directions can be
025         * moved in. The default.
026         */
027        MANHATTAN,
028        /**
029         * The distance it takes when diagonal movement costs the same as
030         * cardinal movement.
031         */
032        CHEBYSHEV,
033        /**
034         * The distance it takes as the crow flies.
035         */
036        EUCLIDEAN
037    }
038
039    /**
040     * This affects how distance is measured on diagonal directions vs. orthogonal directions. MANHATTAN should form a
041     * diamond shape on a featureless map, while CHEBYSHEV and EUCLIDEAN will form a square. If you only call
042     * Spill.start() once, you should strongly prefer MANHATTAN, even if the rest of the game uses another
043     * measurement, because CHEBYSHEV and EUCLIDEAN can produce odd, gap-filled flood-fills.  Any case where you have
044     * too many gaps can be corrected to varying extent by calling start() more than once with slowly increasing values.
045     * Because start() will extend from the existing area of the Spill, holes are likely to be filled after a few calls,
046     * but if the last call to start() tries to fill too many more cells than the previous one, it can cause holes on
047     * the periphery of the Spill area.
048     */
049    public Measurement measurement = Measurement.MANHATTAN;

052    /**
053     * Stores which parts of the map are accessible (with a value of true) and which are not (with a value of false,
054     * including both walls and unreachable sections of the map). Should not be changed unless the actual physical
055     * terrain has changed. You should call initialize() with a new map instead of changing this directly.
056     */

240    /**
241     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
242     * one when it was constructed, or because the contents of the terrain have changed permanently.
243     * @param level the level as a 2D rectangular boolean array, using {@code false} to represent walls
244     * @return this Spill after initialization has completed, for chaining
245     */
246    public Spill initialize(final boolean[][] level) {
247        fresh = new OrderedSet<>();
248        width = level.length;
249        height = level[0].length;
250        spillMap = new boolean[width][height];
251        physicalMap = new boolean[width][height];
252        for (int y = 0; y < height; y++) {
253            for (int x = 0; x < width; x++) {
254                spillMap[x][y] = level[x][y];
255                physicalMap[x][y] = level[x][y];
256            }
257        }
258        initialized = true;
259        return this;
260    }

266     * @return this Spill after initialization has completed, for chaining

310     */
311    public void resetMap() {
312        if(!initialized) return;
313        for (int y = 0; y < height; y++) {
314            for (int x = 0; x < width; x++) {
315                spillMap[x][y] = false;
316            }
317        }
318    }

367    /**
368     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to entry will be true,
369     * the cells near that will be true if chosen at random from all passable cells adjacent to a
370     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
371     * to fill (the volume parameter), and will fill less if it has completely exhausted all passable cells.
372     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
373     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
374     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
375     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
376     *
377     * @param entry The first cell to spread from, which should really be passable.
378     * @param volume The total number of cells to attempt to fill, which must be non-negative.
379     * @param impassable A Set of Position keys representing the locations of moving obstacles to a
380     *                   path that cannot be moved through; this can be null if there are no such obstacles.
381     * @return An ArrayList of Points that this will enter, in order starting with entry at index 0, until it
382     * reaches its volume or fills its boundaries completely.
383     */
384    public ArrayList<Coord> start(Coord entry, int volume, Set<Coord> impassable) {
385        if(!initialized) return null;
386        if(impassable == null)
387            impassable = new OrderedSet<>();
388        if(!physicalMap[entry.x][entry.y] || impassable.contains(entry))
389            return null;
390        spreadPattern = new ArrayList<>(volume);
391        spillMap[entry.x][entry.y] = true;
392        Coord temp;
393        for(int x = 0; x < spillMap.length; x++)
394        {
395            for(int y = 0; y < spillMap[x].length; y++)
396            {
397                temp = Coord.get(x, y);
398                if(spillMap[x][y] && !impassable.contains(temp))
399                    fresh.add(temp);
400            }
401        }
402
403        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
404        while (!fresh.isEmpty() && spreadPattern.size() < volume) {
405            Coord cell = fresh.randomItem(rng);//toArray(new Coord[fresh.size()])[rng.nextInt(fresh.size())];
406            spreadPattern.add(cell);
407            spillMap[cell.x][cell.y] = true;
408            for (int d = 0; d < dirs.length; d++) {
409                Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
410                double h = heuristic(dirs[d]);
411                if (physicalMap[adj.x][adj.y] && !spillMap[adj.x][adj.y] && !impassable.contains(adj) && rng.nextDouble() <= 1.0 / h) {
412                    setFresh(adj);
413                }
414            }
415            fresh.remove(cell);
416        }
417        filled = spreadPattern.size();
418        return spreadPattern;
419    }
420
421    private static final double root2 = Math.sqrt(2.0);
422    private double heuristic(Direction target) {
423        switch (measurement) {
424            case MANHATTAN:
425            case CHEBYSHEV:
426                return 1.0;
427            default:
428                switch (target) {
429                    case DOWN_LEFT:
430                    case DOWN_RIGHT:
431                    case UP_LEFT:
432                    case UP_RIGHT:
433                        return root2;
434                    default:
435                        return  1.0;
436                }
437        }
438    }

014/**
015 * Creates a dungeon in the style of the original Rogue game. It will always
016 * make a grid style of rooms where there are a certain number horizontally and
017 * vertically and it will link them only next to each other.
018 *
019 * This dungeon generator is based on a port of the rot.js version.
020 *
021 * @author hyakugei
022 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
023 */
024public class ClassicRogueMapGenerator implements IDungeonGenerator{

032        private int x, y, width, height, cellx, celly;
033        private final OrderedSet<ClassicRogueRoom> connections = new OrderedSet<>(4);
034
035        ClassicRogueRoom(int x, int y, int width, int height, int cellx, int celly) {
036            this.x = x;
037            this.y = y;
038            this.width = width;
039            this.height = height;
040            this.cellx = cellx;
041            this.celly = celly;
042        }
043
044        @Override
045        public int hashCode() {
046            int hash = 5;
047            hash = 89 * hash + cellx;
048            hash = 89 * hash + celly;
049            return hash;
050        }
051
052        @Override
053        public boolean equals(Object obj) {
054            if (obj == null) {
055                return false;
056            }
057            if (getClass() != obj.getClass()) {
058                return false;
059            }
060            final ClassicRogueRoom other = (ClassicRogueRoom) obj;
061            if (cellx != other.cellx) {
062                return false;
063            }
064            return celly == other.celly;
065        }
066    }

112    public ClassicRogueMapGenerator(int horizontalRooms, int verticalRooms, int dungeonWidth, int dungeonHeight,

114    {
115        this.rng = rng;
116        this.horizontalRooms = horizontalRooms;
117        this.verticalRooms = verticalRooms;
118        this.dungeonWidth = dungeonWidth;
119        this.dungeonHeight = dungeonHeight;
120        this.minRoomWidth = minRoomWidth;
121        this.maxRoomWidth = maxRoomWidth;
122        this.minRoomHeight = minRoomHeight;
123        this.maxRoomHeight = maxRoomHeight;
124
125        sanitizeRoomDimensions();
126    }

128    private void sanitizeRoomDimensions() {
129        int test = (dungeonWidth - 3 * horizontalRooms) / horizontalRooms;//have to leave space for hallways
130        maxRoomWidth = Math.min(test, maxRoomWidth);
131        minRoomWidth = Math.max(minRoomWidth, 2);
132        minRoomWidth = Math.min(minRoomWidth, maxRoomWidth);
133
134        test = (dungeonHeight - 3 * verticalRooms) / verticalRooms;//have to leave space for hallways
135        maxRoomHeight = Math.min(test, maxRoomHeight);
136        minRoomHeight = Math.max(minRoomHeight, 2);
137        minRoomHeight = Math.min(minRoomHeight, maxRoomHeight);
138    }

140    /**
141     * Builds and returns a map in the Classic Rogue style.
142     * <br>
143     * Only includes rooms, corridors and doors.
144     * <br>
145     * There is also a generate method that produces a 2D char array, which may be more suitable.
146     * @return a 2D array of Terrain objects
147     */
148    public Terrain[][] create() {
149        initRooms();
150        connectRooms();
151        connectUnconnectedRooms();
152        fullyConnect();
153        createRooms();
154        createCorridors();
155        return map;
156    }
157    /**
158     * Builds and returns a map in the Classic Rogue style, returned as a 2D char array.
159     *
160     * Only includes rooms ('.' for floor and '#' for walls), corridors (using the same chars as rooms) and doors ('+'
161     * for closed doors, does not generate open doors).
162     * <br>
163     * There is also a create method that produces a 2D array of Terrain objects, which could (maybe) be what you want.
164     * More methods in SquidLib expect char 2D arrays than Terrain anything, particularly in DungeonUtility.
165     * @return a 2D char array version of the map
166     */
167    public char[][] generate()
168    {
169        create();
170        if(map.length <= 0)
171            return new char[0][0];
172        if(dungeon == null || dungeon.length != map.length || dungeon[0].length != map[0].length)
173            dungeon = new char[map.length][map[0].length];
174        for (int x = 0; x < map.length; x++) {
175            for (int y = 0; y < map[x].length; y++) {
176                dungeon[x][y] = map[x][y].symbol();
177            }
178        }
179        return dungeon;
180    }
181
182    public char[][] getDungeon() {
183        return dungeon;
184    }

186    private void initRooms() {
187        rooms = new ClassicRogueRoom[horizontalRooms][verticalRooms];
188        map = new Terrain[dungeonWidth][dungeonHeight];
189        for (int x = 0; x < horizontalRooms; x++) {
190            for (int y = 0; y < verticalRooms; y++) {
191                rooms[x][y] = new ClassicRogueRoom(0, 0, 0, 0, x, y);
192            }
193        }
194        for (int x = 0; x < dungeonWidth; x++) {
195            for (int y = 0; y < dungeonHeight; y++) {
196                map[x][y] = Terrain.WALL;
197            }
198        }
199    }

201    private void connectRooms() {
202        List<ClassicRogueRoom> unconnected = new LinkedList<>();
203        for (int x = 0; x < horizontalRooms; x++) {
204            for (int y = 0; y < verticalRooms; y++) {
205                unconnected.add(rooms[x][y]);
206            }
207        }
208        unconnected = rng.shuffle(unconnected);
209
210        Direction[]  dirToCheck = new Direction[4];
211        for (ClassicRogueRoom room : unconnected) {
212            rng.shuffle(Direction.CARDINALS, dirToCheck);
213            for (Direction dir : dirToCheck) {
214                int nextX = room.x + dir.deltaX;
215                int nextY = room.y + dir.deltaY;
216                if (nextX < 0 || nextX >= horizontalRooms || nextY < 0 || nextY >= verticalRooms) {
217                    continue;
218                }
219                ClassicRogueRoom otherRoom = rooms[nextX][nextY];
220
221                if (room.connections.contains(otherRoom)) {
222                    break;//already connected to this room
223                }
224
225                if (!otherRoom.connections.isEmpty()) {
226                    room.connections.add(otherRoom);
227                    break;
228                }
229            }
230        }
231    }
232
233    private void connectUnconnectedRooms() {
234        for (int x = 0; x < horizontalRooms; x++) {
235            for (int y = 0; y < verticalRooms; y++) {
236                ClassicRogueRoom room = rooms[x][y];
237
238                if (room.connections.isEmpty()) {
239                    List<Direction> dirToCheck = Arrays.asList(Direction.CARDINALS);
240                    dirToCheck = rng.shuffle(dirToCheck);
241
242                    boolean validRoom = false;
243                    ClassicRogueRoom otherRoom = null;
244
245                    do {
246                        Direction dir = dirToCheck.remove(0);
247
248                        int nextX = x + dir.deltaX;
249                        if (nextX < 0 || nextX >= horizontalRooms) {
250                            continue;
251                        }
252                        int nextY = y + dir.deltaY;
253                        if (nextY < 0 || nextY >= verticalRooms) {
254                            continue;
255                        }
256
257                        otherRoom = rooms[nextX][nextY];
258                        validRoom = true;
259
260                        if (otherRoom.connections.contains(room)) {
261                            validRoom = false;
262                        }
263                        else {
264                            break;
265                        }
266                    } while (!dirToCheck.isEmpty());
267
268                    if (validRoom) {
269                        room.connections.add(otherRoom);
270                    }
271                }
272            }
273        }
274    }

276    private void fullyConnect() {
277        boolean allGood;
278        do {
279            LinkedList<ClassicRogueRoom> deq = new LinkedList<>();
280            for (int x = 0; x < horizontalRooms; x++) {
281                for (int y = 0; y < verticalRooms; y++) {
282                    deq.offer(rooms[x][y]);
283                }
284            }
285            LinkedList<ClassicRogueRoom> connected = new LinkedList<>();
286            connected.add(deq.removeFirst());
287            boolean changed = true;
288            testing:
289            while (changed) {
290                changed = false;
291                for (ClassicRogueRoom test : deq) {
292                    for (ClassicRogueRoom r : connected) {
293                        if (test.connections.contains(r) || r.connections.contains(test)) {
294                            connected.offer(test);
295                            deq.remove(test);
296                            changed = true;
297                            continue testing;
298                        }
299                    }
300                }
301            }
302
303            allGood = true;
304            if (!deq.isEmpty()) {
305                testing:
306                for (ClassicRogueRoom room : deq) {
307                    for (Direction dir : Direction.CARDINALS) {
308                        int x = room.cellx + dir.deltaX;
309                        int y = room.celly + dir.deltaY;
310                        if (x >= 0 && y >= 0 && x < horizontalRooms && y < verticalRooms) {
311                            ClassicRogueRoom otherRoom = rooms[x][y];
312                            if (connected.contains(otherRoom)) {
313                                room.connections.add(otherRoom);
314                                allGood = false;
315                                break testing;
316                            }
317                        }
318                    }
319                }
320            }
321
322        } while (!allGood);
323    }

325    private void createRooms() {
326        int cwp = dungeonWidth / horizontalRooms;
327        int chp = dungeonHeight / verticalRooms;
328
329        ClassicRogueRoom otherRoom;
330
331        for (int x = 0; x < horizontalRooms; x++) {
332            for (int y = 0; y < verticalRooms; y++) {
333                int sx = cwp * x;
334                int sy = chp * y;
335
336                sx = Math.max(sx, 2);
337                sy = Math.max(sy, 2);
338
339                int roomw = rng.between(minRoomWidth, maxRoomWidth + 1);
340                int roomh = rng.between(minRoomHeight, maxRoomHeight + 1);
341
342                if (y > 0) {
343                    otherRoom = rooms[x][y - 1];
344                    while (sy - (otherRoom.y + otherRoom.height) < 3) {
345                        sy++;
346                    }
347                }
348
349                if (x > 0) {
350                    otherRoom = rooms[x - 1][y];
351                    while (sx - (otherRoom.x + otherRoom.width) < 3) {
352                        sx++;
353                    }
354                }
355
356                int sxOffset = Math.round(rng.nextInt(cwp - roomw) / 2);
357                int syOffset = Math.round(rng.nextInt(chp - roomh) / 2);
358
359                while (sx + sxOffset + roomw >= dungeonWidth) {
360                    if (sxOffset > 0) {
361                        sxOffset--;
362                    } else {
363                        roomw--;
364                    }
365                }
366                while (sy + syOffset + roomh >= dungeonHeight) {
367                    if (syOffset > 0) {
368                        syOffset--;
369                    } else {
370                        roomh--;
371                    }
372                }
373
374                sx += sxOffset;
375                sy += syOffset;
376
377                ClassicRogueRoom r = rooms[x][y];
378                r.x = sx;
379                r.y = sy;
380                r.width = roomw;
381                r.height = roomh;
382
383                for (int xx = sx; xx < sx + roomw; xx++) {
384                    for (int yy = sy; yy < sy + roomh; yy++) {
385                        map[xx][yy] = Terrain.FLOOR;
386                    }
387                }
388            }
389        }
390    }

392    private Coord randomWallPosition(ClassicRogueRoom room, Direction dir) {
393        int x, y;
394        Coord p = null;
395
396        switch (dir) {
397            case LEFT:
398                y = rng.between(room.y + 1, room.y + room.height);
399                x = room.x - 1;
400                map[x][y] = Terrain.CLOSED_DOOR;
401                p = Coord.get(x - 1, y);
402                break;
403            case RIGHT:
404                y = rng.between(room.y + 1, room.y + room.height);
405                x = room.x + room.width;
406                map[x][y] = Terrain.CLOSED_DOOR;
407                p = Coord.get(x + 1, y);
408                break;
409            case UP:
410                x = rng.between(room.x + 1, room.x + room.width);
411                y = room.y - 1;
412                map[x][y] = Terrain.CLOSED_DOOR;
413                p = Coord.get(x, y - 1);
414                break;
415            case DOWN:
416                x = rng.between(room.x + 1, room.x + room.width);
417                y = room.y + room.height;
418                map[x][y] = Terrain.CLOSED_DOOR;
419                p = Coord.get(x, y + 1);
420                break;
421        case NONE:
422                break;
423                case DOWN_LEFT:
424                case DOWN_RIGHT:
425                case UP_LEFT:
426                case UP_RIGHT:
427                        throw new IllegalStateException("There should only be cardinal positions here");
428        }
429
430        return p;
431    }

433    /**
434     * Draws a corridor between the two points with a zig-zag in between.
435     *
436     * @param start
437     * @param end
438     */
439    private void digPath(Coord start, Coord end) {
440        int xOffset = end.x - start.x;
441        int yOffset = end.y - start.y;
442        int xpos = start.x;
443        int ypos = start.y;
444
445        List<Magnitude> moves = new LinkedList<>();
446
447        int xAbs = Math.abs(xOffset);
448        int yAbs = Math.abs(yOffset);
449
450        double firstHalf = rng.nextDouble();
451        double secondHalf = 1 - firstHalf;
452
453        Direction xDir = xOffset < 0 ? Direction.LEFT : Direction.RIGHT;
454        Direction yDir = yOffset > 0 ? Direction.DOWN : Direction.UP;
455
456        if (xAbs < yAbs) {
457            int tempDist = (int) Math.ceil(yAbs * firstHalf);
458            moves.add(new Magnitude(yDir, tempDist));
459            moves.add(new Magnitude(xDir, xAbs));
460            tempDist = (int) Math.floor(yAbs * secondHalf);
461            moves.add(new Magnitude(yDir, tempDist));
462        } else {
463            int tempDist = (int) Math.ceil(xAbs * firstHalf);
464            moves.add(new Magnitude(xDir, tempDist));
465            moves.add(new Magnitude(yDir, yAbs));
466            tempDist = (int) Math.floor(xAbs * secondHalf);
467            moves.add(new Magnitude(xDir, tempDist));
468        }
469
470        map[xpos][ypos] = Terrain.FLOOR;
471
472        while (!moves.isEmpty()) {
473            Magnitude move = moves.remove(0);
474            Direction dir = move.dir;
475            int dist = move.distance;
476            while (dist > 0) {
477                xpos += dir.deltaX;
478                ypos += dir.deltaY;
479                map[xpos][ypos] = Terrain.FLOOR;
480                dist--;
481            }
482        }
483    }

485    private void createCorridors() {
486        for (int x = 0; x < horizontalRooms; x++) {
487            for (int y = 0; y < verticalRooms; y++) {
488                ClassicRogueRoom room = rooms[x][y];
489                for (ClassicRogueRoom otherRoom : room.connections) {
490                    Direction dir = Direction.getCardinalDirection(otherRoom.cellx - room.cellx, otherRoom.celly - room.celly);
491                    digPath(randomWallPosition(room, dir), randomWallPosition(otherRoom, dir.opposite()));
492                }
493            }
494        }
495    }

497    private class Magnitude {

499        public Direction dir;
500        public int distance;

502        public Magnitude(Direction dir, int distance) {

504            this.distance = distance;
505        }

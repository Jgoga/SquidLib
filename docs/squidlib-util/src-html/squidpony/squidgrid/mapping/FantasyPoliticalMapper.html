012/**
013 * When you have a world map as produced by {@link WorldMapGenerator}, you may want to fill it with claims by various
014 * factions, where each faction may be hand-made and may consist of humans or some fantasy species, such as goblins,
015 * elves, or demons. This can assign contiguous areas of land to various factions, while acknowledging any preferences
016 * some species may have for specific types of land (elves may strongly prefer forest terrain, or flying demons may be
017 * the ideal residents for difficult mountainous terrain). This needs both a {@link WorldMapGenerator}, preferably
018 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.TilingMap} or
019 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.EllipticalMap} because they distribute area equally across the
020 * map, and a {@link squidpony.squidgrid.mapping.WorldMapGenerator.BiomeMapper} to allocate biomes and height/moisture
021 * info.
022 */
023@Beta
024public class FantasyPoliticalMapper implements Serializable {

148    /**
149     * @param wmg a WorldMapGenerator, which must have produced a map by calling its generate() method
150     * @param biomeMapper a WorldMapGenerator.BiomeMapper, which must have been initialized with wmg and refer to the same world
151     * @param factionCount the number of factions to have claiming land, cannot be negative or more than 253
152     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
153     */
154    public char[][] generate(long seed, WorldMapGenerator wmg, WorldMapGenerator.BiomeMapper biomeMapper,
155                                  Collection<Faction> factions, int factionCount, double controlledFraction) {
156        rng.setState(seed);
157        factionCount = Math.abs(factionCount % 253);
158        Thesaurus th = new Thesaurus(rng.nextLong());
159        th.addKnownCategories();
160        ArrayList<Faction> fact = factions == null ? new ArrayList<Faction>() : rng.shuffle(factions);
161        for (int i = fact.size(); i < factionCount; i++) {
162            String name = th.makeNationName(), shortName = th.latestGenerated;
163            FakeLanguageGen lang;
164            if(th.randomLanguages == null || th.randomLanguages.isEmpty())
165                lang = FakeLanguageGen.randomLanguage(rng);
166            else
167                lang = th.randomLanguages.get(0);
168            fact.add(new Faction(lang, name, shortName));
169        }
170        if(factionCount > 0)
171            rng.shuffleInPlace(fact);
172        fact.add(0, new Faction(FakeLanguageGen.DEMONIC, "The Lost Wilderness", "Wilderness"));
173        fact.add(0, new Faction(FakeLanguageGen.ALIEN_O, "The Vast Domain of the Seafolk", "Seafolk", new String[]{"Ocean"}));
174        atlas = new OrderedMap<>(letters.subList(0, factionCount + 2), fact.subList(0, factionCount + 2), 0.5f);
175        this.wmg = wmg;
176        this.biomeMapper = biomeMapper;
177        width = wmg.width;
178        height = wmg.height;
179        GreasedRegion land = new GreasedRegion(wmg.heightCodeData, 4, 999);
180        politicalMap = land.toChars('%', '~');
181        int controlled = (int) (land.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
182
183        int[] centers = land.copy().randomScatter(rng, (int) (Math.sqrt(width * height) * 0.1 + 0.999), factionCount).asTightEncoded();
184        int cen, cx, cy, cx2, cy2, biome, high, hot, moist, count = centers.length, re;
185        String biomeName;
186        String[] biomeTable = biomeMapper.getBiomeNameTable();
187        int[] reorder = new int[count];
188        Faction current;
189        int[] factionIndices = new int[count];
190        for (int c = 0; c < count; c++) {
191            cen = centers[c];
192            cx = cen % width;
193            cy = cen / width;
194            biome = biomeMapper.getBiomeCode(cx, cy);
195            biomeName = biomeTable[biome];
196            high = wmg.heightCodeData[cx][cy];
197            hot = biomeMapper.getHeatCode(cx, cy);
198            moist = biomeMapper.getMoistureCode(cx, cy);
199            rng.randomOrdering(count, reorder);
200            PER_FACTION:
201            for (int i = 0; i < count; i++) {
202                current = fact.get(re = reorder[i]);
203                if(current.preferredBiomes == null || current.preferredBiomes.contains(biomeName))
204                {
205                    factionIndices[c] = re;
206                    break;
207                }
208                if(current.blockedBiomes != null && current.blockedBiomes.contains(biomeName))
209                    continue;
210                if(current.preferredHeight != null)
211                {
212                    for (int j = 0; j < current.preferredHeight.length; j++) {
213                        if(high == current.preferredHeight[j])
214                        {
215                            factionIndices[c] = re;
216                            break PER_FACTION;
217                        }
218                    }
219                }
220                if(current.preferredHeat != null)
221                {
222                    for (int j = 0; j < current.preferredHeat.length; j++) {
223                        if(hot == current.preferredHeat[j])
224                        {
225                            factionIndices[c] = re;
226                            break PER_FACTION;
227                        }
228                    }
229                }
230                if(current.preferredMoisture != null)
231                {
232                    for (int j = 0; j < current.preferredMoisture.length; j++) {
233                        if(moist == current.preferredMoisture[j])
234                        {
235                            factionIndices[c] = re;
236                            break PER_FACTION;
237                        }
238                    }
239                }
240            }
241        }
242        //ArrayList<Coord> spillers = new ArrayList<>(entries); // centers
243        IntVLA[] fresh = new IntVLA[count];
244        int filled = 0;
245        boolean hasFresh = false;
246        int approximateArea = (controlled * 4) / (count * 3);
247        char[] keys = new char[count];
248        double[] biases = new double[count];
249        for (int i = 0; i < count; i++) {
250            fresh[i] = new IntVLA(approximateArea);
251            cen = centers[i];
252            fresh[i].add(cen);
253            cx = cen % width;
254            cy = cen / width;
255            politicalMap[cx][cy] = keys[i] = atlas.keyAt(factionIndices[i] + 2);
256            biases[i] = rng.nextDouble() * rng.nextDouble() + rng.nextDouble() + 0.03125;
257            hasFresh = true;
258        }
259        Direction[] dirs = Direction.CARDINALS;
260        IntVLA currentFresh;
261        GreasedRegion anySpillMap = new GreasedRegion(width, height),
262                anyFreshMap = new GreasedRegion(width, height);
263
264        while (hasFresh && filled < controlled) {
265            hasFresh = false;
266            for (int i = 0; i < count && filled < controlled; i++) {
267                currentFresh = fresh[i];
268                if (currentFresh.isEmpty())
269                    continue;
270                else
271                    hasFresh = true;
272                if (rng.nextDouble() < biases[i]) {
273                    int index = rng.nextIntHasty(currentFresh.size), cell = currentFresh.get(index);
274                    cx = cell % width;
275                    cy = cell / width;
276
277
278                    politicalMap[cx][cy] = keys[i];
279                    filled++;
280                    anySpillMap.insert(cx, cy);
281
282                    for (int d = 0; d < dirs.length; d++) {
283                        cx2 = wmg.wrapX(cx + dirs[d].deltaX, cy);
284                        cy2 = wmg.wrapY(cx, cy + dirs[d].deltaY);
285                        if (cx == cx2 && cy == cy2)
286                            continue;
287                        if (land.contains(cx2, cy2) && !anySpillMap.contains(cx2, cy2)) {
288                            if(!anyFreshMap.contains(cx2, cy2)) {
289                                currentFresh.add(cx2 + cy2 * width);
290                                anyFreshMap.insert(cx2, cy2);
291                            }
292                        }
293                    }
294                    currentFresh.removeIndex(index);
295                    anyFreshMap.remove(cx, cy);
296                }
297            }
298        }
299        return politicalMap;
300    }

008/**
009 * Generate dungeons based on a random, winding, looping path through 2D space. Uses techniques from MixedGenerator.
010 * Uses a Moore Curve, which is related to Hilbert Curves but loops back to its starting point, and stretches and
011 * distorts the grid to make sure a visual correlation isn't obvious. This supports the getEnvironment() method, which
012 * can be used in conjunction with RoomFinder to find where separate room, corridor, and cave areas have been placed.
013 * <br>
014 * To get a sense of what kinds of map this generates, you can look at a sample map on
015 * https://gist.github.com/tommyettinger/93b47048fc8a209a9712 , which also includes a snippet of Java code that can
016 * generate that map.
017 * <br>
018 * The name comes from a vivid dream I had about gigantic, multi-colored snakes that completely occupied a roguelike
019 * dungeon. Shortly after, I made the connection to the Australian mythology I'd heard about the Rainbow Serpent, which
020 * in some stories dug water-holes and was similarly gigantic.
021 * Created by Tommy Ettinger on 10/24/2015.
022 */
023public class SerpentMapGenerator {
024    private MixedGenerator mix;
025    private int[] columns, rows;
026    private RNG random;

030     * The intended purpose is to carve a long path that loops through the whole dungeon, while hopefully maximizing
031     * the amount of rooms the player encounters. You call the different carver-adding methods to affect what the
032     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
033     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
034     *
035     * @param width  the width of the final map in cells
036     * @param height the height of the final map in cells
037     * @param rng    an RNG object to use for random choices; this make a lot of random choices.
038     * @see MixedGenerator
039     */
040    public SerpentMapGenerator(int width, int height, RNG rng) {
041        this(width, height, rng, false);
042    }

044    /**
045     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
046     * The intended purpose is to carve a long path that loops through the whole dungeon, while hopefully maximizing
047     * the amount of rooms the player encounters. You call the different carver-adding methods to affect what the
048     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
049     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
050     *
051     * @param width       the width of the final map in cells
052     * @param height      the height of the final map in cells
053     * @param rng         an RNG object to use for random choices; this make a lot of random choices.
054     * @param symmetrical true if this should generate a bi-radially symmetric map, false for a typical map
055     * @see MixedGenerator
056     */
057    public SerpentMapGenerator(int width, int height, RNG rng, boolean symmetrical) {
058        if (width <= 2 || height <= 2)
059            throw new IllegalArgumentException("width and height must be greater than 2");
060        random = rng;
061        long columnAlterations = random.nextLong(0x1000000000000L);
062        float columnBase = width / (Long.bitCount(columnAlterations) + 48.0f);
063        long rowAlterations = random.nextLong(0x1000000000000L);
064        float rowBase = height / (Long.bitCount(rowAlterations) + 48.0f);
065
066        columns = new int[16];
067        rows = new int[16];
068        int csum = 0, rsum = 0;
069        long b = 7;
070        for (int i = 0; i < 16; i++, b <<= 3) {
071            columns[i] = csum + (int) (columnBase * 0.5f * (3 + Long.bitCount(columnAlterations & b)));
072            csum += (int) (columnBase * (3 + Long.bitCount(columnAlterations & b)));
073            rows[i] = rsum + (int) (rowBase * 0.5f * (3 + Long.bitCount(rowAlterations & b)));
074            rsum += (int) (rowBase * (3 + Long.bitCount(rowAlterations & b)));
075        }
076        int cs = width - csum;
077        int rs = height - rsum;
078        int cs2 = cs, rs2 = rs, cs3 = cs, rs3 = rs;
079        for (int i = 0; i <= 7; i++) {
080            cs2 = cs2 * i / 7;
081            rs2 = rs2 * i / 7;
082            columns[i] -= cs2;
083            rows[i] -= rs2;
084        }
085        for (int i = 15; i >= 8; i--) {
086            cs3 = cs3 * (i - 8) / 8;
087            rs3 = rs3 * (i - 8) / 8;
088            columns[i] += cs3;
089            rows[i] += rs3;
090        }
091
092        List<Coord> points = new ArrayList<>(80);
093        Coord temp;
094        for (int i = 0, m = random.nextInt(64), r; i < 256; r = random.between(4, 12), i += r, m += r) {
095            temp = CoordPacker.mooreToCoord(m);
096            points.add(Coord.get(columns[temp.x], rows[temp.y]));
097        }
098        points.add(points.get(0));
099        if (symmetrical) {
100            mix = new SymmetryDungeonGenerator(width, height, random,
101                    SymmetryDungeonGenerator.removeSomeOverlap(width, height, points));
102        } else
103            mix = new MixedGenerator(width, height, random, points);
104    }

123    /**
124     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
125     * The intended purpose is to carve a long path that loops through the whole dungeon, while hopefully maximizing
126     * the amount of rooms the player encounters. You call the different carver-adding methods to affect what the
127     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
128     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
129     *
130     * @param width           the width of the final map in cells
131     * @param height          the height of the final map in cells
132     * @param rng             an RNG object to use for random choices; this make a lot of random choices.
133     * @param branchingChance the chance from 0.0 to 1.0 that each room will branch at least once
134     * @param symmetrical     true if this should generate a bi-radially symmetric map, false for a typical map
135     * @see MixedGenerator
136     */
137    public SerpentMapGenerator(int width, int height, RNG rng, double branchingChance, boolean symmetrical) {
138        if (width <= 2 || height <= 2)
139            throw new IllegalArgumentException("width and height must be greater than 2");
140        random = rng;
141        long columnAlterations = random.nextLong(0x1000000000000L);
142        float columnBase = width / (Long.bitCount(columnAlterations) + 48.0f);
143        long rowAlterations = random.nextLong(0x1000000000000L);
144        float rowBase = height / (Long.bitCount(rowAlterations) + 48.0f);
145
146        columns = new int[16];
147        rows = new int[16];
148        int csum = 0, rsum = 0;
149        long b = 7;
150        for (int i = 0; i < 16; i++, b <<= 3) {
151            columns[i] = csum + (int) (columnBase * 0.5f * (3 + Long.bitCount(columnAlterations & b)));
152            csum += (int) (columnBase * (3 + Long.bitCount(columnAlterations & b)));
153            rows[i] = rsum + (int) (rowBase * 0.5f * (3 + Long.bitCount(rowAlterations & b)));
154            rsum += (int) (rowBase * (3 + Long.bitCount(rowAlterations & b)));
155        }
156        int cs = width - csum;
157        int rs = height - rsum;
158        int cs2 = cs, rs2 = rs, cs3 = cs, rs3 = rs;
159        for (int i = 0; i <= 7; i++) {
160            cs2 = cs2 * i / 7;
161            rs2 = rs2 * i / 7;
162            columns[i] -= cs2;
163            rows[i] -= rs2;
164        }
165        for (int i = 15; i >= 8; i--) {
166            cs3 = cs3 * (i - 8) / 8;
167            rs3 = rs3 * (i - 8) / 8;
168            columns[i] += cs3;
169            rows[i] += rs3;
170        }
171
172        OrderedMap<Coord, List<Coord>> connections = new OrderedMap<>(80);
173        Coord temp, t;
174        int m = random.nextInt(64), r = random.between(4, 12);
175        temp = CoordPacker.mooreToCoord(m);
176        Coord starter = CoordPacker.mooreToCoord(m);
177        m += r;
178        for (int i = r; i < 256; i += r, m += r) {
179            List<Coord> cl = new ArrayList<>(4);
180            cl.add(Coord.get(columns[temp.x], rows[temp.y]));
181            temp = CoordPacker.mooreToCoord(m);
182            r = random.between(4, 12);
183            for (int j = 0, p = r - 1;
184                 j < 3 && p > 2 && Math.min(random.nextDouble(), random.nextDouble()) < branchingChance;
185                 j++, p -= random.between(1, p)) {
186                t = CoordPacker.mooreToCoord(m + p);
187                cl.add(Coord.get(columns[t.x], rows[t.y]));
188            }
189            connections.put(Coord.get(columns[temp.x], rows[temp.y]), cl);
190        }
191        connections.get(Coord.get(columns[temp.x], rows[temp.y])).add(
192                Coord.get(columns[starter.x], rows[starter.y]));
193        if (symmetrical) {
194            mix = new SymmetryDungeonGenerator(width, height, random,
195                    SymmetryDungeonGenerator.removeSomeOverlap(width, height, connections));
196        } else
197            mix = new MixedGenerator(width, height, random, connections);
198
199    }

201    /**
202     * Changes the number of "carvers" that will create caves from one room to the next. If count is 0 or less, no caves
203     * will be made. If count is at least 1, caves are possible, and higher numbers relative to the other carvers make
204     * caves more likely. Carvers are shuffled when used, then repeat if exhausted during generation. Since typically
205     * about 30-40 rooms are carved, large totals for carver count aren't really needed; aiming for a total of 10
206     * between the count of putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers() is reasonable.
207     *
208     * @param count the number of carvers making caves between rooms; only matters in relation to other carvers
209     * @see MixedGenerator
210     */
211    public void putCaveCarvers(int count) {
212        mix.putCaveCarvers(count);
213    }

285    /**
286     * This generates a new map by stretching a 16x16 grid of potential rooms to fit the width and height passed to the
287     * constructor, randomly expanding columns and rows before contracting the whole to fit perfectly. This uses the
288     * Moore Curve, a space-filling curve that loops around on itself, to guarantee that the rooms will always have a
289     * long path through the dungeon that, if followed completely, will take you back to your starting room. Some small
290     * branches are possible, and large rooms may merge with other rooms nearby. This uses MixedGenerator.
291     *
292     * @return a char[][] where '#' is a wall and '.' is a floor or corridor; x first y second
293     * @see MixedGenerator
294     */
295    public char[][] generate() {
296        return mix.generate();
297    }

299    /**
300     * Gets a 2D array of int constants, each representing a type of environment corresponding to a static field of
301     * MixedGenerator. This array will have the same size as the last char 2D array produced by generate(); the value
302     * of this method if called before generate() is undefined, but probably will be a 2D array of all 0 (UNTOUCHED).
303     * <ul>
304     * <li>MixedGenerator.UNTOUCHED, equal to 0, is used for any cells that aren't near a floor.</li>
305     * <li>MixedGenerator.ROOM_FLOOR, equal to 1, is used for floor cells inside wide room areas.</li>
306     * <li>MixedGenerator.ROOM_WALL, equal to 2, is used for wall cells around wide room areas.</li>
307     * <li>MixedGenerator.CAVE_FLOOR, equal to 3, is used for floor cells inside rough cave areas.</li>
308     * <li>MixedGenerator.CAVE_WALL, equal to 4, is used for wall cells around rough cave areas.</li>
309     * <li>MixedGenerator.CORRIDOR_FLOOR, equal to 5, is used for floor cells inside narrow corridor areas.</li>
310     * <li>MixedGenerator.CORRIDOR_WALL, equal to 6, is used for wall cells around narrow corridor areas.</li>
311     * </ul>
312     *
313     * @return a 2D int array where each element is an environment type constant in MixedGenerator
314     */
315    public int[][] getEnvironment() {
316        return mix.getEnvironment();
317    }

319    public static ArrayList<Coord> pointPath(int width, int height, RNG rng) {
320        if (width <= 2 || height <= 2)
321            throw new IllegalArgumentException("width and height must be greater than 2");
322        long columnAlterations = rng.nextLong(0x1000000000000L);
323        float columnBase = width / (Long.bitCount(columnAlterations) + 48.0f);
324        long rowAlterations = rng.nextLong(0x1000000000000L);
325        float rowBase = height / (Long.bitCount(rowAlterations) + 48.0f);
326
327        int[] columns = new int[16], rows = new int[16];
328        int csum = 0, rsum = 0;
329        long b = 7;
330        for (int i = 0; i < 16; i++, b <<= 3) {
331            columns[i] = csum + (int) (columnBase * 0.5f * (3 + Long.bitCount(columnAlterations & b)));
332            csum += (int) (columnBase * (3 + Long.bitCount(columnAlterations & b)));
333            rows[i] = rsum + (int) (rowBase * 0.5f * (3 + Long.bitCount(rowAlterations & b)));
334            rsum += (int) (rowBase * (3 + Long.bitCount(rowAlterations & b)));
335        }
336        int cs = width - csum;
337        int rs = height - rsum;
338        int cs2 = cs, rs2 = rs, cs3 = cs, rs3 = rs;
339        for (int i = 0; i <= 7; i++) {
340            cs2 = cs2 * i / 7;
341            rs2 = rs2 * i / 7;
342            columns[i] -= cs2;
343            rows[i] -= rs2;
344        }
345        for (int i = 15; i >= 8; i--) {
346            cs3 = cs3 * (i - 8) / 8;
347            rs3 = rs3 * (i - 8) / 8;
348            columns[i] += cs3;
349            rows[i] += rs3;
350        }
351
352        ArrayList<Coord> points = new ArrayList<>(80);
353        int m = rng.nextInt(64);
354        Coord temp = CoordPacker.mooreToCoord(m), next;
355        temp = Coord.get(columns[temp.x], rows[temp.y]);
356        for (int i = 0, r; i < 256; r = rng.between(4, 12), i += r, m += r) {
357            next = CoordPacker.mooreToCoord(m);
358            next = Coord.get(columns[next.x], rows[next.y]);
359            points.addAll(OrthoLine.line(temp, next));
360            temp = next;
361        }
362        points.add(points.get(0));
363        return points;
364    }

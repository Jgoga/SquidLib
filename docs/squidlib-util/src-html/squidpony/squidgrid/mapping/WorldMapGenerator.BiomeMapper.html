011/**
012 * Can be used to generate world maps with a wide variety of data, starting with height, temperature and moisture.
013 * From there, you can determine biome information in as much detail as your game needs, with a default implementation
014 * available that assigns a single biome to each cell based on heat/moisture. The maps this produces are valid
015 * for spherical or toroidal world projections, and will wrap from edge to opposite edge seamlessly thanks to a
016 * technique from the Accidental Noise Library ( https://www.gamedev.net/blog/33/entry-2138456-seamless-noise/ ) that
017 * involves getting a 2D slice of 4D Simplex noise. Because of how Simplex noise works, this also allows extremely high
018 * zoom levels as long as certain parameters are within reason. You can access the height map with the
019 * {@link #heightData} field, the heat map with the {@link #heatData} field, the moisture map with the
020 * {@link #moistureData} field, and a special map that stores ints representing the codes for various ranges of
021 * elevation (0 to 8 inclusive, with 0 the deepest ocean and 8 the highest mountains) with {@link #heightCodeData}. The
022 * last map should be noted as being the simplest way to find what is land and what is water; any height code 4 or
023 * greater is land, and any height code 3 or less is water. This can produce rivers, and keeps that information in a
024 * GreasedRegion (alongside a GreasedRegion containing lake positions) instead of in the other map data. This class does
025 * not use Coord at all, but if you want maps with width and/or height greater than 256, and you want to use the river
026 * or lake data as a Collection of Coord, then you should call {@link Coord#expandPoolTo(int, int)}
027 * with your width and height so the Coords remain safely pooled. If you're fine with keeping rivers and lakes as
028 * GreasedRegions and not requesting Coord values from them, then you don't need to do anything with Coord. Certain
029 * parts of this class are not necessary to generate, just in case you want river-less maps or something similar;
030 * setting {@link #generateRivers} to false will disable river generation (it defaults to true).
031 * <br>
032 * The main trade-off this makes to obtain better quality is reduced speed; generating a 512x512 map on a circa-2016
033 * laptop (i7-6700HQ processor at 2.6 GHz) takes about 1 second (about 1.15 seconds for an un-zoomed map, 0.95 or so
034 * seconds to increase zoom at double resolution). If you don't need a 512x512 map, this takes commensurately less time
035 * to generate less grid cells, with 64x64 maps generating faster than they can be accurately seen on the same hardware.
036 * River positions are produced using a different method, and do not involve the Simplex noise parts other than using
037 * the height map to determine flow. Zooming with rivers is tricky, and generally requires starting from the outermost
038 * zoom level and progressively enlarging and adding detail to all rivers as zoom increases on specified points.
039 */

154    /**
155     * Generates a world using the specified RNG state as a long, with specific water and cooling modifiers that affect
156     * the land-water ratio and the average temperature, respectively.
157     * The worlds this produces will always have width and height as specified in the constructor (default 256x256).
158     * You can call {@link #zoomIn(int, int, int)} to double the resolution and center on the specified area, but the width
159     * and height of the 2D arrays this changed, such as {@link #heightData} and {@link #moistureData} will be the same.
160     * @param waterMod should be between 0.85 and 1.2; a random value will be used if this is negative
161     * @param coolMod should be between 0.85 and 1.4; a random value will be used if this is negative
162     * @param state the state to give this generator's RNG; if the same as the last call, this will reuse data
163     */
164    public void generate(double waterMod, double coolMod, long state)
165    {
166        if(cachedState != state || waterMod != waterModifier || coolMod != coolingModifier)
167        {
168            seed = state;
169            zoom = 0;
170            startCacheX.clear();
171            startCacheY.clear();
172            startCacheX.add(0);
173            startCacheY.add(0);
174            zoomStartX = width >> 1;
175            zoomStartY = height >> 1;
176
177        }
178        //System.out.printf("generate, zoomStartX: %d, zoomStartY: %d\n", zoomStartX, zoomStartY);
179
180        regenerate(startX = (zoomStartX >> zoom) - (width >> 1 + zoom), startY = (zoomStartY >> zoom) - (height >> 1 + zoom),
181                //startCacheX.peek(), startCacheY.peek(),
182                usedWidth = (width >> zoom), usedHeight = (height >> zoom), waterMod, coolMod, state);
183    }

186     * Halves the resolution of the map and doubles the area it covers; the 2D arrays this uses keep their sizes. This
187     * version of zoomOut always zooms out from the center of the currently used area.
188     * <br>
189     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
190     */
191    public void zoomOut()
192    {
193        zoomOut(1, width >> 1, height >> 1);
194    }
195    /**
196     * Halves the resolution of the map and doubles the area it covers repeatedly, halving {@code zoomAmount} times; the
197     * 2D arrays this uses keep their sizes. This version of zoomOut allows you to specify where the zoom should be
198     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
199     * 255, and will refer to the currently used area and not necessarily the full world size).
200     * <br>
201     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
202     * @param zoomCenterX the center X position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
203     * @param zoomCenterY the center Y position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
204     */
205    public void zoomOut(int zoomAmount, int zoomCenterX, int zoomCenterY)
206    {
207        zoomAmount = Math.min(zoom, zoomAmount);
208        if(zoomAmount == 0) return;
209        if(zoomAmount < 0) {
210            zoomIn(-zoomAmount, zoomCenterX, zoomCenterY);
211            return;
212        }
213        if(zoom > 0)
214        {
215            if(seed != cachedState)
216            {
217                generate(rng.nextLong());
218            }
219            zoomStartX = Math.min(Math.max(
220                    (zoomStartX + (zoomCenterX - (width >> 1))) >> zoomAmount,
221                    width >> 1), (width << zoom - zoomAmount) - (width >> 1));
222            zoomStartY = Math.min(Math.max(
223                    (zoomStartY + (zoomCenterY - (height >> 1))) >> zoomAmount,
224                    height >> 1), (height << zoom - zoomAmount) - (height >> 1));
225//            System.out.printf("zoomOut, zoomStartX: %d, zoomStartY: %d\n", zoomStartX, zoomStartY);
226            zoom -= zoomAmount;
227            startCacheX.pop();
228            startCacheY.pop();
229            startCacheX.add(Math.min(Math.max(startCacheX.pop() + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
230                    0), width - (width >> zoom)));
231            startCacheY.add(Math.min(Math.max(startCacheY.pop() + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
232                    0), height - (height >> zoom)));
233//            zoomStartX = Math.min(Math.max((zoomStartX >> 1) + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
234//                    0), width - (width >> zoom));
235//            zoomStartY = Math.min(Math.max((zoomStartY >> 1) + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
236//                    0), height - (height >> zoom));
237            regenerate(startX = (zoomStartX >> zoom) - (width >> zoom + 1), startY = (zoomStartY >> zoom) - (height >> zoom + 1),
238                    //startCacheX.peek(), startCacheY.peek(),
239                    usedWidth = width >> zoom,  usedHeight = height >> zoom,
240                    waterModifier, coolingModifier, cachedState);
241            rng.setState(cachedState);
242        }
243
244    }
245    /**
246     * Doubles the resolution of the map and halves the area it covers; the 2D arrays this uses keep their sizes. This
247     * version of zoomIn always zooms in to the center of the currently used area.
248     * <br>
249     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
250     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
251     * normal landscape.
252     */
253    public void zoomIn()
254    {
255        zoomIn(1, width >> 1, height >> 1);
256    }
257    /**
258     * Doubles the resolution of the map and halves the area it covers repeatedly, doubling {@code zoomAmount} times;
259     * the 2D arrays this uses keep their sizes. This version of zoomIn allows you to specify where the zoom should be
260     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
261     * 255, and will refer to the currently used area and not necessarily the full world size).
262     * <br>
263     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
264     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
265     * normal landscape.
266     * @param zoomCenterX the center X position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
267     * @param zoomCenterY the center Y position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
268     */
269    public void zoomIn(int zoomAmount, int zoomCenterX, int zoomCenterY)
270    {
271        if(zoomAmount == 0) return;
272        if(zoomAmount < 0)
273        {
274            zoomOut(-zoomAmount, zoomCenterX, zoomCenterY);
275            return;
276        }
277        if(seed != cachedState)
278        {
279            generate(rng.nextLong());
280        }
281        zoomStartX = Math.min(Math.max(
282                (zoomStartX + zoomCenterX - (width >> 1) << zoomAmount),
283                width >> 1), (width << zoom + zoomAmount) - (width >> 1));
284//        int oldZoomY = zoomStartY;
285        zoomStartY = Math.min(Math.max(
286                (zoomStartY + zoomCenterY - (height >> 1) << zoomAmount),
287                height >> 1), (height << zoom + zoomAmount) - (height >> 1));
288//        System.out.printf("zoomIn, zoomStartX: %d, zoomStartY: %d, oldZoomY: %d, unedited: %d, upperCap: %d\n", zoomStartX, zoomStartY,
289//                oldZoomY, (oldZoomY + zoomCenterY - (height >> 1) << zoomAmount), (height << zoom + zoomAmount) - (height >> 1));
290        zoom += zoomAmount;
291        if(startCacheX.isEmpty())
292        {
293            startCacheX.add(0);
294            startCacheY.add(0);
295        }
296        else {
297            startCacheX.add(Math.min(Math.max(startCacheX.peek() + (zoomCenterX >> zoom - 1) - (width >> zoom + 1),
298                    0), width - (width >> zoom)));
299            startCacheY.add(Math.min(Math.max(startCacheY.peek() + (zoomCenterY >> zoom - 1) - (height >> zoom + 1),
300                    0), height - (height >> zoom)));
301        }
302        regenerate(startX = (zoomStartX >> zoom) - (width >> 1 + zoom), startY = (zoomStartY >> zoom) - (height >> 1 + zoom),
303                //startCacheX.peek(), startCacheY.peek(),
304                usedWidth = width >> zoom, usedHeight = height >> zoom,
305                waterModifier, coolingModifier, cachedState);
306        rng.setState(cachedState);
307    }

312    public int codeHeight(final double high)
313    {
314        if(high < deepWaterUpper)
315            return 0;
316        if(high < mediumWaterUpper)
317            return 1;
318        if(high < shallowWaterUpper)
319            return 2;
320        if(high < coastalWaterUpper)
321            return 3;
322        if(high < sandUpper)
323            return 4;
324        if(high < grassUpper)
325            return 5;
326        if(high < forestUpper)
327            return 6;
328        if(high < rockUpper)
329            return 7;
330        return 8;
331    }

776        @Override
777        public void makeBiomes(WorldMapGenerator world) {
778            if(world == null || world.width <= 0 || world.height <= 0)
779                return;
780            if(heatCodeData == null || (heatCodeData.length != world.width || heatCodeData[0].length != world.height))
781                heatCodeData = new int[world.width][world.height];
782            if(moistureCodeData == null || (moistureCodeData.length != world.width || moistureCodeData[0].length != world.height))
783                moistureCodeData = new int[world.width][world.height];
784            if(biomeCodeData == null || (biomeCodeData.length != world.width || biomeCodeData[0].length != world.height))
785                biomeCodeData = new int[world.width][world.height];
786            final double i_hot = (world.maxHeat == world.minHeat) ? 1.0 : 1.0 / (world.maxHeat - world.minHeat);
787            for (int x = 0; x < world.width; x++) {
788                for (int y = 0; y < world.height; y++) {
789                    final double hot = (world.heatData[x][y] - world.minHeat) * i_hot, moist = world.moistureData[x][y],
790                            fresh = world.freshwaterData[x][y];
791                    final int heightCode = world.heightCodeData[x][y];
792                    if(heightCode == 1000) {
793                        biomeCodeData[x][y] = 60;
794                        continue;
795                    }
796                    int hc, mc;
797                    boolean isLake = world.generateRivers && heightCode >= 4 && fresh > 0.65 && fresh + moist * 2.35 > 2.75,//world.partialLakeData.contains(x, y) && heightCode >= 4,
798                            isRiver = world.generateRivers && !isLake && heightCode >= 4 && fresh > 0.55 && fresh + moist * 2.2 > 2.15;//world.partialRiverData.contains(x, y) && heightCode >= 4;
799                    if(heightCode < 4) {
800                        mc = 9;
801                    }
802                    else if (moist > wetterValueUpper) {
803                        mc = 5;
804                    } else if (moist > wetValueUpper) {
805                        mc = 4;
806                    } else if (moist > dryValueUpper) {
807                        mc = 3;
808                    } else if (moist > drierValueUpper) {
809                        mc = 2;
810                    } else if (moist > driestValueUpper) {
811                        mc = 1;
812                    } else {
813                        mc = 0;
814                    }
815
816                    if (hot > warmerValueUpper) {
817                        hc = 5;
818                    } else if (hot > warmValueUpper) {
819                        hc = 4;
820                    } else if (hot > coldValueUpper) {
821                        hc = 3;
822                    } else if (hot > colderValueUpper) {
823                        hc = 2;
824                    } else if (hot > coldestValueUpper) {
825                        hc = 1;
826                    } else {
827                        hc = 0;
828                    }
829
830                    heatCodeData[x][y] = hc;
831                    moistureCodeData[x][y] = mc;
832                    biomeCodeData[x][y] = heightCode < 4 ? hc + 54 // 54 == 9 * 6, 9 is used for Ocean groups
833                            : isLake ? hc + 48 : (isRiver ? hc + 42 : ((heightCode == 4) ? hc + 36 : hc + mc * 6));
834                }
835            }

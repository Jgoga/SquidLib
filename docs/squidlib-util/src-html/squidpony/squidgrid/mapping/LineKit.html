096    /**
097     * Produces a 4x4 2D char array by interpreting the bits of the given long as line information. Uses the box drawing
098     * chars from {@link #light}, which are compatible with most fonts.
099     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
100     * @return a 4x4 2D char array containing elements from symbols assigned based on encoded
101     */
102    public static char[][] decode4x4(long encoded)
103    {
104        return decode4x4(encoded, light);
105    }
106    /**
107     * Produces a 4x4 2D char array by interpreting the bits of the given long as line information. Uses the given char
108     * array, which must have at least 16 elements and is usually one of {@link #light}, {@link #heavy},
109     * {@link #lightAlt}, or {@link #heavyAlt}, with the last two usable only if using a font that supports the chars
110     * {@code ╴╵╶╷} (this is true for Iosevka and Source Code Pro, for instance, but not Inconsolata or GoMono).
111     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
112     * @param symbols a 16-element-or-larger char array; usually a constant in this class like {@link #light}
113     * @return a 4x4 2D char array containing elements from symbols assigned based on encoded
114     */
115    public static char[][] decode4x4(long encoded, char[] symbols)
116    {
117        char[][] v = new char[4][4];
118        for (int i = 0; i < 16; i++) {
119            v[i & 3][i >> 2] = symbols[(int) (encoded >>> (i << 2) & 15L)];
120        }
121        return v;
122    }
123    /**
124     * Fills a 4x4 area of the given 2D char array {@code into} by interpreting the bits of the given long as line
125     * information. Uses the given char array {@code symbols}, which must have at least 16 elements and is usually one
126     * of {@link #light}, {@link #heavy}, {@link #lightAlt}, or {@link #heavyAlt}, with the last two usable only if
127     * using a font that supports the chars {@code ╴╵╶╷} (this is true for Iosevka and Source Code Pro, for instance,
128     * but not Inconsolata or GoMono).
129     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
130     * @param symbols a 16-element-or-larger char array; usually a constant in this class like {@link #light}
131     * @param into a 2D char array that will be modified in a 4x4 area
132     * @param startX the first x position to modify in into
133     * @param startY the first y position to modify in into
134     * @return into, after modification
135     */
136    public static char[][] decodeInto4x4(long encoded, char[] symbols, char[][] into, int startX, int startY)
137    {
138        for (int i = 0; i < 16; i++) {
139            into[(i & 3) + startX][(i >> 2) + startY] = symbols[(int) (encoded >>> (i << 2) & 15L)];
140        }
141        return into;
142    }

144    /**
145     * Reads a 2D char array {@code decoded}, which must be at least 4x4 in size, and returns a long that encodes the cells from 0,0 to
146     * 3,3 in a way that this class can interpret and manipulate. The 2D array {@code decoded} must contain box drawing
147     * symbols, which can be any of those from {@link #light}, {@link #heavy}, {@link #lightAlt}, or {@link #heavyAlt}.
148     * Valid chars are {@code ╴╵┘╶─└┴╷┐│┤┌┬├┼╸╹┛╺━┗┻╻┓┃┫┏┳┣╋}; any other chars will be treated as empty space.
149     * @param decoded a 2D char array that must be at least 4x4 and should usually contain box drawing characters
150     * @return a long that encodes the box drawing information in decoded so this class can manipulate it
151     */
152    public static long encode4x4(char[][] decoded)
153    {
154        long v = 0L;
155        for (int i = 0; i < 16; i++) {
156            switch (decoded[i & 3][i >> 2])
157            {
158                // ╴╵┘╶─└┴╷┐│┤┌┬├┼
159                // ╸╹┛╺━┗┻╻┓┃┫┏┳┣╋
160                //0123456789ABCDEF
161                case '─':
162                case '━':
163                    v |= 5L << (i << 2);
164                    break;
165                case '│':
166                case '┃':
167                    v |= 10L << (i << 2);
168                    break;
169                case '┘':
170                case '┛':
171                    v |= 3L << (i << 2);
172                    break;
173                case '└':
174                case '┗':
175                    v |= 6L << (i << 2);
176                    break;
177                case '┐':
178                case '┓':
179                    v |= 9L << (i << 2);
180                    break;
181                case '┌':
182                case '┏':
183                    v |= 12L << (i << 2);
184                    break;
185                case '┴':
186                case '┻':
187                    v |= 7L << (i << 2);
188                    break;
189                case '┤':
190                case '┫':
191                    v |= 11L << (i << 2);
192                    break;
193                case '┬':
194                case '┳':
195                    v |= 13L << (i << 2);
196                    break;
197                case '├':
198                case '┣':
199                    v |= 14L << (i << 2);
200                    break;
201                case '┼':
202                case '╋':
203                    v |= 15L << (i << 2);
204                    break;
205                case '╴':
206                case '╸':
207                    v |= 1L << (i << 2);
208                    break;
209                case '╵':
210                case '╹':
211                    v |= 2L << (i << 2);
212                    break;
213                case '╶':
214                case '╺':
215                    v |= 4L << (i << 2);
216                    break;
217                case '╷':
218                case '╻':
219                    v |= 8L << (i << 2);
220                    break;
221            }
222        }
223        return v;
224    }

226    /**
227     * Makes a variant on the given encoded 4x4 pattern so the left side is flipped to the right side and vice versa.
228     * @param encoded an encoded pattern long that represents a 4x4 area
229     * @return a different encoded pattern long that represents the argument flipped left-to-right
230     */
231    public static long flipHorizontal4x4(long encoded)
232    {
233        long v = 0L;
234        for (int i = 0, i4 = 0; i < 16; i++, i4 += 4) {
235            v |= ((encoded >>> i4 & 10L) | ((encoded >>> i4 & 1) << 2L) | ((encoded >>> i4 & 4L) >>> 2)) << (i + 3 - ((i & 3) << 1) << 2);
236        }
237        return v;
238    }
239    /**
240     * Makes a variant on the given encoded 4x4 pattern so the top side is flipped to the bottom side and vice versa.
241     * @param encoded an encoded pattern long that represents a 4x4 area
242     * @return a different encoded pattern long that represents the argument flipped top-to-bottom
243     */
244    public static long flipVertical4x4(long encoded)
245    {
246        long v = 0L;
247        for (int i = 0, i4 = 0; i < 16; i++, i4 += 4) {
248            v |= ((encoded >>> i4 & 5L) | ((encoded >>> i4 & 2L) << 2) | ((encoded >>> i4 & 8L) >>> 2)) << (i + 12 - ((i >> 2) << 3) << 2);
249        }
250        return v;
251    }
252
253    /**
254     * Makes a variant on the given encoded 4x4 pattern so the x and y axes are interchanged, making the top side become
255     * the left side and vice versa, while the bottom side becomes the right side and vice versa.
256     * @param encoded an encoded pattern long that represents a 4x4 area
257     * @return a different encoded pattern long that represents the argument transposed top-to-left and bottom-to-right
258     */
259    public static long transpose4x4(long encoded)
260    {
261        long v = 0L;
262        for (int i4 = 0; i4 < 64; i4 += 4) {
263            v |= (((encoded >>> i4 & 5L) << 1) | ((encoded >>> i4 & 10L) >>> 1)) << ((i4 >>> 2 & -4L) | ((i4 & 12) << 2));
264        }
265        return v;
266    }

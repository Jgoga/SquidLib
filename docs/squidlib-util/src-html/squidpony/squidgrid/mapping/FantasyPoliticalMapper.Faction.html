014/**
015 * When you have a world map as produced by {@link WorldMapGenerator}, you may want to fill it with claims by various
016 * factions, where each faction may be hand-made and may consist of humans or some fantasy species, such as goblins,
017 * elves, or demons. This can assign contiguous areas of land to various factions, while acknowledging any preferences
018 * some species may have for specific types of land (elves may strongly prefer forest terrain, or flying demons may be
019 * the ideal residents for difficult mountainous terrain). This needs both a {@link WorldMapGenerator}, preferably
020 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.TilingMap} or
021 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.EllipticalMap} because they distribute area equally across the
022 * map, and a {@link squidpony.squidgrid.mapping.WorldMapGenerator.BiomeMapper} to allocate biomes and height/moisture
023 * info.
024 */
025@Beta
026public class FantasyPoliticalMapper implements Serializable {

150    /**
151     * @param wmg a WorldMapGenerator, which must have produced a map by calling its generate() method
152     * @param biomeMapper a WorldMapGenerator.BiomeMapper, which must have been initialized with wmg and refer to the same world
153     * @param factionCount the number of factions to have claiming land, cannot be negative or more than 253
154     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
155     */
156    public char[][] generate(long seed, WorldMapGenerator wmg, WorldMapGenerator.BiomeMapper biomeMapper,
157                                  Collection<Faction> factions, int factionCount, double controlledFraction) {
158        rng.setState(seed);
159        factionCount = Math.abs(factionCount % 253);
160        Thesaurus th = new Thesaurus(rng.nextLong());
161        th.addKnownCategories();
162        ArrayList<Faction> fact = factions == null ? new ArrayList<Faction>() : rng.shuffle(factions);
163        for (int i = fact.size(); i < factionCount; i++) {
164            String name = th.makeNationName(), shortName = th.latestGenerated;
165            FakeLanguageGen lang;
166            if(th.randomLanguages == null || th.randomLanguages.isEmpty())
167                lang = FakeLanguageGen.randomLanguage(rng);
168            else
169                lang = th.randomLanguages.get(0);
170            fact.add(new Faction(lang, name, shortName));
171        }
172        if(factionCount > 0)
173            rng.shuffleInPlace(fact);
174        fact.add(0, new Faction(FakeLanguageGen.DEMONIC, "The Lost Wilderness", "Wilderness"));
175        fact.add(0, new Faction(FakeLanguageGen.ALIEN_O, "The Vast Domain of the Seafolk", "Seafolk", new String[]{"Ocean"}));
176        atlas = new OrderedMap<>(letters.subList(0, factionCount + 2), fact.subList(0, factionCount + 2), 0.5f);
177        this.wmg = wmg;
178        this.biomeMapper = biomeMapper;
179        width = wmg.width;
180        height = wmg.height;
181        GreasedRegion land = new GreasedRegion(wmg.heightCodeData, 4, 999);
182        politicalMap = land.toChars('%', '~');
183        int controlled = (int) (land.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
184
185        int[] centers = land.copy().randomScatter(rng, (int) (Math.sqrt(width * height) * 0.1 + 0.999), factionCount).asTightEncoded();
186        int cen, cx, cy, cx2, cy2, biome, high, hot, moist, count = centers.length, re;
187        String biomeName;
188        String[] biomeTable = biomeMapper.getBiomeNameTable();
189        int[] reorder = new int[count];
190        Faction current;
191        int[] factionIndices = new int[count];
192        for (int c = 0; c < count; c++) {
193            cen = centers[c];
194            cx = cen % width;
195            cy = cen / width;
196            biome = biomeMapper.getBiomeCode(cx, cy);
197            biomeName = biomeTable[biome];
198            high = wmg.heightCodeData[cx][cy];
199            hot = biomeMapper.getHeatCode(cx, cy);
200            moist = biomeMapper.getMoistureCode(cx, cy);
201            rng.randomOrdering(count, reorder);
202            PER_FACTION:
203            for (int i = 0; i < count; i++) {
204                current = fact.get(re = reorder[i]);
205                if(current.preferredBiomes == null || current.preferredBiomes.contains(biomeName))
206                {
207                    factionIndices[c] = re;
208                    break;
209                }
210                if(current.blockedBiomes != null && current.blockedBiomes.contains(biomeName))
211                    continue;
212                if(current.preferredHeight != null)
213                {
214                    for (int j = 0; j < current.preferredHeight.length; j++) {
215                        if(high == current.preferredHeight[j])
216                        {
217                            factionIndices[c] = re;
218                            break PER_FACTION;
219                        }
220                    }
221                }
222                if(current.preferredHeat != null)
223                {
224                    for (int j = 0; j < current.preferredHeat.length; j++) {
225                        if(hot == current.preferredHeat[j])
226                        {
227                            factionIndices[c] = re;
228                            break PER_FACTION;
229                        }
230                    }
231                }
232                if(current.preferredMoisture != null)
233                {
234                    for (int j = 0; j < current.preferredMoisture.length; j++) {
235                        if(moist == current.preferredMoisture[j])
236                        {
237                            factionIndices[c] = re;
238                            break PER_FACTION;
239                        }
240                    }
241                }
242            }
243        }
244        //ArrayList<Coord> spillers = new ArrayList<>(entries); // centers
245        IntVLA[] fresh = new IntVLA[count];
246        int filled = 0;
247        boolean hasFresh = false;
248        int approximateArea = (controlled * 4) / (count * 3);
249        char[] keys = new char[count];
250        double[] biases = new double[count];
251        for (int i = 0; i < count; i++) {
252            fresh[i] = new IntVLA(approximateArea);
253            cen = centers[i];
254            fresh[i].add(cen);
255            cx = cen % width;
256            cy = cen / width;
257            politicalMap[cx][cy] = keys[i] = atlas.keyAt(factionIndices[i] + 2);
258            biases[i] = rng.nextDouble() * rng.nextDouble() + rng.nextDouble() + 0.03125;
259            hasFresh = true;
260        }
261        Direction[] dirs = Direction.CARDINALS;
262        IntVLA currentFresh;
263        GreasedRegion anySpillMap = new GreasedRegion(width, height),
264                anyFreshMap = new GreasedRegion(width, height);
265
266        while (hasFresh && filled < controlled) {
267            hasFresh = false;
268            for (int i = 0; i < count && filled < controlled; i++) {
269                currentFresh = fresh[i];
270                if (currentFresh.isEmpty())
271                    continue;
272                else
273                    hasFresh = true;
274                if (rng.nextDouble() < biases[i]) {
275                    int index = rng.nextIntHasty(currentFresh.size), cell = currentFresh.get(index);
276                    cx = cell % width;
277                    cy = cell / width;
278
279
280                    politicalMap[cx][cy] = keys[i];
281                    filled++;
282                    anySpillMap.insert(cx, cy);
283
284                    for (int d = 0; d < dirs.length; d++) {
285                        cx2 = wmg.wrapX(cx + dirs[d].deltaX, cy);
286                        cy2 = wmg.wrapY(cx, cy + dirs[d].deltaY);
287                        if (cx == cx2 && cy == cy2)
288                            continue;
289                        if (land.contains(cx2, cy2) && !anySpillMap.contains(cx2, cy2)) {
290                            if(!anyFreshMap.contains(cx2, cy2)) {
291                                currentFresh.add(cx2 + cy2 * width);
292                                anyFreshMap.insert(cx2, cy2);
293                            }
294                        }
295                    }
296                    currentFresh.removeIndex(index);
297                    anyFreshMap.remove(cx, cy);
298                }
299            }
300        }
301        return politicalMap;
302    }

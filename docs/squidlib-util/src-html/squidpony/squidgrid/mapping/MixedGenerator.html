013/**
014 * A dungeon generator that can use a mix of techniques to have part-cave, part-room dungeons. Not entirely intended for
015 * normal use outside of this library, though it can be very useful when you want to make a dungeon match a specific
016 * path and existing generators that use MixedGenerator aren't sufficient. You may want to use a simpler generator based
017 * on this, like SerpentMapGenerator, which generates a long, winding path that loops around on itself. This supports
018 * the getEnvironment() method, which can be used in conjunction with RoomFinder to find where separate room, corridor,
019 * and cave areas have been placed.
020 * <br>
021 * Based on Michael Patraw's excellent Drunkard's Walk dungeon generator.
022 * http://mpatraw.github.io/libdrunkard/
023 * @see squidpony.squidgrid.mapping.SerpentMapGenerator a normal use for MixedGenerator that makes winding dungeons
024 * @see squidpony.squidgrid.mapping.SerpentDeepMapGenerator uses MixedGenerator as it makes a multi-level dungeon
025 * Created by Tommy Ettinger on 10/22/2015.
026 */
027public class MixedGenerator implements IDungeonGenerator {

090    /**
091     * Mainly for internal use; this is used by {@link #MixedGenerator(int, int, IRNG)} to get its room positions.
092     * This is (and was) the default for generating a List of Coord if no other collection of Coord was supplied to the
093     * constructor. For some time this was not the default, and {@link #cleanPoints(int, int, IRNG)} was used instead.
094     * Both are still options, but this technique seems to keep corridors shorter and makes connections between rooms
095     * more relevant to the current area.
096     * <br>
097     * <a href="https://gist.githubusercontent.com/tommyettinger/be0ed51858cb492bc7e8cda43a04def1/raw/dae9d8e4f45dd3a3577bdd5f58b419ea5f9ed570/PoissonDungeon.txt">Preview map.</a>
098     * @param width dungeon width in cells
099     * @param height dungeon height in cells
100     * @param rng rng to use
101     * @return evenly spaced Coord points in a list made by PoissonDisk, trimmed down so they aren't all used
102     * @see PoissonDisk used to make the list
103     */
104    public static OrderedSet<Coord> basicPoints(int width, int height, IRNG rng)
105    {
106        return PoissonDisk.sampleRectangle(Coord.get(2, 2), Coord.get(width - 3, height - 3),
107                8.5f * (width + height) / 120f, width, height, 35, rng);
108    }
109    /**
110     * Mainly for internal use; this was used by {@link #MixedGenerator(int, int, IRNG)} to get its room positions, and
111     * you can choose to use it with {@code new MixedGenerator(width, height, rng, cleanPoints(width, height, rng))}.
112     * It produces a fairly rigid layout of rooms that should have less overlap between rooms and corridors; the
113     * downside to this is that corridors can become extremely long. The exact technique used here is to get points from
114     * a Halton-like sequence, formed using {@link VanDerCorputQRNG} to get a van der Corput sequence, for the x axis
115     * and a scrambled van der Corput sequence for the y axis. MixedGenerator will connect these points in pairs. The
116     * current method is much better at avoiding "clumps" of closely-positioned rooms in the center of the map.
117     * <br>
118     * <a href="https://gist.githubusercontent.com/tommyettinger/2745e6fc16fc2acebe2fc959fb4e4c2e/raw/77e1f4dbc844d8892c1a686754535c02cadaa270/TenDungeons.txt">Preview maps, with and without box drawing characters.</a>
119     * Note that one of the dungeons in that gist was produced as a fallback by {@link DungeonGenerator} with no
120     * arguments (using DEFAULT_DUNGEON as the dungeon style), because the map this method produced in one case had too
121     * few floor cells to be used.
122     * @param width dungeon width in cells
123     * @param height dungeon height in cells
124     * @param rng rng to use
125     * @return erratically-positioned but generally separated Coord points to pass to a MixedGenerator constructor
126     * @see VanDerCorputQRNG used to get separated positions
127     */
128    public static List<Coord> cleanPoints(int width, int height, IRNG rng)
129    {
130        width -= 2;
131        height -= 2;
132        //float mx = rng.nextFloat() * 0.2f, my = rng.nextFloat() * 0.2f;
133        int blocks = width * height / 640, sz = blocks * 5, index = 0,
134                seed = rng.nextInt()|1, seed2 = rng.nextInt()|1;
135        //System.out.println("mx: " + mx + ", my: " + my + ", index: " + index);
136        List<Coord> list = new ArrayList<>(sz);
137        for (int i = 0; i < blocks; i++) {
138            Coord area = VanDerCorputQRNG.haltoid(seed, width * 3 >> 2, height * 3 >> 2, 1, 1, i);
139            for (int j = 0; j < 5; j++) {
140                list.add(VanDerCorputQRNG.haltoid(seed2, width >> 2, height >> 2, area.x, area.y, index++));
141            }
142        }
143        return list;
144    }

146    /**
147     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
148     * This version of the constructor uses a sub-random point sequence to generate the points it will draw caves and
149     * corridors between, helping to ensure a minimum distance between points, but it does not ensure that paths between
150     * points  will avoid overlapping with rooms or other paths. You call the different carver-adding methods to affect
151     * what the dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to
152     * only caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
153     * @param width the width of the final map in cells
154     * @param height the height of the final map in cells
155     * @param rng an RNG object to use for random choices; this make a lot of random choices.
156     * @see PoissonDisk used to ensure spacing for the points.
157     */
158    public MixedGenerator(int width, int height, IRNG rng) {
159        this(width, height, rng, cleanPoints(width, height, rng));
160    }

162    /**
163     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
164     * This version of the constructor uses a List of Coord points from some other source to determine the path to add
165     * rooms or caves to and then connect. You call the different carver-adding methods to affect what the
166     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
167     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
168     * @param width the width of the final map in cells
169     * @param height the height of the final map in cells
170     * @param rng an RNG object to use for random choices; this make a lot of random choices.
171     * @param sequence a List of Coord to connect in order; index 0 is the start, index size() - 1 is the end.
172     * @see SerpentMapGenerator a class that uses this technique
173     */
174    public MixedGenerator(int width, int height, IRNG rng, List<Coord> sequence) {
175        this.width = width;
176        this.height = height;
177        this.roomWidth = width / 64.0f;
178        this.roomHeight = height / 64.0f;
179        if(width <= 2 || height <= 2)
180            throw new IllegalStateException("width and height must be greater than 2");
181        this.rng = rng;
182        dungeon = new char[width][height];
183        environment = new int[width][height];
184        marked = new boolean[width][height];
185        walled = new boolean[width][height];
186        fixedRooms = new boolean[width][height];
187        Arrays.fill(dungeon[0], '#');
188        Arrays.fill(environment[0], DungeonUtility.UNTOUCHED);
189        for (int i = 1; i < width; i++) {
190            System.arraycopy(dungeon[0], 0, dungeon[i], 0, height);
191            System.arraycopy(environment[0], 0, environment[i], 0, height);
192        }
193        totalPoints = sequence.size() - 1;
194        points = new IntVLA(totalPoints);
195        for (int i = 0; i < totalPoints; i++) {
196            Coord c1 = sequence.get(i), c2 = sequence.get(i + 1);
197            points.add(((c1.x & 0xff) << 24) | ((c1.y & 0xff) << 16) | ((c2.x & 0xff) << 8) | (c2.y & 0xff));
198        }
199        carvers = new double[5];
200    }

218    /**
219     * This prepares a map generator that will generate a map with the given width and height, using the given IRNG.
220     * This version of the constructor uses a Map with Coord keys and Coord array values to determine a
221     * branching path for the dungeon to take; each key will connect once to each of the Coords in its value, and you
222     * usually don't want to connect in both directions. You call the different carver-adding methods to affect what the
223     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
224     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
225     * @param width the width of the final map in cells
226     * @param height the height of the final map in cells
227     * @param rng an IRNG object to use for random choices; this make a lot of random choices.
228     * @param connections a Map of Coord keys to arrays of Coord to connect to next; shouldn't connect both ways
229     * @param roomSizeMultiplier a float multiplier that will be applied to each room's width and height
230     * @see SerpentMapGenerator a class that uses this technique
231     */
232    public MixedGenerator(int width, int height, IRNG rng, Map<Coord, List<Coord>> connections,
233                          float roomSizeMultiplier) {
234        this.width = width;
235        this.height = height;
236        roomWidth = (width / 64.0f) * roomSizeMultiplier;
237        roomHeight = (height / 64.0f) * roomSizeMultiplier;
238        if(width <= 2 || height <= 2)
239            throw new IllegalStateException("width and height must be greater than 2");
240        this.rng = rng;
241        dungeon = new char[width][height];
242        environment = new int[width][height];
243        marked = new boolean[width][height];
244        walled = new boolean[width][height];
245        fixedRooms = new boolean[width][height];
246        Arrays.fill(dungeon[0], '#');
247        Arrays.fill(environment[0], DungeonUtility.UNTOUCHED);
248        for (int i = 1; i < width; i++) {
249            System.arraycopy(dungeon[0], 0, dungeon[i], 0, height);
250            System.arraycopy(environment[0], 0, environment[i], 0, height);
251        }
252        totalPoints = 0;
253        for(List<Coord> vals : connections.values())
254        {
255            totalPoints += vals.size();
256        }
257        points = new IntVLA(totalPoints);
258        for (Map.Entry<Coord, List<Coord>> kv : connections.entrySet()) {
259            Coord c1 = kv.getKey();
260            for (Coord c2 : kv.getValue()) {
261                points.add(((c1.x & 0xff) << 24) | ((c1.y & 0xff) << 16) | ((c2.x & 0xff) << 8) | (c2.y & 0xff));
262            }
263        }
264        carvers = new double[5];
265    }

267    /**
268     * This prepares a map generator that will generate a map with the given width and height, using the given IRNG.
269     * This version of the constructor uses an {@link squidpony.squidgrid.mapping.locks.IRoomLayout} to set up rooms,
270     * almost always produced by {@link squidpony.squidgrid.mapping.locks.generators.LayoutGenerator}. This method does
271     * alter the individual Room objects inside layout, making the center of each room match where that center is placed
272     * in the dungeon this generates. You call the different carver-adding methods to affect what the dungeon will look
273     * like, i.e. {@link #putCaveCarvers(int)}, {@link #putBoxRoomCarvers(int)} , {@link #putRoundRoomCarvers(int)},
274     * {@link #putWalledBoxRoomCarvers(int)}, and {@link #putWalledRoundRoomCarvers(int)}, defaulting to only caves if
275     * none are called (using rooms is recommended for this constructor). You call generate() after adding carvers,
276     * which returns a char[][] for a map and sets the environment to be fetched with {@link #getEnvironment()}, which
277     * is usually needed for {@link SectionDungeonGenerator} to correctly place doors and various other features.
278     * @param width the width of the final map in cells
279     * @param height the height of the final map in cells
280     * @param rng an IRNG object to use for random choices; this make a lot of random choices.
281     * @param layout an IRoomLayout that will almost always be produced by LayoutGenerator; the rooms will be altered
282     * @param roomSizeMultiplier a float multiplier that will be applied to each room's width and height
283     */
284    public MixedGenerator(int width, int height, IRNG rng, IRoomLayout layout,
285                          float roomSizeMultiplier) {
286        this.width = width;
287        this.height = height;
288        Rect2I bounds = layout.getExtentBounds();
289        int offX = bounds.getBottomLeft().x, offY = bounds.getBottomLeft().y;
290        float rw = (width) / (bounds.width+1f), rh = (height) / (bounds.height+1f);
291        this.roomWidth = roomSizeMultiplier * rw * 0.125f;
292        this.roomHeight = roomSizeMultiplier * rh * 0.125f;
293        if(width <= 2 || height <= 2)
294            throw new IllegalStateException("width and height must be greater than 2");
295        this.rng = rng;
296        dungeon = new char[width][height];
297        environment = new int[width][height];
298        marked = new boolean[width][height];
299        walled = new boolean[width][height];
300        fixedRooms = new boolean[width][height];
301        ArrayTools.fill(dungeon, '#');
302        ArrayTools.fill(environment, DungeonUtility.UNTOUCHED);
303        totalPoints = layout.roomCount();
304        points = new IntVLA(totalPoints);
305        Coord c2;
306        Set<Room> rooms = layout.getRooms(), removing = new HashSet<>(rooms);
307        Room t;
308        for (Room room : rooms) {
309            Coord c1 = room.getCenter();
310            if (!bounds.contains(c1)) {
311                removing.remove(room);
312            }
313            else {
314                room.setCenter(Coord.get(
315                        (int) ((c1.x - offX + 0.75f) * (rw)) & 0xff,
316                        (int) ((c1.y - offY + 0.75f) * (rh)) & 0xff));
317            }
318        }
319
320        for (Room room : rooms) {
321            Coord c1 = room.getCenter();
322            for (Edge e : room.getEdges()) {
323                if (removing.contains(t = layout.get(e.getTargetRoomId()))) {
324                    c2 = t.getCenter();
325                    points.add((c1.x << 24)
326                            | (c1.y << 16)
327                            | (c2.x << 8)
328                            | c2.y);
329                }
330            }
331            removing.remove(room);
332        }
333        totalPoints = points.size;
334        carvers = new double[5];
335    }

338     * Changes the number of "carvers" that will create caves from one room to the next. If count is 0 or less, no caves
339     * will be made. If count is at least 1, caves are possible, and higher numbers relative to the other carvers make
340     * caves more likely. Carvers are shuffled when used, then repeat if exhausted during generation. Since typically
341     * about 30-40 rooms are carved, large totals for carver count aren't really needed; aiming for a total of 10
342     * between the count of putCaveCarvers(), putBoxRoomCarvers(), putRoundRoomCarvers(), putWalledBoxRoomCarvers(), and
343     * putWalledRoundRoomCarvers() is reasonable.
344     * @param count the number of carvers making caves between rooms; only matters in relation to other carvers
345     */

347    {
348        carvers[CAVE] = count;

452            int pair = points.get(p);
453            Coord start = Coord.get(pair >>> 24 & 0xff, pair >>> 16 & 0xff),
454                  end   = Coord.get(pair >>> 8 & 0xff, pair & 0xff);
455            int ct = carverTable.random(rng.nextLong());
456            Direction dir;
457            switch (ct)
458            {
459                case CAVE:
460                    markPiercing(end);
461                    markEnvironmentCave(end.x, end.y);
462                    store();
463                    double weight = 0.75;
464                    do {
465                        Coord cent = markPlusCave(start);
466                        if(cent != null)
467                        {
468                            markPiercingCave(cent);
469                            markPiercingCave(cent.translate(1, 0));
470                            markPiercingCave(cent.translate(-1, 0));
471                            markPiercingCave(cent.translate(0, 1));
472                            markPiercingCave(cent.translate(0, -1));
473                            weight = 0.95;
474                        }
475                        dir = stepWobbly(start, end, weight);
476                        start = start.translate(dir);
477                    }while (dir != Direction.NONE);
478                    break;
479                case BOX:
480                    markRectangle(end, rng.between(1, 5), rng.between(1, 5));
481                    markRectangle(start, rng.between(1, 4), rng.between(1, 4));
482                    store();
483                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
484                    if(dir.isDiagonal())
485                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
486                                : Direction.getCardinalDirection(0, -dir.deltaY);
487                    while (start.x != end.x && start.y != end.y)
488                    {
489                        markPiercing(start);
490                        markEnvironmentCorridor(start.x, start.y);
491                        start = start.translate(dir);
492                    }
493                    markRectangle(start, 1, 1);
494                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
495                    while (!(start.x == end.x && start.y == end.y))
496                    {
497                        markPiercing(start);
498                        markEnvironmentCorridor(start.x, start.y);
499                        start = start.translate(dir);
500                    }
501                    break;
502                case BOX_WALLED:
503                    markRectangleWalled(end, rng.between(1, 5), rng.between(1, 5));
504                    markRectangleWalled(start, rng.between(1, 4), rng.between(1, 4));
505                    store();
506                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
507                    if(dir.isDiagonal())
508                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
509                                : Direction.getCardinalDirection(0, -dir.deltaY);
510                    while (start.x != end.x && start.y != end.y)
511                    {
512                        markPiercing(start);
513                        markEnvironmentCorridor(start.x, start.y);
514                        start = start.translate(dir);
515                    }
516                    markRectangleWalled(start, 1, 1);
517                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
518                    while (!(start.x == end.x && start.y == end.y))
519                    {
520                        markPiercing(start);
521                        markEnvironmentCorridor(start.x, start.y);
522                        start = start.translate(dir);
523                    }
524                    break;
525                case ROUND:
526                    markCircle(end, rng.between(2, 6));
527                    markCircle(start, rng.between(2, 6));
528                    store();
529                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
530                    if(dir.isDiagonal())
531                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
532                                : Direction.getCardinalDirection(0, -dir.deltaY);
533                    while (start.x != end.x && start.y != end.y)
534                    {
535                        markPiercing(start);
536                        markEnvironmentCorridor(start.x, start.y);
537                        start = start.translate(dir);
538                    }
539                    markCircle(start, 2);
540                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
541                    while (!(start.x == end.x && start.y == end.y))
542                    {
543                        markPiercing(start);
544                        markEnvironmentCorridor(start.x, start.y);
545                        start = start.translate(dir);
546                    }
547                    break;
548                case ROUND_WALLED:
549                    markCircleWalled(end, rng.between(2, 6));
550                    markCircleWalled(start, rng.between(2, 6));
551                    store();
552                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
553                    if(dir.isDiagonal())
554                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
555                                : Direction.getCardinalDirection(0, -dir.deltaY);
556                    while (start.x != end.x && start.y != end.y)
557                    {
558                        markPiercing(start);
559                        markEnvironmentCorridor(start.x, start.y);
560                        start = start.translate(dir);
561                    }
562                    markCircleWalled(start, 2);
563                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
564                    while (!(start.x == end.x && start.y == end.y))
565                    {
566                        markPiercing(start);
567                        markEnvironmentCorridor(start.x, start.y);
568                        start = start.translate(dir);
569                    }
570                    break;
571            }
572            store();
573        }
574        for (int x = 0; x < width; x++) {
575            for (int y = 0; y < height; y++) {
576                if(fixedRooms[x][y])
577                    markPiercingRoom(x, y);
578            }
579        }
580        store();
581        markEnvironmentWalls();
582        generated = true;
583        return dungeon;
584    }

608    /**
609     * Internal use. Takes cells that have been previously marked and permanently stores them as floors in the dungeon.
610     */
611    protected void store()
612    {
613        for (int i = 0; i < width; i++) {
614            for (int j = 0; j < height; j++) {
615                if(marked[i][j])
616                {
617                    dungeon[i][j] = '.';
618                    marked[i][j] = false;
619                }
620            }
621        }
622    }

624    /**
625     * Internal use. Finds all floor cells by environment and marks untouched adjacent (8-way) cells as walls, using the
626     * appropriate type for the nearby floor.
627     */
628    protected void markEnvironmentWalls()
629    {
630        for (int i = 0; i < width; i++) {
631            for (int j = 0; j < height; j++) {
632                if(environment[i][j] == DungeonUtility.UNTOUCHED)
633                {
634                    boolean allWalls = true;
635                    //lowest precedence, also checks for any floors
636                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
637
638                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
639                            if (environment[x][y] == DungeonUtility.CORRIDOR_FLOOR) {
640                                markEnvironment(i, j, DungeonUtility.CORRIDOR_WALL);
641                            }
642                            if(dungeon[x][y] == '.')
643                                allWalls = false;
644                        }
645                    }
646                    //if there are no floors we don't need to check twice again.
647                    if(allWalls)
648                        continue;
649                    //more precedence
650                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
651
652                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
653                            if (environment[x][y] == DungeonUtility.CAVE_FLOOR) {
654                                markEnvironment(i, j, DungeonUtility.CAVE_WALL);
655                            }
656                        }
657                    }
658                    //highest precedence
659                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
660
661                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
662                            if (environment[x][y] == DungeonUtility.ROOM_FLOOR) {
663                                markEnvironment(i, j, DungeonUtility.ROOM_WALL);
664                            }
665                        }
666                    }
667
668                }
669            }
670        }
671    }

676     * @param y y position to mark

788    /**
789     * Internal use. Marks a point and the four cells orthogonally adjacent to it.
790     * @param pos center position to mark
791     * @return null if the center of the plus shape wasn't blocked by wall, otherwise the Coord of the center
792     */
793    private Coord markPlus(Coord pos) {
794        Coord block = null;
795        if (mark(pos.x, pos.y))
796            block = pos;
797        mark(pos.x + 1, pos.y);
798        mark(pos.x - 1, pos.y);
799        mark(pos.x, pos.y + 1);
800        mark(pos.x, pos.y - 1);
801        return block;

805     * Internal use. Marks a point and the four cells orthogonally adjacent to it, and also marks any cells that weren't
806     * blocked as cave floors.
807     * @param pos center position to mark
808     * @return null if the center of the plus shape wasn't blocked by wall, otherwise the Coord of the center
809     */
810    private Coord markPlusCave(Coord pos) {
811        Coord block = null;
812        if (mark(pos.x, pos.y))
813            block = pos;
814        else
815            markEnvironmentCave(pos.x, pos.y);
816        if(!mark(pos.x + 1, pos.y))
817            markEnvironmentCave(pos.x + 1, pos.y);
818        if(!mark(pos.x - 1, pos.y))
819            markEnvironmentCave(pos.x - 1, pos.y);
820        if(!mark(pos.x, pos.y + 1))
821            markEnvironmentCave(pos.x, pos.y + 1);
822        if(!mark(pos.x, pos.y - 1))
823            markEnvironmentCave(pos.x, pos.y - 1);
824        return block;
825    }
826    /**
827     * Internal use. Marks a rectangle of points centered on pos, extending halfWidth in both x directions and
828     * halfHeight in both vertical directions. Marks all cells in the rectangle as room floors.
829     * @param pos center position to mark
830     * @param halfWidth the distance from the center to extend horizontally
831     * @param halfHeight the distance from the center to extend vertically
832     * @return null if no points in the rectangle were blocked by walls, otherwise a Coord blocked by a wall
833     */
834    private Coord markRectangle(Coord pos, int halfWidth, int halfHeight)
835    {
836        halfWidth = Math.max(1, Math.round(halfWidth * roomWidth));
837        halfHeight = Math.max(1, Math.round(halfHeight * roomHeight));
838        Coord block = null;
839        for (int i = pos.x - halfWidth; i <= pos.x + halfWidth; i++) {
840            for (int j = pos.y - halfHeight; j <= pos.y + halfHeight; j++) {
841                if(mark(i, j))
842                    block = Coord.get(i, j);
843                else
844                    markEnvironmentRoom(i, j);
845            }
846        }
847        return block;
848    }
849    /**
850     * Internal use. Marks a rectangle of points centered on pos, extending halfWidth in both x directions and
851     * halfHeight in both vertical directions. Also considers the area just beyond each wall, but not corners, to be
852     * a blocking wall that can only be passed by corridors and small cave openings. Marks all cells in the rectangle as
853     * room floors.
854     * @param pos center position to mark
855     * @param halfWidth the distance from the center to extend horizontally
856     * @param halfHeight the distance from the center to extend vertically
857     * @return null if no points in the rectangle were blocked by walls, otherwise a Coord blocked by a wall
858     */
859    private Coord markRectangleWalled(Coord pos, int halfWidth, int halfHeight)
860    {
861        halfWidth = Math.max(1, Math.round(halfWidth * roomWidth));
862        halfHeight = Math.max(1, Math.round(halfHeight * roomHeight));
863        Coord block = null;
864        for (int i = pos.x - halfWidth; i <= pos.x + halfWidth; i++) {
865            for (int j = pos.y - halfHeight; j <= pos.y + halfHeight; j++) {
866                markPiercing(i, j);
867                markEnvironmentRoom(i, j);
868            }
869        }
870        for (int i = Math.max(0, pos.x - halfWidth - 1); i <= Math.min(width - 1, pos.x + halfWidth + 1); i++) {
871            for (int j = Math.max(0, pos.y - halfHeight - 1); j <= Math.min(height - 1, pos.y + halfHeight + 1); j++)
872            {
873                wallOff(i, j);
874            }
875        }
876        return block;
877    }

879    /**
880     * Internal use. Marks a circle of points centered on pos, extending out to radius in Euclidean measurement. Marks
881     * all cells in the circle as room floors.
882     * @param pos center position to mark
883     * @param radius radius to extend in all directions from center
884     * @return null if no points in the circle were blocked by walls, otherwise a Coord blocked by a wall
885     */
886    private Coord markCircle(Coord pos, int radius)
887    {
888        Coord block = null;
889        int high;
890        radius = Math.max(1, Math.round(radius * Math.min(roomWidth, roomHeight)));
891        for (int dx = -radius; dx <= radius; ++dx)
892        {
893            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
894            for (int dy = -high; dy <= high; ++dy)
895            {
896                if(mark(pos.x + dx, pos.y + dy))
897                    block = pos.translate(dx, dy);
898                else
899                    markEnvironmentRoom(pos.x + dx, pos.y + dy);
900            }
901        }
902        return block;
903    }
904    /**
905     * Internal use. Marks a circle of points centered on pos, extending out to radius in Euclidean measurement.
906     * Also considers the area just beyond each wall, but not corners, to be a blocking wall that can only be passed by
907     * corridors and small cave openings. Marks all cells in the circle as room floors.
908     * @param pos center position to mark
909     * @param radius radius to extend in all directions from center
910     * @return null if no points in the circle were blocked by walls, otherwise a Coord blocked by a wall
911     */
912    private Coord markCircleWalled(Coord pos, int radius)
913    {
914        Coord block = null;
915        int high;
916        radius = Math.max(1, Math.round(radius * Math.min(roomWidth, roomHeight)));
917        for (int dx = -radius; dx <= radius; ++dx)
918        {
919            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
920            for (int dy = -high; dy <= high; ++dy)
921            {
922                markPiercing(pos.x + dx, pos.y + dy);
923                markEnvironmentRoom(pos.x + dx, pos.y + dy);
924            }
925        }
926        for (int dx = -radius; dx <= radius; ++dx)
927        {
928            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
929            int dx2 = Math.max(1, Math.min(pos.x + dx, width - 2));
930            for (int dy = -high; dy <= high; ++dy)
931            {
932                int dy2 = Math.max(1, Math.min(pos.y + dy, height - 2));
933
934                wallOff(dx2, dy2-1);
935                wallOff(dx2+1, dy2-1);
936                wallOff(dx2-1, dy2-1);
937                wallOff(dx2, dy2);
938                wallOff(dx2+1, dy2);
939                wallOff(dx2-1, dy2);
940                wallOff(dx2, dy2+1);
941                wallOff(dx2+1, dy2+1);
942                wallOff(dx2-1, dy2+1);
943
944            }
945        }
946        return block;
947    }

949    /**
950     * Internal use. Drunkard's walk algorithm, single step. Based on Michael Patraw's C code, used for cave carving.
951     * http://mpatraw.github.io/libdrunkard/
952     * @param current the current point
953     * @param target the point to wobble towards
954     * @param weight between 0.5 and 1.0, usually. 0.6 makes very random caves, 0.9 is almost a straight line.
955     * @return a Direction, either UP, DOWN, LEFT, or RIGHT if we should move, or NONE if we have reached our target
956     */
957    private Direction stepWobbly(Coord current, Coord target, double weight)
958    {
959        int dx = target.x - current.x;
960        int dy = target.y - current.y;
961
962        if (dx >  1) dx = 1;
963        if (dx < -1) dx = -1;
964        if (dy >  1) dy = 1;
965        if (dy < -1) dy = -1;
966
967        double r = rng.nextDouble();
968        Direction dir;
969        if (dx == 0 && dy == 0)
970        {
971            return Direction.NONE;
972        }
973        else if (dx == 0 || dy == 0)
974        {
975            int dx2 = (dx == 0) ? dx : dy, dy2 = (dx == 0) ? dy : dx;
976            if (r >= (weight * 0.5))
977            {
978                r -= weight * 0.5;
979                if (r < weight * (1.0 / 6) + (1 - weight) * (1.0 / 3))
980                {
981                    dx2 = -1;
982                    dy2 = 0;
983                }
984                else if (r < weight * (2.0 / 6) + (1 - weight) * (2.0 / 3))
985                {
986                    dx2 = 1;
987                    dy2 = 0;
988                }
989                else
990                {
991                    dx2 = 0;
992                    dy2 *= -1;
993                }
994            }
995            dir = Direction.getCardinalDirection(dx2, -dy2);
996
997        }
998        else
999        {
1000            if (r < weight * 0.5)
1001            {
1002                dy = 0;
1003            }
1004            else if (r < weight)
1005            {
1006                dx = 0;
1007            }
1008            else if (r < weight + (1 - weight) * 0.5)
1009            {
1010                dx *= -1;
1011                dy = 0;
1012            }
1013            else
1014            {
1015                dx = 0;
1016                dy *= -1;
1017            }
1018            dir = Direction.getCardinalDirection(dx, -dy);
1019        }
1020        if(current.x + dir.deltaX <= 0 || current.x + dir.deltaX >= width - 1) {
1021            if (current.y < target.y) dir = Direction.DOWN;
1022            else if (current.y > target.y) dir = Direction.UP;
1023        }
1024        else if(current.y + dir.deltaY <= 0 || current.y + dir.deltaY >= height - 1) {
1025            if (current.x < target.x) dir = Direction.RIGHT;
1026            else if (current.x > target.x) dir = Direction.LEFT;
1027        }
1028        return dir;
1029    }

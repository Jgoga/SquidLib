016/**
017 * Class that builds up a dictionary of words in a source text to words generated by a FakeLanguageGen, and can
018 * "translate" a source text to a similarly-punctuated, similarly-capitalized fake text. Uses a hash of each word in the
019 * source text to determine the RNG seed that FakeLanguageGen will use, so the translation is not random. Can cipher a
020 * typically English text and generate a text with FakeLanguageGen, but also decipher such a generated text with a
021 * fully-complete, partially-complete, or partially-incorrect vocabulary.
022 * <br>
023 * This defaults to caching source-language words to their generated-language word translations in the field table, as
024 * well as the reverse translation in reverse. This can be changed to reduce memory usage for large vocabularies with
025 * {@code setCacheLevel()}, where it starts at 2 (writing to table and reverse), and can be lowered to 1 (writing to
026 * table only) if you don't need reverse to decipher a language easily, or to 0 (writing to neither) if you expect that
027 * memory will be at a premium and don't mind re-generating the same word each time it occurs in a source text. If
028 * cacheLevel is 1 or less, then this will not check for overlap between previously-generated words (it won't have an
029 * easy way to look up previously-generated ones) and so may be impossible to accurately decipher. As an example, one
030 * test of level 1 generated "he" as the translation for both "a" and "at", so every time "a" had been ciphered and then
031 * deciphered, the reproduced version said "at" instead. This won't happen by default, but the default instead relies on
032 * words being entered as inputs to cipher() or lookup() in the same order. If words are entered in two different orders
033 * to different runs of the program, they may have different generated results if cacheLevel is 2. One way to handle
034 * this is to use cacheLevel 2 and cipher the whole game script, or just the unique words in it (maybe just a large word
035 * list, such as http://wordlist.aspell.net/12dicts/ ), then serialize the LanguageCipher for later usage.
036 * Created by Tommy Ettinger on 5/1/2016.
037 * @author Tommy Ettinger
038 * @see NaturalLanguageCipher NaturalLanguageCipher offers similar features but handles common prefixes and suffixes.
039 */
040public class LanguageCipher implements Serializable{
041    private static final long serialVersionUID = 1287835632461186341L;
042    /**
043     * The FakeLanguageGen this will use to construct words; normally one of the static fields in FakeLanguageGen or a
044     * FakeLanguageGen produced by using the mix() method of one of them. Manually constructing FakeLanguageGen objects
045     * isn't especially easy, and if you decide to do that it's recommended you look at SquidLib's source to see how the
046     * existing calls to constructors work.
047     */
048    public FakeLanguageGen language;
049    private StatefulRNG rng;
050    // not an OrderedMap because this should never be need a random element to be requested
051    /**
052     * The mapping of lower-case word keys to lower-case word values, where keys are in the source language and values
053     * are generated by language.
054     */
055    public HashMap<String, String> table,
056    /**
057     * The mapping of lower-case word keys to lower-case word values, where keys are generated by language and values
058     * are in the source language. Can be used as a complete vocabulary when passed to decipher.
059     */
060    reverse;
061    private static final Pattern wordMatch = Pattern.compile("(\\pL+)|(\\pL[\\pL-]*\\pL)");
062
063    /**
064     * The degree of vocabulary to cache to speed up future searches at the expense of memory usage.
065     * <ul>
066     * <li>2 will cache source words to generated words in table, and generated to source in reverse.</li>
067     * <li>1 will cache source words to generated words in table, and won't write to reverse.</li>
068     * <li>0 won't write to table or reverse.</li>
069     * </ul>
070     * Defaults to 2, writing to both table and reverse.
071     */
072    public int cacheLevel = 2;
073
074    public final long shift;
075
076    /**
077     * Constructs a LanguageCipher that will generate English-like or Dutch-like text by default.
078     */
079    public LanguageCipher()
080    {
081        this(FakeLanguageGen.ENGLISH);
082    }

084    /**
085     * Constructs a LanguageCipher that will use the given style of language generator to produce its text.
086     * @param language a FakeLanguageGen, typically one of the static constants in that class or a mix of them.
087     */
088    public LanguageCipher(FakeLanguageGen language)
089    {
090        this(language, 0);
091    }

093    /**
094     * Constructs a LanguageCipher that will use the given style of language generator to produce its text.
095     * @param language a FakeLanguageGen, typically one of the static constants in that class or a mix of them.
096     * @param shift any long; this will be used to alter the specific words generated unless it is 0
097     */
098    public LanguageCipher(FakeLanguageGen language, long shift)
099    {
100        this.shift = shift;
101        this.language = language.copy();
102        rng = new StatefulRNG();
103        table = new HashMap<>(512);
104        reverse = new HashMap<>(512);
105    }
106
107    /**
108     * Copies another LanguageCipher and constructs this one with the information in the other. Copies the dictionary
109     * of known words, as well as the FakeLanguageGen style and everything else.
110     * @param other a previously-constructed LanguageCipher.
111     */
112    public LanguageCipher(LanguageCipher other)
113    {
114        this.language = other.language.copy();
115        this.rng = new StatefulRNG();
116        this.table = new HashMap<>(other.table);
117        this.reverse = new HashMap<>(other.reverse);
118        this.shift = other.shift;
119    }
120
121    /**
122     * Given a word in the source language (usually English), looks up an existing translation for that word, or if none
123     * exists, generates a new word based on the hash of the source word and this LanguageCipher's FakeLanguageGen.
124     * @param source a word in the source language
125     * @return a word in the fake language
126     */
127    public String lookup(String source)
128    {
129        if(source == null || source.isEmpty())
130            return "";
131        String s2 = source.toLowerCase(), ciphered;
132        if(table.containsKey(s2))
133            ciphered = table.get(s2);
134        else {
135            long h = CrossHash.hash64(s2) + shift, frustration = 0;
136            rng.setState(h);
137            do {
138                ciphered = language.word(rng, false, (int) Math.ceil(s2.length() / (2.2 + rng.nextDouble())));
139                if(cacheLevel < 2 || frustration++ > 9)
140                    break;
141            }while (reverse.containsKey(ciphered));
142            switch (cacheLevel) {
143                case 2: reverse.put(ciphered, s2);
144                case 1: table.put(s2, ciphered);
145            }
146        }
147        char[] chars = ciphered.toCharArray();
148        // Lu is the upper case letter category in Unicode; we're using regexodus for this because GWT won't
149        // respect unicode case data on its own (see
150        // https://github.com/gwtproject/gwt/blob/2.6.1/user/super/com/google/gwt/emul/java/lang/Character.java#L54-L61
151        // ). We are using GWT to capitalize, though, which appears to work in practice and the docs agree.
152        if(Category.Lu.contains(source.charAt(0)))
153            chars[0] = Character.toUpperCase(chars[0]);
154        if(source.length() > 1 && Category.Lu.contains(source.charAt(1))) {
155            for (int i = 1; i < chars.length; i++) {
156                chars[i] = Character.toUpperCase(chars[i]);
157            }
158        }
159        return new String(chars);
160    }
161
162    /**
163     * Given a String, StringBuilder, or other CharSequence that should contain words in the source language, this
164     * translates each word to the fake language, using existing translations if previous calls to cipher() or lookup()
165     * had translated that word.
166     * @param text a CharSequence, such as a String, that contains words in the source language
167     * @return a String of the translated text.
168     */
169    public String cipher(CharSequence text)
170    {
171        Replacer rep = wordMatch.replacer(new CipherSubstitution());
172        return rep.replace(text);
173    }
174
175    private class CipherSubstitution implements Substitution
176    {
177        @Override
178        public void appendSubstitution(MatchResult match, TextBuffer dest) {
179            dest.append(lookup(match.group(0)));
180        }
181    }
182    private class DecipherSubstition implements Substitution
183    {
184        private final Map<String, String> vocabulary;
185        DecipherSubstition(final Map<String, String> vocabulary)
186        {
187            this.vocabulary = vocabulary;
188        }
189        public void appendSubstitution(MatchResult match, TextBuffer dest) {
190            String translated = match.group(0);
191            if(translated == null) {
192                return;
193            }
194            translated = translated.toLowerCase();
195            translated = vocabulary.get(translated);
196            if(translated == null) {
197                dest.append(match.group(0));
198                return;
199            }
200            char[] chars = translated.toCharArray();
201            if(Category.Lu.contains(match.charAt(0)))
202                chars[0] = Character.toUpperCase(chars[0]);
203            if(match.length() > 1 && Category.Lu.contains(match.charAt(1))) {
204                for (int i = 1; i < chars.length; i++) {
205                    chars[i] = Character.toUpperCase(chars[i]);
206                }
207            }
208            dest.append(chars, 0, chars.length);
209        }
210    }

212    /**
213     * Deciphers words in an already-ciphered text with a given String-to-String Map for a vocabulary. This Map could be
214     * the reverse field of this LanguageCipher, which would give a complete translation, or it could be a
215     * partially-complete or partially-correct vocabulary of words the player has learned. The vocabulary should
216     * typically have entries added using the quick and accurate learnTranslations() method, unless you want to add
217     * translations one word at a time (then use learnTranslation() ) or you want incorrect or biased translations added
218     * (then use mismatchTranslation() ). You don't need to use one of these methods if you just pass the whole of the
219     * reverse field as a vocabulary, which will translate every word. If making your own vocabulary without the learn
220     * methods, the keys need to be lower-case because while regex Patterns can be case-insensitive, Map lookups cannot.
221     * @param text a text in the fake language
222     * @param vocabulary a Map of Strings in the fake language to Strings in the source language
223     * @return a deciphered version of text that has any words as keys in vocabulary translated to the source language
224     */
225    public String decipher(String text, final Map<String, String> vocabulary)
226    {
227        Pattern pat;
228        Replacer rep;
229        StringBuilder sb = new StringBuilder(128);
230        sb.append("(?:");
231        for(String k : vocabulary.keySet())
232        {
233            sb.append("(?:\\Q");
234            sb.append(k);
235            sb.append("\\E)|");
236        }
237        sb.deleteCharAt(sb.length() - 1);
238        sb.append(')');
239
240        pat = Pattern.compile("\\b" + sb + "\\b", "ui");
241
242        rep = pat.replacer(new DecipherSubstition(vocabulary));
243        return rep.replace(text);
244    }

246    /**
247     * Adds a translation pair to vocabulary so it can be used in decipher, giving a correct translation for sourceWord.
248     * Modifies vocabulary in-place and returns this LanguageCipher for chaining. Can be used to correct a mismatched
249     * translation added to vocabulary with mismatchTranslation.
250     * @param vocabulary a Map of String keys to String values that will be modified in-place
251     * @param sourceWord a word in the source language, typically English; the meaning will be "learned" for decipher
252     * @return this, for chaining
253     */
254    public LanguageCipher learnTranslation(Map<String, String> vocabulary, String sourceWord)
255    {
256        vocabulary.put(lookup(sourceWord.toLowerCase()), sourceWord);
257        return this;
258    }

260    /**
261     * Adds translation pairs to vocabulary so it can be used in decipher, giving a correct translation for sourceWords.
262     * Modifies vocabulary in-place and returns this LanguageCipher for chaining. Can be used to correct mismatched
263     * translations added to vocabulary with mismatchTranslation.
264     * @param vocabulary a Map of String keys to String values that will be modified in-place
265     * @param sourceWords an array or vararg of words in the source language, typically English; their meanings will
266     *                    be "learned" for decipher
267     * @return this, for chaining
268     */
269    public LanguageCipher learnTranslations(Map<String, String> vocabulary, String... sourceWords)
270    {
271        for (int i = 0; i < sourceWords.length; i++) {
272            learnTranslation(vocabulary, sourceWords[i]);
273        }
274        return this;
275    }

277    /**
278     * Adds translation pairs to vocabulary so it can be used in decipher, giving a correct translation for sourceWords.
279     * Modifies vocabulary in-place and returns this LanguageCipher for chaining. Can be used to correct mismatched
280     * translations added to vocabulary with mismatchTranslation.
281     * @param vocabulary a Map of String keys to String values that will be modified in-place
282     * @param sourceWords an Iterable of words in the source language, typically English; their meanings will be
283     *                   "learned" for decipher
284     * @return this, for chaining
285     */
286    public LanguageCipher learnTranslations(Map<String, String> vocabulary, Iterable<String> sourceWords)
287    {
288        for (String s : sourceWords) {
289            learnTranslation(vocabulary, s);
290        }
291        return this;
292    }

294    /**
295     * Adds a translation pair to vocabulary so it can be used in decipher, giving a typically-incorrect translation for
296     * correctWord where it provides mismatchWord instead when the ciphered version of correctWord appears.
297     * Modifies vocabulary in-place and returns this LanguageCipher for chaining. You can use learnTranslation() to
298     * correct a mismatched vocabulary word, or mismatchTranslation() again to change the mismatched word.
299     * @param vocabulary a Map of String keys to String values that will be modified in-place
300     * @param correctWord a word in the source language, typically English; where the ciphered version of this
301     *                    appears and the text is deciphered, mismatchWord will be used instead
302     * @param mismatchWord a String that will be used for deciphering in place of the translation of correctWord.
303     * @return this, for chaining
304     */
305    public LanguageCipher mismatchTranslation(Map<String, String> vocabulary, String correctWord, String mismatchWord)
306    {
307        vocabulary.put(lookup(correctWord.toLowerCase()), mismatchWord);
308        return this;
309    }

311    public int getCacheLevel() {
312        return cacheLevel;
313    }
314
315    public void setCacheLevel(int cacheLevel) {
316        if(cacheLevel >= 2) this.cacheLevel = 2;
317        else if(cacheLevel <= 0) this.cacheLevel = 0;
318        else this.cacheLevel = cacheLevel;
319    }

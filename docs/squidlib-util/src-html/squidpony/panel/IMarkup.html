006public interface IMarkup<T> {
007    String getMarkup(T value);
008    String closeMarkup();
009    class StringMarkup implements IMarkup<String>
010    {
011        @Override
012        public String getMarkup(String value) {
013            return "[" + value + "]";
014        }
015
016        @Override
017        public String closeMarkup() {
018            return "[]";
019        }
020    }
021}

014/**
015 * A simple format parser for String-based configuration or data files where JSON is overkill.
016 * Supports only one type, String, but allows each String to have arbitrary nested levels of
017 * String children as if in sub-lists. You can interpret the Strings however you want, and
018 * quoting each String isn't necessary if they are just one word ("bare words" are allowed).
019 * This stores its items in an inner class, {@link ObTextEntry}, which only has a "primary"
020 * String and may have a List of "associated" ObTextEntry values, each of which must have
021 * their own primary String and which may have their own associated List.
022 * <br>
023 * You can use this like any other List, though it will be contain ObTextEntry objects instead
024 * of Strings directly. This allows you to control whether you want to iterate through a
025 * particular primary String's associated entries, if there are any, or to skip over them and
026 * go to the next String in the current List.
027 * <br>
028 * This implements List of String and is modifiable; it extends the List interface with
029 * {@link #parse(CharSequence)} to read a String in the following format, as well as with some
030 * overloads of add() to add just a String with no associated values. A quirk of how this
031 * implements List is that it only considers the top-level Strings to be part of the List for
032 * length and for {@link #contains(Object)}, and will ignore child strings unless you access
033 * them via the {@link ObTextEntry#associated} List on an entry that has associated entries.
034 * <br>
035 * Format example:
036 * <pre>
037 * hello world
038 * 'how are you today?' [just great thanks]
039 * hooray!
040 *
041 * complexity?
042 * [it is possible [yes this is a good example]
043 * 'escapes like \[\'\] all work'
044 * ]
045 *
046 * comments are allowed // like this
047 * comments can have different forms # like this
048 * // block comments like in c are allowed
049 * / * but because this example is in javadoc, this example is not actually a comment * /
050 * // remove the spaces between each slash and asterisk to make the last line a comment.
051 * /[delimit/or block comments with delimiters/delimit]/
052 *
053 * '''
054 * raw strings (heredocs) look like this normally.
055 *     they permit characters without escapes, ]][][[ \/\/\ ,
056 *     except for triple quotes.
057 *     they keep newlines and indentation intact,
058 * except for up to one newline ignored adjacent to each triple quote.
059 * '''
060 *
061 * [[different[
062 * if you may need triple quotes
063 *     in the raw string, use a different syntax that allows delimiters.
064 * here, the delimiter is '''different''', just to be different.]different]]
065 * </pre>
066 * <br>
067 * Inspired strongly by STOB, http://stobml.org/ , but no code is shared and the format is
068 * slightly different. The main differences are that ObText supports nested block comments
069 * using the syntax {@code /[delimiter/contents/delimiter]/} where delimiter may be empty
070 * but must match on both sides, and contents is the body of the comment. ObText uses Python-
071 * like "heredoc" syntax for raw strings surrounded by triple-apostrophes '''like so''' with
072 * optional initial and final newlines in the raw string ignored. An alternate raw string
073 * syntax is present that allows delimiters, using the syntax
074 * {@code [[delimiter[contents]delimiter]]}, where again delimiter may be empty and contents
075 * is the body of the raw string. We use square brackets in place of STOB's curly braces to
076 * mark children associated with a string.
077 */

208    /**
209     * Parses the given text (a String or other CharSequence) and appends it into this ObText.
210     * @param text a CharSequence (such as a String) using ObText formatting, as described in this class' JavaDocs
211     * @return this ObText object after appending the parsed text, for chaining
212     */
213    public ObText parse(CharSequence text)
214    {
215        m.setTarget(text);
216        ObTextEntry current = null;
217        List<ObTextEntry> ls = entries;
218        IntVLA nesting = new IntVLA(4);
219        nesting.add(-1);
220        int t = -1, depth = 0;
221        while (m.find()) {
222            if (m.isCaptured(stringId)) {
223                ls.add(current = new ObTextEntry(m.group(stringId)));
224                nesting.incr(depth, 1);
225            }
226            else if(m.isCaptured(openId))
227            {
228                if(current == null) throw new UnsupportedOperationException("ObText entries can't have associated items without a primary String.");
229                nesting.add(-1);
230                ls = current.openAssociated();
231                depth++;
232            }
233            else if(m.isCaptured(closeId))
234            {
235                if(nesting.size <= 1) throw new UnsupportedOperationException("Associated item sequences in ObText can't end more times than they start.");
236                nesting.pop();
237                depth--;
238                ls = entries;
239                for (int i = 1; i < nesting.size; i++) {
240                    ls = ls.get(nesting.get(i)).associated;
241                }
242                nesting.incr(depth, 1);
243            }
244        }
245        return this;
246    }

383    // Used to generate randomized delimiters using up to 9 non-English letters.
384    // call while assigning your state with randomChars(state += 0x9E3779B97F4A7C15L, myChars)
385    // that assumes you have a 9-element char[] called myChars
386    // as long as z/state is deterministic (i.e. based on a hash), this should be too
387    private static void randomChars(long z, char[] mut)
388    {
389        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
390        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
391        z ^= (z >>> 31);
392
393        mut[0] = letters[(int)(128 + (z & 127))];
394        mut[1] = letters[(int)(128 + (z >>> 7 & 127))];
395        mut[2] = letters[(int)(128 + (z >>> 14 & 127))];
396        mut[3] = letters[(int)(128 + (z >>> 21 & 127))];
397        mut[4] = letters[(int)(128 + (z >>> 28 & 127))];
398        mut[5] = letters[(int)(128 + (z >>> 35 & 127))];
399        mut[6] = letters[(int)(128 + (z >>> 42 & 127))];
400        mut[7] = letters[(int)(128 + (z >>> 49 & 127))];
401        mut[8] = letters[(int)(128 + (z >>> 56 & 127))];
402    }

413    protected static void appendQuoted(StringBuilder sb, String text, Matcher bareFinder)
414    {
415        if(text == null || text.isEmpty()) {
416            sb.append("''");
417            return;
418        }
419        bareFinder.setTarget(text);
420        if(!bareFinder.find())
421            sb.append(text);
422        else
423        {
424            raw.setTarget(text);
425            if(raw.find()) {
426
427                if (text.contains("'''")) {
428                    long state = CrossHash.hash64(text);
429                    char[] myChars = new char[9];
430                    int count;
431                    do {
432                        randomChars(state += 0x9E3779B97F4A7C15L, myChars);
433                        count = StringKit.containsPart(text, myChars, "]", "]]");
434                    } while (count == 12);
435                    sb.append("[[").append(myChars, 0, count).append("[\n").append(text).append("\n]")
436                            .append(myChars, 0, count).append("]]");
437                } else {
438                    sb.append("'''\n").append(text).append("\n'''");
439                }
440            }
441            else if(!text.contains("'"))
442            {
443                sb.append('\'').append(text).append('\'');
444            }
445            else
446            {
447                if(text.contains("\""))
448                {
449                    if(text.contains("'''"))
450                    {
451                        long state = CrossHash.hash64(text);
452                        char[] myChars = new char[9];
453                        int count;
454                        do
455                        {
456                            randomChars(state += 0x9E3779B97F4A7C15L, myChars);
457                            count = StringKit.containsPart(text, myChars);
458                        }while(count == 9);
459                        sb.append("[[").append(myChars, 0, count).append("[\n").append(text).append("\n]")
460                                .append(myChars, 0, count).append("]]");
461                    }
462                    else
463                    {
464                        sb.append("'''\n").append(text).append("\n'''");
465                    }
466                }
467                else
468                {
469                    sb.append('"').append(text).append('"');
470                }
471            }
472        }
473    }

503    private static void iterate(StringBuilder sb, List<ObTextEntry> it)
504    {
505        int len = it.size();
506        ObTextEntry entry;
507        for (int i = 0; i < len; i++) {
508            appendQuotedObText(sb, (entry = it.get(i)).primary);
509            sb.append('\n');
510            if(entry.hasAssociated())
511            {
512                sb.append("[\n");
513                iterate(sb, entry.associated);
514                sb.append("]\n");
515            }
516        }
517    }

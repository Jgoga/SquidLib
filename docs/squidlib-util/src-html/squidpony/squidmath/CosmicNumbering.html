005/**
006 * Like a kind of RNG, but fully deterministic in a way that depends on certain connected variables.
007 * Intended as a way to produce similar values when small changes occur in the connections, while potentially producing
008 * larger changes when the changes are more significant (unlike an RNG or hashing function, which can and should produce
009 * very different output given even slightly different seeds/input). This might be useful to produce procedural story
010 * data that is similar when most of the connected inputs are similar, or for terrain generation/population. This can
011 * produce ints and doubles, and does not produce a different output unless its input is changed (usually by altering a
012 * shared reference to {@code connections}).
013 * <br>
014 * Created by Tommy Ettinger on 5/18/2017.
015 */

040    /*
041     * Quintic-interpolates between start and end (valid floats), with a between 0 (yields start) and 1 (yields end).
042     * Will smoothly transition toward start or end as a approaches 0 or 1, respectively.
043     * @param start a valid float
044     * @param end a valid float
045     * @param a a float between 0 and 1 inclusive
046     * @return a float between x and y inclusive
047     */
048    public static double querp(final double start, final double end, double a){
049        return (1.0 - (a *= a * a * (a * (a * 6.0 - 15.0) + 10.0))) * start + a * end;
050    }
051    /*
052     * Linearly interpolates between start and end (valid floats), with a between 0 (yields start) and 1 (yields end).
053     * @param start a valid float
054     * @param end a valid float
055     * @param a a float between 0 and 1 inclusive
056     * @return a float between x and y inclusive
057     */
058    public static double interpolate(final double start, final double end, final double a)
059    {
060        return (1.0 - a) * start + a * end;
061    }

063    /**
064     * Gets a double determined by the current values in the connections, accessible via {@link #getConnections()}.
065     * Returns a value between -1.0 and 1.0 (exclusive on 1.0). Used as the basis for other methods in this class.
066     * @return a double between -1.0 and 1.0; will be the same value until/unless connections change
067     */
068    public double getDoubleBase()
069    {
070        double[] connections = this.connections;
071        final int len = connections.length;
072        long floor, seed = 1234567;
073        double diff, conn, result = 0.0;
074        for (int i = 0; i < len; i++) {
075            diff = (conn = connections[i]) - (floor = fastFloor(conn));
076            seed += 10000;
077            result += querp(
078                    NumberTools.formCurvedFloat(NumberTools.splitMix64(floor * seed + 100 * (i + 1))),
079                    NumberTools.formCurvedFloat(NumberTools.splitMix64((floor + 1L) * seed + 100 * (i + 1))),
080                    diff
081            );
082        }
083        return NumberTools.bounce(5.0 + 2.4 * result);
084    }

126    /**
127     * Gets an int determined by the current values in the connections, accessible via {@link #getConnections()}.
128     * Returns a value in the full range of ints, but is less likely to produce ints close to {@link Integer#MAX_VALUE}
129     * or {@link Integer#MIN_VALUE} (expect very few values in the bottom and top quarters of the range).
130     * @return an int which can be positive or negative; will be the same value until/unless connections change
131     */
132    public int getInt()
133    {
134        return (int)(0x80000000 * getDoubleBase());
135    }

137    /**
138     * Like {@link Math#floor}, but returns an int. Doesn't consider weird doubles like INFINITY and NaN.
139     *
140     * @param t the double to find the floor for
141     * @return the floor of t, as an int
142     */
143    public static long fastFloor(double t) {
144        return t >= 0 ? (long) t : (long) t - 1L;
145    }

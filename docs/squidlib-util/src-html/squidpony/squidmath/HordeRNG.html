001package squidpony.squidmath;
002
003import squidpony.StringKit;
004import squidpony.annotation.Beta;
005
006import java.util.Arrays;
007
008/**
009 * A mix of something like the fast algorithm of LapRNG with the larger state size of LongPeriodRNG, in the hopes of
010 * improving Lap's period without seriously reducing speed. It sorta works, since {@link #nextLong()} is faster than
011 * {@link LongPeriodRNG#nextLong()} by a fair amount, but only "sorta" because methods like {@link #next(int)} slow
012 * down when they are called by classes like RNG, though not as severely as with the related {@link HerdRNG}. There's
013 * some behavior of the JVM at play here, and it may be different across machines and installations. This has 1024 bits
014 * of {@link #state} in an long array with 16 elements, plus 32 bits of semi-state in the {@link #choice} field (used to
015 * decide which of the 16 longs in state to update and query). The period is known to be fairly good, and must be at
016 * least (2 to the 80) but is almost certainly much higher, since testing a variant of this with significantly fewer
017 * bits of state (using 4 shorts instead of 16 longs, with the same int for choice) still had a period greater than 2 to
018 * the 38, implying the period here may be greater than (2 to the 512), and potentially as high as (2 to the 1024),
019 * though this last possibility is very unlikely.
020 * Created by Tommy Ettinger on 6/4/2017.
021 */
022@Beta
023public class HordeRNG implements RandomnessSource {
024    public final long[] state = new long[16];
025    public int choice = 0;
026    public HordeRNG() {
027        this((long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000L)
028                ^ (long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000000L));
029    }
030
031    public HordeRNG(final long seed) {
032        for (int i = 0; i < 16; i++) {
033            state[i] = LightRNG.determine(seed + i);
034        }
035        choice = (int) (state[0] ^ seed);
036    }
037
038    public HordeRNG(final long a, final long b, final long c, final long d,
039                    final long e, final long f, final long g, final long h,
040                    final long i, final long j, final long k, final long l,
041                    final long m, final long n, final long o, final long p)
042    {
043        state[0]  = a;
044        state[1]  = b;
045        state[2]  = c;
046        state[3]  = d;
047        state[4]  = e;
048        state[5]  = f;
049        state[6]  = g;
050        state[7]  = h;
051        state[8]  = i;
052        state[9]  = j;
053        state[10] = k;
054        state[11] = l;
055        state[12] = m;
056        state[13] = n;
057        state[14] = o;
058        state[15] = p;
059        choice = (int)(a ^ ~p);
060    }
061
062    public HordeRNG(final long[] seed) {
063        int len;
064        if (seed == null || (len = seed.length) == 0) {
065            for (int i = 0; i < 16; i++) {
066                state[i] = LightRNG.determine(0xC6BC279692B5C483L + i);
067            }
068            choice = (int) (state[0]);
069        } else if (len < 16) {
070            for (int i = 0, s = 0; i < 16; i++, s++) {
071                if(s == len) s = 0;
072                state[i] ^= seed[s];
073            }
074            choice = (int) (state[0] * len);
075        } else {
076            for (int i = 0, s = 0; s < len; s++, i = (i + 1) & 15) {
077                state[i] ^= seed[s];
078            }
079            choice = (int) (state[0] * len);
080        }
081    }
082
083    public void setState(final long seed) {
084        for (int i = 0; i < 16; i++) {
085            state[i] = LightRNG.determine(seed + i);
086        }
087        choice = (int) (state[0] ^ seed);
088    }
089
090    public final long nextLong() {
091        final int c = (choice += 0x9CBC278D);
092        return (state[c & 15] += (state[c >>> 28] >>> 1) + 0x8E3779B97F4A7C15L);
093        // 0x632AE59B69B3C209L
094
095        //        + high ^ (0x9E3779B97F4A7C15L * ((high += low & (low += 0xAB79B96DCD7FE75EL)) >> 20))); // thunder
096        //        + ((low = (low >>> 1 ^ (-(low & 1L) & 0x6000000000000000L)))) // LFSR, 63-bit
097        ///        ^ (high = high >>> 1 ^ (-(high & 1L) & 0xD800000000000000L))); // LFSR, 64-bit;
098    }
099
100    public final int nextInt() {
101        final int c = (choice += 0x9CBC278D);
102        return (int)(state[c & 15] += (state[c >>> 28] >>> 1) + 0x8E3779B97F4A7C15L);
103        //0x9E3779B97F4A7C15L
104        //0xBE377BB97F4A7C17L
105        /*
106        return (int) ((state1 += (state0 += 0x632AE59B69B3C209L) * 0x9E3779B97F4A7C15L)
107                + (low = (low >>> 1 ^ (-(low & 1L) & 0x6000000000000000L))) // LFSR, 63-bit
108                ^ (high = high >>> 1 ^ (-(high & 1L) & 0xD800000000000000L))); // LFSR, 64-bit;
109        */
110    }
111
112    public final int next(final int bits) {
113        final int c = (choice += 0x9CBC278D);
114        return (int) ((state[c & 15] += (state[c >>> 28] >>> 1) + 0x8E3779B97F4A7C15L) >>> (64 - bits));
115    }
116
117    /**
118     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
119     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
120     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
121     *
122     * @return a copy of this RandomnessSource
123     */
124    @Override
125    public RandomnessSource copy() {
126        HordeRNG hr = new HordeRNG(state);
127        hr.choice = choice;
128        return hr;
129    }
130
131    @Override
132    public String toString() {
133        return "HordeRNG{" +
134                "state=" + StringKit.hex(state) +
135                ", choice=" + choice +
136                '}';
137    }
138
139    @Override
140    public boolean equals(Object o) {
141        if (this == o) return true;
142        if (o == null || getClass() != o.getClass()) return false;
143
144        HordeRNG hordeRNG = (HordeRNG) o;
145
146        if (choice != hordeRNG.choice) return false;
147        return Arrays.equals(state, hordeRNG.state);
148    }
149
150    @Override
151    public int hashCode() {
152        return 31 * choice + CrossHash.Wisp.hash(state);
153    }
154}

005/**
006 * A different kind of noise that has spotted and striped areas, like a tabby cat.
007 * Highly experimental and expected to change; currently has significant spiral-shaped artifacts from stretching.
008 * Created by Tommy Ettinger on 9/2/2017.
009 */
010@Beta
011public class TabbyNoise implements Noise.Noise2D, Noise.Noise3D, Noise.Noise4D, Noise.Noise6D {
012    public static final TabbyNoise instance = new TabbyNoise();
013    public int seed;
014    public TabbyNoise()
015    {
016        seed = 0x1337BEEF;
017    }
018    public TabbyNoise(final int seed)
019    {
020        this.seed = seed;
021    }
022    /*
023     * Quintic-interpolates between start and end (valid floats), with a between 0 (yields start) and 1 (yields end).
024     * Will smoothly transition toward start or end as a approaches 0 or 1, respectively.
025     * @param start a valid float
026     * @param end a valid float
027     * @param a a float between 0 and 1 inclusive
028     * @return a float between x and y inclusive
029     */
030    public static double querp(final double start, final double end, double a){
031        return (1.0 - (a *= a * a * (a * (a * 6.0 - 15.0) + 10.0))) * start + a * end;
032    }
033    /**
034     * Like {@link Math#floor}, but returns an int. Doesn't consider weird doubles like INFINITY and NaN.
035     * @param t the double to find the floor for
036     * @return the floor of t, as an int
037     */
038    public static int fastFloor(double t) {
039        return t >= 0 ? (int) t : (int) t - 1;
040    }

042    @Override

044        return getNoiseWithSeed(x, y, seed);
045    }

047    @Override
048    public double getNoiseWithSeed(double x, double y, int seed) {
049        final int sx = seed | 5, sy = sx + 22222;
050        final long
051                rx = NumberTools.splitMix64(sx),
052                ry = NumberTools.splitMix64(sy);
053        final double
054                mx = ((((rx & 0x1FL) << 4 | 15L) - 255.5) * 0x1.4p-8) * x + ((((rx & 0x1F00L) >>> 4 | 15L) - 255.5) * 0x0.89p-8) * y,
055                my = ((((ry & 0x1FL) << 4 | 15L) - 255.5) * 0x1.4p-8) * y + ((((ry & 0x1F00L) >>> 4 | 15L) - 255.5) * 0x0.89p-8) * x;
056        final long
057                xf = fastFloor(mx),
058                yf = fastFloor(my);
059        return NumberTools.bounce(5.5 + 2.5 *
060                (
061                        querp(NumberTools.formCurvedFloat(NumberTools.splitMix64(xf * sx + 100)),
062                                NumberTools.formCurvedFloat(NumberTools.splitMix64((xf+1) * sx + 100)),
063                                mx - xf)
064                                + querp(NumberTools.formCurvedFloat(NumberTools.splitMix64(yf * sy + 200)),
065                                NumberTools.formCurvedFloat(NumberTools.splitMix64((yf+1) * sy + 200)),
066                                my - yf)));
067    }

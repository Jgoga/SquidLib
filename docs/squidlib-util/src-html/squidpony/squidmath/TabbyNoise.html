022    /*
023     * Quintic-interpolates between start and end (valid floats), with a between 0 (yields start) and 1 (yields end).
024     * Will smoothly transition toward start or end as a approaches 0 or 1, respectively.
025     * @param start a valid float
026     * @param end a valid float
027     * @param a a float between 0 and 1 inclusive
028     * @return a float between x and y inclusive
029     */
030    public static double querp(final double start, final double end, double a){
031        return (1.0 - (a *= a * a * (a * (a * 6.0 - 15.0) + 10.0))) * start + a * end;
032    }
033    /**
034     * Like {@link Math#floor}, but returns an int. Doesn't consider weird doubles like INFINITY and NaN.
035     * @param t the double to find the floor for
036     * @return the floor of t, as an int
037     */
038    public static long fastFloor(double t) {
039        return t >= 0 ? (long) t : (long) t - 1;
040    }
041    public static double gauss(final long state) {
042        final long s1 = state + 0x9E3779B97F4A7C15L,
043                s2 = s1 + 0x9E3779B97F4A7C15L,
044                y = (s1 ^ s1 >>> 30) * 0x5851F42D4C957F2DL,
045                z = (s2 ^ s2 >>> 30) * 0x5851F42D4C957F2DL;
046        return ((((y ^ y >>> 28) & 0x7FFFFFL) + ((y ^ y >>> 28) >>> 41))
047                + (((z ^ z >>> 28) & 0x7FFFFFL) + ((z ^ z >>> 28) >>> 41))) * 0x1p-24 - 1.0;
048    }

055    public double getNoiseWithSeed(double x, double y, int seed) {
056        final long
057                s = ThrustRNG.determine(seed ^ (long)~seed << 32),
058                rx = (seed - s) ^ (s >>> 29 ^ s << 23),
059                ry = (seed + rx) ^ (s >>> 17 ^ s << 36);
060        final double
061                mx = ((((rx & 0x1F0L) | 0x2FL) - 0x1FFp-1) * 0x1.4p-8) * x + ((((rx & 0x1F000L) | 0x2F00L) - 0x1FF00p-1) * 0x1.1p-16) * y,
062                my = ((((ry & 0x1F0L) | 0x2FL) - 0x1FFp-1) * 0x1.4p-8) * y + ((((ry & 0x1F000L) | 0x2F00L) - 0x1FF00p-1) * 0x1.1p-16) * x;
063
064        final long
065                xf = fastFloor(mx),
066                yf = fastFloor(my);
067        return
068                NumberTools.bounce(5.875 + 2.0625 * (
069                        querp(gauss(xf * 0xAE3779B97F4A7E35L),
070                                gauss((xf+1) * 0xAE3779B97F4A7E35L),
071                                mx - xf)
072                                + querp(gauss(yf * 0xBE3779B97F4A7C55L),
073                                gauss((yf+1) * 0xBE3779B97F4A7C55L),
074                                my - yf)));
075    }

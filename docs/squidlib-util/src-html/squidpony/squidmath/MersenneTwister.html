041package squidpony.squidmath;
042
043import java.io.Serializable;
044/**
045 * Mersenne Twister, 64-bit version as a RandomnessSource.
046 * <br>
047 * Similar to the regular Mersenne Twister but is implemented with 64-bit
048 * registers (Java <code>long</code>) and produces different output.
049 * <br>
050 * This is mostly a straight port of the
051 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/VERSIONS/C-LANG/mt19937-64.c">
052 * C-code (mt19937-64.c)</a>, but made more "java-like". This version was originally found
053 * at <a href="https://github.com/javabeanz/javarng/blob/master/com/modp/random/MersenneTwister64.java">
054 * an archived version of a Google Code repo</a>, and it is licensed under the 3-clause BSD license, like
055 * the Mersenne Twister.
056 * <br>
057 * References:
058 * <br>
059 * <ul>
060 * <li>
061 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html">
062 * The Mersenne Twister Home Page </a>
063 * </li>
064 * <li>  Makato Matsumoto and Takuji Nishimura,
065 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/ARTICLES/mt.pdf">"Mersenne Twister:A 623-Dimensionally Equidistributed Uniform Pseudo-Random Number Generator"</a>,
066 * <i>ACM Transactions on Modeling and Computer Simulation, </i> Vol. 8, No. 1,
067 * January 1998, pp 3--30.</li>
068 * </ul>
069 *
070 * @author Nick Galbreath -- nickg [at] modp [dot] com
071 * @author Tommy Ettinger
072 * @version 1.1 -- 07-Oct-2017
073 */
074public class MersenneTwister implements Serializable, RandomnessSource {
075
076    private static final long serialVersionUID = -8219700664442619525L;
077
078    private static final int NN = 312;
079
080    private static final int MM = 156;
081
082    private static final long MATRIX_A = 0xB5026F5AA96619E9L;
083
084    /**
085     * Mask: Most significant 33 bits
086     */
087    private static final long UM = 0xFFFFFFFF80000000L;
088
089    /**
090     * Mask: Least significant 31 bits
091     */
092    private static final long LM = 0x7FFFFFFFL;
093
094    /**
095     * Mersenne Twister data.
096     */
097    private long[] mt = new long[NN];
098
099    /**
100     * Mersenne Twister Index.
101     */
102    private int mti = NN + 1;
103
104    /**
105     * Internal to hold 64 bits, that might
106     * used to generate two 32 bit values.
107     */
108    private long extra;
109
110    /**
111     * Set to true if we need to generate another 64 bits, false if we have enough bits available for an int.
112     */
113    private boolean bitState = true;
114
115    /**
116     * Seeds this using two calls to Math.random().
117     */
118    public MersenneTwister() {
119        setSeed((long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000L)
120                ^ (long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000000L));
121    }
122
123    /**
124     * Seeds this with the given long, which will be used to affect the large state.
125     *
126     * @param seed any long
127     */
128    public MersenneTwister(final long seed) {
129        setSeed(seed);
130    }
131
132    /**
133     * Seeds this with the given long array, which will be used to affect the large state, and not used directly.
134     *
135     * @param seed a long array; generally should be non-null
136     */
137    public MersenneTwister(final long[] seed) {
138        setSeed(seed);
139    }
140
141    /**
142     * Initalize the pseudo random number generator with 64 bits.
143     * Not the same as setState() in StatefulRandomness; this changes the seed quite a bit.
144     *
145     * @param seed any long
146     */
147    public void setSeed(final long seed) {
148        mt[0] = seed;
149        for (mti = 1; mti < NN; mti++) {
150            mt[mti] = (6364136223846793005L * (mt[mti - 1] ^ (mt[mti - 1] >>> 62)) + mti);
151        }
152    }
153
154    /**
155     * Initalize the pseudo random number generator with a long array of any size, which should not be null but can be.
156     * Not the same as setState() in StatefulRandomness; this changes the seed quite a bit.
157     *
158     * @param array any long array
159     */
160    public void setSeed(final long[] array) {
161        setSeed(19650218L);
162        if (array == null)
163            return;
164        int i = 1;
165        int j = 0;
166        int k = (NN > array.length ? NN : array.length);
167        for (; k != 0; k--) {
168            mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >>> 62)) * 3935559000370003845L))
169                    + array[j] + j;
170            i++;
171            j++;
172            if (i >= NN) {
173                mt[0] = mt[NN - 1];
174                i = 1;
175            }
176            if (j >= array.length)
177                j = 0;
178        }
179        for (k = NN - 1; k != 0; k--) {
180            mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >>> 62)) * 2862933555777941757L))
181                    - i;
182            i++;
183            if (i >= NN) {
184                mt[0] = mt[NN - 1];
185                i = 1;
186            }
187        }
188
189        mt[0] = 0x8000000000000000L; /* MSB is 1; assuring non-zero initial array */
190    }
191
192    /**
193     * Returns up to 32 random bits.
194     * <p>
195     * <p>the implementation splits a 64-bit long into
196     * two 32-bit chunks.
197     * </p>
198     */
199    @Override
200    public int next(final int bits) {
201        //return ((int)nextLong()) >>> (32 - numbits);
202
203        if (bitState) {
204            extra = nextLong();
205            bitState = false;
206            return (int) (extra >>> (64 - bits));
207        } else {
208            bitState = true;
209            return ((int) extra) >>> (32 - bits);
210        }
211    }
212
213    /**
214     * Returns 64 random bits.
215     */
216    @Override
217    public long nextLong() {
218        int i;
219        long x;
220        if (mti >= NN) { /* generate NN words at one time */
221
222            for (i = 0; i < NN - MM; i++) {
223                x = (mt[i] & UM) | (mt[i + 1] & LM);
224                mt[i] = mt[i + MM] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
225            }
226            for (; i < NN - 1; i++) {
227                x = (mt[i] & UM) | (mt[i + 1] & LM);
228                mt[i] = mt[i + (MM - NN)] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
229            }
230            x = (mt[NN - 1] & UM) | (mt[0] & LM);
231            mt[NN - 1] = mt[MM - 1] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
232
233            mti = 0;
234        }
235
236        x = mt[mti++];
237
238        x ^= (x >>> 29) & 0x5555555555555555L;
239        x ^= (x << 17) & 0x71D67FFFEDA60000L;
240        x ^= (x << 37) & 0xFFF7EEE000000000L;
241        x ^= (x >>> 43);
242
243        return x;
244    }
245
246    @Override
247    public final MersenneTwister copy() {
248        MersenneTwister f = new MersenneTwister(MATRIX_A);
249        f.mti = mti;
250        f.extra = extra;
251        f.bitState = bitState;
252        System.arraycopy(mt, 0, f.mt, 0, mt.length);
253        return f;
254    }
255

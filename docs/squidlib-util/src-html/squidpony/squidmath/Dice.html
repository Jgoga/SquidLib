001package squidpony.squidmath;
002
003import regexodus.Matcher;
004import regexodus.Pattern;
005import squidpony.StringKit;
006import squidpony.annotation.Beta;

008import java.io.Serializable;

010/**
011 * Class for emulating various traditional RPG-style dice rolls.
012 *
013 * Based on code from the Blacken library.
014 *
015 * @author yam655
016 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
017 */
018@Beta
019public class Dice implements Serializable {
020
021    private static final long serialVersionUID = -488902743486431146L;
022
023    private static final Pattern guessPattern = Pattern.compile("\\s*(\\d+)?\\s*(?:([:><])\\s*(\\d+))??\\s*(?:([d:])\\s*(\\d+))?\\s*(?:([+/*-])\\s*(\\d+))?\\s*");
024    private RNG rng;
025    private transient IntVLA temp = new IntVLA(20);
026    /**
027     * Creates a new dice roller that uses a random RNG seed for an RNG that it owns.
028     */
029    public Dice() {
030        rng = new RNG();
031    }
032
033    /**
034     * Creates a new dice roller that uses the given RNG, which can be seeded before it's given here. The RNG will be
035     * shared, not copied, so requesting a random number from the same RNG in another place may change the value of the
036     * next die roll this makes, and dice rolls this makes will change the state of the shared RNG.
037     * @param rng an RNG that can be seeded; will be shared (dice rolls will change the RNG state outside here)
038     */
039    public Dice(RNG rng)
040    {
041        this.rng = rng;
042    }
043
044    /**
045     * Creates a new dice roller that will use its own RNG, seeded with the given seed.
046     * @param seed a long to use as a seed for a new RNG (can also be an int, short, or byte)
047     */
048    public Dice(long seed)
049    {
050        rng = new RNG(seed);
051    }
052    /**
053     * Creates a new dice roller that will use its own RNG, seeded with the given seed.
054     * @param seed a String to use as a seed for a new RNG
055     */
056    public Dice(String seed)
057    {
058        rng = new RNG(seed);
059    }
060    /**
061     * Sets the random number generator to be used.
062     *
063     * This method does not need to be called before using the methods of this
064     * class.
065     *
066     * @param rng the source of randomness
067     */
068    public void setRandom(RNG rng) {
069        this.rng = rng;
070    }
071
072    /**
073     * Rolls the given number of dice with the given number of sides and returns
074     * the total of the best n dice.
075     *
076     * @param n number of best dice to total
077     * @param dice total number of dice to roll
078     * @param sides number of sides on the dice
079     * @return sum of best n out of <em>dice</em><b>d</b><em>sides</em>
080     */
081    private int bestOf(int n, int dice, int sides) {
082        int rolls = Math.min(n, dice);
083        temp.clear();
084        for (int i = 0; i < dice; i++) {
085            temp.add(rollDice(1, sides));
086        }
087
088        return bestOf(rolls, temp);
089    }
090
091    /**
092     * Rolls the given number of dice with the given number of sides and returns
093     * the total of the lowest n dice.
094     *
095     * @param n number of worst dice to total
096     * @param dice total number of dice to roll
097     * @param sides number of sides on the dice
098     * @return sum of best n out of <em>dice</em><b>d</b><em>sides</em>
099     */
100    private int worstOf(int n, int dice, int sides) {
101        int rolls = Math.min(n, dice);
102        temp.clear();
103        for (int i = 0; i < dice; i++) {
104            temp.add(rollDice(1, sides));
105        }
106
107        return worstOf(rolls, temp);
108    }
109
110    /**
111     * Totals the highest n numbers in the pool.
112     *
113     * @param n the number of dice to be totaled
114     * @param pool the dice to pick from
115     * @return the sum
116     */
117    private int bestOf(int n, IntVLA pool) {
118        int rolls = Math.min(n, pool.size);
119        pool.sort();
120
121        int ret = 0;
122        for (int i = pool.size - 1, r = 0;  r < rolls && i >= 0; i--, r++) {
123            ret += pool.get(i);
124        }
125        return ret;
126    }
127
128    /**
129     * Totals the lowest n numbers in the pool.
130     *
131     * @param n the number of dice to be totaled
132     * @param pool the dice to pick from
133     * @return the sum
134     */
135    private int worstOf(int n, IntVLA pool) {
136        int rolls = Math.min(n, pool.size);
137        pool.sort();
138
139        int ret = 0;
140        for (int r = 0;  r < rolls; r++) {
141            ret += pool.get(r);
142        }
143        return ret;
144    }
145
146    /**
147     * Find the best n totals from the provided number of dice rolled according
148     * to the roll group string.
149     *
150     * @param n number of roll groups to total
151     * @param dice number of roll groups to roll
152     * @param group string encoded roll grouping
153     * @return the sum
154     */
155    public int bestOf(int n, int dice, String group) {
156        int rolls = Math.min(n, dice);
157        temp.clear();
158
159        for (int i = 0; i < dice; i++) {
160            temp.add(roll(group));
161        }
162
163        return bestOf(rolls, temp);
164    }
165
166    /**
167     * Find the worst n totals from the provided number of dice rolled according
168     * to the roll group string.
169     *
170     * @param n number of roll groups to total
171     * @param dice number of roll groups to roll
172     * @param group string encoded roll grouping
173     * @return the sum
174     */
175    public int worstOf(int n, int dice, String group) {
176        int rolls = Math.min(n, dice);
177        temp.clear();
178
179        for (int i = 0; i < dice; i++) {
180            temp.add(roll(group));
181        }
182
183        return worstOf(rolls, temp);
184    }
185
186    /**
187     * Emulate a dice roll and return the sum.
188     *
189     * @param n number of dice to sum
190     * @param sides number of sides on the rollDice
191     * @return sum of rollDice
192     */
193    public int rollDice(int n, int sides) {
194        int ret = 0;
195        for (int i = 0; i < n; i++) {
196            ret += rng.nextIntHasty(sides) + 1;
197        }
198        return ret;
199    }
200
201    /**
202     * Get a list of the independent results of n rolls of dice with the given
203     * number of sides.
204     *
205     * @param n number of dice used
206     * @param sides number of sides on each die
207     * @return list of results
208     */
209    public IntVLA independentRolls(int n, int sides) {
210        IntVLA ret = new IntVLA(n);
211        for (int i = 0; i < n; i++) {
212            ret.add(rng.nextIntHasty(sides) + 1);
213        }
214        return ret;
215    }
216
217    /**
218     * Evaluate the String {@code rollCode} as dice roll notation and roll to get a random result of that dice roll.
219     * This is the main way of using the Dice class. The following notation is supported:
220     * <ul>
221     *     <li>{@code 42} : simple absolute string</li>
222     *     <li>{@code 10:20} : simple random range (inclusive between 10 and 20)</li>
223     *     <li>{@code 3d6} : sum of 3 6-sided dice</li>
224     *     <li>{@code d6} : synonym for {@code 1d6}</li>
225     *     <li>{@code 3>4d6} : best 3 of 4 6-sided dice</li>
226     *     <li>{@code 3:4d6} : synonym for {@code 3>4d6}; older syntax</li>
227     *     <li>{@code 2<5d6} : worst 2 of 5 6-sided dice</li>
228     * </ul>
229     * The following types of suffixes are supported:
230     * <ul>
231     *     <li>{@code +4} : add 4 to the value</li>
232     *     <li>{@code -3} : subtract 3 from the value</li>
233     *     <li>{@code *100} : multiply value by 100</li>
234     *     <li>{@code /8} : divide value by 8</li>
235     * </ul>
236     * @param rollCode string using dice roll notation
237     * @return random number
238     */
239    public int roll(String rollCode) {//TODO -- rework to tokenize and allow multiple chained operations
240        Matcher mat = guessPattern.matcher(rollCode);
241        int ret = 0;
242
243        if (mat.matches()) {
244            String num1 = mat.group(1); // number constant
245            String wmode = mat.group(2); // between notation
246            String wnum = mat.group(3); // number constant
247            String mode = mat.group(4); // d or colon
248            String num2 = mat.group(5); // number constant
249            String pmode = mat.group(6); // math operation
250            String pnum = mat.group(7); // number constant
251
252            int a = num1 == null ? 0 : StringKit.intFromDec(num1);
253            int b = num2 == null ? 0 : StringKit.intFromDec(num2);
254            int w = wnum == null ? 0 : StringKit.intFromDec(wnum);
255
256            if (num1 != null && num2 != null) {
257                if (wnum != null) {
258                    if (">".equals(wmode) || ":".equals(wmode)) {
259                        if ("d".equals(mode)) {
260                            ret = bestOf(a, w, b);
261                        }
262                    }
263                    else if("<".equals(wmode))
264                    {
265                        if ("d".equals(mode)) {
266                            ret = worstOf(a, w, b);
267                        }
268                    }
269                } else if ("d".equals(mode)) {
270                    ret = rollDice(a, b);
271                } else if (":".equals(mode)) {
272                    ret = a + rng.nextIntHasty(b + 1 - a);
273                }
274            } else if (num1 != null) {
275                if (":".equals(wmode)) {
276                    ret = a + rng.nextIntHasty(w + 1 - a);
277                } else {
278                    ret = a;
279                }
280            } else if (num2 != null) {
281                if (mode != null) {
282                    switch (mode) {
283                        case "d":
284                            ret = rollDice(1, b);
285                            break;
286                        case ":":
287                            ret = rng.nextIntHasty(b + 1);
288                            break;
289                    }
290                }
291            }
292            if (pmode != null) {
293                int p = pnum == null ? 0 : StringKit.intFromDec(pnum);
294                switch (pmode) {
295                    case "+":
296                        ret += p;
297                        break;
298                    case "-":
299                        ret -= p;
300                        break;
301                    case "*":
302                        ret *= p;
303                        break;
304                    case "/":
305                        ret /= p;
306                        break;
307                }
308            }
309        }
310        return ret;
311    }

003import squidpony.StringKit;

005import java.io.Serializable;

007/**
008 * A variant on {@link ThrustRNG} that gives up a small amount of speed to attain better quality. ThrustAltRNG is
009 * expected to pass BigCrush, which is a difficult statistical quality test that is part of TestU01, because it does so
010 * well on other statistical tests. On <a href="http://gjrand.sourceforge.net/">gjrand</a>'s "testunif" checks, this
011 * does very well on 100GB of tested data, with the "Overall summary one sided P-value P = 0.904", where 1 is perfect
012 * and 0.1 or less is a failure. On <a href="http://pracrand.sourceforge.net/">PractRand</a>, this runs past 1TB of
013 * generated numbers without finding any failures, and this version avoids issues with Gap-16 tests that cause ThrustRNG
014 * to fail at 32GB and can cause slight variations on the code here to fail at 256GB. Like ThrustRNG and LightRNG, this
015 * changes its state with a steady fixed increment, and does cipher-like adjustments to the current state to randomize
016 * it. Unlike those two RNGs, ThrustAltRNG uses both the current and subsequent state in its calculations, which appears
017 * critical to improve quality given how simple the algorithm is. The period on ThrustAltRNG is 2 to the 64.
018 * ThrustAltRNG is a bit slower than ThrustRNG (but seems to have better quality), while it is faster than LightRNG,
019 * XoRoRNG, and most other very-high-quality generators (not counting ThrustRNG). Similarly to other cipher-like PRNGs,
020 * ThrustAltRNG has a {@link #determine(long)} method that takes a state as a long and returns a deterministic random
021 * number (each input has one output). Unlike some generators (like PermutedRNG), changing the seed even slightly
022 * generally produces completely different results, which applies primarily to determine() but also the first number
023 * generated in a series of nextLong() calls.
024 * <br>
025 * As an aside, this generator has probably been adjusted more by me than any other generator in the library, and if
026 * quality is only somewhat more important than speed, this should be ideal, though it isn't the fastest and doesn't
027 * have the highest quality. For some purposes, ThrustRNG will be better because it is faster. For others, XoRoRNG or
028 * IsaacRNG may be better because they may have more-provably-good quality. Note than XoRoRNG fails binary rank tests,
029 * which may be important for some usage, while LightRNG and ThrustAltRNG don't. ThrustAltRNG actually does better than
030 * LightRNG on gjrand's tests despite LightRNG using significantly more operations (LightRNG has "P = 0.305" on gjrand
031 * with 100GB tested and fails 2 of 13 tests with grade 1 failures, while ThrustAltRNG fails none). IsaacRNG is also
032 * substantially slower than most other generators, though it offers better promise of security.
033 * <br>
034 * Created by Tommy Ettinger on 10/18/2017.
035 */
036public final class ThrustAltRNG implements StatefulRandomness, Serializable {
037    private static final long serialVersionUID = 3L;
038    /**
039     * Can be any long value.
040     */
041    public long state;
042
043    /**
044     * Creates a new generator seeded using Math.random.
045     */
046    public ThrustAltRNG() {
047        this((long) ((Math.random() - 0.5) * 0x10000000000000L)
048                ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L));
049    }
050
051    public ThrustAltRNG(final long seed) {
052        state = seed;
053    }
054
055    /**
056     * Get the current internal state of the StatefulRandomness as a long.
057     *
058     * @return the current internal state of this object.
059     */
060    @Override
061    public long getState() {
062        return state;
063    }
064
065    /**
066     * Set the current internal state of this StatefulRandomness with a long.
067     *
068     * @param state a 64-bit long
069     */
070    @Override
071    public void setState(long state) {
072        this.state = state;
073    }
074
075    /**
076     * Using this method, any algorithm that might use the built-in Java Random
077     * can interface with this randomness source.
078     *
079     * @param bits the number of bits to be returned
080     * @return the integer containing the appropriate number of bits
081     */
082    @Override
083    public final int next(final int bits) {
084        long z = state;
085        z ^= (((state += 0xA99635D5B8597AE5L) ^ z >>> 23) * 0xAD5DE9A61A9C3D95L);
086        return (int)(z ^ (z >>> 29)) >>> (32 - bits);
087    }
088    /**
089     * Using this method, any algorithm that needs to efficiently generate more
090     * than 32 bits of random data can interface with this randomness source.
091     * <p>
092     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
093     *
094     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
095     */
096    @Override
097    public final long nextLong() {
098        long z = state;
099        z ^= (((state += 0xA99635D5B8597AE5L) ^ z >>> 23) * 0xAD5DE9A61A9C3D95L);
100        return z ^ (z >>> 29);
101    }
102
103    /**
104     * Advances or rolls back the ThrustAltRNG's state without actually generating each number. Skips forward
105     * or backward a number of steps specified by advance, where a step is equal to one call to nextLong(),
106     * and returns the random number produced at that step (you can get the state with {@link #getState()}).
107     *
108     * @param advance Number of future generations to skip over; can be negative to backtrack, 0 gets the most-recently-generated number
109     * @return the random long generated after skipping forward or backwards by {@code advance} numbers
110     */
111    public final long skip(long advance) {
112        long z = state + 0xA99635D5B8597AE5L * (advance - 1L);
113        z ^= (((state += 0xA99635D5B8597AE5L * advance) ^ z >>> 23) * 0xAD5DE9A61A9C3D95L);
114        return z ^ (z >>> 29);
115    }
116
117
118    /**
119     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
120     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
121     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
122     *
123     * @return a copy of this RandomnessSource
124     */
125    @Override
126    public RandomnessSource copy() {
127        return new ThrustAltRNG(state);
128    }
129    @Override
130    public String toString() {
131        return "ThrustAltRNG with state 0x" + StringKit.hex(state) + 'L';
132    }
133
134    @Override
135    public boolean equals(Object o) {
136        if (this == o) return true;
137        if (o == null || getClass() != o.getClass()) return false;
138
139        ThrustAltRNG thrustAltRNG = (ThrustAltRNG) o;
140
141        return state == thrustAltRNG.state;
142    }
143
144    @Override
145    public int hashCode() {
146        return (int) (state ^ (state >>> 32));
147    }
148
149    /**
150     * Returns a random permutation of state; if state is the same on two calls to this, this will return the same
151     * number. This is expected to be called with some changing variable, e.g. {@code determine(++state)}, where
152     * the increment for state should be odd but otherwise doesn't really matter. This multiplies state by
153     * {@code 0xA99635D5B8597AE5L} within this method, so using a small increment won't be much different from using a
154     * very large one, as long as it is odd. The period is 2 to the 64.
155     * @param state a variable that should be different every time you want a different random result;
156     *              using {@code determine(++state)} is recommended to go forwards or {@code determine(--state)} to
157     *              generate numbers in reverse order
158     * @return a pseudo-random permutation of state
159     */
160    public static long determine(long state) {
161        final long z = (state *= 0xA99635D5B8597AE5L) ^ (((state + 0xA99635D5B8597AE5L) ^ state >>> 23) * 0xAD5DE9A61A9C3D95L);
162        return z ^ (z >>> 29);
163    }
164    //for quick one-line pastes of how the algo can be used with "randomize(++state)"
165    //public static long randomize(long state) { final long z = (state *= 0xA99635D5B8597AE5L) ^ (((state + 0xA99635D5B8597AE5L) ^ state >>> 23) * 0xAD5DE9A61A9C3D95L); return z ^ (z >>> 29); }
166
167    /**
168     * Returns a random permutation of state; if state is the same on two calls to this, this will return the same
169     * number. This is expected to be called with a changing variable using a specific pattern, namely
170     * {@code determine(state += 0xA99635D5B8597AE5L)}, which was specifically matched up to the rest of the generator.
171     * You can add the given number to go forwards or subtract it to go backwards in the sequence. The period
172     * is 2 to the 64. This method may rarely be preferable over {@link #determine(long)} if your code controls how
173     * state is updated, and can make sure it updates by the given amount or a similar value; this version saves a small
174     * amount of time by not needing to multiply the local state.
175     * @param state a variable that should be different every time you want a different random result;
176     *              using {@code determine(state += 0xA99635D5B8597AE5L)} is recommended to go forwards or
177     *              {@code determine(state -= 0xA99635D5B8597AE5L)} to generate numbers in reverse order
178     * @return a pseudo-random permutation of state
179     */
180    public static long determineBare(long state)
181    {
182        final long z = state ^ (((state + 0xA99635D5B8597AE5L) ^ state >>> 23) * 0xAD5DE9A61A9C3D95L);
183        return z ^ (z >>> 29);
184    }
185
186    /**
187     * Given a state that should usually change each time this is called, and a bound that limits the result to some
188     * (typically fairly small) int, produces a pseudo-random int between 0 and bound (exclusive). The bound can be
189     * negative, which will cause this to produce 0 or a negative int; otherwise this produces 0 or a positive int.
190     * The state should change each time this is called, generally by incrementing by an odd number (not an even number,
191     * especially not 0). It's fine to use {@code determineBounded(++state, bound)} to get a different int each time.
192     * The period is usually 2 to the 64, but some bounds may reduce the period (in the extreme case, a bound of 1 would
193     * force only 0 to be generated, so that would make the period 1).
194     * @param state a variable that should be different every time you want a different random result;
195     *              using {@code determineBounded(++state, bound)} is recommended to go forwards or
196     *              {@code determineBounded(--state, bound)} to generate numbers in reverse order
197     * @param bound the outer exclusive bound for the int this produces; can be negative or positive
198     * @return a pseudo-random int between 0 (inclusive) and bound (exclusive)
199     */
200    public static int determineBounded(long state, final int bound)
201    {
202        final long z = (state *= 0xA99635D5B8597AE5L) ^ (((state + 0xA99635D5B8597AE5L) ^ state >>> 23) * 0xAD5DE9A61A9C3D95L);
203        return (int)((bound * ((z ^ (z >>> 29)) & 0xFFFFFFFFL)) >> 32);
204    }
205}

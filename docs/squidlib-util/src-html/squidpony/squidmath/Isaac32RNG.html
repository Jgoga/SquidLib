003/**
004 * This is a port of the public domain Isaac (cryptographic) random number generator to Java, by Bob Jenkins.
005 * It is a RandomnessSource here, so it should generally be used to make an RNG, which has more features.
006 * Isaac32RNG is slower than the non-cryptographic RNGs in SquidLib, but much faster than cryptographic RNGs
007 * that need SecureRandom, and it's compatible with GWT and Android to boot! Isaac32RNG should perform better
008 *  when you specifically need a large amount of int values to be set using {@link #fillBlock(int[])}. If you
009 *  don't need that, then {@link IsaacRNG} will have better properties.
010 * Created by Tommy Ettinger on 8/1/2016.
011 */
012public class Isaac32RNG implements RandomnessSource {
013    private int count;                           /* count through the results in results[] */
014    private int results[];                                /* the results given to the user */
015    private int mem[];                                   /* the internal state */
016    private int a;                                              /* accumulator */
017    private int b;                                          /* the last result */
018    private int c;              /* counter, guarantees cycle is at least 2^^40 */
019
020
021    /* no seed, equivalent to randinit(ctx,FALSE) in C */
022    public Isaac32RNG() {
023        mem = new int[256];
024        results = new int[256];
025        init(false);
026    }
027
028    /* equivalent to randinit(ctx, TRUE) after putting seed in randctx in C */
029    public Isaac32RNG(final int seed[]) {
030        mem = new int[256];
031        results = new int[256];
032        if(seed == null)
033            init(false);
034        else {
035            System.arraycopy(seed, 0, results, 0, Math.min(256, seed.length));
036            init(true);
037        }
038    }
039    /**
040     * Constructs an IsaacRNG with its state filled by the value of seed, run through the LightRNG algorithm.
041     * @param seed any long; will have equal influence on all bits of state
042     */
043    public Isaac32RNG(long seed) {
044        mem = new int[256];
045        results = new int[256];
046        long z;
047        for (int i = 0; i < 256; i++) {
048            z = seed += 0x9E3779B97F4A7C15L;
049            z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
050            z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
051            results[i++] = (int) ((z ^ (z >>> 31)) & 0xffffffffL);
052            results[i] = (int) ((z ^ (z >>> 31)) >>> 32);
053        }
054        init(true);
055    }
056
057    /**
058     * Constructs an IsaacRNG with its state filled by repeated hashing of seed.
059     * @param seed a String that should be exceptionally long to get the best results.
060     */
061    public Isaac32RNG(String seed) {
062        mem = new int[256];
063        results = new int[256];
064        if(seed == null)
065            init(false);
066        else {
067            char[] chars = seed.toCharArray();
068            int slen = chars.length, i = 0;
069            for (; i < 256 && i < slen; i++) {
070                results[i] = CrossHash.hash(chars, i, slen);
071            }
072            for (; i < 256; i++) {
073                results[i] = CrossHash.hash(results);
074            }
075            init(true);
076        }
077    }
078
079    private Isaac32RNG(Isaac32RNG other)
080    {
081        this(other.results);
082    }
083

085    /**
086     * Generates 256 results to be used by later calls to next() or nextLong().
087     * This is a fast (not small) implementation.
088     */
089    public final void regen() {
090        int i, j, x, y;
091
092        b += ++c;
093        for (i = 0, j = 128; i < 128; ) {
094            x = mem[i];
095            a = (a ^ a << 13) + mem[j++];
096            mem[i] = y = mem[x >> 2 & 255] + a + b;
097            results[i++] = b = mem[y >> 10 & 255] + x;
098
099            x = mem[i];
100            a = (a ^ a >>> 6) + mem[j++];
101            mem[i] = y = mem[x >> 2 & 255] + a + b;
102            results[i++] = b = mem[y >> 10 & 255] + x;
103
104            x = mem[i];
105            a = (a ^ a << 2) + mem[j++];
106            mem[i] = y = mem[x >> 2 & 255] + a + b;
107            results[i++] = b = mem[y >> 10 & 255] + x;
108
109            x = mem[i];
110            a = (a ^ a >>> 16) + mem[j++];
111            mem[i] = y = mem[x >> 2 & 255] + a + b;
112            results[i++] = b = mem[y >> 10 & 255] + x;
113        }
114
115        for (j = 0; j < 128; ) {
116            x = mem[i];
117            a = (a ^ a << 13) + mem[j++];
118            mem[i] = y = mem[x >> 2 & 255] + a + b;
119            results[i++] = b = mem[y >> 10 & 255] + x;
120
121            x = mem[i];
122            a = (a ^ a >>> 6) + mem[j++];
123            mem[i] = y = mem[x >> 2 & 255] + a + b;
124            results[i++] = b = mem[y >> 10 & 255] + x;
125
126            x = mem[i];
127            a = (a ^ a << 2) + mem[j++];
128            mem[i] = y = mem[x >> 2 & 255] + a + b;
129            results[i++] = b = mem[y >> 10 & 255] + x;
130
131            x = mem[i];
132            a = (a ^ a >>> 16) + mem[j++];
133            mem[i] = y = mem[x >> 2 & 255] + a + b;
134            results[i++] = b = mem[y >> 10 & 255] + x;
135        }
136    }

139    /**
140     * Initializes this IsaacRNG; typically used from the constructor but can be called externally.
141     *
142     * @param flag if true, use data from seed; if false, initializes this to unseeded random state
143     */
144    public final void init(boolean flag) {
145        int i;
146        int a, b, c, d, e, f, g, h;
147        a = b = c = d = e = f = g = h = 0x9e3779b9;                        /* the golden ratio */
148
149        for (i = 0; i < 4; ++i) {
150            a ^= b << 11;
151            d += a;
152            b += c;
153            b ^= c >>> 2;
154            e += b;
155            c += d;
156            c ^= d << 8;
157            f += c;
158            d += e;
159            d ^= e >>> 16;
160            g += d;
161            e += f;
162            e ^= f << 10;
163            h += e;
164            f += g;
165            f ^= g >>> 4;
166            a += f;
167            g += h;
168            g ^= h << 8;
169            b += g;
170            h += a;
171            h ^= a >>> 9;
172            c += h;
173            a += b;
174        }
175
176        for (i = 0; i < 256; i += 8) {              /* fill in mem[] with messy stuff */
177            if (flag) {
178                a += results[i];
179                b += results[i + 1];
180                c += results[i + 2];
181                d += results[i + 3];
182                e += results[i + 4];
183                f += results[i + 5];
184                g += results[i + 6];
185                h += results[i + 7];
186            }
187            a ^= b << 11;
188            d += a;
189            b += c;
190            b ^= c >>> 2;
191            e += b;
192            c += d;
193            c ^= d << 8;
194            f += c;
195            d += e;
196            d ^= e >>> 16;
197            g += d;
198            e += f;
199            e ^= f << 10;
200            h += e;
201            f += g;
202            f ^= g >>> 4;
203            a += f;
204            g += h;
205            g ^= h << 8;
206            b += g;
207            h += a;
208            h ^= a >>> 9;
209            c += h;
210            a += b;
211            mem[i] = a;
212            mem[i + 1] = b;
213            mem[i + 2] = c;
214            mem[i + 3] = d;
215            mem[i + 4] = e;
216            mem[i + 5] = f;
217            mem[i + 6] = g;
218            mem[i + 7] = h;
219        }
220
221        if (flag) {           /* second pass makes all of seed affect all of mem */
222            for (i = 0; i < 256; i += 8) {
223                a += mem[i];
224                b += mem[i + 1];
225                c += mem[i + 2];
226                d += mem[i + 3];
227                e += mem[i + 4];
228                f += mem[i + 5];
229                g += mem[i + 6];
230                h += mem[i + 7];
231                a ^= b << 11;
232                d += a;
233                b += c;
234                b ^= c >>> 2;
235                e += b;
236                c += d;
237                c ^= d << 8;
238                f += c;
239                d += e;
240                d ^= e >>> 16;
241                g += d;
242                e += f;
243                e ^= f << 10;
244                h += e;
245                f += g;
246                f ^= g >>> 4;
247                a += f;
248                g += h;
249                g ^= h << 8;
250                b += g;
251                h += a;
252                h ^= a >>> 9;
253                c += h;
254                a += b;
255                mem[i] = a;
256                mem[i + 1] = b;
257                mem[i + 2] = c;
258                mem[i + 3] = d;
259                mem[i + 4] = e;
260                mem[i + 5] = f;
261                mem[i + 6] = g;
262                mem[i + 7] = h;
263            }
264        }
265
266        regen();
267        count = 256;
268    }
269
270    public final int nextInt() {
271        if (0 == count--) {
272            regen();
273            count = 255;
274        }
275        return results[count];
276    }
277    /**
278     * Generates and returns a block of 256 pseudo-random int values.
279     * @return a freshly-allocated array of 256 pseudo-random ints, with all bits possible
280     */
281    public final int[] nextBlock()
282    {
283        regen();
284        final int[] block = new int[256];
285        System.arraycopy(results, 0, block, 0, 256);
286        count = 0;
287        return block;
288    }
289
290    /**
291     * Generates enough pseudo-random int values to fill {@code data} and assigns them to it.
292     */
293    public final void fillBlock(final int[] data)
294    {
295        int len, i;
296        if(data == null || (len = data.length) == 0) return;
297        for (i = 0; len > 256; i += 256, len -= 256) {
298            regen();
299            System.arraycopy(results, 0, data, i, 256);
300        }
301        regen();
302        System.arraycopy(results, 0, data, i, len);
303        count = len & 255;
304    }

306    /**
307     * Generates enough pseudo-random float values between 0f and 1f to fill {@code data} and assigns them to it.
308     * Inclusive on 0f, exclusive on 1f. Intended for cases where you need some large source of randomness to be checked
309     * later repeatedly, such as how permutation tables are used in Simplex noise.
310     */
311    public final void fillFloats(final float[] data)
312    {
313        int len, n;
314        if(data == null || (len = data.length) == 0) return;
315        for (int i = 0; i < len; i++) {
316            data[i] = NumberTools.intBitsToFloat(((n = nextInt()) >>> 9 ^ (n & 0x7fffff)) | 0x3f800000) - 1f;
317        }
318    }
319
320    /**
321     * Generates enough pseudo-random float values between -1f and 1f to fill {@code data} and assigns them to it.
322     * Inclusive on -1f, exclusive on 1f. Intended for cases where you need some large source of randomness to be
323     * checked later repeatedly, such as how permutation tables are used in Simplex noise.
324     */
325    public final void fillSignedFloats(final float[] data)
326    {
327        int len, n;
328        if(data == null || (len = data.length) == 0) return;
329        for (int i = 0; i < len; i++) {
330            data[i] = NumberTools.intBitsToFloat(((n = nextInt()) >>> 9 ^ (n & 0x7fffff)) | 0x40000000) - 3f;
331        }
332    }
333
334    @Override
335    public final int next( int bits ) {
336        return nextInt() >>> 32 - bits;
337    }
338
339    /**
340     * Using this method, any algorithm that needs to efficiently generate more
341     * than 32 bits of random data can interface with this randomness source.
342     * <p>
343     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
344     *
345     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
346     */
347    @Override
348    public long nextLong() {
349        return (nextInt() & 0xffffffffL) | (nextInt() & 0xffffffffL) << 32;
350    }

353    /**
354     * Produces another RandomnessSource, but the new one will not produce the same data as this one.
355     * This is meant to be a "more-secure" generator, so this helps reduce the ability to guess future
356     * results from a given sequence of output.
357     * @return another RandomnessSource with the same implementation but no guarantees as to generation
358     */
359    @Override
360    public RandomnessSource copy() {
361        return new Isaac32RNG(results);
362    }
363    
364    @Override
365    public String toString()
366    {
367        return "Isaac32RNG with a hidden state (id is " + System.identityHashCode(this) + ')';
368    }

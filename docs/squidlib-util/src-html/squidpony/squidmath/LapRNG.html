001package squidpony.squidmath;
002
003import squidpony.StringKit;
004import squidpony.annotation.Beta;
005
006import java.io.Serializable;
007
008/**
009 * Like FlapRNG, this sacrifices quality for speed, but it uses 64-bit math and as such has a larger period (FlapRNG
010 * will repeat the cycle of random numbers it produces after about 8 billion numbers, while this will only repeat after
011 * many more, at least 2 to the 64). As such, LapRNG is like FlapRNG but can generate for a longer period of time (more
012 * laps) without repeating, and in some cases (mainly generating 64-bit values) at even higher speed. It is currently
013 * the fastest RandomnessSource in this library at generating 64-bit longs, which is most of what {@link RNG} uses, and
014 * is second-fastest at generating 32-bit ints, with FlapRNG just slightly faster at that.
015 * <br>
016 * Created by Tommy Ettinger on 5/25/2017.
017 */
018@Beta
019public class LapRNG implements RandomnessSource, Serializable {
020    private static final long serialVersionUID = 1L;
021
022    /**
023     * Constructs a LapRNG with a random internal state, using {@link Math#random()} four times to get enough bits.
024     */
025    public LapRNG(){
026        this((long)((Math.random() * 2.0 - 1.0) * 0x8000000000000L)
027                ^ (long)((Math.random() * 2.0 - 1.0) * 0x8000000000000000L),
028                (long)((Math.random() * 2.0 - 1.0) * 0x8000000000000L)
029                        ^ (long)((Math.random() * 2.0 - 1.0) * 0x8000000000000000L));
030    }
031
032    /**
033     * Given one long for a seed, this fills the two internal longs of state with different values, each produced by
034     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
035     * ends up using with {@link #getState0()} and {@link #getState1()}.
036     * @param seed any long; will not be used verbatim for either internal state
037     */
038    public LapRNG(final long seed) {
039        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
040        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
041    }
042    /**
043     * Given one int for a seed, this fills the two internal longs of state with different values, each produced by
044     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
045     * ends up using with {@link #getState0()} and {@link #getState1()}.
046     * @param seed any int; will not be used verbatim for either internal state
047     */
048    public LapRNG(final int seed) {
049        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
050        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
051    }
052
053    /**
054     * The only constructor that does not modify its seeds in some way; you should ensure seed0 is large enough (at
055     * least 25 bits should be needed to represent it, so greater than 33554432 or less than -33554433). If seed0 is too
056     * small, then this will appear highly repetitive for many cycles before changing its output pattern.
057     * @param seed0 a long that should be greater than 33554432 or less than -33554433, preferably by a large amount
058     * @param seed1 any long
059     */
060    public LapRNG(final long seed0, final long seed1) {
061        state0 = seed0;
062        state1 = seed1;
063    }
064
065    /**
066     * This constructor takes ints for seeds, but internally LapRNG uses longs, so it multiplies each input by a large
067     * constant and adds another to get the state it actually will start with. You can obtain the state that this ends
068     * up using with {@link #getState0()} and {@link #getState1()}.
069     * @param seed0 any int, will not be used verbatim
070     * @param seed1 any int, will not be used verbatim
071     */
072    public LapRNG(final int seed0, final int seed1) {
073        state0 = seed0 * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
074        state1 = seed1 * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
075    }
076
077    /**
078     * This constructor gets a 64-bit hash code from the given String or other CharSequence and gives it to the
079     * constructor that takes one long, {@link #LapRNG(long)}.
080     * @param seed any CharSequence, such as a String
081     */
082    public LapRNG(final CharSequence seed)
083    {
084        this(CrossHash.Wisp.hash64(seed));
085    }
086
087    private long state0, state1;
088
089    /**
090     * Using this method, any algorithm that might use the built-in Java Random
091     * can interface with this randomness source.
092     *
093     * @param bits the number of bits to be returned
094     * @return the integer containing the appropriate number of bits
095     */
096    @Override
097    public final int next( final int bits ) {
098        return (int) ((state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L) >>> (64 - bits));
099    }
100
101    /**
102     * Gets a pseudo-random int, which can be positive or negative but is likely to be drawn from less possible options
103     * than the full range of {@link Integer#MIN_VALUE} to {@link Integer#MAX_VALUE}. Very fast, though.
104     * @return a pseudo-random 32-bit int
105     */
106    public final int nextInt()
107    {
108        return (int) ((state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L) >>> 32);
109    }
110    /**
111     * Using this method, any algorithm that needs to efficiently generate more
112     * than 32 bits of random data can interface with this randomness source.
113     * This implementation is the primary one, and {@link #nextInt()} and {@link #next(int)} just use some portion of
114     * the bits this produces. It should be very, very fast on most desktops, with lower speed relative to its closest
115     * competitor ({@link FlapRNG}) on 32-bit operating systems or 32-bit JREs or JDKs, and possibly lower speed on
116     * mobile devices as well, but at the time of writing LapRNG is the fastest source SquidLib has of pseudo-random
117     * long values.
118     * <p>
119     * Pseudo-random results may be between between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
120     *
121     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
122     */
123    @Override
124    public final long nextLong() {
125        return (state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
126        //return (state0 += (((state1 += 0xC6BC279692B5C483L) >>> 59) + 124) * 0x632AE59B79B4E319L);
127        //0x9E3779B97F4A7C15L
128    }
129
130    /**
131     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
132     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
133     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
134     *
135     * @return a copy of this RandomnessSource
136     */
137    @Override
138    public RandomnessSource copy() {
139        return new LapRNG(state0, state1);
140    }
141
142    /**
143     * Gets the linearly-changing part of the state.
144     * @return one of the two parts of the state as a long
145     */
146    public long getState0() {
147        return state0;
148    }
149
150    /**
151     * Sets the linearly-changing part of the state; if the parameter is too small, the generator will behave poorly.
152     * At minimum, the parameter should have significant bits beyond the 24th bit (greater than 33554432 for positive
153     * longs), and should ideally be larger than can be represented in a 32-bit int.
154     * @param state0 a large-enough long, either greater than 33554432 or less than -33554433, preferably greatly so
155     */
156    public void setState0(long state0) {
157        this.state0 = state0;
158    }
159
160    /**
161     * Gets the irregularly-changing part of the state.
162     * @return one of the two parts of the state as a long
163     */
164    public long getState1() {
165        return state1;
166    }
167
168    /**
169     * Sets the irregularly-changing part of the state; just about any long can be passed as a reasonable value here.
170     * @param state1 any long
171     */
172    public void setState1(long state1) {
173        this.state1 = state1;
174    }
175
176    /**
177     * @param state0 any long
178     * @param state1 any long
179     * @return any long, from the full range
180     */
181    public static long determine(final long state0, final long state1)
182    {
183        return (state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
184    }
185    /**
186     * @param state0 any long
187     * @param state1 any long
188     * @return a pseudo-random double from 0.0 (inclusive) to 1.0 (exclusive)
189     */
190    public static double randomDouble(final long state0, final long state1)
191    {
192        return NumberTools.longBitsToDouble(((state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L)
193                >>> 12) | 0x3ff00000) - 1f;
194    }
195
196    /**
197     * @param state0 any long
198     * @param state1 any long
199     * @return a pseudo-random double from -1.0 (inclusive) to 1.0 (exclusive)
200     */
201    public static double randomSignedDouble(final long state0, final long state1) {
202        return NumberTools.longBitsToDouble(((state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L)
203                >>> 12) | 0x40000000) - 3.0;
204    }
205
206    @Override
207    public String toString() {
208        return "FlapRNG with state0 0x" + StringKit.hex(state0) + ", state1 0x" + StringKit.hex(state1);
209    }
210
211    @Override
212    public int hashCode() {
213        long h = (0x632BE5ABL * state0 ^ state1);
214        return (int) (h ^ h >>> 32);
215    }
216
217    @Override
218    public boolean equals(Object o) {
219        if (this == o) return true;
220        if (o == null || getClass() != o.getClass()) return false;
221
222        LapRNG lapRNG = (LapRNG) o;
223
224        if (state0 != lapRNG.state0) return false;
225        return state1 == lapRNG.state1;
226    }
227}

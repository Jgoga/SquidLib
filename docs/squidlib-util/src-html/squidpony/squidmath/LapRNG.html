003import squidpony.StringKit;
004import squidpony.annotation.Beta;

006import java.io.Serializable;

008/**
009 * Like FlapRNG, this sacrifices quality for speed, but it uses 64-bit math and as such has a larger period (FlapRNG
010 * will repeat the cycle of random numbers it produces after about 8 billion numbers, while this will only repeat after
011 * many more, probably 2 to the 65). As such, LapRNG is like FlapRNG but can generate for a longer period of time (more
012 * laps) without repeating, and in some cases (mainly generating 64-bit values) at even higher speed. It is currently
013 * the fastest RandomnessSource in this library at generating 64-bit longs, which is most of what {@link RNG} uses, and
014 * is second-fastest at generating 32-bit ints, with FlapRNG just slightly faster at that. While the period must be at
015 * least 2 to the 64, it is likely that it is actually 2 to the 65. This is one of three RandomnessSources recommended
016 * for frequent use, but only if speed is the only consideration; the quality of the numbers this produces is very bad
017 * (though it appears random to even a trained human eye, a computer could detect patterns).
018 * <br>
019 * Created by Tommy Ettinger on 5/25/2017.
020 * @see FlapRNG FlapRNG is similar to LapRNG but is implemented using int math; LapRNG should usually be preferred
021 */
022@Beta
023public class LapRNG implements RandomnessSource, Serializable {
024    private static final long serialVersionUID = 1L;
025    /**
026     * Constructs a LapRNG with a random internal state, using {@link Math#random()} three times to get enough bits and
027     * passing responsibility off to {@link #LapRNG(int, int, int)}.
028     */
029    public LapRNG(){
030        this((int) ((Math.random() * 2.0 - 1.0) * 0x80000000),
031                (int) ((Math.random() * 2.0 - 1.0) * 0x80000000),
032                (int) ((Math.random() * 2.0 - 1.0) * 0x80000000));
033    }
034
035    /**
036     * Given one long for a seed, this fills the two internal longs of state with different values, each produced by
037     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
038     * ends up using with {@link #getState0()} and {@link #getState1()}.
039     * @param seed any long; will not be used verbatim for either internal state
040     */
041    public LapRNG(final long seed) {
042        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
043        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
044    }
045    /**
046     * Given one int for a seed, this fills the two internal longs of state with different values, each produced by
047     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
048     * ends up using with {@link #getState0()} and {@link #getState1()}.
049     * @param seed any int; will not be used verbatim for either internal state
050     */
051    public LapRNG(final int seed) {
052        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
053        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
054    }
055
056    /**
057     * The only constructor that does not modify its seeds in some way; you should ensure seed0 is large enough (at
058     * least 25 bits should be needed to represent it, so greater than 33554432 or less than -33554433). If seed0 is too
059     * small, then this will appear highly repetitive for many cycles before changing its output pattern.
060     * @param seed0 a long that should be greater than 33554432 or less than -33554433, preferably by a large amount
061     * @param seed1 any long
062     */
063    public LapRNG(final long seed0, final long seed1) {
064        state0 = seed0;
065        state1 = seed1;
066    }
067
068    /**
069     * This constructor takes ints for seeds, but internally LapRNG uses longs, so it multiplies each input by a large
070     * constant and adds another to get the state it actually will start with. You can obtain the state that this ends
071     * up using with {@link #getState0()} and {@link #getState1()}.
072     * @param seed0 any int, will not be used verbatim
073     * @param seed1 any int, will not be used verbatim
074     */
075    public LapRNG(final int seed0, final int seed1) {
076        state0 = seed0 * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
077        state1 = seed1 * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
078    }
079    /**
080     * This constructor takes 3 ints for seeds, but internally LapRNG uses 2 longs (one, state0, only considers its
081     * upper 48 bits significant and has closely correlated results if inputs don't differ on significant bits). To
082     * ensure the state works fairly well, this uses seed0 to affect the upper 48 bits of state0 (this also uses the
083     * product of seed1 and seed2), while employing all 3 seed parameters again to set state1 (but these can affect all
084     * 64 bits of state1). Note that only 96 bits of parameter are supplied here to set 128 bits of state, which is all
085     * right in this case because LapRNG only has a period of 2 to the 65. Anything more than 65 bits of state isn't
086     * strictly necessary, though giving more should allow different, non-overlapping sequences to be produced.
087     * You can obtain the state that this ends up using with {@link #getState0()} and {@link #getState1()}.
088     * @param seed0 any int, will not be used verbatim; should probably not be 0
089     * @param seed1 any int, will not be used verbatim; should probably not be 0
090     * @param seed2 any int, will not be used verbatim; should probably not be 0
091     */
092    public LapRNG(final int seed0, final int seed1, final int seed2) {
093        state0 = (seed0 * 8219L + seed1 * 4999L * seed2 << 20) ^ 0x8329C6EB9E6AD3E3L;
094        state1 = (seed1 * 0x8329C6EB9E6AD3E3L ^ seed2 - 0xC6BC279692B5C483L) + seed0 * 0x9E3779B97F4A7C15L;
095    }
096
097    /**
098     * This constructor gets three differently-calculated 32-bit hash codes from the given String or other
099     * CharSequence and gives them to the constructor that takes three ints, {@link #LapRNG(int, int, int)}. You can
100     * pass a null seed and this will still work.
101     * @param seed any CharSequence, such as a String
102     */
103    public LapRNG(final CharSequence seed)
104    {
105        this(CrossHash.Mist.alpha.hash(seed), CrossHash.Mist.beta.hash(seed), CrossHash.Mist.gamma.hash(seed));
106    }
107
108    private long state0, state1;
109
110    /**
111     * Using this method, any algorithm that might use the built-in Java Random
112     * can interface with this randomness source.
113     *
114     * @param bits the number of bits to be returned
115     * @return the integer containing the appropriate number of bits
116     */
117    @Override
118    public final int next( final int bits ) {
119        return (int) ((state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L) >>> (64 - bits));
120    }
121
122    /**
123     * Gets a pseudo-random int, which can be positive or negative but is likely to be drawn from less possible options
124     * than the full range of {@link Integer#MIN_VALUE} to {@link Integer#MAX_VALUE}. Very fast, though.
125     * @return a pseudo-random 32-bit int
126     */
127    public final int nextInt()
128    {
129        return (int) (state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
130    }
131    /**
132     * Using this method, any algorithm that needs to efficiently generate more
133     * than 32 bits of random data can interface with this randomness source.
134     * This implementation is the primary one, and {@link #nextInt()} and {@link #next(int)} just use some portion of
135     * the bits this produces. It should be very, very fast on most desktops, with lower speed relative to its closest
136     * competitor ({@link FlapRNG}) on 32-bit operating systems or 32-bit JREs or JDKs, and possibly lower speed on
137     * mobile devices as well, but at the time of writing LapRNG is the fastest source SquidLib has of pseudo-random
138     * long values.
139     * <p>
140     * Pseudo-random results may be between between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
141     *
142     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
143     */
144    @Override
145    public final long nextLong() {
146        return (state1 += ((state0 += 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
147
148        //return (state0 += (((state1 += 0xC6BC279692B5C483L) >>> 59) + 124) * 0x632AE59B79B4E319L);
149        //0x9E3779B97F4A7C15L
150    }
151
152    /**
153     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
154     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
155     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
156     *
157     * @return a copy of this RandomnessSource
158     */
159    @Override
160    public LapRNG copy() {
161        return new LapRNG(state0, state1);
162    }
163
164    /**
165     * Gets the linearly-changing part of the state.
166     * @return one of the two parts of the state as a long
167     */
168    public long getState0() {
169        return state0;
170    }
171
172    /**
173     * Sets the linearly-changing part of the state; if the parameter is too small, the generator will behave poorly.
174     * At minimum, the parameter should have significant bits beyond the 24th bit (greater than 33554432 for positive
175     * longs), and should ideally be larger than can be represented in a 32-bit int.
176     * @param state0 a large-enough long, either greater than 33554432 or less than -33554433, preferably greatly so
177     */
178    public void setState0(long state0) {
179        this.state0 = state0;
180    }
181
182    /**
183     * Gets the irregularly-changing part of the state.
184     * @return one of the two parts of the state as a long
185     */
186    public long getState1() {
187        return state1;
188    }
189
190    /**
191     * Sets the irregularly-changing part of the state; just about any long can be passed as a reasonable value here.
192     * @param state1 any long
193     */
194    public void setState1(long state1) {
195        this.state1 = state1;
196    }
197
198    /**
199     * @param state any long
200     * @return any long, from the full range
201     */
202    public static long determine(final long state)
203    {
204        return (state + ((state * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
205    }
206
207    /**
208     * @param state0 any long
209     * @param state1 any long
210     * @return any long, from the full range
211     */
212    public static long determine(final long state0, final long state1)
213    {
214        return (state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L);
215    }
216
217    /**
218     * @param state any long
219     * @return any long, from the full range
220     */
221    public static int determineBounded(final long state, final int bound)
222    {
223        return (int)((bound * ((state + ((state * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L) & 0x7FFFFFFFL)) >> 31);
224    }
225
226    /**
227     * @param state0 any long
228     * @param state1 any long
229     * @return any long, from the full range
230     */
231    public static int determineBounded(final long state0, final long state1, final int bound)
232    {
233        return (int)((bound * ((state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L) & 0x7FFFFFFFL)) >> 31);
234
235    }
236
237    /**
238     * @param state0 any long
239     * @param state1 any long
240     * @return a pseudo-random double from 0.0 (inclusive) to 1.0 (exclusive)
241     */
242    public static double randomDouble(final long state0, final long state1)
243    {
244        return NumberTools.longBitsToDouble(((state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L)
245                >>> 12) | 0x3ff00000) - 1f;
246    }
247
248    /**
249     * @param state0 any long
250     * @param state1 any long
251     * @return a pseudo-random double from -1.0 (inclusive) to 1.0 (exclusive)
252     */
253    public static double randomSignedDouble(final long state0, final long state1) {
254        return NumberTools.longBitsToDouble(((state1 + ((state0 * 0x9E3779B97F4A7C15L) >> 24) * 0x632AE59B69B3C209L)
255                >>> 12) | 0x40000000) - 3.0;
256    }
257
258    @Override
259    public String toString() {
260        return "LapRNG with state0 0x" + StringKit.hex(state0) + ", state1 0x" + StringKit.hex(state1);
261    }
262
263    @Override
264    public int hashCode() {
265        long h = (0x632BE5ABL * state0 ^ state1);
266        return (int) (h ^ h >>> 32);
267    }
268
269    @Override
270    public boolean equals(Object o) {
271        if (this == o) return true;
272        if (o == null || getClass() != o.getClass()) return false;
273
274        LapRNG lapRNG = (LapRNG) o;
275
276        if (state0 != lapRNG.state0) return false;
277        return state1 == lapRNG.state1;
278    }

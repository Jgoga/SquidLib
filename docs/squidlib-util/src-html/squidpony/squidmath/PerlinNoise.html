003/**
004 * This is Ken Perlin's third revision of his noise function. It is sometimes
005 * referred to as "Simplex Noise". Results are bound by (-1, 1) inclusive.
006 *
007 *
008 * It is significantly faster than his earlier versions. This particular version
009 * was originally from Stefan Gustavson. This is much preferred to the earlier
010 * versions of Perlin Noise due to the reasons noted in the articles:
011 * <ul>
012 * <li>http://www.itn.liu.se/~stegu/simplexnoise/simplexnoise.pdf</li>
013 * <li>http://webstaff.itn.liu.se/~stegu/TNM022-2005/perlinnoiselinks/ch02.pdf</li>
014 * </ul>
015 * But, Gustavson's paper is not without its own issues, particularly for 2D noise.
016 * More detail is noted here,
017 * http://stackoverflow.com/questions/18885440/why-does-simplex-noise-seem-to-have-more-artifacts-than-classic-perlin-noise#21568753
018 * and some changes have been made to 2D noise generation to reduce angular artifacts.
019 * Specifically for the 2D gradient table, code based on Gustavson's paper used 12
020 * points, with some duplicates, and not all on the unit circle. In this version,
021 * points are used on the unit circle starting at (1,0) and moving along the circle
022 * in increments of 1.61803398875 radians, that is, the golden ratio phi, getting the
023 * sin and cosine of 15 points after the starting point and storing them as constants.
024 * This definitely doesn't have a noticeable 45 degree angle artifact, though it does
025 * have, to a lesser extent, some other minor artifacts.
026 * <br>
027 * You can also consider {@link WhirlingNoise} as an alternative, which can be faster
028 * and also reduces the likelihood of angular artifacts. WhirlingNoise does not scale its
029 * input (it doesn't need to), so it won't produce the same results as PerlinNoise for the
030 * same inputs, but it will produce similar shape, density, and aesthetic quality of noise.
031 * @see WhirlingNoise A subclass that has a faster implementation and some different qualities.
032 */

402    /**
403     * 2D simplex noise.
404     * This doesn't use its parameters verbatim; xin and yin are both effectively divided by
405     * ({@link Math#E} * {@link Math#PI}), because without a step like that, any integer parameters would return 0 and
406     * only doubles with a decimal component would produce actual noise. This step allows integers to be passed in a
407     * arguments, and changes the cycle at which 0 is repeated to multiples of (E*PI).
408     *
409     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
410     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
411     * @return noise from -1.0 to 1.0, inclusive
412     */
413    public static double noise(double xin, double yin) {
414        xin *= epi;
415        yin *= epi;
416        double noise0, noise1, noise2; // from the three corners
417        // Skew the input space to determine which simplex cell we're in
418        double skew = (xin + yin) * F2; // Hairy factor for 2D
419        int i = (int) Math.floor(xin + skew);
420        int j = (int) Math.floor(yin + skew);
421        double t = (i + j) * G2;
422        double X0 = i - t; // Unskew the cell origin back to (x,y) space
423        double Y0 = j - t;
424        double x0 = xin - X0; // The x,y distances from the cell origin
425        double y0 = yin - Y0;
426        // For the 2D case, the simplex shape is an equilateral triangle.
427        // Determine which simplex we are in.
428        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
429        // coords
430        if (x0 > y0) {
431            i1 = 1;
432            j1 = 0;
433        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
434        else {
435            i1 = 0;
436            j1 = 1;
437        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
438        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
439        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
440        // where
441        // c = (3-sqrt(3))/6
442        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
443        // unskewed coords
444        double y1 = y0 - j1 + G2;
445        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
446        // unskewed coords
447        double y2 = y0 - 1.0 + 2.0 * G2;
448        // Work out the hashed gradient indices of the three simplex corners
449        int ii = i & 255;
450        int jj = j & 255;
451        int gi0 = perm[ii + perm[jj]];
452        int gi1 = perm[ii + i1 + perm[jj + j1]];
453        int gi2 = perm[ii + 1 + perm[jj + 1]];
454        // Calculate the contribution from the three corners
455        double t0 = 0.5 - x0 * x0 - y0 * y0;
456        if (t0 < 0) {
457            noise0 = 0.0;
458        } else {
459            t0 *= t0;
460            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0); // (x,y) of grad3 used
461            // for 2D gradient
462        }
463        double t1 = 0.5 - x1 * x1 - y1 * y1;
464        if (t1 < 0) {
465            noise1 = 0.0;
466        } else {
467            t1 *= t1;
468            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
469        }
470        double t2 = 0.5 - x2 * x2 - y2 * y2;
471        if (t2 < 0) {
472            noise2 = 0.0;
473        } else {
474            t2 *= t2;
475            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
476        }
477        // Add contributions from each corner to get the final noise value.
478        // The result is scaled to return values in the interval [-1,1].
479        return 70.0 * (noise0 + noise1 + noise2);
480    }

482    /**
483     * 3D simplex noise.
484     *
485     * @param xin X input
486     * @param yin Y input
487     * @param zin Z input
488     * @return noise from -1.0 to 1.0, inclusive
489     */
490    public static double noise(double xin, double yin, double zin) {
491        xin *= epi;
492        yin *= epi;
493        zin *= epi;
494        double n0, n1, n2, n3; // Noise contributions from the four corners
495        // Skew the input space to determine which simplex cell we're in
496        double s = (xin + yin + zin) * F3; // Very nice and simple skew
497        // factor for 3D
498        int i = (int) Math.floor(xin + s);
499        int j = (int) Math.floor(yin + s);
500        int k = (int) Math.floor(zin + s);
501        double t = (i + j + k) * G3;
502        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
503        double Y0 = j - t;
504        double Z0 = k - t;
505        double x0 = xin - X0; // The x,y,z distances from the cell origin
506        double y0 = yin - Y0;
507        double z0 = zin - Z0;
508        // For the 3D case, the simplex shape is a slightly irregular
509        // tetrahedron.
510        // Determine which simplex we are in.
511        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
512        // coords
513        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
514        // coords
515        if (x0 >= y0) {
516            if (y0 >= z0) {
517                i1 = 1;
518                j1 = 0;
519                k1 = 0;
520                i2 = 1;
521                j2 = 1;
522                k2 = 0;
523            } // X Y Z order
524            else if (x0 >= z0) {
525                i1 = 1;
526                j1 = 0;
527                k1 = 0;
528                i2 = 1;
529                j2 = 0;
530                k2 = 1;
531            } // X Z Y order
532            else {
533                i1 = 0;
534                j1 = 0;
535                k1 = 1;
536                i2 = 1;
537                j2 = 0;
538                k2 = 1;
539            } // Z X Y order
540        } else { // x0<y0
541            if (y0 < z0) {
542                i1 = 0;
543                j1 = 0;
544                k1 = 1;
545                i2 = 0;
546                j2 = 1;
547                k2 = 1;
548            } // Z Y X order
549            else if (x0 < z0) {
550                i1 = 0;
551                j1 = 1;
552                k1 = 0;
553                i2 = 0;
554                j2 = 1;
555                k2 = 1;
556            } // Y Z X order
557            else {
558                i1 = 0;
559                j1 = 1;
560                k1 = 0;
561                i2 = 1;
562                j2 = 1;
563                k2 = 0;
564            } // Y X Z order
565        }
566        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
567        // (x,y,z),
568        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
569        // (x,y,z), and
570        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
571        // (x,y,z), where
572        // c = 1/6.
573        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
574        // coords
575        double y1 = y0 - j1 + G3;
576        double z1 = z0 - k1 + G3;
577        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in
578        // (x,y,z) coords
579        double y2 = y0 - j2 + 2.0 * G3;
580        double z2 = z0 - k2 + 2.0 * G3;
581        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in
582        // (x,y,z) coords
583        double y3 = y0 - 1.0 + 3.0 * G3;
584        double z3 = z0 - 1.0 + 3.0 * G3;
585        // Work out the hashed gradient indices of the four simplex corners
586        int ii = i & 255;
587        int jj = j & 255;
588        int kk = k & 255;
589        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
590        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
591        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
592        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
593        // Calculate the contribution from the four corners
594        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
595        if (t0 < 0) {
596            n0 = 0.0;
597        } else {
598            t0 *= t0;
599            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
600        }
601        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
602        if (t1 < 0) {
603            n1 = 0.0;
604        } else {
605            t1 *= t1;
606            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
607        }
608        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
609        if (t2 < 0) {
610            n2 = 0.0;
611        } else {
612            t2 *= t2;
613            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
614        }
615        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
616        if (t3 < 0) {
617            n3 = 0.0;
618        } else {
619            t3 *= t3;
620            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
621        }
622        // Add contributions from each corner to get the final noise value.
623        // The result is scaled to stay just inside [-1,1]
624        return 32.0 * (n0 + n1 + n2 + n3);
625    }

627    /**
628     * 4D simplex noise.
629     *
630     * @param x X position
631     * @param y Y position
632     * @param z Z position
633     * @param w Fourth-dimensional position. It is I, the Fourth-Dimensional Ziltoid the Omniscient!
634     * @return noise from -1.0 to 1.0, inclusive
635     */
636    public static double noise(double x, double y, double z, double w) {
637        x *= epi;
638        y *= epi;
639        z *= epi;
640        w *= epi;
641        // The skewing and unskewing factors are hairy again for the 4D case
642        double n0, n1, n2, n3, n4; // Noise contributions from the five
643        // corners
644        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
645        // we're in
646        double s = (x + y + z + w) * F4; // Factor for 4D skewing
647        int i = (int) Math.floor(x + s);
648        int j = (int) Math.floor(y + s);
649        int k = (int) Math.floor(z + s);
650        int l = (int) Math.floor(w + s);
651        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
652        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
653        double Y0 = j - t;
654        double Z0 = k - t;
655        double W0 = l - t;
656        double x0 = x - X0; // The x,y,z,w distances from the cell origin
657        double y0 = y - Y0;
658        double z0 = z - Z0;
659        double w0 = w - W0;
660        // For the 4D case, the simplex is a 4D shape I won't even try to
661        // describe.
662        // To find out which of the 24 possible simplices we're in, we need
663        // to
664        // determine the magnitude ordering of x0, y0, z0 and w0.
665        // The method below is a good way of finding the ordering of x,y,z,w
666        // and
667        // then find the correct traversal order for the simplex we’re in.
668        // First, six pair-wise comparisons are performed between each
669        // possible pair
670        // of the four coordinates, and the results are used to add up binary
671        // bits
672        // for an integer index.
673        int c =
674                (x0 > y0 ? 32 : 0)
675              + (x0 > z0 ? 16 : 0)
676              + (y0 > z0 ? 8 : 0 )
677              + (x0 > w0 ? 4 : 0 )
678              + (y0 > w0 ? 2 : 0 )
679              + (z0 > w0 ? 1 : 0 );
680        int i1, j1, k1, l1; // The integer offsets for the second simplex
681        // corner
682        int i2, j2, k2, l2; // The integer offsets for the third simplex
683        // corner
684        int i3, j3, k3, l3; // The integer offsets for the fourth simplex
685        // corner
686        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
687        // order.
688        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
689        // y<w and x<w
690        // impossible. Only the 24 indices which have non-zero entries make
691        // any sense.
692        // We use a thresholding to set the coordinates in turn from the
693        // largest magnitude.
694        // The number 3 in the "simplex" array is at the position of the
695        // largest coordinate.
696        i1 = simplex[c][0] >= 3 ? 1 : 0;
697        j1 = simplex[c][1] >= 3 ? 1 : 0;
698        k1 = simplex[c][2] >= 3 ? 1 : 0;
699        l1 = simplex[c][3] >= 3 ? 1 : 0;
700        // The number 2 in the "simplex" array is at the second largest
701        // coordinate.
702        i2 = simplex[c][0] >= 2 ? 1 : 0;
703        j2 = simplex[c][1] >= 2 ? 1 : 0;
704        k2 = simplex[c][2] >= 2 ? 1 : 0;
705        l2 = simplex[c][3] >= 2 ? 1 : 0;
706        // The number 1 in the "simplex" array is at the second smallest
707        // coordinate.
708        i3 = simplex[c][0] >= 1 ? 1 : 0;
709        j3 = simplex[c][1] >= 1 ? 1 : 0;
710        k3 = simplex[c][2] >= 1 ? 1 : 0;
711        l3 = simplex[c][3] >= 1 ? 1 : 0;
712        // The fifth corner has all coordinate offsets = 1, so no need to
713        // look that up.
714        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w)
715        // coords
716        double y1 = y0 - j1 + G4;
717        double z1 = z0 - k1 + G4;
718        double w1 = w0 - l1 + G4;
719        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in
720        // (x,y,z,w) coords
721        double y2 = y0 - j2 + 2.0 * G4;
722        double z2 = z0 - k2 + 2.0 * G4;
723        double w2 = w0 - l2 + 2.0 * G4;
724        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in
725        // (x,y,z,w) coords
726        double y3 = y0 - j3 + 3.0 * G4;
727        double z3 = z0 - k3 + 3.0 * G4;
728        double w3 = w0 - l3 + 3.0 * G4;
729        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in
730        // (x,y,z,w) coords
731        double y4 = y0 - 1.0 + 4.0 * G4;
732        double z4 = z0 - 1.0 + 4.0 * G4;
733        double w4 = w0 - 1.0 + 4.0 * G4;
734        // Work out the hashed gradient indices of the five simplex corners
735        int ii = i & 255;
736        int jj = j & 255;
737        int kk = k & 255;
738        int ll = l & 255;
739        int gi0 = perm[ii + perm[jj + perm[kk + perm[ll]]]] & 31;
740        int gi1
741                = perm[ii + i1 + perm[jj + j1 + perm[kk + k1 + perm[ll + l1]]]] & 31;
742        int gi2
743                = perm[ii + i2 + perm[jj + j2 + perm[kk + k2 + perm[ll + l2]]]] & 31;
744        int gi3
745                = perm[ii + i3 + perm[jj + j3 + perm[kk + k3 + perm[ll + l3]]]] & 31;
746        int gi4
747                = perm[ii + 1 + perm[jj + 1 + perm[kk + 1 + perm[ll + 1]]]] & 31;
748        // Calculate the contribution from the five corners
749        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0;
750        if (t0 < 0) {
751            n0 = 0.0;
752        } else {
753            t0 *= t0;
754            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
755        }
756        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1;
757        if (t1 < 0) {
758            n1 = 0.0;
759        } else {
760            t1 *= t1;
761            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
762        }
763        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2;
764        if (t2 < 0) {
765            n2 = 0.0;
766        } else {
767            t2 *= t2;
768            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
769        }
770        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3;
771        if (t3 < 0) {
772            n3 = 0.0;
773        } else {
774            t3 *= t3;
775            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
776        }
777        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4;
778        if (t4 < 0) {
779            n4 = 0.0;
780        } else {
781            t4 *= t4;
782            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
783        }
784        // Sum up and scale the result to cover the range [-1,1]
785        return 27.0 * (n0 + n1 + n2 + n3 + n4);
786    }

007/**
008 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
009 * periodic results. This is still considered experimental because the exact output may change in future versions, along
010 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
011 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
012 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
013 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
014 * over PerlinNoise using those two methods). Sound good?
015 * <br>
016 * Created by Tommy Ettinger on 12/14/2016.
017 */

1140     * 4D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
1141     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double, double)} and this
1142     * method. Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in
1143     * chunks because it uses a pseudo-random function (curiously, {@link ThrustRNG#determine(long)}, which has rather
1144     * good distribution and is fast) instead of a number chosen by hash from a single 256-element array.
1145     * @param x X input
1146     * @param y Y input
1147     * @param z Z input
1148     * @param w W input (fourth-dimensional)
1149     * @param seed any int; will be used to completely alter the noise
1150     * @return noise from -1.0 to 1.0, inclusive
1151     */
1152    public static double noise(final double x, final double y, final double z, final double w, final int seed)
1153    {
1154        // The skewing and unskewing factors are hairy again for the 4D case
1155
1156        // Skew the (x,y,z,w) space to figure out which cell of 24 simplices
1157        // we're in
1158        double s = (x + y + z + w) * F4; // Factor for 4D skewing
1159        int i = fastFloor(x + s);
1160        int j = fastFloor(y + s);
1161        int k = fastFloor(z + s);
1162        int l = fastFloor(w + s);
1163        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
1164        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
1165        double Y0 = j - t;
1166        double Z0 = k - t;
1167        double W0 = l - t;
1168        double x0 = x - X0; // The x,y,z,w distances from the cell origin
1169        double y0 = y - Y0;
1170        double z0 = z - Z0;
1171        double w0 = w - W0;
1172        // For the 4D case, the simplex is a 4D shape I won't even try to
1173        // describe.
1174        // To find out which of the 24 possible simplices we're in, we need
1175        // to figure out the magnitude ordering of x0, y0, z0 and w0.
1176        // The method below is a good way of finding the ordering of x,y,z,w
1177        // and
1178        // then find the correct traversal order for the simplex we’re in.
1179        // First, six pair-wise comparisons are performed between each
1180        // possible pair
1181        // of the four coordinates, and the results are used to add up binary
1182        // bits
1183        // for an integer index.
1184        int c = (x0 > y0 ? 32 : 0) | (x0 > z0 ? 16 : 0) | (y0 > z0 ? 8 : 0) |
1185                (x0 > w0 ? 4 : 0) | (y0 > w0 ? 2 : 0) | (z0 > w0 ? 1 : 0);
1186
1187        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
1188        // order.
1189        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
1190        // y<w and x<w
1191        // impossible. Only the 24 indices which have non-zero entries make
1192        // any sense.
1193        // We use a thresholding to set the coordinates in turn from the
1194        // largest magnitude.
1195        // The number 3 in the "simplex" array is at the position of the
1196        // largest coordinate.
1197
1198        // The integer offsets for the second simplex corner
1199        int i1 = simplex[c][0] >= 3 ? 1 : 0;
1200        int j1 = simplex[c][1] >= 3 ? 1 : 0;
1201        int k1 = simplex[c][2] >= 3 ? 1 : 0;
1202        int l1 = simplex[c][3] >= 3 ? 1 : 0;
1203        // The number 2 in the "simplex" array is at the second largest
1204        // coordinate.
1205
1206        // The integer offsets for the third simplex corner
1207        int i2 = simplex[c][0] >= 2 ? 1 : 0;
1208        int j2 = simplex[c][1] >= 2 ? 1 : 0;
1209        int k2 = simplex[c][2] >= 2 ? 1 : 0;
1210        int l2 = simplex[c][3] >= 2 ? 1 : 0;
1211        // The number 1 in the "simplex" array is at the second smallest
1212        // coordinate.
1213
1214        // The integer offsets for the fourth simplex corner
1215        int i3 = simplex[c][0] >= 1 ? 1 : 0;
1216        int j3 = simplex[c][1] >= 1 ? 1 : 0;
1217        int k3 = simplex[c][2] >= 1 ? 1 : 0;
1218        int l3 = simplex[c][3] >= 1 ? 1 : 0;
1219        // The fifth corner has all coordinate offsets = 1, so no need to
1220        // look that up.
1221        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
1222        double y1 = y0 - j1 + G4;
1223        double z1 = z0 - k1 + G4;
1224        double w1 = w0 - l1 + G4;
1225        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in (x,y,z,w) coords
1226        double y2 = y0 - j2 + 2.0 * G4;
1227        double z2 = z0 - k2 + 2.0 * G4;
1228        double w2 = w0 - l2 + 2.0 * G4;
1229        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in (x,y,z,w) coords
1230        double y3 = y0 - j3 + 3.0 * G4;
1231        double z3 = z0 - k3 + 3.0 * G4;
1232        double w3 = w0 - l3 + 3.0 * G4;
1233        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in (x,y,z,w) coords
1234        double y4 = y0 - 1.0 + 4.0 * G4;
1235        double z4 = z0 - 1.0 + 4.0 * G4;
1236        double w4 = w0 - 1.0 + 4.0 * G4;
1237
1238        int gi0 = (int)(determine(seed + i + determine(j + determine(k + determine(l)))) & 255);
1239        int gi1 = (int)(determine(seed + i + i1 + determine(j + j1 + determine(k + k1 + determine(l + l1)))) & 255);
1240        int gi2 = (int)(determine(seed + i + i2 + determine(j + j2 + determine(k + k2 + determine(l + l2)))) & 255);
1241        int gi3 = (int)(determine(seed + i + i3 + determine(j + j3 + determine(k + k3 + determine(l + l3)))) & 255);
1242        int gi4 = (int)(determine(seed + i + 1 + determine(j + 1 + determine(k + 1 + determine(l + 1)))) & 255);
1243
1244        // Noise contributions from the five corners are n0 to n4
1245
1246        // Calculate the contribution from the five corners
1247        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0, n0;
1248        if (t0 < 0) {
1249            n0 = 0.0;
1250        } else {
1251            t0 *= t0;
1252            n0 = t0 * t0 * dot(grad4f[gi0], x0, y0, z0, w0);
1253        }
1254        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1, n1;
1255        if (t1 < 0) {
1256            n1 = 0.0;
1257        } else {
1258            t1 *= t1;
1259            n1 = t1 * t1 * dot(grad4f[gi1], x1, y1, z1, w1);
1260        }
1261        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2,  n2;
1262        if (t2 < 0) {
1263            n2 = 0.0;
1264        } else {
1265            t2 *= t2;
1266            n2 = t2 * t2 * dot(grad4f[gi2], x2, y2, z2, w2);
1267        }
1268        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3, n3;
1269        if (t3 < 0) {
1270            n3 = 0.0;
1271        } else {
1272            t3 *= t3;
1273            n3 = t3 * t3 * dot(grad4f[gi3], x3, y3, z3, w3);

1275        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4, n4;

1281        }
1282        // Sum up and scale the result to cover the range [-1,1]
1283        return 27.0 * (n0 + n1 + n2 + n3 + n4);

008/**
009 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
010 * periodic results. This is still considered experimental because the exact output may change in future versions, along
011 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
012 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
013 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
014 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
015 * over PerlinNoise using those two methods). Sound good? This also performs minimal (if any) arithmetic on 64-bit long
016 * numbers, instead using double and int, a style that is ideal for GWT (this is true of PerlinNoise too).
017 * <br>
018 * Created by Tommy Ettinger on 12/14/2016.
019 */
020@Beta
021public class WhirlingNoise extends PerlinNoise implements Noise.Noise2D, Noise.Noise3D, Noise.Noise4D {

225    /**
226     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
227     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
228     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
229     * it uses a pseudo-random function (curiously, {@link LightRNG#determine(long)}, which has rather good distribution
230     * and is fast) instead of a number chosen by hash from a single 256-element array.
231     *
232     * @param xin X input; works well if between 0.0 and 1.0, but anything is accepted
233     * @param yin Y input; works well if between 0.0 and 1.0, but anything is accepted
234     * @return noise from -1.0 to 1.0, inclusive
235     */
236    public static double noise(final double xin, final double yin, final int seed) {
237        //xin *= epi;
238        //yin *= epi;
239        double noise0, noise1, noise2; // from the three corners
240        // Skew the input space to figure out which simplex cell we're in
241        double skew = (xin + yin) * F2; // Hairy factor for 2D
242        int i = fastFloor(xin + skew);
243        int j = fastFloor(yin + skew);
244        double t = (i + j) * G2;
245        double X0 = i - t; // Unskew the cell origin back to (x,y) space
246        double Y0 = j - t;
247        double x0 = xin - X0; // The x,y distances from the cell origin
248        double y0 = yin - Y0;
249        // For the 2D case, the simplex shape is an equilateral triangle.
250        // Determine which simplex we are in.
251        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
252        // coords
253        if (x0 > y0) {
254            i1 = 1;
255            j1 = 0;
256        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
257        else {
258            i1 = 0;
259            j1 = 1;
260        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
261        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
262        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
263        // where
264        // c = (3-sqrt(3))/6
265        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
266        // unskewed coords
267        double y1 = y0 - j1 + G2;
268        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
269        // unskewed coords
270        double y2 = y0 - 1.0 + 2.0 * G2;
271        // Work out the hashed gradient indices of the three simplex corners
272        /*
273        int ii = i & 255;
274        int jj = j & 255;
275        int gi0 = perm[ii + perm[jj]] & 15;
276        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
277        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
278        */
279        /*
280        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
281                i + i1 + ((j + j1) * 0x9E3779B9),
282                i + 1 + ((j + 1) * 0x9E3779B9),
283                seed);
284        int gi0 = hash & 15;
285        int gi1 = (hash >>>= 4) & 15;
286        int gi2 = (hash >>> 4) & 15;
287        */
288        int gi0 = determineBounded(seed + i + determine(j), 16);
289        int gi1 = determineBounded(seed + i + i1 + determine(j + j1), 16);
290        int gi2 = determineBounded(seed + i + 1 + determine(j + 1), 16);
291
292        // Calculate the contribution from the three corners
293        double t0 = 0.5 - x0 * x0 - y0 * y0;
294        if (t0 < 0) {
295            noise0 = 0.0;
296        } else {
297            t0 *= t0;
298            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0);
299            // for 2D gradient
300        }
301        double t1 = 0.5 - x1 * x1 - y1 * y1;
302        if (t1 < 0) {
303            noise1 = 0.0;
304        } else {
305            t1 *= t1;
306            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
307        }
308        double t2 = 0.5 - x2 * x2 - y2 * y2;
309        if (t2 < 0) {
310            noise2 = 0.0;
311        } else {
312            t2 *= t2;
313            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
314        }
315        // Add contributions from each corner to get the final noise value.
316        // The result is scaled to return values in the interval [-1,1].
317        return 70.0 * (noise0 + noise1 + noise2);
318    }

331        //xin *= epi;
332        //yin *= epi;
333        float noise0, noise1, noise2; // from the three corners
334        float xin = (float)x, yin = (float)y;
335        // Skew the input space to figure out which simplex cell we're in
336        float skew = (xin + yin) * F2f; // Hairy factor for 2D
337        int i = fastFloor(xin + skew);
338        int j = fastFloor(yin + skew);
339        float t = (i + j) * G2f;
340        float X0 = i - t; // Unskew the cell origin back to (x,y) space
341        float Y0 = j - t;
342        float x0 = xin - X0; // The x,y distances from the cell origin
343        float y0 = yin - Y0;
344        // For the 2D case, the simplex shape is an equilateral triangle.
345        // Determine which simplex we are in.
346        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
347        // coords
348        if (x0 > y0) {
349            i1 = 1;
350            j1 = 0;
351        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
352        else {
353            i1 = 0;
354            j1 = 1;
355        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
356        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
357        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
358        // where
359        // c = (3-sqrt(3))/6
360        float x1 = x0 - i1 + G2f; // Offsets for middle corner in (x,y)
361        // unskewed coords
362        float y1 = y0 - j1 + G2f;
363        float x2 = x0 - 1f + 2f * G2f; // Offsets for last corner in (x,y)
364        // unskewed coords
365        float y2 = y0 - 1f + 2f * G2f;
366        // Work out the hashed gradient indices of the three simplex corners
367        /*
368        int ii = i & 255;
369        int jj = j & 255;
370        int gi0 = perm[ii + perm[jj]] & 15;
371        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
372        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
373        */
374        /*
375        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
376                i + i1 + ((j + j1) * 0x9E3779B9),
377                i + 1 + ((j + 1) * 0x9E3779B9),
378                seed);
379        int gi0 = hash & 15;
380        int gi1 = (hash >>>= 4) & 15;
381        int gi2 = (hash >>> 4) & 15;
382        */
383        int gi0 = determineBounded(i + determine(j), 16);
384        int gi1 = determineBounded(i + i1 + determine(j + j1), 16);
385        int gi2 = determineBounded(i + 1 + determine(j + 1), 16);
386
387        // Calculate the contribution from the three corners
388        float t0 = 0.5f - x0 * x0 - y0 * y0;
389        if (t0 < 0) {
390            noise0 = 0f;
391        } else {
392            t0 *= t0;
393            //noise0 = t0 * t0 * dotterize(x0, y0, i, j);
394            noise0 = t0 * t0 * dotf(phiGrad2f[gi0], x0, y0);
395        }
396        float t1 = 0.5f - x1 * x1 - y1 * y1;
397        if (t1 < 0) {
398            noise1 = 0f;
399        } else {
400            t1 *= t1;
401            //noise1 = t1 * t1 * dotterize(x1, y1, i + i1, j + j1);
402            noise1 = t1 * t1 * dotf(phiGrad2f[gi1], x1, y1);
403        }
404        float t2 = 0.5f - x2 * x2 - y2 * y2;
405        if (t2 < 0) {
406            noise2 = 0f;
407        } else {
408            t2 *= t2;
409            //noise2 = t2 * t2 * dotterize(x2, y2, i+1, j+1);
410            noise2 = t2 * t2 * dotf(phiGrad2f[gi2], x2, y2);
411        }
412        // Add contributions from each corner to get the final noise value.
413        // The result is scaled to return values in the interval [-1,1].
414        return 70f * (noise0 + noise1 + noise2);
415    }

433     * 3D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
434     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double)} and this method.
435     * Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks
436     * because it uses a pseudo-random function (curiously, {@link LightRNG#determine(long)}, which has rather
437     * good distribution and is fast) instead of a number chosen by hash from a single 256-element array.
438     * @param xin X input
439     * @param yin Y input
440     * @param zin Z input
441     * @return noise from -1.0 to 1.0, inclusive
442     */
443    public static double noise(final double xin, final double yin, final double zin, final int seed){
444        //xin *= epi;
445        //yin *= epi;
446        //zin *= epi;
447        double n0, n1, n2, n3; // Noise contributions from the four corners
448        // Skew the input space to figure out which simplex cell we're in
449        double s = (xin + yin + zin) * F3; // Very nice and simple skew
450        // factor for 3D
451        int i = fastFloor(xin + s);
452        int j = fastFloor(yin + s);
453        int k = fastFloor(zin + s);
454        double t = (i + j + k) * G3;
455        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
456        double Y0 = j - t;
457        double Z0 = k - t;
458        double x0 = xin - X0; // The x,y,z distances from the cell origin
459        double y0 = yin - Y0;
460        double z0 = zin - Z0;
461        // For the 3D case, the simplex shape is a slightly irregular
462        // tetrahedron.
463        // Determine which simplex we are in.
464        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
465        // coords

467        // coords

505                k1 = 0;
506                i2 = 0;

508                k2 = 1;
509            } // Y Z X order

511                i1 = 0;

516                k2 = 0;

518        }

520        // (x,y,z),
521        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
522        // (x,y,z), and

526        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
527        // coords
528        double y1 = y0 - j1 + G3;
529        double z1 = z0 - k1 + G3;

531        // (x,y,z) coords
532        double y2 = y0 - j2 + F3;
533        double z2 = z0 - k2 + F3;
534        double x3 = x0 - 0.5; // Offsets for last corner in
535        // (x,y,z) coords
536        double y3 = y0 - 0.5;
537        double z3 = z0 - 0.5;
538        // Work out the hashed gradient indices of the four simplex corners
539
540        /*

552        int gi2 = determineBounded(seed + i + i2 + determine(j + j2 + determine(k + k2)), 12);
553        int gi3 = determineBounded(seed + i + 1 + determine(j + 1 + determine(k + 1)), 12);

559                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),

569        // Calculate the contribution from the four corners
570        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
571        if (t0 < 0) {
572            n0 = 0.0;
573        } else {
574            t0 *= t0;
575            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
576        }
577        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
578        if (t1 < 0) {
579            n1 = 0.0;
580        } else {
581            t1 *= t1;
582            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
583        }
584        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
585        if (t2 < 0) {
586            n2 = 0.0;
587        } else {
588            t2 *= t2;

592        if (t3 < 0) {
593            n3 = 0.0;
594        } else {
595            t3 *= t3;
596            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
597        }
598        // Add contributions from each corner to get the final noise value.
599        // The result is scaled to stay just inside [-1,1]
600        return 32.0 * (n0 + n1 + n2 + n3);
601    }

792    /**
793     * 4D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
794     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double, double)} and this
795     * method. Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in
796     * chunks because it uses a pseudo-random function (curiously, {@link LightRNG#determine(long)}, which has rather
797     * good distribution and is fast) instead of a number chosen by hash from a single 256-element array.
798     * @param x X input
799     * @param y Y input
800     * @param z Z input
801     * @param w W input (fourth-dimensional)
802     * @param seed any int; will be used to completely alter the noise
803     * @return noise from -1.0 to 1.0, inclusive
804     */
805    public static double noise(final double x, final double y, final double z, final double w, final int seed)
806    {
807        // The skewing and unskewing factors are hairy again for the 4D case
808
809        // Skew the (x,y,z,w) space to figure out which cell of 24 simplices
810        // we're in
811        double s = (x + y + z + w) * F4; // Factor for 4D skewing
812        int i = fastFloor(x + s);
813        int j = fastFloor(y + s);
814        int k = fastFloor(z + s);
815        int l = fastFloor(w + s);
816        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
817        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
818        double Y0 = j - t;
819        double Z0 = k - t;
820        double W0 = l - t;
821        double x0 = x - X0; // The x,y,z,w distances from the cell origin
822        double y0 = y - Y0;
823        double z0 = z - Z0;
824        double w0 = w - W0;
825        // For the 4D case, the simplex is a 4D shape I won't even try to
826        // describe.
827        // To find out which of the 24 possible simplices we're in, we need
828        // to figure out the magnitude ordering of x0, y0, z0 and w0.
829        // The method below is a good way of finding the ordering of x,y,z,w
830        // and
831        // then find the correct traversal order for the simplex we’re in.
832        // First, six pair-wise comparisons are performed between each
833        // possible pair
834        // of the four coordinates, and the results are used to add up binary
835        // bits
836        // for an integer index.
837        int c = (x0 > y0 ? 32 : 0) | (x0 > z0 ? 16 : 0) | (y0 > z0 ? 8 : 0) |
838                (x0 > w0 ? 4 : 0) | (y0 > w0 ? 2 : 0) | (z0 > w0 ? 1 : 0);
839
840        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
841        // order.
842        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
843        // y<w and x<w
844        // impossible. Only the 24 indices which have non-zero entries make
845        // any sense.
846        // We use a thresholding to set the coordinates in turn from the
847        // largest magnitude.
848        // The number 3 in the "simplex" array is at the position of the
849        // largest coordinate.
850
851        // The integer offsets for the second simplex corner
852        int i1 = simplex[c][0] >= 3 ? 1 : 0;
853        int j1 = simplex[c][1] >= 3 ? 1 : 0;
854        int k1 = simplex[c][2] >= 3 ? 1 : 0;
855        int l1 = simplex[c][3] >= 3 ? 1 : 0;
856        // The number 2 in the "simplex" array is at the second largest
857        // coordinate.
858
859        // The integer offsets for the third simplex corner
860        int i2 = simplex[c][0] >= 2 ? 1 : 0;
861        int j2 = simplex[c][1] >= 2 ? 1 : 0;
862        int k2 = simplex[c][2] >= 2 ? 1 : 0;
863        int l2 = simplex[c][3] >= 2 ? 1 : 0;
864        // The number 1 in the "simplex" array is at the second smallest
865        // coordinate.
866
867        // The integer offsets for the fourth simplex corner
868        int i3 = simplex[c][0] >= 1 ? 1 : 0;
869        int j3 = simplex[c][1] >= 1 ? 1 : 0;
870        int k3 = simplex[c][2] >= 1 ? 1 : 0;
871        int l3 = simplex[c][3] >= 1 ? 1 : 0;
872        // The fifth corner has all coordinate offsets = 1, so no need to
873        // look that up.
874        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
875        double y1 = y0 - j1 + G4;
876        double z1 = z0 - k1 + G4;
877        double w1 = w0 - l1 + G4;
878        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in (x,y,z,w) coords
879        double y2 = y0 - j2 + 2.0 * G4;
880        double z2 = z0 - k2 + 2.0 * G4;
881        double w2 = w0 - l2 + 2.0 * G4;
882        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in (x,y,z,w) coords
883        double y3 = y0 - j3 + 3.0 * G4;
884        double z3 = z0 - k3 + 3.0 * G4;
885        double w3 = w0 - l3 + 3.0 * G4;
886        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in (x,y,z,w) coords
887        double y4 = y0 - 1.0 + 4.0 * G4;
888        double z4 = z0 - 1.0 + 4.0 * G4;
889        double w4 = w0 - 1.0 + 4.0 * G4;
890
891        int gi0 = determineBounded(seed + i + determine(j + determine(k + determine(l))), 32);
892        int gi1 = determineBounded(seed + i + i1 + determine(j + j1 + determine(k + k1 + determine(l + l1))), 32);
893        int gi2 = determineBounded(seed + i + i2 + determine(j + j2 + determine(k + k2 + determine(l + l2))), 32);
894        int gi3 = determineBounded(seed + i + i3 + determine(j + j3 + determine(k + k3 + determine(l + l3))), 32);
895        int gi4 = determineBounded(seed + i + 1 + determine(j + 1 + determine(k + 1 + determine(l + 1))), 32);
896
897        // Noise contributions from the five corners are n0 to n4
898
899        // Calculate the contribution from the five corners
900        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0, n0;
901        if (t0 < 0) {
902            n0 = 0.0;
903        } else {
904            t0 *= t0;
905            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
906        }
907        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1, n1;
908        if (t1 < 0) {
909            n1 = 0.0;
910        } else {
911            t1 *= t1;
912            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
913        }
914        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2,  n2;
915        if (t2 < 0) {
916            n2 = 0.0;
917        } else {
918            t2 *= t2;
919            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
920        }
921        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3, n3;
922        if (t3 < 0) {
923            n3 = 0.0;
924        } else {
925            t3 *= t3;
926            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
927        }
928        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4, n4;
929        if (t4 < 0) {
930            n4 = 0.0;
931        } else {
932            t4 *= t4;
933            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
934        }
935        // Sum up and scale the result to cover the range [-1,1]
936        return 27.0 * (n0 + n1 + n2 + n3 + n4);
937    }

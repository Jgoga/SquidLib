849     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
850     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
851     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
852     * it uses a pseudo-random function (curiously, {@link ThrustAltRNG#determine(long)}, which has very good distribution
853     * and is fast) instead of a number chosen by hash from a single 256-element array.
854     *
855     * @param xin X input; works well if between 0.0 and 1.0, but anything is accepted
856     * @param yin Y input; works well if between 0.0 and 1.0, but anything is accepted
857     * @return noise from -1.0 to 1.0, inclusive
858     */
859    public static double noise(final double xin, final double yin, final int seed) {
860        //xin *= epi;
861        //yin *= epi;
862        double noise0, noise1, noise2; // from the three corners
863        // Skew the input space to figure out which simplex cell we're in
864        double skew = (xin + yin) * F2; // Hairy factor for 2D
865        int i = fastFloor(xin + skew);
866        int j = fastFloor(yin + skew);
867        double t = (i + j) * G2;
868        double X0 = i - t; // Unskew the cell origin back to (x,y) space
869        double Y0 = j - t;
870        double x0 = xin - X0; // The x,y distances from the cell origin
871        double y0 = yin - Y0;
872        // For the 2D case, the simplex shape is an equilateral triangle.
873        // determine which simplex we are in.
874        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
875        // coords
876        if (x0 > y0) {
877            i1 = 1;
878            j1 = 0;
879        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
880        else {
881            i1 = 0;
882            j1 = 1;
883        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
884        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
885        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
886        // where
887        // c = (3-sqrt(3))/6
888        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
889        // unskewed coords
890        double y1 = y0 - j1 + G2;
891        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
892        // unskewed coords
893        double y2 = y0 - 1.0 + 2.0 * G2;
894        // Work out the hashed gradient indices of the three simplex corners
895        /*
896        int ii = i & 255;
897        int jj = j & 255;
898        int gi0 = perm[ii + perm[jj]] & 15;
899        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
900        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
901        */
902        /*
903        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
904                i + i1 + ((j + j1) * 0x9E3779B9),
905                i + 1 + ((j + 1) * 0x9E3779B9),
906                seed);
907        int gi0 = hash & 15;
908        int gi1 = (hash >>>= 4) & 15;
909        int gi2 = (hash >>> 4) & 15;
910        */
911        int gi0 = determine256(seed + i + determine(j));
912        int gi1 = determine256(seed + i + i1 + determine(j + j1));
913        int gi2 = determine256(seed + i + 1 + determine(j + 1));
914
915        // Calculate the contribution from the three corners
916        double t0 = 0.75 - x0 * x0 - y0 * y0;
917        if (t0 < 0) {
918            noise0 = 0.0;
919        } else {
920            t0 *= t0;
921            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0);
922            // for 2D gradient
923        }
924        double t1 = 0.75 - x1 * x1 - y1 * y1;
925        if (t1 < 0) {
926            noise1 = 0.0;
927        } else {
928            t1 *= t1;
929            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
930        }
931        double t2 = 0.75 - x2 * x2 - y2 * y2;
932        if (t2 < 0) {
933            noise2 = 0.0;
934        } else {
935            t2 *= t2;
936            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
937        }
938        // Add contributions from each corner to get the final noise value.
939        // The result is scaled to return values in the interval [-1,1].
940        return  9.125 * (noise0 + noise1 + noise2);
941    }

065    public static int rawNoise2D(final int x, final int y) {
066        return ((x * 0x9E3779B9 ^ 0xC6BC2796) * (y * 0x92B5CC83 + 0xD9B4E019) - (x ^ ~y) * 0x632BE59B * (~x + y) * 0x7F4A7C15) >>> 8 & 0xff;
067    }

073    }

075
076        int s,
077                result = (s = 0x632BE5AB ^ x * 0x85157AF5) + 0xC5C5581A // equal to 0x62E2AC0D + 0x62E2AC0D

097        return (rawNoise2D(x - 1, y - 1) & 43) + (rawNoise2D(x - 1, y + 1) & 43) +
098                (rawNoise2D(x + 1, y - 1) & 43) + (rawNoise2D(x + 1, y + 1) & 43) +

100                (rawNoise2D(x, y - 1) & 53) + (rawNoise2D(x, y - 1) & 53) +
101                (rawNoise2D(x, y) & 127) >>> 1;

231        if (zoom < 1)
232            return 0;
233        int v = 0, a = x << 1, b = y << 1;
234        for (int s = zoom; s > 0; s--) {
235            v += noise2D((a += 379) / s, (b += 379) / s) + noise2D(a / s, (b + 1) / s) + noise2D((a + 1) / s, b / s);

243        adj *= adj;

250        for (int s = zoom, u = 1; s > 0; s--, u++) {

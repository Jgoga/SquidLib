005/**
006 * Simple somewhat-continuous noise functions that use long coordinates instead of the traditional double (this approach
007 * works better on a grid). It's called this because it should be possible to replace PerlinNoise with MerlinNoise in
008 * some cases, and because working with noise functions makes me feel like a wizard.
009 */
010@Beta
011public class MerlinNoise {
012
013    public long seed;
014    protected int bits = 56, resolution = 3;
015    public MerlinNoise() {
016        seed = 1L;
017    }
018    public MerlinNoise(long seed, int bits, int resolution)
019    {
020        this.seed = seed;
021        this.bits = (-bits & 63);
022        this.resolution = resolution & 63;
023    }
024
025    public long getSeed() {
026        return seed;
027    }
028
029    public void setSeed(long seed) {
030        this.seed = seed;
031    }
032
033    public int getBits() {
034        return 64 - bits;
035    }
036
037    public void setBits(int bits) {
038        this.bits = (-bits & 63);
039    }
040
041    public int getResolution() {
042        return resolution;
043    }
044
045    public void setResolution(int resolution) {
046        this.resolution = resolution & 63;
047    }
048
049    private static long lorp(long start, long end, long a, long resolution) {
050        a = a * a * ((3L << resolution) - (a << 1));
051        end = ((1L << resolution * 3) - a) * start + a * end >> resolution * 3;
052        return end;
053    }
054    /**
055     * 2D merlin noise.
056     *
057     * @param x x input
058     * @param y y input
059     * @param state state to adjust the output
060     * @param resolution the number of cells between "vertices" where one random number is used fully
061     * @param bits how many bits should be used for each (signed long) output; often this is 8 to output a byte
062     * @return noise from {@code -(1L << bits)} to {@code (1L << bits) - 1L}, both inclusive
063     */
064    public static long noise2D(long x, long y, long state, int resolution, int bits) {
065        long xb = x >>> resolution, yb = y >>> resolution, xr = (x & ~(-1L << resolution)), yr = (y & ~(-1L << resolution)),
066                x0y0 = Noise.PointHash.hashAll(xb, yb, state), x1y0 = Noise.PointHash.hashAll(xb + 1, yb, state),
067                x0y1 = Noise.PointHash.hashAll(xb, yb + 1, state), x1y1 = Noise.PointHash.hashAll(xb + 1, yb + 1, state);
068//                x0y0 = (x0y0b >> 2) + (ThrustAltRNG.determine(x0y0b + 1) >> 2)
069//                        + (ThrustAltRNG.determine(x0y0b + 2) >> 2) + (ThrustAltRNG.determine(x0y0b + 3) >> 2),
070//                x1y0 = (x1y0b >> 2) + (ThrustAltRNG.determine(x1y0b + 1) >> 2)
071//                        + (ThrustAltRNG.determine(x1y0b + 2) >> 2) + (ThrustAltRNG.determine(x1y0b + 3) >> 2),
072//                x0y1 = (x0y1b >> 2) + (ThrustAltRNG.determine(x0y1b + 1) >> 2)
073//                        + (ThrustAltRNG.determine(x0y1b + 2) >> 2) + (ThrustAltRNG.determine(x0y1b + 3) >> 2),
074//                x1y1 = (x1y1b >> 2) + (ThrustAltRNG.determine(x1y1b + 1) >> 2)
075//                        + (ThrustAltRNG.determine(x1y1b + 2) >> 2) + (ThrustAltRNG.determine(x1y1b + 3) >> 2)
076        long xly0 = lorp(x0y0, x1y0, xr, resolution), xly1 = lorp(x0y1, x1y1, xr, resolution),
077        yl = lorp(xly0, xly1, yr, resolution);
078        return yl >> (-bits & 63);
079    }

081    /**
082     * 3D merlin noise.
083     *
084     * @param x x input
085     * @param y y input
086     * @param z z input
087     * @param state state to adjust the output
088     * @param resolution the number of cells between "vertices" where one random number is used fully
089     * @param bits how many bits should be used for each (signed long) output; often this is 8 to output a byte
090     * @return noise from {@code -(1L << bits)} to {@code (1L << bits) - 1L}, both inclusive
091     */
092    public static long noise3D(long x, long y, long z, long state, int resolution, int bits) {
093        long xb = x >> resolution, yb = y >> resolution, zb = z >> resolution,
094                xr = x & ~(-1L << resolution), yr = y & ~(-1L << resolution), zr = z & ~(-1L << resolution),
095                x0y0z0b = Noise.PointHash.hashAll(xb, yb, zb, state), x1y0z0b = Noise.PointHash.hashAll(xb + 1, yb, zb, state),
096                x0y1z0b = Noise.PointHash.hashAll(xb, yb + 1, zb, state), x1y1z0b = Noise.PointHash.hashAll(xb + 1, yb + 1, zb, state),
097                x0y0z1b = Noise.PointHash.hashAll(xb, yb, zb + 1, state), x1y0z1b = Noise.PointHash.hashAll(xb + 1, yb, zb + 1, state),
098                x0y1z1b = Noise.PointHash.hashAll(xb, yb + 1, zb + 1, state), x1y1z1b = Noise.PointHash.hashAll(xb + 1, yb + 1, zb + 1, state),
099                x0y0z0 = (x0y0z0b >> 2) + (ThrustAltRNG.determine(x0y0z0b + 1) >> 2)
100                        + (ThrustAltRNG.determine(x0y0z0b + 2) >> 2) + (ThrustAltRNG.determine(x0y0z0b + 3) >> 2),
101                x1y0z0 = (x1y0z0b >> 2) + (ThrustAltRNG.determine(x1y0z0b + 1) >> 2)
102                        + (ThrustAltRNG.determine(x1y0z0b + 2) >> 2) + (ThrustAltRNG.determine(x1y0z0b + 3) >> 2),
103                x0y1z0 = (x0y1z0b >> 2) + (ThrustAltRNG.determine(x0y1z0b + 1) >> 2)
104                        + (ThrustAltRNG.determine(x0y1z0b + 2) >> 2) + (ThrustAltRNG.determine(x0y1z0b + 3) >> 2),
105                x1y1z0 = (x1y1z0b >> 2) + (ThrustAltRNG.determine(x1y1z0b + 1) >> 2)
106                        + (ThrustAltRNG.determine(x1y1z0b + 2) >> 2) + (ThrustAltRNG.determine(x1y1z0b + 3) >> 2),
107                x0y0z1 = (x0y0z1b >> 2) + (ThrustAltRNG.determine(x0y0z1b + 1) >> 2)
108                        + (ThrustAltRNG.determine(x0y0z1b + 2) >> 2) + (ThrustAltRNG.determine(x0y0z1b + 3) >> 2),
109                x1y0z1 = (x1y0z1b >> 2) + (ThrustAltRNG.determine(x1y0z1b + 1) >> 2)
110                        + (ThrustAltRNG.determine(x1y0z1b + 2) >> 2) + (ThrustAltRNG.determine(x1y0z1b + 3) >> 2),
111                x0y1z1 = (x0y1z1b >> 2) + (ThrustAltRNG.determine(x0y1z1b + 1) >> 2)
112                        + (ThrustAltRNG.determine(x0y1z1b + 2) >> 2) + (ThrustAltRNG.determine(x0y1z1b + 3) >> 2),
113                x1y1z1 = (x1y1z1b >> 2) + (ThrustAltRNG.determine(x1y1z1b + 1) >> 2)
114                        + (ThrustAltRNG.determine(x1y1z1b + 2) >> 2) + (ThrustAltRNG.determine(x1y1z1b + 3) >> 2);
115        return lorp(lorp(lorp(x0y0z0, x1y0z0, xr, resolution), lorp(x0y1z0, x1y1z0, xr, resolution), yr, resolution),
116                lorp(lorp(x0y0z1, x1y0z1, xr, resolution), lorp(x0y1z1, x1y1z1, xr, resolution), yr, resolution), zr, resolution) >> (-bits & 63);
117    }

119    /**
120     * Generates higher-quality continuous-style noise than the other methods, but requires pre-calculating a grid.
121     * Does allow taking a seed because internally it uses a ThrustRNG to quickly generate initial white noise before
122     * processing it into more continuous noise. This generates a lot of random numbers (at least 1 + 14 * height, or
123     * more if width is greater than 64), so ThrustRNG's high speed and fairly good period are both assets here.
124     * <br>
125     * The 2D int array this produces has ints ranging from 1 to 255, with extreme values very unlikely. Because 0 is
126     * impossible for this to generate, it should be fine to use values from this as denominators in division.
127     *
128     * @param width  the width of the 2D int array to generate
129     * @param height the height of the 2D int array to generate
130     * @param seed   the RNG seed to use when pseudo-randomly generating the initial white noise this then processes
131     * @return a 2D int array where each int should be between 1 and 255, inclusive
132     */
133    public static int[][] preCalcNoise2D(int width, int height, long seed) {
134        ThrustRNG random = new ThrustRNG(seed);
135        int w = (width << 1) + 2, h = (height << 1) + 2;
136        GreasedRegion[] regions = new GreasedRegion[]{
137                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
138                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
139                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
140                new GreasedRegion(random, w, h).retract().expand(3)
141        };
142        int[][] data = GreasedRegion.bitSum(regions);
143
144        regions = new GreasedRegion[]{
145                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
146                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
147                new GreasedRegion(random, w, h).retract().expand(3), new GreasedRegion(random, w, h).retract().expand(3),
148                new GreasedRegion(random, w, h).retract().expand(3)
149        };
150        int[][] data2 = GreasedRegion.bitSum(regions), data3 = new int[width][height];
151        for (int x = 0; x < w; x++) {
152            for (int y = 0; y < h; y++) {
153                data[x][y] += 128 - data2[x][y];
154            }
155        }
156        for (int x = 0, dx = 1; x < width; x++, dx += 2) {
157            for (int y = 0, dy = 1; y < height; y++, dy += 2) {
158                data3[x][y] = ((data[dx][dy] << 2) + data[dx - 1][dy] + data[dx + 1][dy] + data[dx][dy + 1] + data[dx][dy - 1]) >>> 3;
159            }
160        }
161        return data3;
162        /*
163        int[][] data = GreasedRegion.bitSum(regions);
164        return GreasedRegion.selectiveNegate(data, new GreasedRegion(random, width, height), 0xff);
165        */
166    }

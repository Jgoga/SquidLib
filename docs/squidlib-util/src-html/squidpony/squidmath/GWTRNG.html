001package squidpony.squidmath;
002
003import squidpony.StringKit;
004
005import java.io.Serializable;
006
007/**
008 * An IRNG implementation that is meant to provide random numbers very quickly when targeting GWT but also to produce
009 * the same numbers when used on desktop, Android, or other platforms, and that can have its state read as a
010 * StatefulRandomness. This uses the same algorithm as {@link Lathe32RNG}, which means it has two 32-bit ints for state
011 * and a period of 0xFFFFFFFFFFFFFFFF (2 to the 64 minus 1), while passing 32TB of PractRand tests (so its quality is
012 * very good). Unlike {@link RNG}, there is no RandomnessSource that can be swapped out, but also somewhat less
013 * indirection on common calls like {@link #nextInt()} and {@link #nextFloat()}. Although this implements
014 * {@link StatefulRandomness}, it is not recommended to use this as the RandomnessSource for a StatefulRNG; you should
015 * use {@link Lathe32RNG} if you want the larger API provided by StatefulRNG and/or RNG while keeping similar, though
016 * probably slightly weaker, GWT performance relative to this class.
017 * <br>
018 * <a href="http://xoroshiro.di.unimi.it/xoroshiro128plus.c">Original version here for xorshiro128+</a>; this version
019 * uses <a href="https://groups.google.com/d/msg/prng/Ll-KDIbpO8k/bfHK4FlUCwAJ">different constants</a> by the same
020 * author, Sebastiano Vigna.
021 * <br>
022 * Written in 2016 by David Blackman and Sebastiano Vigna (vigna@acm.org)
023 * Ported and modified in 2018 by Tommy Ettinger
024 * @author Sebastiano Vigna
025 * @author David Blackman
026 * @author Tommy Ettinger (if there's a flaw, use SquidLib's issues and don't bother Vigna or Blackman, the algorithm here has been adjusted from their work)
027 */
028public final class GWTRNG extends AbstractRNG implements StatefulRandomness, Serializable {
029    private static final long serialVersionUID = 1L;
030
031    private int stateA, stateB;
032
033    /**
034     * Creates a new generator seeded using two calls to Math.random().
035     */
036    public GWTRNG()
037    {
038        setState((int)((Math.random() * 2.0 - 1.0) * 0x80000000), (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
039    }
040    /**
041     * Constructs this GWTRNG by dispersing the bits of seed using {@link #setSeed(int)} across the two parts of state
042     * this has.
043     * @param seed an int that won't be used exactly, but will affect both components of state
044     */
045    public GWTRNG(final int seed) {
046        setSeed(seed);
047    }
048    /**
049     * Constructs this GWTRNG by splitting the given seed across the two parts of state this has with
050     * {@link #setState(long)}.
051     * @param seed a long that will be split across both components of state
052     */
053    public GWTRNG(final long seed) {
054        setState(seed);
055    }
056    /**
057     * Constructs this GWTRNG by calling {@link #setState(int, int)} on stateA and stateB as given; see that method
058     * for the specific details (stateA and stateB are kept as-is unless they are both 0).
059     * @param stateA the number to use as the first part of the state; this will be 1 instead if both seeds are 0
060     * @param stateB the number to use as the second part of the state
061     */
062    public GWTRNG(final int stateA, final int stateB) {
063        setState(stateA, stateB);
064    }
065
066    /**
067     * Get up to 32 bits (inclusive) of random output; the int this produces
068     * will not require more than {@code bits} bits to represent.
069     *
070     * @param bits an int between 1 and 32, both inclusive
071     * @return a random number that fits in the specified number of bits
072     */
073    @Override
074    public final int next(int bits) {
075        final int s0 = stateA;
076        int s1 = stateB;
077        final int result = s0 + s1;
078        s1 ^= s0;
079        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
080        stateB = (s1 << 28 | s1 >>> 4); // c
081        return (result << 10 | result >>> 22) + s0 >>> (32 - bits);
082    }
083
084    /**
085     * Get a random integer between Integer.MIN_VALUE to Integer.MAX_VALUE (both inclusive).
086     *
087     * @return a 32-bit random int.
088     */
089    @Override
090    public final int nextInt() {
091        final int s0 = stateA;
092        int s1 = stateB;
093        final int result = s0 + s1;
094        s1 ^= s0;
095        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
096        stateB = (s1 << 28 | s1 >>> 4); // c
097        return (result << 10 | result >>> 22) + s0;
098    }
099
100    /**
101     * Get a random long between Long.MIN_VALUE to Long.MAX_VALUE (both inclusive).
102     *
103     * @return a 64-bit random long.
104     */
105    @Override
106    public final long nextLong() {
107        final int s0 = stateA;
108        int s1 = stateB;
109        final int high = s0 + s1;
110        s1 ^= s0;
111        final int s00 = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
112        s1 = (s1 << 28 | s1 >>> 4); // c
113        final int low = s00 + s1;
114        s1 ^= s00;
115        stateA = (s00 << 13 | s00 >>> 19) ^ s1 ^ (s1 << 5); // a, b
116        stateB = (s1 << 28 | s1 >>> 4); // c
117        return (long)((high << 10 | high >>> 22) + s0) << 32 ^ ((low << 10 | low >>> 22) + s00);
118    }
119
120    /**
121     * Get a random bit of state, interpreted as true or false with approximately equal likelihood.
122     * This implementation uses a sign check as a safeguard, since its algorithm is based on (but is not equivalent to)
123     * xoroshiro, which recommends a sign check instead of using the least significant bit.
124     *
125     * @return a random boolean.
126     */
127    @Override
128    public final boolean nextBoolean() {
129        final int s0 = stateA;
130        int s1 = stateB;
131        final int result = s0 + s1;
132        s1 ^= s0;
133        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
134        stateB = (s1 << 28 | s1 >>> 4); // c
135        return (result << 10 | result >>> 22) + s0 < 0;
136    }
137
138    /**
139     * Gets a random double between 0.0 inclusive and 1.0 exclusive.
140     * This returns a maximum of 0.9999999999999999 because that is the largest double value that is less than 1.0 .
141     * <br>
142     * This is abstract because some generators may natively work with double or float values, but others may need to
143     * convert a long to a double as with {@code (nextLong() & 0x1fffffffffffffL) * 0x1p-53}, which is recommended if
144     * longs are fast to produce.
145     *
146     * @return a double between 0.0 (inclusive) and 0.9999999999999999 (inclusive)
147     */
148    @Override
149    public final double nextDouble() {
150        final int s0 = stateA;
151        int s1 = stateB;
152        final int high = s0 + s1;
153        s1 ^= s0;
154        final int s00 = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
155        s1 = (s1 << 28 | s1 >>> 4); // c
156        final int low = s00 + s1;
157        s1 ^= s00;
158        stateA = (s00 << 13 | s00 >>> 19) ^ s1 ^ (s1 << 5); // a, b
159        stateB = (s1 << 28 | s1 >>> 4); // c
160        return (((long)((high << 10 | high >>> 22) + s0) << 32 ^ ((low << 10 | low >>> 22) + s00))
161                & 0x1fffffffffffffL) * 0x1p-53;
162    }
163
164    /**
165     * Gets a random float between 0.0f inclusive and 1.0f exclusive.
166     * This returns a maximum of 0.99999994 because that is the largest float value that is less than 1.0f .
167     * <br>
168     * This is abstract because some generators may natively work with double or float values, but others may need to
169     * convert an int or long to a float as with {@code (nextInt() & 0xffffff) * 0x1p-24f},
170     * {@code (nextLong() & 0xffffffL) * 0x1p-24f}, or {@code next(24) * 0x1p-24f}, any of which can work when the
171     * method they call is high-quality and fast. You probably would want to use nextInt() or next() if your
172     * implementation is natively 32-bit and is slower at producing longs, for example.
173     *
174     * @return a float between 0f (inclusive) and 0.99999994f (inclusive)
175     */
176    @Override
177    public final float nextFloat() {
178        final int s0 = stateA;
179        int s1 = stateB;
180        final int result = s0 + s1;
181        s1 ^= s0;
182        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5);
183        stateB = (s1 << 28 | s1 >>> 4);
184        return ((result << 10 | result >>> 22) + s0 & 0xffffff) * 0x1p-24f;
185    }
186
187    /**
188     * Creates a copy of this IRNG; it will generate the same random numbers, given the same calls in order, as this
189     * IRNG at the point copy() is called. The copy will not share references with this IRNG. This implementation will
190     * faithfully reproduce this generator as a copied GWTRNG.
191     * 
192     * @return a copy of this IRNG
193     */
194    @Override
195    public GWTRNG copy() {
196        return new GWTRNG(stateA, stateB);
197    }
198
199    /**
200     * Gets a view of this IRNG in a way that implements {@link Serializable}, which is simply this IRNG.
201     * @return a {@link Serializable} view of this IRNG or a similar one; always {@code this}
202     */
203    @Override
204    public Serializable toSerializable() {
205        return this;
206    }
207    /**
208     * Sets the state of this generator using one int, running it through Zog32RNG's algorithm two times to get 
209     * two ints. If the states would both be 0, state A is assigned 1 instead.
210     * @param seed the int to use to produce this generator's state
211     */
212    public void setSeed(final int seed) {
213        int z = seed + 0xC74EAD55, a = seed ^ z;
214        a ^= a >>> 14;
215        z = (z ^ z >>> 10) * 0xA5CB3;
216        a ^= a >>> 15;
217        stateA = (z ^ z >>> 20) + (a ^= a << 13);
218        z = seed + 0x8E9D5AAA;
219        a ^= a >>> 14;
220        z = (z ^ z >>> 10) * 0xA5CB3;
221        a ^= a >>> 15;
222        stateB = (z ^ z >>> 20) + (a ^ a << 13);
223        if((stateA | stateB) == 0)
224            stateA = 1;
225    }
226
227    public int getStateA()
228    {
229        return stateA;
230    }
231    /**
232     * Sets the first part of the state to the given int. As a special case, if the parameter is 0 and stateB is
233     * already 0, this will set stateA to 1 instead, since both states cannot be 0 at the same time. Usually, you
234     * should use {@link #setState(int, int)} to set both states at once, but the result will be the same if you call
235     * setStateA() and then setStateB() or if you call setStateB() and then setStateA().
236     * @param stateA any int
237     */
238
239    public void setStateA(int stateA)
240    {
241        this.stateA = (stateA | stateB) == 0 ? 1 : stateA;
242    }
243    public int getStateB()
244    {
245        return stateB;
246    }
247
248    /**
249     * Sets the second part of the state to the given int. As a special case, if the parameter is 0 and stateA is
250     * already 0, this will set stateA to 1 and stateB to 0, since both cannot be 0 at the same time. Usually, you
251     * should use {@link #setState(int, int)} to set both states at once, but the result will be the same if you call
252     * setStateA() and then setStateB() or if you call setStateB() and then setStateA().
253     * @param stateB any int
254     */
255    public void setStateB(int stateB)
256    {
257        this.stateB = stateB;
258        if((stateB | stateA) == 0) stateA = 1;
259    }
260
261    /**
262     * Sets the current internal state of this GWTRNG with three ints, where stateA and stateB can each be any int
263     * unless they are both 0 (which will be treated as if stateA is 1 and stateB is 0).
264     * @param stateA any int (if stateA and stateB are both 0, this will be treated as 1)
265     * @param stateB any int
266     */
267    public void setState(int stateA, int stateB)
268    {
269        this.stateA = stateA == 0 && stateB == 0 ? 1 : stateA;
270        this.stateB = stateB;
271    }
272
273    /**
274     * Get the current internal state of the StatefulRandomness as a long.
275     *
276     * @return the current internal state of this object.
277     */
278    @Override
279    public long getState() {
280        return stateA & 0xFFFFFFFFL | ((long)stateB) << 32;
281    }
282
283    /**
284     * Set the current internal state of this StatefulRandomness with a long.
285     *
286     * @param state a 64-bit long. You should avoid passing 0; this implementation will treat it as 1.
287     */
288    @Override
289    public void setState(long state) {
290        stateA = state == 0 ? 1 : (int)(state & 0xFFFFFFFFL);
291        stateB = (int)(state >>> 32);
292    }
293
294    @Override
295    public boolean equals(Object o) {
296        if (this == o) return true;
297        if (o == null || getClass() != o.getClass()) return false;
298
299        GWTRNG gwtrng = (GWTRNG) o;
300
301        if (stateA != gwtrng.stateA) return false;
302        return stateB == gwtrng.stateB;
303    }
304
305    @Override
306    public int hashCode() {
307        return 31 * stateA + stateB;
308    }
309    
310    @Override
311    public String toString() {
312        return "GWTRNG with stateA 0x" + StringKit.hex(stateA) + " and stateB 0x" + StringKit.hex(stateB);
313    }

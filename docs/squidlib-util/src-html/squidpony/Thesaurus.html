068    /**
069     * Allows this Thesaurus to find the exact words in synonyms and, when requested, replace each occurrence with a
070     * different word from the same Collection. Each word in synonyms should have the same part of speech, so "demon"
071     * and "devils" should not be in the same list of synonyms (singular noun and plural noun), but "demon" and "devil"
072     * could be (each is a singular noun). The Strings in synonyms should all be lower-case, since case is picked up
073     * from the text as it is being replaced and not from the words themselves. Proper nouns should normally not be used
074     * as synonyms, since this could get very confusing if it changed occurrences of "Germany" to "France" at random and
075     * a character's name, like "Dorothy", to "Anne", "Emily", "Cynthia", etc. in the middle of a section about Dorothy.
076     * The word matching pattern this uses only matches all-letter words, not words that contain hyphens, apostrophes,
077     * or other punctuation.
078     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
079     * @return this for chaining
080     */
081    public Thesaurus addSynonyms(Collection<String> synonyms)
082    {
083        if(synonyms.isEmpty())
084            return this;
085        long prevState = rng.getState();
086        rng.setState(CrossHash.hash64(synonyms));
087        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
088        for(String syn : synonyms)
089        {
090            mappings.put(syn, shuffler);
091        }
092        rng.setState(prevState);
093        return this;
094    }
095
096    public Thesaurus addReplacement(CharSequence before, String after)
097    {
098        mappings.put(before, new GapShuffler<String>(after));
099        return this;
100    }

102    /**
103     * Allows this Thesaurus to replace a specific keyword, typically containing multiple backtick characters ('`') so
104     * it can't be confused with a "real word," with one of the words in synonyms (chosen in shuffled order). The
105     * backtick is the only punctuation character that this class' word matcher considers part of a word, both for this
106     * reason and because it is rarely used in English text.
107     * @param keyword a word (typically containing backticks, '`') that will be replaced by a word from synonyms
108     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
109     * @return this for chaining
110     */
111    public Thesaurus addCategory(String keyword, Collection<String> synonyms)
112    {
113        if(synonyms.isEmpty())
114            return this;
115        long prevState = rng.getState();
116        rng.setState(CrossHash.hash64(synonyms));
117        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
118        mappings.put(keyword, shuffler);
119        rng.setState(prevState);
120        return this;
121    }

273    private StringBuilder modify(CharSequence text)
274    {
275        Matcher m;
276        StringBuilder sb = new StringBuilder(text);
277        Replacer.StringBuilderBuffer tb, working = Replacer.wrap(sb);
278        StringBuilder tmp;
279        boolean found;
280        FakeLanguageGen.Alteration alt;
281        for (int a = 0; a < alterations.size(); a++) {
282            alt = alterations.get(a);
283            tmp = working.sb;
284            tb = Replacer.wrap(new StringBuilder(tmp.length()));
285            m = alt.replacer.getPattern().matcher(tmp);
286
287            found = false;
288            while (true) {
289                if (rng.nextDouble() < alt.chance) {
290                    if (!Replacer.replaceStep(m, alt.replacer.getSubstitution(), tb))
291                        break;
292                    found = true;
293                } else {
294                    if (!m.find())
295                        break;
296                    found = true;
297                    m.getGroup(MatchResult.PREFIX, tb);
298                    m.getGroup(MatchResult.MATCH, tb);
299                    m.setTarget(m, MatchResult.SUFFIX);
300                }
301            }
302            if (found) {
303                m.getGroup(MatchResult.TARGET, tb);
304                working = tb;
305            }
306        }
307        return working.sb;
308
309    }
310    /**
311     * Given a String, StringBuilder, or other CharSequence that should contain words this knows synonyms for, this
312     * replaces each occurrence of such a known word with one of its synonyms, leaving unknown words untouched. Words
313     * that were learned together as synonyms with addSynonyms() will be replaced in such a way that an individual
314     * replacement word should not occur too close to a previous occurrence of the same word; that is, replacing the
315     * text "You fiend! You demon! You despoiler of creation; devil made flesh!", where "fiend", "demon", and "devil"
316     * are all synonyms, would never produce a string that contained "fiend" as the replacement for all three of those.
317     * @param text a CharSequence, such as a String, that contains words in the source language
318     * @return a String of the translated text.
319     */
320    public String process(CharSequence text)
321    {
322        Replacer rep = wordMatch.replacer(new SynonymSubstitution());
323        if(alterations.isEmpty())
324            return rep.replace(text);
325        else
326            return modify(rep.replace(text)).toString();
327    }

329    public String lookup(String word)
330    {
331        if(word.isEmpty())
332            return word;
333        String word2 = word.toLowerCase();
334        if(mappings.containsKey(word2))
335        {
336            String nx = mappings.get(word2).next();
337            if(nx.isEmpty())
338                return nx;
339            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
340                return nx.toUpperCase();
341            if(Category.Lu.contains(word.charAt(0)))
342            {
343                return Character.toUpperCase(nx.charAt(0)) + nx.substring(1, nx.length());
344            }
345            return nx;
346        }
347        else if(languages.containsKey(word2))
348        {
349            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
350                return languages.get(word2).word(rng, false, rng.between(2, 4)).toUpperCase();
351            if(Category.Lu.contains(word.charAt(0)))
352            {
353                return languages.get(word2).word(rng, true, rng.between(2, 4));
354            }
355            return languages.get(word2).word(rng, false, rng.between(2, 4));
356        }
357        return word;
358    }

360    private class SynonymSubstitution implements Substitution
361    {
362        private StringBuilder temp = new StringBuilder(64);
363        @Override
364        public void appendSubstitution(MatchResult match, TextBuffer dest) {
365            //dest.append(lookup(match.group(0)));
366            temp.setLength(0);
367            match.getGroup(0, temp);
368            writeLookup(dest, temp);
369        }

372    private void writeLookup(TextBuffer dest, StringBuilder word) {
373        if(word.length() <= 0)
374            return;
375        if(mappings.containsKey(word))
376        {
377            String nx = mappings.get(word).next();
378            if(nx.isEmpty())
379                return;
380            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
381            {
382                dest.append(nx.toUpperCase());
383                return;
384            }
385            if(Category.Lu.contains(word.charAt(0)))
386            {
387                dest.append(Character.toUpperCase(nx.charAt(0)));
388                dest.append(nx.substring(1, nx.length()));
389                return;
390            }
391            dest.append(nx);
392            return;
393        }
394        else if(languages.containsKey(word))
395        {
396            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
397            {
398                dest.append(languages.get(word).word(rng, false, rng.between(2, 4)).toUpperCase());
399            }
400            else if(Category.Lu.contains(word.charAt(0)))
401            {
402                dest.append(languages.get(word).word(rng, true, rng.between(2, 4)));
403            }
404            else
405            {
406                dest.append(languages.get(word).word(rng, false, rng.between(2, 4)));
407            }
408            return;
409        }
410        if(dest instanceof Replacer.StringBuilderBuffer)
411        {
412            ((Replacer.StringBuilderBuffer)dest).sb.append(word);
413        }
414        else
415            dest.append(word.toString());
416
417    }

419    private class RandomLanguageSubstitution implements Substitution
420    {
421        @Override
422        public void appendSubstitution(MatchResult match, TextBuffer dest) {
423            FakeLanguageGen lang = FakeLanguageGen.randomLanguage(rng.nextLong());
424            randomLanguages.add(lang);
425            if(match.isCaptured(1))
426            {
427                lang = FakeLanguageGen.randomLanguage(rng.nextLong());
428                randomLanguages.add(lang);
429                do {
430                    latestGenerated = randomLanguages.get(0).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)))
431                            + "-" + randomLanguages.get(1).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)));
432                }while (latestGenerated.length() <= 5 || latestGenerated.length() >= 17);
433                dest.append(latestGenerated);
434            }

477     * calling it again, you can retrieve the generated random languages, if any were used while making nation names, by
478     * getting the FakeLanguageGen elements of this class' {@link #randomLanguages} field. Using one of these
479     * FakeLanguageGen objects, you can produce many more words with a similar style to the nation name, like "Drayo" in

499        return replacer.replace(working);

066    /**
067     * Allows this Thesaurus to find the exact words in synonyms and, when requested, replace each occurrence with a
068     * different word from the same Collection. Each word in synonyms should have the same part of speech, so "demon"
069     * and "devils" should not be in the same list of synonyms (singular noun and plural noun), but "demon" and "devil"
070     * could be (each is a singular noun). The Strings in synonyms should all be lower-case, since case is picked up
071     * from the text as it is being replaced and not from the words themselves. Proper nouns should normally not be used
072     * as synonyms, since this could get very confusing if it changed occurrences of "Germany" to "France" at random and
073     * a character's name, like "Dorothy", to "Anne", "Emily", "Cynthia", etc. in the middle of a section about Dorothy.
074     * The word matching pattern this uses only matches all-letter words, not words that contain hyphens, apostrophes,
075     * or other punctuation.
076     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
077     * @return this for chaining
078     */
079    public Thesaurus addSynonyms(Collection<String> synonyms)
080    {
081        if(synonyms.isEmpty())
082            return this;
083        long prevState = rng.getState();
084        rng.setState(CrossHash.hash64(synonyms));
085        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
086        for(String syn : synonyms)
087        {
088            mappings.put(syn, shuffler);
089        }
090        rng.setState(prevState);
091        return this;
092    }

094    /**
095     * Allows this Thesaurus to replace a specific keyword, typically containing multiple backtick characters ('`') so
096     * it can't be confused with a "real word," with one of the words in synonyms (chosen in shuffled order). The
097     * backtick is the only punctuation character that this class' word matcher considers part of a word, both for this
098     * reason and because it is rarely used in English text.
099     * @param keyword a word (typically containing backticks, '`') that will be replaced by a word from synonyms
100     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
101     * @return this for chaining
102     */
103    public Thesaurus addCategory(String keyword, Collection<String> synonyms)
104    {
105        if(synonyms.isEmpty())
106            return this;
107        long prevState = rng.getState();
108        rng.setState(CrossHash.hash64(synonyms));
109        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
110        mappings.put(keyword, shuffler);
111        rng.setState(prevState);
112        return this;
113    }

265    /**
266     * Given a String, StringBuilder, or other CharSequence that should contain words this knows synonyms for, this
267     * replaces each occurrence of such a known word with one of its synonyms, leaving unknown words untouched. Words
268     * that were learned together as synonyms with addSynonyms() will be replaced in such a way that an individual
269     * replacement word should not occur too close to a previous occurrence of the same word; that is, replacing the
270     * text "You fiend! You demon! You despoiler of creation; devil made flesh!", where "fiend", "demon", and "devil"
271     * are all synonyms, would never produce a string that contained "fiend" as the replacement for all three of those.
272     * @param text a CharSequence, such as a String, that contains words in the source language
273     * @return a String of the translated text.
274     */
275    public String process(CharSequence text)
276    {
277        Replacer rep = wordMatch.replacer(new SynonymSubstitution());
278        return rep.replace(text);
279    }

281    public String lookup(String word)
282    {
283        if(word.isEmpty())
284            return word;
285        String word2 = word.toLowerCase();
286        if(mappings.containsKey(word2))
287        {
288            String nx = mappings.get(word2).next();
289            if(nx.isEmpty())
290                return nx;
291            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
292                return nx.toUpperCase();
293            if(Category.Lu.contains(word.charAt(0)))
294            {
295                return Character.toUpperCase(nx.charAt(0)) + nx.substring(1, nx.length());
296            }
297            return nx;
298        }
299        else if(languages.containsKey(word2))
300        {
301            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
302                return languages.get(word2).word(rng, false, rng.between(2, 4)).toUpperCase();
303            if(Category.Lu.contains(word.charAt(0)))
304            {
305                return languages.get(word2).word(rng, true, rng.between(2, 4));
306            }
307            return languages.get(word2).word(rng, false, rng.between(2, 4));
308
309        }
310        return word;
311    }

321    private class RandomLanguageSubstitution implements Substitution
322    {
323        @Override
324        public void appendSubstitution(MatchResult match, TextBuffer dest) {
325            FakeLanguageGen lang = FakeLanguageGen.randomLanguage(rng.nextLong());
326            randomLanguages.add(lang);
327            if(match.isCaptured(1))
328            {
329                lang = FakeLanguageGen.randomLanguage(rng.nextLong());
330                randomLanguages.add(lang);
331                do {
332                    latestGenerated = randomLanguages.get(0).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)))
333                            + "-" + randomLanguages.get(1).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)));
334                }while (latestGenerated.length() <= 5 || latestGenerated.length() >= 17);
335                dest.append(latestGenerated);
336            }
337            else
338            {
339                do{
340                    latestGenerated = lang.word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)));
341                }while (latestGenerated.length() <= 2 || latestGenerated.length() >= 11);
342                dest.append(latestGenerated);
343            }
344        }
345    }

373    /**
374     * Generates a random possible name for a nation, such as "Iond-Gouccief Alliance" or "The Last Drayo Commonwealth".
375     * Needs {@link #addKnownCategories()} to be called on this Thesaurus first. May use accented characters, as in
376     * "Thùdshù-Hyóttiálb Hegemony" or "The Glorious Chô Empire"; if you want to strip these out and replace accented
377     * chars with their un-accented counterparts, you can use {@link FakeLanguageGen#removeAccents(CharSequence)}, which
378     * returns a CharSequence that can be converted to String if needed. Shortly after calling this method, but before
379     * calling it again, you can retrieve the generated random languages, if any were used while making nation names, by
380     * getting the FakeLanguageGen elements of this class' {@link #randomLanguages} field. Using one of these
381     * FakeLanguageGen objects, you can produce many more words with a similar style to the nation name, like "Drayo" in
382     * "The Last Drayo Commonwealth". If more than one language was used in the nation name, as in "Thùdshù-Hyóttiálb
383     * Hegemony", you will have two languages in randomLanguages, so here "Thùdshù" would be generated by the first
384     * language, and "Hyóttiálb" by the second language. Calling this method replaces the current contents of
385     * randomLanguages, so if you want to use those languages, get them while you can. This also assigns the
386     * {@link #latestGenerated} field to contain the part of the nation name without any larger titles; in the case of
387     * "The Glorious Chô Empire", the latestGenerated field would be assigned "Chô" at the same time the longer name
388     * would be returned. This field will be reassigned if this method is called again.
389     *
390     * @return a random name for a nation or a loose equivalent to a nation, as a String
391     */
392    public String makeNationName()
393    {
394        String working = process(rng.getRandomElement(nationTerms));
395        int frustration = 0;
396        while (frustration++ < 8 && similarFinder.matches(working))
397            working = process(rng.getRandomElement(nationTerms));
398        randomLanguages.clear();
399        RandomLanguageSubstitution sub = new RandomLanguageSubstitution();
400        Replacer replacer = Pattern.compile("@(-@)?").replacer(sub);
401        return replacer.replace(working);
402    }

405     * with the FakeLanguageGen already available instead of randomly created. Needs {@link #addKnownCategories()} to be
406     * called on this Thesaurus first. May use accented characters, as in "Thùdshù Hegemony" or "The Glorious Chô
407     * Empire", if the given language can produce them; if you want to strip these out and replace accented chars
408     * with their un-accented counterparts, you can use {@link FakeLanguageGen#removeAccents(CharSequence)}, which
409     * returns a CharSequence that can be converted to String if needed, or simply get an accent-less language by
410     * calling {@link FakeLanguageGen#removeAccents()} on the FakeLanguageGen you would give this. This assigns the
411     * {@link #latestGenerated} field to contain the part of the nation name without any larger titles; in the case of
412     * "The Glorious Chô Empire", the latestGenerated field would be assigned "Chô" at the same time the longer name
413     * would be returned. This field will be reassigned if this method is called again.
414     * <br>
415     * Some nation names use a hyphenated pairing of what would normally be names in two different languages; if one of
416     * those names is produced by this it will produce two names in the same linguistic style. The randomLanguages field
417     * is not populated by this method; it is assumed that since you are passing this a FakeLanguageGen, you already
418     * have the one you want to use anyway.
419     *
420     * @param language a FakeLanguageGen that will be used to construct any non-English names
421     * @return a random name for a nation or a loose equivalent to a nation, as a String
422     */
423    public String makeNationName(FakeLanguageGen language)
424    {
425        String working = process(rng.getRandomElement(nationTerms));
426        int frustration = 0;
427        while (frustration++ < 8 && similarFinder.matches(working))
428            working = process(rng.getRandomElement(nationTerms));
429        randomLanguages.clear();
430        KnownLanguageSubstitution sub = new KnownLanguageSubstitution(language);
431        Replacer replacer = Pattern.compile("@(-@)?").replacer(sub);
432        return replacer.replace(working);
433    }

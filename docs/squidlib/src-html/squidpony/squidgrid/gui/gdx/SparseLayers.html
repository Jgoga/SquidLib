101    public SparseLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
102        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
103        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
104        backgrounds = ArrayTools.fill(0f, this.gridWidth, this.gridHeight);
105        layers = new ArrayList<>(4);
106        if(font.initialized())
107            this.font = font;
108        else
109            this.font = font.width(cellWidth).height(cellHeight).initBySize();
110        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
111        mapping = new IntIntMap(4);
112        mapping.put(0, 0);
113        glyphs = new ArrayList<>(16);
114        scc = DefaultResources.getSCC();
115        setBounds(xOffset, yOffset,
116                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
117    }

149     * Sets the SparseTextMap associated with the given layerNumber to the given contents. If layerNumber is too high
150     * to set an existing layer, this will add contents as a new layer on top of the others.
151     * @param layerNumber must be 0 or greater
152     * @param contents a SparseTextMap, possibly obtained with {@link #getLayer(int)}

154    public void setLayer(int layerNumber, SparseTextMap contents)
155    {
156        if(layerNumber < 0)
157            return;
158        layerNumber = mapping.get(layerNumber, layerNumber);
159        if(layerNumber >= layers.size())
160        {
161            mapping.put(layerNumber, layers.size());
162            layers.add(contents);
163        }
164        else
165        {
166            layers.set(layerNumber, contents);
167        }

184    /**
185     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added. The layer will generally be
186     * associated with a layer number equal to the number of layers currently present, and will be rendered over the
187     * existing layers. It might not be associated with the same number if layers were added out-of-order or with skips
188     * between associations. If you want to add a layer at some alternate layer number (which can be confusing but may
189     * be needed for some reason), there is another overload that allows you to specify the association number.
190     * If you may have added some layers out of order or skipped some numbers, you can use
191     * {@link #findLayer(SparseTextMap)} on the returned SparseTextMap to get its association number.
192     * @return a SparseTextMap that was just added; null if something went wrong
193     */
194    public SparseTextMap addLayer()
195    {
196        int association = layers.size();
197        while (mapping.containsKey(association)) {
198            ++association;
199        }
200        return addLayer(association);
201    }
202    /**
203     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added, or returns an existing layer
204     * if one is already associated with the given number. The layer can be greater than the number of layers currently
205     * present, which will add a layer to be rendered over the existing layers, but the number that refers to that layer
206     * will not change. It is recommended that to add a layer, you only add at the value equal to
207     * {@link #getLayerCount()}, which will maintain the order and layer numbers in a sane way, and this is the behavior
208     * of the addLayer overload that does not take a parameter.
209     * @param association the number to associate the layer with; should usually be between 0 and {@link #getLayerCount()} inclusive
210     * @return a SparseTextMap that either was just added or was already associated with the given layer number; null if something went wrong
211     */
212    public SparseTextMap addLayer(int association)
213    {
214        if(association < 0)
215            return null;
216        association = mapping.get(association, association);
217        if(association >= layers.size())
218        {
219            mapping.put(association, layers.size());
220            SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
221            layers.add(stm);
222            return stm;
223        }
224        else
225        {
226            return layers.get(association);
227        }
228    }

319    /**
320     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
321     * setting the background colors to match the given Color 2D array. If the colors argument is null, does nothing. If
322     * the chars argument is null, only affects the background colors. This will filter each Color in colors if the
323     * color center this uses has a filter.
324     * @param chars Can be {@code null}, indicating that only colors must be put.
325     * @param colors the background colors for the given chars
326     */
327    @Override
328    public void put(char[][] chars, Color[][] colors) {
329        if(chars == null)
330        {
331            if(colors != null)
332            {
333                for (int i = 0; i < colors.length; i++) {
334                    if (colors[i] == null)
335                        continue;
336                    for (int j = 0; j < colors[i].length; j++) {
337                        backgrounds[i][j] = (colors[i][j] == null) ? 0f : scc.filter(colors[i][j]).toFloatBits();
338                    }
339                }
340            }
341        }
342        else
343        {
344            if(colors == null)
345            {
346                for (int i = 0; i < chars.length; i++) {
347                    if(chars[i] == null)
348                        continue;
349                    for (int j = 0; j < chars[i].length; j++) {
350                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
351                    }
352                }
353            }
354            else
355            {
356                for (int i = 0; i < chars.length && i < colors.length; i++) {
357                    if(colors[i] == null || chars[i] == null)
358                        continue;
359                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
360                        if(colors[i][j] == null)
361                            put(i, j, chars[i][j], defaultPackedForeground, 0f);
362                        else
363                            put(i, j, chars[i][j], defaultPackedForeground, scc.filter(colors[i][j]).toFloatBits());
364                    }
365                }
366            }
367        }
368    }
369    /**
370     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
371     * setting the background colors to match the given 2D array of colors as packed floats. If the colors argument is
372     * null, does nothing. If the chars argument is null, only affects the background colors. This will not filter the
373     * passed colors at all.
374     * @param chars Can be {@code null}, indicating that only colors must be put.
375     * @param colors the background colors for the given chars
376     */
377    public void put(char[][] chars, float[][] colors) {
378        if(chars == null)
379        {
380            if(colors != null) {
381                for (int i = 0; i < colors.length; i++) {
382                    if (colors[i] == null)
383                        continue;
384                    System.arraycopy(colors[i], 0, backgrounds[i], 0, colors[i].length);
385                }
386            }
387        }
388        else
389        {
390            if(colors == null)
391            {
392                for (int i = 0; i < chars.length; i++) {
393                    if(chars[i] == null)
394                        continue;
395                    for (int j = 0; j < chars[i].length; j++) {
396                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
397                    }
398                }
399            }
400            else
401            {
402                for (int i = 0; i < chars.length && i < colors.length; i++) {
403                    if(colors[i] == null || chars[i] == null)
404                        continue;
405                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
406                        put(i, j, chars[i][j], defaultPackedForeground, colors[i][j]);
407                    }
408                }
409            }
410        }
411    }

584    /**
585     * Puts the char c at the position x,y in the requested layer with the given foreground and background colors. If
586     * foreground is null or is fully transparent (all channels 0), then this does not change
587     * the foreground contents. If background is null or is fully transparent, this does not change the background. Uses
588     * the color center to potentially filter the given colors; this can be changed with
589     * {@link #setColorCenter(IColorCenter)}. The layer can be greater than the number of layers currently present,
590     * which will add a layer to be rendered over the existing layers, but the number that refers to that layer will not
591     * change. It is recommended that to add a layer, you only add at the value equal to {@link #getLayerCount()}, which
592     * will maintain the order and layer numbers in a sane way.
593     * @param x the x position to place the char at
594     * @param y the y position to place the char at
595     * @param c the char to place
596     * @param foreground the color to use for c; if null or fully transparent, nothing will change in the foreground
597     * @param background the color to use for the cell; if null or fully transparent, nothing will change in the background
598     * @param layer the layer to place the colorful char into; should usually be between 0 and {@link #getLayerCount()} inclusive
599     */

1504     * {@link #summon(int, int, int, int, char, float, float, float)} repeatedly with
1505     * different parameters. As with summon(), this creates temporary Glyphs that change color and position.
1506     * The distance is how many cells away to move the created Actors away from (x,y). The measurement determines
1507     * whether this produces Glyphs in 4 (cardinal) directions for DIAMOND or OCTAHEDRON, or 8 (cardinal and diagonal)
1508     * directions for any other enum value for Radius; CIRCLE and SPHERE will position the 8 glyphs in a circle, while
1509     * SQUARE and CUBE will position their 8 glyphs in a square.
1510     * <br>
1511     * Unlike {@link SquidPanel#burst(float, int, int, int, boolean, char, Color, Color, boolean, float, float)}, this
1512     * does not rotate the individual Glyphs it produces.
1513     * @param x the starting, center, x-position to create all Actors at
1514     * @param y the starting, center, y-position to create all Actors at
1515     * @param distance how far away, in cells, to move each actor from the center (Chebyshev distance, forming a square)
1516     * @param measurement a Radius enum that determines if 4 (DIAMOND, OCTAHEDRON) or 8 (anything else) Glyphs are
1517     *                    created, and the shape they will take
1518     * @param shown the char to use for Glyphs; should definitely be visible
1519     * @param startColor the encoded color to start the effect with
1520     * @param endColor the encoded color to end the effect on
1521     * @param duration how long, in seconds, the effect should last
1522     */
1523    public void burst(int x, int y, int distance, Radius measurement, char shown,
1524                      float startColor, float endColor, float duration)
1525    {
1526        Direction d;
1527        switch (measurement)
1528        {
1529            case SQUARE:
1530            case CUBE:
1531                for (int i = 0; i < 8; i++) {
1532                    d = Direction.CLOCKWISE[i];
1533                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1534                            shown, startColor, endColor, duration);
1535                }
1536                break;
1537            case CIRCLE:
1538            case SPHERE:
1539                float xf = worldX(x), yf = worldY(y);
1540                for (int i = 0; i < 4; i++) {
1541                    d = Direction.DIAGONALS[i];
1542                    summon(xf, yf, xf - d.deltaX * font.actualCellWidth * distance * 0.7071067811865475f, // the constant is 1.0 / Math.sqrt(2.0)
1543                            yf + d.deltaY * font.actualCellHeight * distance * 0.7071067811865475f,
1544                            shown, startColor, endColor, duration);
1545                }
1546                // break intentionally absent
1547            default:
1548                for (int i = 0; i < 4; i++) {
1549                    d = Direction.CARDINALS_CLOCKWISE[i];
1550                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1551                            shown, startColor, endColor, duration);
1552                }
1553                break;
1554        }
1555    }

1578                      float startColor, float endColor, float duration)
1579    {
1580        Direction d;
1581        final int len = choices.length();
1582        final StatefulRNG rng = DefaultResources.getGuiRandom();
1583        switch (measurement)
1584        {
1585            case SQUARE:
1586            case CUBE:
1587                for (int i = 0; i < 8; i++) {
1588                    d = Direction.CLOCKWISE[i];
1589                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1590                            choices.charAt(rng.nextIntHasty(len)), startColor, endColor, duration);
1591                }
1592                break;
1593            case CIRCLE:
1594            case SPHERE:
1595                float xf = worldX(x), yf = worldY(y);
1596                for (int i = 0; i < 4; i++) {
1597                    d = Direction.DIAGONALS[i];
1598                    summon(xf, yf, xf - d.deltaX * font.actualCellWidth * distance * 0.7071067811865475f,
1599                            yf + d.deltaY * font.actualCellHeight * distance * 0.7071067811865475f,
1600                            choices.charAt(rng.nextIntHasty(len)), startColor, endColor, duration);
1601                }
1602                // break intentionally absent
1603            default:
1604                for (int i = 0; i < 4; i++) {
1605                    d = Direction.CARDINALS_CLOCKWISE[i];
1606                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1607                            choices.charAt(rng.nextIntHasty(len)), startColor, endColor, duration);

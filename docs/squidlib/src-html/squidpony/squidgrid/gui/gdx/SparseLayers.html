057    public SparseLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
058        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
059        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
060        backgrounds = ArrayTools.fill(0f, this.gridWidth, this.gridHeight);
061        layers = new Array<>(true, 4, SparseTextMap.class);
062        this.font = font.width(cellWidth).height(cellHeight).initBySize();
063        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
064        mapping = new IntIntMap(4);
065        mapping.put(0, 0);
066        glyphs = new SnapshotArray<>(true, 16, TextCellFactory.Glyph.class);
067        scc = DefaultResources.getSCC();
068        setBounds(xOffset, yOffset,
069                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
070    }

116    /**
117     * Puts the given string horizontally with the first character at the given
118     * offset, using the colors that {@code cs} provides. Does not change the
119     * background colors. Does filter the colors from cs if the color center this
120     * uses is set up to filter colors.
121     * <p>
122     * Does not word wrap. Characters that are not renderable (due to being at
123     * negative offsets or offsets greater than the grid size) will not be shown
124     * but will not cause any malfunctions.
125     *
126     * @param xOffset the x coordinate of the first character
127     * @param yOffset the y coordinate of the first character
128     * @param cs an {@link IColoredString} with potentially multiple colors
129     */
130    @Override
131    public void put(int xOffset, int yOffset, IColoredString<? extends Color> cs) {
132        int x = xOffset;
133        for (IColoredString.Bucket<? extends Color> fragment : cs) {
134            final String s = fragment.getText();
135            final Color color = fragment.getColor();
136            put(x, yOffset, s, color == null ? getDefaultForegroundColor() : scc.filter(color), null);
137            x += s.length();
138        }
139    }

155    /**
156     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
157     * setting the background colors to match the given Color 2D array. If the colors argument is null, does nothing. If
158     * the chars argument is null, only affects the background colors.
159     * @param chars Can be {@code null}, indicating that only colors must be put.
160     * @param colors the background colors for the given chars
161     */
162    @Override
163    public void put(char[][] chars, Color[][] colors) {
164        if(colors == null)
165            return;
166        if(chars == null)
167        {
168            for (int i = 0; i < colors.length; i++) {
169                if(colors[i] == null)
170                    continue;
171                for (int j = 0; j < colors.length; j++) {
172                    backgrounds[i][j] = (colors[i][j] == null) ? 0f : scc.filter(colors[i][j]).toFloatBits();
173                }
174            }
175        }
176        else
177        {
178            for (int i = 0; i < chars.length && i < colors.length; i++) {
179                if(colors[i] == null || chars[i] == null)
180                    continue;
181                for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
182                    if(colors[i][j] == null)
183                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
184                    else
185                        put(i, j, chars[i][j], defaultPackedForeground, scc.filter(colors[i][j]).toFloatBits());
186                }
187            }
188        }
189    }

342     * Puts the char c at the position x,y in the requested layer with the given foreground and background colors as
343     * encoded floats, such as those produced by {@link Color#toFloatBits()}. If foreground is 0f, then this does not
344     * change the foreground contents. If background is 0f, this does not change the
345     * background. Does not filter the given colors. The layer can be greater than the number of layers currently
346     * present, which will add a layer to be rendered over the existing layers, but the number that refers to that layer
347     * will not change. It is recommended that to add a layer, you only add at the value equal to
348     * {@link #getLayerCount()}, which will maintain the order and layer numbers in a sane way.
349     * @param x the x position to place the char at
350     * @param y the y position to place the char at
351     * @param c the char to place
352     * @param foreground the color to use for c; if 0f, nothing will change in the foreground
353     * @param background the color to use for the cell; if null or fully transparent, nothing will change in the background
354     * @param layer the layer to place the colorful char into; should usually be between 0 and {@link #getLayerCount()} inclusive
355     */
356    public void put(int x, int y, char c, float foreground, float background, int layer)
357    {
358        if(x < 0 || x >= gridWidth || y < 0 || y >= gridHeight || layer < 0)
359            return;
360        if(background != 0f)
361            backgrounds[x][y] = background;
362        layer = mapping.get(layer, layer);
363        if(foreground != 0f)
364        {
365            if(layer >= layers.size)
366            {
367                mapping.put(layer, layers.size);
368                SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
369                stm.place(x, y, c, foreground);
370                layers.add(stm);
371            }
372            else
373            {
374                layers.get(layer).place(x, y, c, foreground);
375            }
376        }
377    }

432     * floats, such as those produced by {@link Color#toFloatBits()}. If foreground is 0f, then this does not change the
433     * foreground contents. If background is 0f, this does not change the background. Does not filter the given colors.
434     * The layer can be greater than the number of layers currently present, which will add a layer to be rendered over
435     * the existing layers, but the number that refers to that layer will not change. It is recommended that to add a
436     * layer, you only add at the value equal to {@link #getLayerCount()}, which will maintain the order and layer
437     * numbers in a sane way.
438     * @param x the x position to place the String at
439     * @param y the y position to place the String at
440     * @param text the String to place
441     * @param foreground the color to use for text; if 0f, nothing will change in the foreground
442     * @param background the color to use for the cells; if null or fully transparent, nothing will change in the background
443     * @param layer the layer to place the colorful char into; should usually be between 0 and {@link #getLayerCount()} inclusive

445    public void put(int x, int y, String text, float foreground, float background, int layer) {
446        if (x < 0 || x >= gridWidth || y < 0 || y >= gridHeight || layer < 0)
447            return;
448        int len = Math.min(text.length(), gridWidth - x);
449        if(background != 0f) {
450            for (int i = 0; i < len; i++) {
451                backgrounds[x + i][y] = background;
452            }
453        }
454        if (foreground != 0f)
455        {
456            layer = mapping.get(layer, layer);
457            if (layer >= layers.size) {
458                mapping.put(layer, layers.size);
459                SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
460                for (int i = 0; i < len; i++) {
461                    stm.place(x + i, y, text.charAt(i), foreground);
462                }
463                layers.add(stm);
464            } else {
465                SparseTextMap stm = layers.get(layer);
466                for (int i = 0; i < len; i++) {
467                    stm.place(x + i, y, text.charAt(i), foreground);
468                }
469            }
470        }

498    /**
499     * Removes the foreground chars, where present, in all layers at the given x,y position.
500     * The backgrounds will be unchanged.
501     * @param x the x-coordinate of the position to remove all chars from
502     * @param y the y-coordinate of the position to remove all chars from
503     */
504    @Override
505    public void clear(int x, int y)
506    {
507        if(x < 0 || x >= gridWidth || y < 0 || y >= gridHeight)
508            return;
509        backgrounds[x][y] = 0f;
510        int code = SparseTextMap.encodePosition(x, y);
511        for (int i = 0; i < layers.size; i++) {
512            layers.get(i).remove(code);
513        }
514    }

529        }

808    /**
809     * Tints the background at position x,y so it becomes the given encodedColor, waiting for {@code delay} (in seconds)
810     * before performing it, then after the tint is complete it returns the cell to its original color, taking duration
811     * seconds. Additionally, enqueue {@code postRunnable} for running after the created action ends.
812     * <br>
813     * This will only behave correctly if you call {@link Stage#act()} before you call {@link Stage#draw()}, but after
814     * any changes to the contents of this SparseLayers. If you change the contents, then draw, and then act, that will
815     * draw the contents without the tint this applies, then apply the tint when you call act(), then quickly overwrite
816     * the tint in the next frame. That visually appears as nothing happening other than a delay.
817     * @param delay how long to wait in seconds before starting the effect
818     * @param x the x-coordinate of the cell to tint
819     * @param y the y-coordinate of the cell to tint
820     * @param encodedColor what to transition the cell's color towards, and then transition back from, as a packed float
821     * @param duration how long the total "round-trip" transition should take in seconds
822     * @param postRunnable a Runnable to execute after the tint completes; may be null to do nothing.
823     */
824    public void tint(float delay, final int x, final int y, final float encodedColor, float duration, Runnable postRunnable) {
825        if(x < 0 || x >= gridWidth || y < 0 || y >= gridHeight)
826            return;
827        duration = Math.max(0.015f, duration);
828        animationCount++;
829        final float ac = backgrounds[x][y];
830        final int nbActions = 3 + (0 < delay ? 1 : 0) + (postRunnable == null ? 0 : 1);
831        final Action[] sequence = new Action[nbActions];
832        int index = 0;
833        if (0 < delay)
834            sequence[index++] = Actions.delay(delay);
835        sequence[index++] = new TemporalAction(duration * 0.3f) {
836            @Override
837            protected void update(float percent) {
838                backgrounds[x][y] = SColor.lerpFloatColors(ac, encodedColor, percent);
839            }
840        };
841        sequence[index++] = new TemporalAction(duration * 0.7f) {
842            @Override
843            protected void update(float percent) {
844                backgrounds[x][y] = SColor.lerpFloatColors(encodedColor, ac, percent);
845            }
846        };
847        if(postRunnable != null)
848        {
849            sequence[index++] = Actions.run(postRunnable);
850        }
851                /* Do this one last, so that hasActiveAnimations() returns true during 'postRunnables' */
852        sequence[index] = Actions.delay(duration, Actions.run(new Runnable() {
853            @Override
854            public void run() {
855                --animationCount;
856            }
857        }));
858
859        addAction(Actions.sequence(sequence));
860    }

942    /**
943     * Convenience method to produce an explosion, splash, or burst effect. Calls
944     * {@link #summon(int, int, int, int, char, float, float, float)} repeatedly with
945     * different parameters. As with summon(), this creates temporary Glyphs that change color and position.
946     * The distance is how many cells away to move the created Actors away from (x,y). The measurement determines
947     * whether this produces Glyphs in 4 (cardinal) directions for DIAMOND or OCTAHEDRON, or 8 (cardinal and diagonal)
948     * directions for any other enum value for Radius; CIRCLE and SPHERE will position the 8 glyphs in a circle, while
949     * SQUARE and CUBE will position their 8 glyphs in a square.
950     * <br>
951     * Unlike {@link SquidPanel#burst(float, int, int, int, boolean, char, Color, Color, boolean, float, float)}, this
952     * does not rotate the individual Glyphs it produces.
953     * @param x the starting, center, x-position to create all Actors at
954     * @param y the starting, center, y-position to create all Actors at
955     * @param distance how far away, in cells, to move each actor from the center (Chebyshev distance, forming a square)
956     * @param measurement a Radius enum that determines if 4 (DIAMOND, OCTAHEDRON) or 8 (anything else) Glyphs are
957     *                    created, and the shape they will take
958     * @param shown the char to use for Glyphs; should definitely be visible
959     * @param startColor the encoded color to start the effect with
960     * @param endColor the encoded color to end the effect on
961     * @param duration how long, in seconds, the effect should last
962     */
963    public void burst(int x, int y, int distance, Radius measurement, char shown,
964                      float startColor, float endColor, float duration)
965    {
966        Direction d;
967        switch (measurement)
968        {
969            case SQUARE:
970            case CUBE:
971                for (int i = 0; i < 8; i++) {
972                    d = Direction.CLOCKWISE[i];
973                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
974                            shown, startColor, endColor, duration);
975                }
976                break;
977            case CIRCLE:
978            case SPHERE:
979                float xf = worldX(x), yf = worldY(y);
980                for (int i = 0; i < 4; i++) {
981                    d = Direction.DIAGONALS[i];
982                    summon(xf, yf, xf - d.deltaX * font.actualCellWidth * distance * 0.7071067811865475f, // the constant is 1.0 / Math.sqrt(2.0)
983                            yf + d.deltaY * font.actualCellHeight * distance * 0.7071067811865475f,
984                            shown, startColor, endColor, duration);
985                }
986                // break intentionally absent
987            default:
988                for (int i = 0; i < 4; i++) {
989                    d = Direction.CARDINALS_CLOCKWISE[i];
990                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
991                            shown, startColor, endColor, duration);
992                }
993                break;
994        }
995    }

1053    /**
1054     * Draws the actor. The batch is configured to draw in the parent's coordinate system.
1055     * {@link Batch#draw(TextureRegion, float, float, float, float, float, float, float, float, float)
1056     * This draw method} is convenient to draw a rotated and scaled TextureRegion. {@link Batch#begin()} has already been called on
1057     * the batch. If {@link Batch#end()} is called to draw without the batch then {@link Batch#begin()} must be called before the
1058     * method returns.
1059     *
1060     * @param batch a Batch such as a SpriteBatch that must be between a begin() and end() call; usually done by Stage
1061     * @param parentAlpha currently ignored
1062     */
1063    @Override
1064    public void draw(Batch batch, float parentAlpha) {
1065        super.draw(batch, parentAlpha);
1066        float xo = getX(), yo = getY(), yOff = yo + 1f + gridHeight * font.actualCellHeight;
1067        font.draw(batch, backgrounds, xo, yo);
1068        int len = layers.size;
1069        for (int i = 0; i < len; i++) {
1070            layers.get(i).draw(batch, font, xo, yOff);
1071        }
1072        TextCellFactory.Glyph[] items = glyphs.begin();
1073        int x, y;
1074        for (int i = 0, n = glyphs.size; i < n; i++) {
1075            TextCellFactory.Glyph glyph = items[i];
1076            if(glyph == null)
1077                continue;
1078            glyph.act(Gdx.graphics.getDeltaTime());
1079            if(
1080                    (x = Math.round((glyph.getX() - xo) / font.actualCellWidth)) < 0 || x >= gridWidth ||
1081                    (y = Math.round((glyph.getY() - yo)  / -font.actualCellHeight + gridHeight)) < 0 || y >= gridHeight ||
1082                    backgrounds[x][y] == 0f)
1083                continue;
1084            glyph.draw(batch, 1f);

093    public SparseLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
094        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
095        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
096        backgrounds = ArrayTools.fill(0f, this.gridWidth, this.gridHeight);
097        layers = new ArrayList<>(4);
098        if(font.initialized())
099            this.font = font;
100        else
101            this.font = font.width(cellWidth).height(cellHeight).initBySize();
102        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
103        mapping = new IntIntMap(4);
104        mapping.put(0, 0);
105        glyphs = new ArrayList<>(16);
106        scc = DefaultResources.getSCC();
107        setBounds(xOffset, yOffset,
108                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
109    }

143     * @param layerNumber must be 0 or greater
144     * @param contents a SparseTextMap, possibly obtained with {@link #getLayer(int)}

147    {
148        if(layerNumber < 0)
149            return;
150        layerNumber = mapping.get(layerNumber, layerNumber);
151        if(layerNumber >= layers.size())
152        {
153            mapping.put(layerNumber, layers.size());

155        }
156        else
157        {
158            layers.set(layerNumber, contents);
159        }

176    /**
177     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added. The layer will generally be
178     * associated with a layer number equal to the number of layers currently present, and will be rendered over the
179     * existing layers. It might not be associated with the same number if layers were added out-of-order or with skips
180     * between associations. If you want to add a layer at some alternate layer number (which can be confusing but may
181     * be needed for some reason), there is another overload that allows you to specify the association number.
182     * If you may have added some layers out of order or skipped some numbers, you can use
183     * {@link #findLayer(SparseTextMap)} on the returned SparseTextMap to get its association number.
184     * @return a SparseTextMap that was just added; null if something went wrong
185     */
186    public SparseTextMap addLayer()
187    {
188        int association = layers.size();
189        while (mapping.containsKey(association)) {
190            ++association;
191        }
192        return addLayer(association);
193    }
194    /**
195     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added, or returns an existing layer
196     * if one is already associated with the given number. The layer can be greater than the number of layers currently
197     * present, which will add a layer to be rendered over the existing layers, but the number that refers to that layer
198     * will not change. It is recommended that to add a layer, you only add at the value equal to
199     * {@link #getLayerCount()}, which will maintain the order and layer numbers in a sane way, and this is the behavior
200     * of the addLayer overload that does not take a parameter.
201     * @param association the number to associate the layer with; should usually be between 0 and {@link #getLayerCount()} inclusive
202     * @return a SparseTextMap that either was just added or was already associated with the given layer number; null if something went wrong
203     */
204    public SparseTextMap addLayer(int association)
205    {
206        if(association < 0)
207            return null;
208        association = mapping.get(association, association);
209        if(association >= layers.size())
210        {
211            mapping.put(association, layers.size());
212            SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
213            layers.add(stm);
214            return stm;
215        }
216        else
217        {
218            return layers.get(association);
219        }
220    }

311    /**
312     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
313     * setting the background colors to match the given Color 2D array. If the colors argument is null, does nothing. If
314     * the chars argument is null, only affects the background colors. This will filter each Color in colors if the
315     * color center this uses has a filter.
316     * @param chars Can be {@code null}, indicating that only colors must be put.
317     * @param colors the background colors for the given chars
318     */
319    @Override
320    public void put(char[][] chars, Color[][] colors) {
321        if(chars == null)
322        {
323            if(colors != null)
324            {
325                for (int i = 0; i < colors.length; i++) {
326                    if (colors[i] == null)
327                        continue;
328                    for (int j = 0; j < colors[i].length; j++) {
329                        backgrounds[i][j] = (colors[i][j] == null) ? 0f : scc.filter(colors[i][j]).toFloatBits();
330                    }
331                }
332            }
333        }
334        else
335        {
336            if(colors == null)
337            {
338                for (int i = 0; i < chars.length; i++) {
339                    if(chars[i] == null)
340                        continue;
341                    for (int j = 0; j < chars[i].length; j++) {
342                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
343                    }
344                }
345            }
346            else
347            {
348                for (int i = 0; i < chars.length && i < colors.length; i++) {
349                    if(colors[i] == null || chars[i] == null)
350                        continue;
351                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
352                        if(colors[i][j] == null)
353                            put(i, j, chars[i][j], defaultPackedForeground, 0f);
354                        else
355                            put(i, j, chars[i][j], defaultPackedForeground, scc.filter(colors[i][j]).toFloatBits());
356                    }
357                }
358            }
359        }
360    }
361    /**
362     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
363     * setting the background colors to match the given 2D array of colors as packed floats. If the colors argument is
364     * null, does nothing. If the chars argument is null, only affects the background colors. This will not filter the
365     * passed colors at all.
366     * @param chars Can be {@code null}, indicating that only colors must be put.
367     * @param colors the background colors for the given chars
368     */
369    public void put(char[][] chars, float[][] colors) {
370        if(chars == null)
371        {
372            if(colors != null) {
373                for (int i = 0; i < colors.length; i++) {
374                    if (colors[i] == null)
375                        continue;
376                    System.arraycopy(colors[i], 0, backgrounds[i], 0, colors[i].length);
377                }
378            }
379        }
380        else
381        {
382            if(colors == null)
383            {
384                for (int i = 0; i < chars.length; i++) {
385                    if(chars[i] == null)
386                        continue;
387                    for (int j = 0; j < chars[i].length; j++) {
388                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
389                    }
390                }
391            }
392            else
393            {
394                for (int i = 0; i < chars.length && i < colors.length; i++) {
395                    if(colors[i] == null || chars[i] == null)
396                        continue;
397                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
398                        put(i, j, chars[i][j], defaultPackedForeground, colors[i][j]);
399                    }
400                }
401            }
402        }
403    }

1308     * Convenience method to produce an explosion, splash, or burst effect. Calls
1309     * {@link #summon(int, int, int, int, char, float, float, float)} repeatedly with
1310     * different parameters. As with summon(), this creates temporary Glyphs that change color and position.
1311     * The distance is how many cells away to move the created Actors away from (x,y). The measurement determines
1312     * whether this produces Glyphs in 4 (cardinal) directions for DIAMOND or OCTAHEDRON, or 8 (cardinal and diagonal)
1313     * directions for any other enum value for Radius; CIRCLE and SPHERE will position the 8 glyphs in a circle, while
1314     * SQUARE and CUBE will position their 8 glyphs in a square.
1315     * <br>
1316     * Unlike {@link SquidPanel#burst(float, int, int, int, boolean, char, Color, Color, boolean, float, float)}, this
1317     * does not rotate the individual Glyphs it produces.
1318     * @param x the starting, center, x-position to create all Actors at
1319     * @param y the starting, center, y-position to create all Actors at
1320     * @param distance how far away, in cells, to move each actor from the center (Chebyshev distance, forming a square)
1321     * @param measurement a Radius enum that determines if 4 (DIAMOND, OCTAHEDRON) or 8 (anything else) Glyphs are
1322     *                    created, and the shape they will take
1323     * @param shown the char to use for Glyphs; should definitely be visible
1324     * @param startColor the encoded color to start the effect with
1325     * @param endColor the encoded color to end the effect on
1326     * @param duration how long, in seconds, the effect should last
1327     */
1328    public void burst(int x, int y, int distance, Radius measurement, char shown,
1329                      float startColor, float endColor, float duration)
1330    {
1331        Direction d;
1332        switch (measurement)
1333        {
1334            case SQUARE:
1335            case CUBE:
1336                for (int i = 0; i < 8; i++) {
1337                    d = Direction.CLOCKWISE[i];
1338                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1339                            shown, startColor, endColor, duration);
1340                }
1341                break;
1342            case CIRCLE:
1343            case SPHERE:
1344                float xf = worldX(x), yf = worldY(y);
1345                for (int i = 0; i < 4; i++) {
1346                    d = Direction.DIAGONALS[i];
1347                    summon(xf, yf, xf - d.deltaX * font.actualCellWidth * distance * 0.7071067811865475f, // the constant is 1.0 / Math.sqrt(2.0)
1348                            yf + d.deltaY * font.actualCellHeight * distance * 0.7071067811865475f,
1349                            shown, startColor, endColor, duration);
1350                }
1351                // break intentionally absent
1352            default:
1353                for (int i = 0; i < 4; i++) {
1354                    d = Direction.CARDINALS_CLOCKWISE[i];
1355                    summon(x, y, x - d.deltaX * distance, y + d.deltaY * distance,
1356                            shown, startColor, endColor, duration);
1357                }
1358                break;
1359        }

1431        float xo = getX(), yo = getY(), yOff = yo + 1f + gridHeight * font.actualCellHeight;
1432        font.draw(batch, backgrounds, xo, yo);
1433        int len = layers.size();
1434        for (int i = 0; i < len; i++) {
1435            layers.get(i).draw(batch, font, xo, yOff);

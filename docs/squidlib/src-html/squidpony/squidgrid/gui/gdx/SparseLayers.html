102    public SparseLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
103        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
104        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
105        backgrounds = ArrayTools.fill(0f, this.gridWidth, this.gridHeight);
106        layers = new ArrayList<>(4);
107        if(font.initialized())
108            this.font = font;
109        else
110            this.font = font.width(cellWidth).height(cellHeight).initBySize();
111        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
112        mapping = new IntIntMap(4);
113        mapping.put(0, 0);
114        glyphs = new ArrayList<>(16);
115        scc = DefaultResources.getSCC();
116        setBounds(xOffset, yOffset,
117                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
118    }

151     * to set an existing layer, this will add contents as a new layer on top of the others.
152     * @param layerNumber must be 0 or greater
153     * @param contents a SparseTextMap, possibly obtained with {@link #getLayer(int)}
154     */
155    public void setLayer(int layerNumber, SparseTextMap contents)
156    {
157        if(layerNumber < 0)
158            return;
159        layerNumber = mapping.get(layerNumber, layerNumber);
160        if(layerNumber >= layers.size())
161        {
162            mapping.put(layerNumber, layers.size());
163            layers.add(contents);
164        }
165        else
166        {
167            layers.set(layerNumber, contents);
168        }

185    /**
186     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added. The layer will generally be
187     * associated with a layer number equal to the number of layers currently present, and will be rendered over the
188     * existing layers. It might not be associated with the same number if layers were added out-of-order or with skips
189     * between associations. If you want to add a layer at some alternate layer number (which can be confusing but may
190     * be needed for some reason), there is another overload that allows you to specify the association number.
191     * If you may have added some layers out of order or skipped some numbers, you can use
192     * {@link #findLayer(SparseTextMap)} on the returned SparseTextMap to get its association number.
193     * @return a SparseTextMap that was just added; null if something went wrong
194     */
195    public SparseTextMap addLayer()
196    {
197        int association = layers.size();
198        while (mapping.containsKey(association)) {
199            ++association;
200        }
201        return addLayer(association);
202    }
203    /**
204     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added, or returns an existing layer
205     * if one is already associated with the given number. The layer can be greater than the number of layers currently
206     * present, which will add a layer to be rendered over the existing layers, but the number that refers to that layer
207     * will not change. It is recommended that to add a layer, you only add at the value equal to
208     * {@link #getLayerCount()}, which will maintain the order and layer numbers in a sane way, and this is the behavior
209     * of the addLayer overload that does not take a parameter.
210     * @param association the number to associate the layer with; should usually be between 0 and {@link #getLayerCount()} inclusive
211     * @return a SparseTextMap that either was just added or was already associated with the given layer number; null if something went wrong
212     */
213    public SparseTextMap addLayer(int association)
214    {
215        if(association < 0)
216            return null;
217        association = mapping.get(association, association);
218        if(association >= layers.size())
219        {
220            mapping.put(association, layers.size());
221            SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
222            layers.add(stm);
223            return stm;
224        }
225        else
226        {
227            return layers.get(association);
228        }
229    }

320    /**
321     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
322     * setting the background colors to match the given Color 2D array. If the colors argument is null, does nothing. If
323     * the chars argument is null, only affects the background colors. This will filter each Color in colors if the
324     * color center this uses has a filter.
325     * @param chars Can be {@code null}, indicating that only colors must be put.
326     * @param colors the background colors for the given chars
327     */
328    @Override
329    public void put(char[][] chars, Color[][] colors) {
330        if(chars == null)
331        {
332            if(colors != null)
333            {
334                for (int i = 0; i < colors.length; i++) {
335                    if (colors[i] == null)
336                        continue;
337                    for (int j = 0; j < colors[i].length; j++) {
338                        backgrounds[i][j] = (colors[i][j] == null) ? 0f : scc.filter(colors[i][j]).toFloatBits();
339                    }
340                }
341            }
342        }
343        else
344        {
345            if(colors == null)
346            {
347                for (int i = 0; i < chars.length; i++) {
348                    if(chars[i] == null)
349                        continue;
350                    for (int j = 0; j < chars[i].length; j++) {
351                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
352                    }
353                }
354            }
355            else
356            {
357                for (int i = 0; i < chars.length && i < colors.length; i++) {
358                    if(colors[i] == null || chars[i] == null)
359                        continue;
360                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
361                        if(colors[i][j] == null)
362                            put(i, j, chars[i][j], defaultPackedForeground, 0f);
363                        else
364                            put(i, j, chars[i][j], defaultPackedForeground, scc.filter(colors[i][j]).toFloatBits());
365                    }
366                }
367            }
368        }
369    }
370    /**
371     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
372     * setting the background colors to match the given 2D array of colors as packed floats. If the colors argument is
373     * null, does nothing. If the chars argument is null, only affects the background colors. This will not filter the
374     * passed colors at all.
375     * @param chars Can be {@code null}, indicating that only colors must be put.
376     * @param colors the background colors for the given chars
377     */
378    public void put(char[][] chars, float[][] colors) {
379        if(chars == null)
380        {
381            if(colors != null) {
382                for (int i = 0; i < colors.length; i++) {
383                    if (colors[i] == null)
384                        continue;
385                    System.arraycopy(colors[i], 0, backgrounds[i], 0, colors[i].length);
386                }
387            }
388        }
389        else
390        {
391            if(colors == null)
392            {
393                for (int i = 0; i < chars.length; i++) {
394                    if(chars[i] == null)
395                        continue;
396                    for (int j = 0; j < chars[i].length; j++) {
397                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
398                    }
399                }
400            }
401            else
402            {
403                for (int i = 0; i < chars.length && i < colors.length; i++) {
404                    if(colors[i] == null || chars[i] == null)
405                        continue;
406                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
407                        put(i, j, chars[i][j], defaultPackedForeground, colors[i][j]);
408                    }
409                }
410            }
411        }
412    }

585    /**
586     * Puts the char c at the position x,y in the requested layer with the given foreground and background colors. If
587     * foreground is null or is fully transparent (all channels 0), then this does not change
588     * the foreground contents. If background is null or is fully transparent, this does not change the background. Uses
589     * the color center to potentially filter the given colors; this can be changed with
590     * {@link #setColorCenter(IColorCenter)}. The layer can be greater than the number of layers currently present,
591     * which will add a layer to be rendered over the existing layers, but the number that refers to that layer will not
592     * change. It is recommended that to add a layer, you only add at the value equal to {@link #getLayerCount()}, which
593     * will maintain the order and layer numbers in a sane way.
594     * @param x the x position to place the char at
595     * @param y the y position to place the char at
596     * @param c the char to place
597     * @param foreground the color to use for c; if null or fully transparent, nothing will change in the foreground
598     * @param background the color to use for the cell; if null or fully transparent, nothing will change in the background
599     * @param layer the layer to place the colorful char into; should usually be between 0 and {@link #getLayerCount()} inclusive

2276                      float startColor, float endColor, float duration, /* @Nullable */ Runnable postRunnable)

2281            case SQUARE:

2293            case SPHERE:

2299                            shown, startColor, endColor, duration, null);
2300                }

2304                    d = Direction.CARDINALS_CLOCKWISE[i];

2306                            shown, startColor, endColor, duration, null);

2347     * the effect wait to start for some amount of time, and a Runnable that will be run once after the burst's full
2348     * duration completes (not once per summoned Glyph).

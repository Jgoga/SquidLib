010    /**
011     * Produces a Color[][] that corresponds to appropriate default colors for the usual meanings of the chars in map.
012     * This takes its values from {@link SColor#LIMITED_PALETTE}, and if that field is changed then the colors this
013     * returns will also change. Tiles containing nothing but a floor ('.') will be silver-gray. Walls are dark gray
014     * (this doesn't care if the walls are '#' marks or box-drawing lines), Doors ('+' and '/' in the map) will be rust
015     * brown. Both shallow and deep water (',' and '~') will be gray-blue. Traps ('^') will be shown in bright orange.
016     * Grass ('"') will be the expected green.
017     *
018     * @param map a char[][] containing foreground characters
019     * @return a 2D array of Colors with the same size as map, that can be used for the corresponding chars
020     */
021    public static Color[][] generateDefaultColors(char[][] map) {
022
023        int width = map.length;
024        int height = map[0].length;
025        Color[][] portion = new Color[width][height];
026        for (int i = 0; i < width; i++) {
027            for (int j = 0; j < height; j++) {
028                switch (map[i][j]) {
029                    case '\1':
030                    case '├':
031                    case '┤':
032                    case '┴':
033                    case '┬':
034                    case '┌':
035                    case '┐':
036                    case '└':
037                    case '┘':
038                    case '│':
039                    case '─':
040                    case '┼':
041                    case '#':
042                        portion[i][j] = SColor.LIMITED_PALETTE[2];
043                        break;
044                    case '.':
045                    case ':':
046                        portion[i][j] = SColor.LIMITED_PALETTE[3];
047                        break;
048                    case '+':
049                    case '/':
050                        portion[i][j] = SColor.LIMITED_PALETTE[4];
051                        break;
052                    case ',':
053                    case '~':
054                        portion[i][j] = SColor.LIMITED_PALETTE[5];
055                        break;
056                    case '"':
057                        portion[i][j] = SColor.LIMITED_PALETTE[20];
058                        break;
059                    case '^':
060                        portion[i][j] = SColor.LIMITED_PALETTE[6];
061                        break;
062                    default:
063                        portion[i][j] = SColor.LIMITED_PALETTE[1];
064                }
065            }
066        }
067        return portion;
068    }

291    /**
292     * Produces an int[][] that can be used with SquidLayers to alter the background colors.
293     *
294     * @param map a char[][] that you want to be find background lightness modifiers for
295     * @return a 2D array of lightness values from -255 to 255 but usually close to 0; can be passed to SquidLayers
296     */
297    public static int[][] generateLightnessModifiers(char[][] map) {
298        int width = map.length;
299        int height = map[0].length;
300        int[][] portion = new int[width][height];
301        for (int i = 0; i < width; i++) {
302            for (int j = 0; j < height; j++) {
303                switch (map[i][j]) {
304                    case '\1':
305                    case '├':
306                    case '┤':
307                    case '┴':
308                    case '┬':
309                    case '┌':
310                    case '┐':
311                    case '└':
312                    case '┘':
313                    case '│':
314                    case '─':
315                    case '┼':
316                    case '#':
317                        portion[i][j] = 30;
318                        break;
319                    case '.':
320                        portion[i][j] = 0;
321                        break;
322                    case ':':
323                        portion[i][j] = -15;
324                        break;
325                    case '+':
326                    case '/':
327                        portion[i][j] = -10;
328                        break;
329                    case ',':
330                        portion[i][j] = (int) (85 * (PerlinNoise.noise(i * 1.45, j * 1.45) * 0.55 - 0.7));
331                        break;
332                    case '~':
333                        portion[i][j] = (int) (100 * (PerlinNoise.noise(i * 1.45, j * 1.45) * 0.4 - 0.65));
334                        break;
335                    case '"':
336                        portion[i][j] = (int) (95 * (PerlinNoise.noise(i * 1.45, j * 1.45) * 0.3 - 1.5));
337                        break;
338                    case '^':
339                        portion[i][j] = 40;
340                        break;
341                    default:
342                        portion[i][j] = 0;
343                }
344            }
345        }
346        return portion;
347    }

415     * to be treated like deep and shallow water regarding the ripple animation.
416     *
417     * @param map           a char[][] that you want to be find background lightness modifiers for

422     * @param deepLiquid    a char that will be treated like deep water when animating ripples
423     * @param shallowLiquid a char that will be treated like shallow water when animating ripples
424     * @return a 2D array of lightness values from -255 to 255 but usually close to 0; can be passed to SquidLayers
425     */

427        int width = map.length;
428        int height = map[0].length;
429        int[][] portion = new int[width][height];
430        for (int i = 0; i < width; i++) {
431            for (int j = 0; j < height; j++) {
432                switch (map[i][j]) {
433                    case '\1':
434                    case '├':
435                    case '┤':
436                    case '┴':
437                    case '┬':
438                    case '┌':
439                    case '┐':
440                    case '└':
441                    case '┘':
442                    case '│':
443                    case '─':
444                    case '┼':
445                    case '#':
446                        portion[i][j] = 30;
447                        break;
448                    case '.':
449                        portion[i][j] = 0;
450                        break;
451                    case ':':
452                        portion[i][j] = -15;
453                        break;
454                    case '+':
455                    case '/':
456                        portion[i][j] = -10;
457                        break;
458                    case ',':
459                        portion[i][j] = (int) (85 * (PerlinNoise.noise(i * 1.45, j * 1.45, frame * 0.45) * 0.55 - 0.7));
460                        break;
461                    case '~':

475                        else portion[i][j] = 0;
476                }
477            }
478        }
479        return portion;
480    }

036     * A Filter that converts all colors passed to it to grayscale, like a black and white film.
037     */
038    public static class GrayscaleFilter implements IFilter<Color>
039    {
040        public GrayscaleFilter()
041        {
042                /* Nothing to do */
043        }
044
045        @Override
046        public Color alter(float r, float g, float b, float a) {
047            float v = (r + g + b) / 3f;
048            return new Color(v, v, v, a);
049        }

053    /**
054     * A Filter that tracks the highest brightness for any component it was assigned and stores it in state as the first
055     * and only element.
056     */
057    public static class MaxValueFilter extends Filter<Color>
058    {
059        public MaxValueFilter()
060        {
061            state = new float[]{0};
062        }
063
064        @Override
065        public Color alter(float r, float g, float b, float a) {
066            state[0] = Math.max(state[0], Math.max(r, Math.max(g, b)));
067            return new Color(r, g, b, a);
068        }

072    /**
073     * A Filter that performs a brightness adjustment to make dark areas lighter and light areas not much less bright.
074     */
075    public static class GammaCorrectFilter extends Filter<Color> {
076        /**
077         * Sets up a GammaCorrectFilter with the desired gamma adjustment.
078         *
079         * @param gamma    should be 1.0 or less, and must be greater than 0. Typical values are between 0.4 to 0.8.
080         * @param maxValue the maximum brightness in the colors this will be passed; use MaxValueFilter for this
081         */
082        public GammaCorrectFilter(float gamma, float maxValue) {
083            state = new float[]{gamma, 1f / (float) Math.pow(maxValue, gamma)};
084        }
085
086        @Override
087        public Color alter(float r, float g, float b, float a) {
088            return new Color(state[1] * (float) Math.pow(r, state[0]),
089                    state[1] * (float) Math.pow(g, state[0]),
090                    state[1] * (float) Math.pow(b, state[0]),
091                    a);
092        }
093    }
094    /**
095     * A Filter that is constructed with a color and linear-interpolates any color it is told to alter toward the color
096     * it was constructed with.
097     */
098    public static class LerpFilter extends Filter<Color> {
099        /**
100         * Sets up a LerpFilter with the desired color to linearly interpolate towards.
101         *
102         * @param r the red component to lerp towards
103         * @param g the green component to lerp towards
104         * @param b the blue component to lerp towards
105         * @param a the opacity component to lerp towards
106         * @param amount the amount to lerp by, should be between 0.0 and 1.0
107         */
108        public LerpFilter(float r, float g, float b, float a, float amount) {
109            state = new float[]{r, g, b, a, amount};
110        }
111        /**
112         * Sets up a LerpFilter with the desired color to linearly interpolate towards.
113         *
114         * @param color the Color to lerp towards
115         * @param amount the amount to lerp by, should be between 0.0 and 1.0
116         */
117        public LerpFilter(Color color, float amount) {
118            state = new float[]{color.r, color.g, color.b, color.a, amount};
119        }
120
121        @Override
122        public Color alter(float r, float g, float b, float a) {
123            return new Color(r, g, b, a).lerp(state[0], state[1], state[2], state[3], state[4]);
124        }
125
126    }
127    /**
128     * A Filter that is constructed with a group of colors and linear-interpolates any color it is told to alter toward
129     * the color it was constructed with that has the closest hue.
130     */
131    public static class MultiLerpFilter extends Filter<Color> {
132        private SquidColorCenter globalSCC;
133        /**
134         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards; the lengths of each given
135         * array should be identical.
136         *
137         * @param r the red components to lerp towards
138         * @param g the green components to lerp towards
139         * @param b the blue components to lerp towards
140         * @param a the opacity components to lerp towards
141         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
142         */
143        public MultiLerpFilter(float[] r, float[] g, float[] b, float[] a, float[] amount) {
144            state = new float[Math.min(r.length, Math.min(g.length, Math.min(b.length,
145                    Math.min(a.length, amount.length)))) * 6];
146            globalSCC = DefaultResources.getSCC();
147            for (int i = 0; i < state.length / 6; i++) {
148                state[i * 6] = r[i];
149                state[i * 6 + 1] = g[i];
150                state[i * 6 + 2] = b[i];
151                state[i * 6 + 3] = a[i];
152                state[i * 6 + 4] = amount[i];
153                state[i * 6 + 5] = globalSCC.getHue(r[i], g[i], b[i]);
154            }
155        }/**
156         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards and their amounts.
157         *
158         * @param colors the Colors to lerp towards
159         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
160         */
161        public MultiLerpFilter(Color[] colors, float[] amount) {
162            state = new float[Math.min(colors.length, amount.length) * 6];
163            globalSCC = DefaultResources.getSCC();
164            for (int i = 0; i < state.length / 6; i++) {
165                state[i * 6] = colors[i].r;
166                state[i * 6 + 1] = colors[i].g;
167                state[i * 6 + 2] = colors[i].b;
168                state[i * 6 + 3] = colors[i].a;
169                state[i * 6 + 4] = amount[i];
170                state[i * 6 + 5] = globalSCC.getHue(colors[i]);
171            }
172        }
173
174        @Override
175        public Color alter(float r, float g, float b, float a) {
176            float givenH = globalSCC.getHue(r, g, b), givenS = globalSCC.getSaturation(r, g, b),
177                    minDiff = 999.0f, temp;
178            if(givenS < 0.05)
179                return new Color(r, g, b, a);
180            int choice = 0;
181            for (int i = 5; i < state.length; i += 6) {
182                temp = state[i] - givenH;
183                temp = (temp >= 0.5f) ? Math.abs(temp - 1f) % 1f : Math.abs(temp);
184                if(temp < minDiff) {
185                    minDiff = temp;
186                    choice = i / 6; // rounds down
187                }
188            }
189            return new Color(r, g, b, a).lerp(state[choice * 6], state[choice * 6 + 1], state[choice * 6 + 2],
190                    state[choice * 6 + 3], state[choice * 6 + 4]);
191        }
192    }

194    /**
195     * A Filter that is constructed with a color and makes any color it is told to alter have the same hue as the given
196     * color, have saturation that is somewhere between the given color's and the altered colors, and chiefly is
197     * distinguishable from other colors by value. Useful for sepia effects, which can be created satisfactorily with
198     * {@code new Filters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
199     */
200    public static class ColorizeFilter extends Filter<Color> {
201        private SquidColorCenter globalSCC;
202        /**
203         * Sets up a ColorizeFilter with the desired color to colorize towards.
204         *
205         * @param r the red component to colorize towards
206         * @param g the green component to colorize towards
207         * @param b the blue component to colorize towards
208         */
209        public ColorizeFilter(float r, float g, float b) {
210            globalSCC = DefaultResources.getSCC();
211
212            state = new float[]{globalSCC.getHue(r, g, b), globalSCC.getSaturation(r, g, b), 1f, 0f};
213        }
214        /**
215         * Sets up a ColorizeFilter with the desired color to colorize towards.
216         *
217         * @param color the Color to colorize towards
218         */
219        public ColorizeFilter(Color color) {
220            globalSCC = DefaultResources.getSCC();
221            state = new float[]{globalSCC.getHue(color), globalSCC.getSaturation(color), 1f, 0f};
222        }
223        /**
224         * Sets up a ColorizeFilter with the desired color to colorize towards.
225         *
226         * @param r the red component to colorize towards
227         * @param g the green component to colorize towards
228         * @param b the blue component to colorize towards
229         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
230         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
231         *                      typically very small, such as in the -0.2f to 0.2f range
232         */
233        public ColorizeFilter(float r, float g, float b, float saturationMultiplier, float valueModifier) {
234            globalSCC = DefaultResources.getSCC();
235
236            state = new float[]{
237                    globalSCC.getHue(r, g, b),
238                    globalSCC.getSaturation(r, g, b),
239                    saturationMultiplier,
240                    valueModifier};
241        }
242        /**
243         * Sets up a ColorizeFilter with the desired color to colorize towards.
244         *
245         * @param color the Color to colorize towards
246         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
247         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
248         *                      typically very small, such as in the -0.2f to 0.2f range
249         */
250        public ColorizeFilter(Color color, float saturationMultiplier, float valueModifier) {
251            globalSCC = DefaultResources.getSCC();
252            state = new float[]{
253                    globalSCC.getHue(color),
254                    globalSCC.getSaturation(color),
255                    saturationMultiplier,
256                    valueModifier};
257        }
258
259        @Override
260        public Color alter(float r, float g, float b, float a) {
261            return globalSCC.getHSV(
262                    state[0],
263                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) + state[1]) * 0.5f * state[2], 1f)),
264                    globalSCC.getValue(r, g, b) * (1f - state[3]) + state[3],
265                    a);
266        }
267    }
268    /**
269     * A Filter that makes the colors requested from it highly saturated, with the original hue, value and a timer that
270     * increments very slowly altering hue, with hue, value and the timer altering saturation, and the original hue,
271     * saturation, and value all altering value. It should look like a hallucination.
272     * <br>
273     * A short (poorly recorded) video can be seen here http://i.imgur.com/SEw2LXe.gifv ; performance should be smoother
274     * during actual gameplay.
275     */
276    public static class HallucinateFilter extends Filter<Color> {
277        private SquidColorCenter globalSCC;
278        /**
279         * Sets up a HallucinateFilter with the timer at 0..
280         */
281        public HallucinateFilter() {
282            globalSCC = DefaultResources.getSCC();
283
284            state = new float[]{0f};
285        }
286        @Override
287        public Color alter(float r, float g, float b, float a) {
288            state[0] += 0.00003f;
289            if(state[0] >= 1.0f)
290                state[0] = 0f;
291            float h = globalSCC.getHue(r, g, b),
292                    s = globalSCC.getSaturation(r, g, b),
293                    v = globalSCC.getValue(r, g, b);
294            return globalSCC.getHSV(
295                    (v * 4f + h + state[0]) % 1.0f,
296                    Math.max(0f, Math.min((h + v) * 0.65f + state[0] * 0.4f, 1f)),
297                    (h + v + s) * 0.35f + 0.7f,
298                    a);
299        }
300
301    }

304    /**
305     * A Filter that multiplies the saturation of any color requested from it by a number given during construction.
306     * When desaturating, you may want to prefer {@link SaturationValueFilter}, because reducing saturation increases
307     * perceived brightness (value), and desaturating an ocean-blue color will make it look like ice if the value
308     * isn't decreased as well. This class is perfectly fine for increasing saturation, in general.
309     */
310    public static class SaturationFilter extends Filter<Color> {
311        private SquidColorCenter globalSCC;
312        /**
313         * Sets up a SaturationFilter with the desired saturation multiplier. Using a multiplier of 0f, as you would
314         * expect, makes the image grayscale. Using a multiplier of 0.5 make the image "muted", with no truly bright
315         * colors, while 1.0f makes no change, and and any numbers higher than 1.0f will make the image more saturated,
316         * with the exception of colors that were already grayscale or close to it. This clamps the result, so there's
317         * no need to worry about using too high of a saturation multiplier.
318         *
319         * @param multiplier the amount to multiply each requested color's saturation by; 1.0f means "no change"
320         */
321        public SaturationFilter(float multiplier) {
322            globalSCC = DefaultResources.getSCC();
323
324            state = new float[]{multiplier};
325        }
326
327        @Override
328        public Color alter(float r, float g, float b, float a) {
329            return globalSCC.getHSV(
330                    globalSCC.getHue(r, g, b),
331                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) * state[0]), 1f)),
332                    globalSCC.getValue(r, g, b),
333                    a);
334        }
335
336    }

338    /**
339     * A Filter that multiplies the saturation and the value of any color requested from it by different numbers given
340     * during construction.
341     */
342    public static class SaturationValueFilter extends Filter<Color> {
343        private SquidColorCenter globalSCC;
344        /**
345         * Sets up a SaturationValueFilter with the desired saturation and value multipliers. Using a multiplier of 0f for
346         * saturation, as you would expect, makes the image grayscale. Using a multiplier of 0f for value makes all
347         * colors identicaly black; this is probably not a good idea. Using a multiplier of 0.5 for saturation will
348         * make the image "muted", with no truly bright colors, while 1.0f for saturation makes no change, and and any
349         * numbers higher than 1.0f for saturation will make the image more saturated, with the exception of colors that
350         * were already grayscale or close to it. Using 0.5f for value will darken the image significantly, 1.0f for
351         * value will keep it the same for brightness, and higher than 1.0f will lighten it. This clamps the result, so
352         * there's no need to worry about using too high of a saturation or value multiplier.
353         *
354         * @param saturation the amount to multiply each requested color's saturation by; 1.0f means "no change"
355         * @param value      the amount to multiply each requested color's value (lightness) by; 1.0f means "no change"
356         */
357        public SaturationValueFilter(float saturation, float value) {
358            globalSCC = DefaultResources.getSCC();
359
360            state = new float[]{saturation, value};
361        }
362
363        @Override
364        public Color alter(float r, float g, float b, float a) {
365            return globalSCC.getHSV(
366                    globalSCC.getHue(r, g, b),
367                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) * state[0]), 1f)),
368                    Math.max(0f, Math.min((globalSCC.getValue(r, g, b) * state[1]), 1f)),
369                    a);
370        }
371
372    }

393    /**
394     * A Filter that is constructed with a group of colors and forces any color it is told to alter to exactly
395     * the color it was constructed with that has the closest red, green, and blue components. A convenient way to
396     * use this is to pass in one of the color series from SColor, such as {@link SColor#RED_SERIES} or
397     * {@link SColor#ACHROMATIC_SERIES}. This can also be used to enforce usage of a limited color palette such as
398     * one of DawnBringer's popular pixel art palettes, {@link SColor#DAWNBRINGER_16} and
399     * {@link SColor#DAWNBRINGER_32}. Beyond 32 colors, the palettes {@link SColor#VARIED_PALETTE} with 56 colors
400     * and {@link SColor#COLOR_WHEEL_PALETTE} with 189 colors can be used.
401     *
402     * Preview using BLUE_GREEN_SERIES foreground, ACHROMATIC_SERIES background: http://i.imgur.com/2HdZpC9.png
403     */
404    public static class PaletteFilter extends Filter<Color> {
405        /**
406         * Sets up a PaletteFilter with the exact colors to use as individual components; the lengths of each given
407         * array should be identical.
408         *
409         * @param r the red components to use
410         * @param g the green components to use
411         * @param b the blue components to use
412         */
413        public PaletteFilter(float[] r, float[] g, float[] b) {
414            state = new float[Math.min(r.length, Math.min(g.length, b.length)) * 3];
415            for (int i = 0; i < state.length / 4; i++) {
416                state[i * 3] = MathUtils.clamp(r[i], 0f, 1f);
417                state[i * 3 + 1] = MathUtils.clamp(g[i], 0f, 1f);
418                state[i * 3 + 2] = MathUtils.clamp(b[i], 0f, 1f);
419            }
420        }/**
421         * Sets up a PaletteFilter with the exact colors to use as Colors. A convenient way to
422         * use this is to pass in one of the color series from SColor, such as RED_SERIES or ACHROMATIC_SERIES.
423         * The alpha component of each color in the palette is ignored, since when a color is requested for
424         * filtering, its alpha is respected and only its red, green, and blue components are changed.
425         * @param colors the Colors to use
426         */
427        public PaletteFilter(Color[] colors) {
428            state = new float[colors.length * 3];
429            for (int i = 0; i < colors.length; i++) {
430                state[i * 3] = colors[i].r;
431                state[i * 3 + 1] = colors[i].g;
432                state[i * 3 + 2] = colors[i].b;
433            }
434        }
435
436        @Override
437        public Color alter(float r, float g, float b, float a) {
438            float diff = 9999.0f, temp;
439            int choice = 0;
440            for (int i = 0; i < state.length; i += 4) {
441                temp = Math.abs(state[i] - r) + Math.abs(state[i + 1] - g) + Math.abs(state[i + 2] - b);
442                if(temp < diff) {
443                    diff = temp;
444                    choice = i;
445                }
446            }
447            return new Color(state[choice], state[choice + 1], state[choice + 2],
448                    a);
449        }
450    }
451    /**
452     * A Filter that alters primarily-red and primarily-green colors so they can be more easily be distinguished by
453     * people with at least some forms of red-green color-blindness (deuteranopia should be handled well, protanopia
454     * very well, and tritanopia may not benefit at all). Causes reds to be darkened and greens to be lightened if the
455     * other of the pair is not present in similar quantities (which is the case for yellows and blues).
456     */
457    public static class DistinctRedGreenFilter implements IFilter<Color> {
458        /**
459         * Constructs a DistinctRedGreenFilter. This class is a simple wrapper around a function that doesn't need
460         * member variables, so there should be little overhead with this filter.
461         */
462        public DistinctRedGreenFilter() {
463        }
464
465        @Override
466        public Color alter(float r, float g, float b, float a) {
467            float diff = g - r;
468            if(diff > 0.4f)
469                return new Color(Math.min(1f, r * (0.8f + diff * 0.5f)), Math.min(1f, g * (0.9f + diff * 0.5f)),
470                        Math.min(1f, b * (0.8f + diff * 0.5f)), a);
471            else if(diff < -0.3f)
472                return new Color(r * (0.6f - diff), g * (0.7f - diff),
473                        b * (0.7f - diff), a);
474            else
475                return new Color(r, g, b, a);
476        }

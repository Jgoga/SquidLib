022    /**
023     * A Filter that does nothing to the colors it is given but pass them along unchanged.
024     */
025    public static class IdentityFilter implements IFilter<Color>
026    {
027        public IdentityFilter()
028        {
029                /* Nothing to do */
030        }
031
032        @Override
033        public Color alter(float r, float g, float b, float a) {
034            return new Color(r, g, b, a);
035        }

037    public static class ChainFilter implements IFilter<Color>
038    {
039        public ArrayList<IFilter<Color>> chains;
040        public ChainFilter(IFilter<Color> firstFilter, IFilter<Color> secondFilter)
041        {
042            chains = new ArrayList<>(2);
043            chains.add(firstFilter);
044            chains.add(secondFilter);
045        }
046        public ChainFilter(IFilter<Color> firstFilter, IFilter<Color> secondFilter, IFilter<Color> thirdFilter)
047        {
048            chains = new ArrayList<>(2);
049            chains.add(firstFilter);
050            chains.add(secondFilter);
051            chains.add(thirdFilter);
052        }
053        public ChainFilter(Collection<IFilter<Color>> filters)
054        {
055            if(filters == null || filters.isEmpty())
056            {
057                chains = new ArrayList<>();
058                chains.add(new IdentityFilter());
059            }
060            else chains = new ArrayList<>(filters);
061        }
062        @Override
063        public Color alter(float r, float g, float b, float a) {
064            Color c = chains.get(0).alter(r, g, b, a);
065            for (int i = 1; i < chains.size(); i++) {
066                c = chains.get(i).alter(c.r, c.g, c.b, c.a);
067            }
068            return c;
069        }
070    }

111    public static class GammaCorrectFilter extends Filter<Color> {
112        /**
113         * Sets up a GammaCorrectFilter with the desired gamma adjustment.
114         *
115         * @param gamma    should be 1.0 or less, and must be greater than 0. Typical values are between 0.4 to 0.8.
116         * @param maxValue the maximum brightness in the colors this will be passed; use MaxValueFilter for this
117         */
118        public GammaCorrectFilter(float gamma, float maxValue) {
119            state = new float[]{gamma, 1f / (float) Math.pow(maxValue, gamma)};
120        }
121
122        @Override
123        public Color alter(float r, float g, float b, float a) {
124            return new Color(state[1] * (float) Math.pow(r, state[0]),
125                    state[1] * (float) Math.pow(g, state[0]),
126                    state[1] * (float) Math.pow(b, state[0]),
127                    a);
128        }

163    /**
164     * A Filter that is constructed with a group of colors and linear-interpolates any color it is told to alter toward
165     * the color it was constructed with that has the closest hue.
166     */
167    public static class MultiLerpFilter extends Filter<Color> {
168        private SquidColorCenter globalSCC;
169        /**
170         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards; the lengths of each given
171         * array should be identical.
172         *
173         * @param r the red components to lerp towards
174         * @param g the green components to lerp towards
175         * @param b the blue components to lerp towards
176         * @param a the opacity components to lerp towards
177         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
178         */
179        public MultiLerpFilter(float[] r, float[] g, float[] b, float[] a, float[] amount) {
180            state = new float[Math.min(r.length, Math.min(g.length, Math.min(b.length,
181                    Math.min(a.length, amount.length)))) * 6];
182            globalSCC = DefaultResources.getSCC();
183            for (int i = 0; i < state.length / 6; i++) {
184                state[i * 6] = r[i];
185                state[i * 6 + 1] = g[i];
186                state[i * 6 + 2] = b[i];
187                state[i * 6 + 3] = a[i];
188                state[i * 6 + 4] = amount[i];
189                state[i * 6 + 5] = globalSCC.getHue(r[i], g[i], b[i]);
190            }
191        }/**
192         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards and their amounts.
193         *
194         * @param colors the Colors to lerp towards
195         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
196         */
197        public MultiLerpFilter(Color[] colors, float[] amount) {
198            state = new float[Math.min(colors.length, amount.length) * 6];
199            globalSCC = DefaultResources.getSCC();
200            for (int i = 0; i < state.length / 6; i++) {
201                state[i * 6] = colors[i].r;
202                state[i * 6 + 1] = colors[i].g;
203                state[i * 6 + 2] = colors[i].b;
204                state[i * 6 + 3] = colors[i].a;
205                state[i * 6 + 4] = amount[i];
206                state[i * 6 + 5] = globalSCC.getHue(colors[i]);
207            }
208        }
209
210        @Override
211        public Color alter(float r, float g, float b, float a) {
212            float givenH = globalSCC.getHue(r, g, b), givenS = globalSCC.getSaturation(r, g, b),
213                    minDiff = 999.0f, temp;
214            if(givenS < 0.05)
215                return new Color(r, g, b, a);
216            int choice = 0;
217            for (int i = 5; i < state.length; i += 6) {
218                temp = state[i] - givenH;
219                temp = (temp >= 0.5f) ? Math.abs(temp - 1f) % 1f : Math.abs(temp);
220                if(temp < minDiff) {
221                    minDiff = temp;
222                    choice = i / 6; // rounds down
223                }
224            }
225            return new Color(r, g, b, a).lerp(state[choice * 6], state[choice * 6 + 1], state[choice * 6 + 2],
226                    state[choice * 6 + 3], state[choice * 6 + 4]);
227        }
228    }

230    /**
231     * A Filter that is constructed with a color and makes any color it is told to alter have the same hue as the given
232     * color, have saturation that is somewhere between the given color's and the altered colors, and chiefly is
233     * distinguishable from other colors by value. Useful for sepia effects, which can be created satisfactorily with
234     * {@code new Filters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
235     */
236    public static class ColorizeFilter extends Filter<Color> {
237        private SquidColorCenter globalSCC;
238        /**
239         * Sets up a ColorizeFilter with the desired color to colorize towards.
240         *
241         * @param r the red component to colorize towards
242         * @param g the green component to colorize towards
243         * @param b the blue component to colorize towards
244         */
245        public ColorizeFilter(float r, float g, float b) {
246            globalSCC = DefaultResources.getSCC();
247
248            state = new float[]{globalSCC.getHue(r, g, b), globalSCC.getSaturation(r, g, b), 1f, 0f};
249        }
250        /**
251         * Sets up a ColorizeFilter with the desired color to colorize towards.
252         *
253         * @param color the Color to colorize towards
254         */
255        public ColorizeFilter(Color color) {
256            globalSCC = DefaultResources.getSCC();
257            state = new float[]{globalSCC.getHue(color), globalSCC.getSaturation(color), 1f, 0f};
258        }
259        /**
260         * Sets up a ColorizeFilter with the desired color to colorize towards.
261         *
262         * @param r the red component to colorize towards
263         * @param g the green component to colorize towards
264         * @param b the blue component to colorize towards
265         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
266         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
267         *                      typically very small, such as in the -0.2f to 0.2f range
268         */
269        public ColorizeFilter(float r, float g, float b, float saturationMultiplier, float valueModifier) {
270            globalSCC = DefaultResources.getSCC();
271
272            state = new float[]{
273                    globalSCC.getHue(r, g, b),
274                    globalSCC.getSaturation(r, g, b),
275                    saturationMultiplier,
276                    valueModifier};
277        }
278        /**
279         * Sets up a ColorizeFilter with the desired color to colorize towards.
280         *
281         * @param color the Color to colorize towards
282         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
283         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
284         *                      typically very small, such as in the -0.2f to 0.2f range
285         */
286        public ColorizeFilter(Color color, float saturationMultiplier, float valueModifier) {
287            globalSCC = DefaultResources.getSCC();
288            state = new float[]{
289                    globalSCC.getHue(color),
290                    globalSCC.getSaturation(color),
291                    saturationMultiplier,
292                    valueModifier};
293        }
294
295        @Override
296        public Color alter(float r, float g, float b, float a) {
297            return globalSCC.getHSV(
298                    state[0],
299                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) + state[1]) * 0.5f * state[2], 1f)),
300                    globalSCC.getValue(r, g, b) * (1f - state[3]) + state[3],
301                    a);
302        }
303    }
304    /**
305     * A Filter that makes the colors requested from it highly saturated, with the original hue, value and a timer that
306     * increments very slowly altering hue, with hue, value and the timer altering saturation, and the original hue,
307     * saturation, and value all altering value. It should look like a hallucination.
308     * <br>
309     * A short (poorly recorded) video can be seen here http://i.imgur.com/SEw2LXe.gifv ; performance should be smoother
310     * during actual gameplay.
311     */
312    public static class HallucinateFilter extends Filter<Color> {
313        private SquidColorCenter globalSCC;
314        /**
315         * Sets up a HallucinateFilter with the timer at 0..
316         */
317        public HallucinateFilter() {
318            globalSCC = DefaultResources.getSCC();
319
320            state = new float[]{0f};
321        }
322        @Override
323        public Color alter(float r, float g, float b, float a) {
324            state[0] += 0.00003f;
325            if(state[0] >= 1.0f)
326                state[0] = 0f;
327            float h = globalSCC.getHue(r, g, b),
328                    s = globalSCC.getSaturation(r, g, b),
329                    v = globalSCC.getValue(r, g, b);
330            return globalSCC.getHSV(
331                    (v * 4f + h + state[0]) % 1.0f,
332                    Math.max(0f, Math.min((h + v) * 0.65f + state[0] * 0.4f, 1f)),
333                    (h + v + s) * 0.35f + 0.7f,
334                    a);
335        }

340    /**
341     * A Filter that multiplies the saturation of any color requested from it by a number given during construction.
342     * When desaturating, you may want to prefer {@link SaturationValueFilter}, because reducing saturation increases
343     * perceived brightness (value), and desaturating an ocean-blue color will make it look like ice if the value
344     * isn't decreased as well. This class is perfectly fine for increasing saturation, in general.
345     */
346    public static class SaturationFilter extends Filter<Color> {
347        private SquidColorCenter globalSCC;
348        /**
349         * Sets up a SaturationFilter with the desired saturation multiplier. Using a multiplier of 0f, as you would
350         * expect, makes the image grayscale. Using a multiplier of 0.5 make the image "muted", with no truly bright
351         * colors, while 1.0f makes no change, and and any numbers higher than 1.0f will make the image more saturated,
352         * with the exception of colors that were already grayscale or close to it. This clamps the result, so there's
353         * no need to worry about using too high of a saturation multiplier.
354         *
355         * @param multiplier the amount to multiply each requested color's saturation by; 1.0f means "no change"
356         */
357        public SaturationFilter(float multiplier) {
358            globalSCC = DefaultResources.getSCC();
359
360            state = new float[]{multiplier};
361        }
362
363        @Override
364        public Color alter(float r, float g, float b, float a) {
365            return globalSCC.getHSV(
366                    globalSCC.getHue(r, g, b),
367                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) * state[0]), 1f)),
368                    globalSCC.getValue(r, g, b),
369                    a);
370        }
371
372    }

374    /**
375     * A Filter that multiplies the saturation and the value of any color requested from it by different numbers given
376     * during construction.
377     */
378    public static class SaturationValueFilter extends Filter<Color> {
379        private SquidColorCenter globalSCC;
380        /**
381         * Sets up a SaturationValueFilter with the desired saturation and value multipliers. Using a multiplier of 0f for
382         * saturation, as you would expect, makes the image grayscale. Using a multiplier of 0f for value makes all
383         * colors identicaly black; this is probably not a good idea. Using a multiplier of 0.5 for saturation will
384         * make the image "muted", with no truly bright colors, while 1.0f for saturation makes no change, and and any
385         * numbers higher than 1.0f for saturation will make the image more saturated, with the exception of colors that
386         * were already grayscale or close to it. Using 0.5f for value will darken the image significantly, 1.0f for
387         * value will keep it the same for brightness, and higher than 1.0f will lighten it. This clamps the result, so
388         * there's no need to worry about using too high of a saturation or value multiplier.
389         *
390         * @param saturation the amount to multiply each requested color's saturation by; 1.0f means "no change"
391         * @param value      the amount to multiply each requested color's value (lightness) by; 1.0f means "no change"
392         */
393        public SaturationValueFilter(float saturation, float value) {
394            globalSCC = DefaultResources.getSCC();
395
396            state = new float[]{saturation, value};
397        }
398
399        @Override
400        public Color alter(float r, float g, float b, float a) {
401            return globalSCC.getHSV(
402                    globalSCC.getHue(r, g, b),
403                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) * state[0]), 1f)),
404                    Math.max(0f, Math.min((globalSCC.getValue(r, g, b) * state[1]), 1f)),
405                    a);
406        }

429    /**
430     * A Filter that is constructed with a group of colors and forces any color it is told to alter to exactly
431     * the color it was constructed with that has the closest red, green, and blue components. A convenient way to
432     * use this is to pass in one of the color series from SColor, such as {@link SColor#RED_SERIES} or
433     * {@link SColor#ACHROMATIC_SERIES}. This can also be used to enforce usage of a limited color palette such as
434     * one of DawnBringer's popular pixel art palettes, {@link SColor#DAWNBRINGER_16} and
435     * {@link SColor#DAWNBRINGER_32}. Beyond 32 colors, the palettes {@link SColor#VARIED_PALETTE} with 56 colors
436     * and {@link SColor#COLOR_WHEEL_PALETTE} with 189 colors can be used.
437     *
438     * Preview using BLUE_GREEN_SERIES foreground, ACHROMATIC_SERIES background: http://i.imgur.com/2HdZpC9.png
439     */
440    public static class PaletteFilter extends Filter<Color> {
441        /**
442         * The array of Color objects this will use as a palette.
443         */
444        public Color[] colorStore;
445        /**
446         * Sets up a PaletteFilter with the exact colors to use as individual components; the lengths of each given
447         * array should be identical.
448         *
449         * @param r the red components to use
450         * @param g the green components to use
451         * @param b the blue components to use
452         */
453        public PaletteFilter(float[] r, float[] g, float[] b) {
454            colorStore = new Color[Math.min(r.length, Math.min(g.length, b.length))];
455            state = new float[colorStore.length * 3];
456            for (int i = 0; i < colorStore.length; i++) {
457                colorStore[i] = new Color(
458                state[i * 3] = MathUtils.clamp(r[i], 0f, 1f),
459                state[i * 3 + 1] = MathUtils.clamp(g[i], 0f, 1f),
460                state[i * 3 + 2] = MathUtils.clamp(b[i], 0f, 1f),
461                        1f);
462            }
463        }/**
464         * Sets up a PaletteFilter with the exact colors to use as Colors. A convenient way to
465         * use this is to pass in one of the color series from SColor, such as RED_SERIES or ACHROMATIC_SERIES.
466         * The alpha component of each color in the palette is ignored, since when a color is requested for
467         * filtering, its alpha is respected and only its red, green, and blue components are changed.
468         * The {@code colors} array is used verbatim (as a reference, not a copy), so changes to the Color values inside
469         * it will change how this PaletteFilter works (possibly badly). If you expect to edit the array you give as a
470         * parameter to this, it may be optimal to give a temporary copy.
471         * @param colors the Colors to use as an array; will be referenced in the Filter, so changing this array changes
472         *               what Colors will be used
473         */
474        public PaletteFilter(Color[] colors) {
475            colorStore = colors;
476            state = new float[colors.length * 3];
477            for (int i = 0; i < colors.length; i++) {
478                state[i * 3] = colors[i].r;
479                state[i * 3 + 1] = colors[i].g;
480                state[i * 3 + 2] = colors[i].b;
481            }
482        }
483
484        @Override
485        public Color alter(float r, float g, float b, float a) {
486            float diff = 9999.0f, temp;
487            int choice = 0;
488            r = (int)(r * 0x1.8p5f) * 0x1.555556p-6f;
489            g = (int)(g * 0x1.8p5f) * 0x1.555556p-6f;
490            b = (int)(b * 0x1.8p5f) * 0x1.555556p-6f;
491            for (int i = 0; i < state.length; i += 3) {
492                temp = Math.abs(state[i] - r) + Math.abs(state[i + 1] - g) + Math.abs(state[i + 2] - b);
493                if(temp < diff) {
494                    diff = temp;
495                    choice = i;
496                }
497            }
498            if(a >= 1f)
499                return colorStore[choice / 3];
500            Color ret = colorStore[choice / 3].cpy();
501            ret.a = a;
502            return ret;
503        }
504    }
505    /**
506     * A Filter that alters primarily-red and primarily-green colors so they can be more easily be distinguished by
507     * people with at least some forms of red-green color-blindness (deuteranopia should be handled well, protanopia
508     * very well, and tritanopia may not benefit at all). Causes reds to be darkened and greens to be lightened if the
509     * other of the pair is not present in similar quantities (which is the case for yellows and blues).
510     */
511    public static class DistinctRedGreenFilter implements IFilter<Color> {
512        /**
513         * Constructs a DistinctRedGreenFilter. This class is a simple wrapper around a function that doesn't need
514         * member variables, so there should be little overhead with this filter.
515         */
516        public DistinctRedGreenFilter() {
517        }
518
519        @Override
520        public Color alter(float r, float g, float b, float a) {
521            float diff = g - r;
522            if(diff > 0.4f)
523                return new Color(Math.min(1f, r * (0.8f + diff * 0.5f)), Math.min(1f, g * (0.9f + diff * 0.5f)),
524                        Math.min(1f, b * (0.8f + diff * 0.5f)), a);
525            else if(diff < -0.3f)
526                return new Color(r * (0.6f - diff), g * (0.7f - diff),
527                        b * (0.7f - diff), a);
528            else
529                return new Color(r, g, b, a);
530        }

010/**
011 * Pre-made FloatFilter classes that you can use to filter colors without producing extra Color objects.
012 * <br>
013 * Created by Tommy Ettinger on 7/22/2018.
014 */
015public final class FloatFilters {
016    private FloatFilters() {
017        // don't build me!
018    }

020    /**
021     * Wraps the functionality of {@link SColor#toEditedFloat(float, float, float, float, float)} so it can be called as
022     * a FloatFilter, adding values to hue, saturation, and value (clamping saturation and value and wrapping hue).
023     */
024    public static class HSVFilter extends FloatFilter {
025        public float hueAddend, saturationAddend, valueAddend;
026
027        public HSVFilter(float saturation, float value) {
028            this(0f, saturation, value);
029        }
030
031        public HSVFilter(float hueAdd, float saturationAdd, float valueAdd) {
032            hueAddend = hueAdd;
033            saturationAddend = saturationAdd;
034            valueAddend = valueAdd;
035        }
036
037        /**
038         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
039         *
040         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
041         * @return a packed float color, as produced by {@link Color#toFloatBits()}
042         */
043        @Override
044        public float alter(float color) {
045            return SColor.toEditedFloat(color, hueAddend, saturationAddend, valueAddend, 0f);
046        }
047    }

049    /**
050     * Changes all colors this alters to have the same hue as, and a closer saturation to, a target color given in the
051     * constructor, as well as optionally multiplying saturation of the result and/or adding lightness/value. A good
052     * example usage of this is to make a sepia-tone effect with
053     * {@code new FloatFilters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
054     */
055    public static class ColorizeFilter extends FloatFilter {
056        public float targetCb, targetCr, lumaAddend;
057
058        public ColorizeFilter(float color) {
059            this(color, 1f, 0f);
060        }
061
062        public ColorizeFilter(Color color) {
063            this(color.toFloatBits(), 1f, 0f);
064        }
065
066        public ColorizeFilter(Color color, float chromaMul, float lumaAdd) {
067            this(color.toFloatBits(), chromaMul, lumaAdd);
068        }
069
070        public ColorizeFilter(float color, float chromaMul, float lumaAdd) {
071            targetCb = SColor.chromaBOfFloat(color) * chromaMul;
072            targetCr = SColor.chromaROfFloat(color) * chromaMul;
073            lumaAddend = lumaAdd;
074        }
075
076        /**
077         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
078         *
079         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
080         * @return a packed float color, as produced by {@link Color#toFloatBits()}
081         */
082        @Override
083        public float alter(float color) {
084            final int bits = NumberTools.floatToIntBits(color);
085            return SColor.floatGetYCbCr((bits & 0x000000ff) * (0x1.010102p-8f  * 0.299f) +
086                    (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
087                    (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f),
088                    targetCb, targetCr,
089                    ((bits & 0xfe000000) >>> 24) * 0x1.020408p-8f);
090        }
091    }

093    /**
094     * A FloatFilter that alters primarily-red and primarily-green colors so they can be more easily be distinguished by
095     * people with at least some forms of red-green color-blindness (deuteranopia should be handled well, protanopia
096     * very well, and tritanopia may not benefit at all). Causes reds to be darkened and greens to be lightened if the
097     * other of the pair is not present in similar quantities (which is the case for yellows and blues).
098     */
099    public static class DistinctRedGreenFilter extends FloatFilter {
100        /**
101         * Constructs a DistinctRedGreenFilter. This class is a simple wrapper around a function that doesn't need
102         * member variables, so there should be little overhead with this filter.
103         */
104        public DistinctRedGreenFilter() {
105        }
106
107        /**
108         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
109         *
110         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
111         * @return a packed float color, as produced by {@link Color#toFloatBits()}
112         */
113        @Override
114        public float alter(float color) {
115            final int bits = NumberTools.floatToIntBits(color),
116                    r = bits & 0xFF, g = bits >>> 8 & 0xFF, b = bits >>> 16 & 0xFF, a = bits >>> 24,
117                    diff = g - r;
118            if (diff > 101)
119                return floatGet(Math.min(1f, 0x1.010102p-16f * r * (203 + (diff >> 1))),
120                        Math.min(1f, 0x1.010102p-16f * g * (228 + (diff >> 1))),
121                        Math.min(1f, 0x1.010102p-16f * b * (203 + (diff >> 1))),
122                        0x1.010102p-8f * a);
123            else if (diff < -75)
124                return floatGet(Math.min(1f, 0x1.010102p-16f * r * (152 - diff)),
125                        Math.min(1f, 0x1.010102p-16f * g * (177 - diff)),
126                        Math.min(1f, 0x1.010102p-16f * b * (177 - diff)),
127                        0x1.010102p-8f * a);
128            else
129                return color;
130        }
131    }

133    /**
134     * A FloatFilter that makes no changes to the colors given to it; useful as a default for when no filter is wanted.
135     */
136    public static class IdentityFilter extends FloatFilter {
137        /**
138         * Takes a packed float color and returns it un-edited.
139         *
140         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
141         * @return a packed float color, as produced by {@link Color#toFloatBits()}
142         */
143        @Override
144        public float alter(float color) {
145            return color;
146        }

149    /**
150     * A FloatFilter that makes all colors given to it grayscale, using only their luma as calculated by
151     * {@link SColor#lumaOfFloat(float)} as the lightness (it does also preserve alpha transparency).
152     */
153    public static class GrayscaleFilter extends FloatFilter {
154        /**
155         * Takes a packed float color and produces a grayscale packed float color that this FloatFilter edited.
156         * Uses the luma calculation from {@link SColor#lumaOfFloat(float)} instead of the value calculation from
157         * {@link SColor#valueOfFloat(float)}; luma tends to be more visually-accurate on modern monitors.
158         *
159         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
160         * @return a packed float color, as produced by {@link Color#toFloatBits()}
161         */
162        @Override
163        public float alter(float color) {
164            final int bits = NumberTools.floatToIntBits(color);
165            color = (bits & 0x000000ff) * (0x1.010102p-8f * 0.299f) +
166                    (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
167                    (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f);
168            return floatGet(color, color, color, (bits >>> 24 & 0xFE) * 0.003937008f);
169        }
170    }

172    /**
173     * Like {@link HSVFilter}, but edits its input colors in YCbCr color space, and multiplies rather than adds.
174     * Y is luma, and affects how bright the color is (luma 1 is white, luma 0 is black). Cb is Chroma(blue) amd Cr is
175     * Chroma(red), two inter-related channels that determine the hue and vividness of a specific color. When Cb and Cr
176     * are both 0, the color is grayscale. When Cb is 0.5 and Cr is -0.5, the color is blue unless Y is very high or
177     * low. When Cb is -0.5 and Cr is 0.5, the color is red with the same caveats re: Y. When Cb and Cr are both -0.5,
178     * the color is green (same caveats), and when both are 0.5, the color is purple. When Y is 0.5, Cb and Cr form a
179     * graph like this:
180     * <br>
181     * <img src="https://upload.wikimedia.org/wikipedia/commons/thumb/9/91/YCbCr-CbCr_Y50.png/240px-YCbCr-CbCr_Y50.png" />
182     * <br>
183     * Valid values for Cb and Cr are from -0.5 to 0.5 at the widest part of the range (it shrinks as Y approaches 0 or
184     * 1), but there aren't really invalid values here because this filter will clamp results with higher or lower
185     * channel values than a color can have. Each of yMul, cbMul, and crMul can have any float value, but yMul should be
186     * positive (unless you want this to only produce solid black). Similarly, cbMul and crMul will not produce
187     * meaningful results if they are very large (either positive or negative); it's recommended to use values between
188     * 0.0 and 1.0 for both if you want to desaturate colors or values somewhat greater than 1.0 to oversaturate them.
189     */
190    public static class YCbCrFilter extends FloatFilter {
191        public float yMul, cbMul, crMul;
192
193        public YCbCrFilter(float yMul) {
194            this(1f, 1f, yMul);
195        }
196
197        public YCbCrFilter(float yMul, float cbMul, float crMul) {
198            this.yMul = yMul;
199            this.cbMul = cbMul;
200            this.crMul = crMul;
201        }
202
203        /**
204         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
205         *
206         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
207         * @return a packed float color, as produced by {@link Color#toFloatBits()}
208         */
209        @Override
210        public float alter(float color) {
211            final int bits = NumberTools.floatToIntBits(color);
212            final float opacity = (bits >>> 24 & 0xFE) * 0.003937008f;
213            float luma = yMul * (
214                    (bits & 0x000000ff) * (0x1.010102p-8f * 0.299f) +
215                            (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
216                            (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f));
217            if (luma <= 0.0039f) {
218                return floatGet(0f, 0f, 0f, opacity);
219            } else if (luma >= 0.9961f) {
220                return floatGet(1f, 1f, 1f, opacity);
221            }
222            final float chromaB = cbMul * (
223                    (bits & 0x000000ff) * (0x1.010102p-8f * -0.168736f) +
224                            (bits & 0x0000ff00) * (0x1.010102p-16f * -0.331264f) +
225                            (bits & 0x00ff0000) * (0x1.010102p-24f * 0.5f));
226            final float chromaR = crMul * (
227                    (bits & 0x000000ff) * (0x1.010102p-8f * 0.5f) +
228                            (bits & 0x0000ff00) * (0x1.010102p-16f * -0.418688f) +
229                            (bits & 0x00ff0000) * (0x1.010102p-24f * -0.081312f));
230
231            if (chromaR >= -0.0039f && chromaR <= 0.0039f && chromaB >= -0.0039f && chromaB <= 0.0039f) {
232                return floatGet(luma, luma, luma, opacity);
233            }
234            return floatGet(MathUtils.clamp(luma + chromaR * 1.402f, 0f, 1f),
235                    MathUtils.clamp(luma - chromaB * 0.344136f - chromaR * 0.714136f, 0f, 1f),
236                    MathUtils.clamp(luma + chromaB * 1.772f, 0f, 1f),
237                    opacity);
238        }
239    }

241    /**
242     * A FloatFilter that chains together one or more FloatFilters one after the next, passing the float output of one
243     * as input to the next until the chain has all been called.
244     */
245    public static class ChainFilter extends FloatFilter {
246        public FloatFilter[] filters;
247
248        /**
249         * Takes a vararg or array of FloatFilter objects and produces a ChainFilter that will call all of them in order
250         * on any color given to this to alter.
251         * @param filters an array or vararg of FloatFilter objects; none can be null
252         */
253        public ChainFilter(FloatFilter... filters)
254        {
255            if(filters == null || filters.length == 0)
256                this.filters = new FloatFilter[]{new IdentityFilter()};
257            this.filters = filters;
258        }
259
260        /**
261         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
262         *
263         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
264         * @return a packed float color, as produced by {@link Color#toFloatBits()}
265         */
266        @Override
267        public float alter(float color) {
268            for (int i = 0; i < filters.length; i++) {
269                color = filters[i].alter(color);
270            }
271            return color;
272        }
273    }

275    /**
276     * A FloatFilter that linearly interpolates (lerps) any color it is given toward a specified color by a specified
277     * amount.
278     */
279    public static class LerpFilter extends FloatFilter {
280        public float target, amount;
281
282        /**
283         * Builds a LerpFilter with a Color (which will be converted to a packed float color) and an amount as a float.
284         * The amount is how much the target color will affect input colors, from 0f to 1f.
285         * @param target a libGDX color; must not be null
286         * @param amount a float that determines how much target will affect an input color; will be clamped between 0f and 1f
287         */
288        public LerpFilter(Color target, float amount) {
289            this.target = target.toFloatBits();
290            this.amount = MathUtils.clamp(amount, 0f, 1f);
291        }
292
293        /**
294         * Builds a LerpFilter with a packed float color and an amount as a float.
295         * The amount is how much the target color will affect input colors, from 0f to 1f.
296         * @param target a packed float color; must not be null
297         * @param amount a float that determines how much target will affect an input color; will be clamped between 0f and 1f
298         */
299
300        public LerpFilter(float target, float amount)
301        {
302            this.target = target;
303            this.amount = MathUtils.clamp(amount, 0f, 1f);
304        }
305
306        /**
307         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
308         *
309         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
310         * @return a packed float color, as produced by {@link Color#toFloatBits()}
311         */
312        @Override
313        public float alter(float color) {
314            return lerpFloatColors(color, target, amount);
315        }
316    }

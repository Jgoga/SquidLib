014/**
015 * Various special effects that can be applied to a {@link SquidPanel} or {@link SparseLayers} as an
016 * {@link com.badlogic.gdx.scenes.scene2d.Action}. The PanelEffect class is abstract and has implementations as static
017 * inner classes, such as {@link ExplosionEffect}. Each PanelEffect specifically affects one {@link IPackedColorPanel}
018 * (an interface that both SquidPanel and SparseLayers implement), which can be a layer in a SquidLayers object or a
019 * SquidPanel/SparseLayers on its own. By adding the PanelEffect to any actor using
020 * {@link com.badlogic.gdx.scenes.scene2d.Actor#addAction(Action)} when the Actor has its act() method called after the
021 * normal map parts of the panel have been placed, the PanelEffect will advance and change what chars/colors are in
022 * the panel as specified in its implementation of the {@link #update(float)} method. Typically the Actor you add this
023 * to is the SquidPanel or SparseLayers this affects, but it could also be a SquidLayers. Most PanelEffect
024 * implementations should allow most configuration to be set in their constructors.
025 * <br>
026 * Created by Tommy Ettinger on 5/24/2017.
027 */
028@Beta
029public abstract class PanelEffect extends TemporalAction{
030    public IPackedColorPanel target;
031    public GreasedRegion validCells;
032
033    protected PanelEffect(IPackedColorPanel targeting)
034    {
035        target = targeting;
036        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
037    }
038    protected PanelEffect(IPackedColorPanel targeting, float duration)
039    {
040        target = targeting;
041        setDuration(duration);
042        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
043    }
044    protected PanelEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid)
045    {
046        target = targeting;
047        setDuration(duration);
048        validCells = valid;
049    }

053        /**
054         * Normally you should set this in the constructor, and not change it later.

056        public Coord center;
057        /**
058         * Normally you should set this in the constructor, and not change it later.
059         */
060        public int radius = 2;
061        /**
062         * The default explosion colors are normal for (non-chemical, non-electrical) fire and smoke, going from orange
063         * at the start to yellow, very light yellow, and then back to a different orange before going to smoke and
064         * fading out to translucent and then transparent by the end.
065         * <br>
066         * If you want to change the colors the explosion uses, you can either pass a List of Color (or SColor or other
067         * subclasses) to the constructor or change this array directly. The float items assigned to this should be the
068         * result of calling {@link Color#toFloatBits()} or possibly the result of mixing multiple existing floats with
069         * {@link SColor#lerpFloatColors(float, float, float)}; other floats that can be directly used by libGDX, that
070         * is, packed as ABGR floats (usually the docs will call this a packed float), can also be used.
071         */
072        public float[] colors = {
073                SColor.floatGet(0xFF4F00FF), // SColor.INTERNATIONAL_ORANGE
074                SColor.floatGet(0xFFB94EFF), // SColor.FLORAL_LEAF
075                SColor.floatGet(0xFDE910FF), // SColor.LEMON
076                SColor.floatGet(0xFFFACDFF), // SColor.LEMON_CHIFFON
077                SColor.floatGet(0xFF6600EE), // SColor.SAFETY_ORANGE
078                SColor.floatGet(0x595652DD), // SColor.DB_SOOT
079                SColor.floatGet(0x59565299)  // SColor.DB_SOOT
080        };
081        /**
082         * The internal representation of how affected each cell is by the explosion, based on proximity to center.
083         */
084        public double[][] lightMap;
085        /**
086         * The raw list of Coords that might be affected by the explosion; may include some cells that aren't going to
087         * show as exploding (it usually has some false positives), but shouldn't exclude any cells that should show as
088         * such (no false negatives). You can edit this if you need to, but it isn't recommended.
089         */
090        public List<Coord> affected;

092         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be

095         * @param center the center of the explosion
096         * @param radius the radius of the explosion, in cells

103        /**
104         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be
105         * the full expanse of the IPackedColorPanel.
106         * @param targeting the IPackedColorPanel to affect
107         * @param duration the duration of this PanelEffect in seconds, as a float
108         * @param center the center of the explosion
109         * @param radius the radius of the explosion, in cells
110         */
111        public ExplosionEffect(IPackedColorPanel targeting, float duration, Coord center, int radius)
112        {
113            super(targeting, duration);
114            this.center = center;
115            this.radius = radius;
116            double[][] resMap = new double[validCells.width][validCells.height];
117            lightMap = new double[validCells.width][validCells.height];
118            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius);
119            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
120        }
121        /**
122         * Constructs an ExplosionEffect with explicit settings for most fields.
123         * @param targeting the IPackedColorPanel to affect
124         * @param duration the duration of this PanelEffect in seconds, as a float
125         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
126         * @param center the center of the explosion
127         * @param radius the radius of the explosion, in cells
128         */
129        public ExplosionEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid, Coord center, int radius)
130        {
131            super(targeting, duration, valid);
132            this.center = center;
133            this.radius = radius;
134            double[][] resMap = ArrayTools.fill(1.0, validCells.width, validCells.height);
135            validCells.writeDoublesInto(resMap, 0.0);
136            lightMap = new double[validCells.width][validCells.height];
137            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius + 0.5);
138            validCells.not().writeDoublesInto(lightMap, 0.0);
139            validCells.not();
140            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
141        }

231        /**
232         * Constructs an ExplosionEffect with explicit settings for most fields but also an alternate group of Color
233         * objects that it will use to color the explosion instead of using fiery/smoke colors; this constructor allows
234         * the case where an explosion is directed in a cone or sector shape. It will center the sector on {@code angle}
235         * (in degrees) and will cover an amount of the circular area (in degrees) equal to {@code span}.
236         * @param targeting the IPackedColorPanel to affect
237         * @param duration the duration of this PanelEffect in seconds, as a float
238         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
239         * @param center the center of the explosion
240         * @param radius the radius of the explosion, in cells
241         * @param angle the angle, in degrees, that will be the center of the sector-shaped effect
242         * @param span the span, in degrees, of the full arc at the end of the sector-shaped effect
243         * @param coloring an array of colors as packed floats that will replace the default fire/smoke colors here
244         */
245        public ExplosionEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid, Coord center, int radius, double angle, double span, float[] coloring)
246        {
247            this(targeting, duration, valid, center, radius, angle, span);
248            if(coloring == null) return;
249            if(colors.length != coloring.length)
250                colors = new float[coloring.length];
251            System.arraycopy(coloring, 0, colors, 0, coloring.length);
252        }
253        /**
254         * Called each frame.
255         *
256         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
257         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
258         */
259        @Override
260        protected void update(float percent) {
261            int len = affected.size();
262            Coord c;
263            float f, color;
264            int idx, seed = System.identityHashCode(this);
265            for (int i = 0; i < len; i++) {
266                c = affected.get(i);
267                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
268                    continue;
269                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
270                        * 0.17f + percent * 1.2f;
271                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
272                    continue;
273                idx = (int) (f * colors.length);
274                if(idx >= colors.length - 1)
275                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
276                else
277                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
278                target.put(c.x, c.y, color);
279            }
280        }

519         * Constructs an ExplosionEffect with explicit settings for most fields but also an alternate group of Color
520         * objects that it will use to color the explosion instead of using purple spark colors; this constructor allows
521         * the case where an explosion is directed in a cone or sector shape. It will center the sector on {@code angle}

523         * @param targeting the IPackedColorPanel to affect
524         * @param duration the duration of this PanelEffect in seconds, as a float
525         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
526         * @param center the center of the explosion
527         * @param radius the radius of the explosion, in cells
528         * @param angle the angle, in degrees, that will be the center of the sector-shaped effect
529         * @param span the span, in degrees, of the full arc at the end of the sector-shaped effect
530         * @param coloring an array of colors as packed floats that will replace the default purple spark colors here
531         */
532        public GibberishEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid, Coord center, int radius, double angle, double span, float[] coloring, char[] choices)
533        {
534            super(targeting, duration, valid, center, radius, angle, span, coloring);
535            this.choices = choices;
536        }
537        /**
538         * Called each frame.
539         *
540         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
541         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
542         */
543        @Override
544        protected void update(float percent) {
545            int len = affected.size();
546            Coord c;
547            float f, color;
548            int idx, seed = System.identityHashCode(this), clen = choices.length;
549            final long tick = LightRNG.determine((System.currentTimeMillis() >>> 7) * seed);
550            for (int i = 0; i < len; i++) {
551                c = affected.get(i);
552                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
553                    continue;
554                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
555                        * 0.17f + percent * 1.2f;
556                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
557                    continue;
558                idx = (int) (f * colors.length);
559                if(idx >= colors.length - 1)
560                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
561                else
562                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
563                target.put(c.x, c.y, choices[(int) ((clen * (LightRNG.determine(tick + i) & 0x7FFFFFFFL)) >> 31)], color);
564            }
565        }

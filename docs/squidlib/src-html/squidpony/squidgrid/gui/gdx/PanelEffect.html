016/**
017 * Various special effects that can be applied to a {@link SquidPanel} or {@link SparseLayers} as an
018 * {@link com.badlogic.gdx.scenes.scene2d.Action}. The PanelEffect class is abstract and has implementations as static
019 * inner classes, such as {@link ExplosionEffect}. Each PanelEffect specifically affects one {@link IPackedColorPanel}
020 * (an interface that both SquidPanel and SparseLayers implement), which can be a layer in a SquidLayers object or a
021 * SquidPanel/SparseLayers on its own. By adding the PanelEffect to any actor using
022 * {@link com.badlogic.gdx.scenes.scene2d.Actor#addAction(Action)} when the Actor has its act() method called after the
023 * normal map parts of the panel have been placed, the PanelEffect will advance and change what chars/colors are in
024 * the panel as specified in its implementation of the {@link #update(float)} method. Typically the Actor you add this
025 * to is the SquidPanel or SparseLayers this affects, but it could also be a SquidLayers. Most PanelEffect
026 * implementations should allow most configuration to be set in their constructors.
027 * <br>
028 * Created by Tommy Ettinger on 5/24/2017.
029 */
030@Beta
031public abstract class PanelEffect extends TemporalAction{
032    public IPackedColorPanel target;
033    public GreasedRegion validCells;
034
035    protected PanelEffect(IPackedColorPanel targeting)
036    {
037        target = targeting;
038        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
039    }
040    protected PanelEffect(IPackedColorPanel targeting, float duration)
041    {
042        target = targeting;
043        setDuration(duration);
044        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
045    }
046    protected PanelEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid)
047    {
048        target = targeting;
049        setDuration(duration);
050        validCells = valid;
051    }

071         * {@link SColor#lerpFloatColors(float, float, float)}; other floats that can be directly used by libGDX, that

126         * @param duration the duration of this PanelEffect in seconds, as a float

134            this.center = center;
135            this.radius = radius;

142            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);

255        /**
256         * Called each frame.
257         *
258         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
259         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
260         */
261        @Override
262        protected void update(float percent) {
263            int len = affected.size();
264            Coord c;
265            float f, color, light;
266            int idx, seed = System.identityHashCode(this);
267            for (int i = 0; i < len; i++) {
268                c = affected.get(i);
269                if((light = (float) lightMap[c.x][c.y]) <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
270                    continue;
271                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
272                        * 0.17f + percent * 1.2f;
273                if(f < 0f || 0.5 * light + f < 0.4)
274                    continue;
275                idx = (int) (f * colors.length);
276                if(idx >= colors.length - 1)
277                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
278                else
279                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
280                target.blend(c.x, c.y, color, SColor.alphaOfFloatF(color) * light * 0.25f + 0.75f);
281            }
282        }

285         * Meant for electrical bursts, this will affect character foregrounds in a GibberishEffect. This should look

295            colors[4] = SColor.floatGet(0xBF00FFDD); // SColor.ELECTRIC_PURPLE

297            colors[6] = SColor.floatGet(0x4B008277); // SColor.INDIGO

524         * Constructs an ExplosionEffect with explicit settings for most fields but also an alternate group of Color
525         * objects that it will use to color the explosion instead of using purple spark colors; this constructor allows
526         * the case where an explosion is directed in a cone or sector shape. It will center the sector on {@code angle}
527         * (in degrees) and will cover an amount of the circular area (in degrees) equal to {@code span}.
528         * @param targeting the IPackedColorPanel to affect
529         * @param duration the duration of this PanelEffect in seconds, as a float
530         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
531         * @param center the center of the explosion
532         * @param radius the radius of the explosion, in cells
533         * @param angle the angle, in degrees, that will be the center of the sector-shaped effect
534         * @param span the span, in degrees, of the full arc at the end of the sector-shaped effect
535         * @param coloring an array of colors as packed floats that will replace the default purple spark colors here
536         */
537        public GibberishEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid, Coord center, int radius, double angle, double span, float[] coloring, char[] choices)
538        {
539            super(targeting, duration, valid, center, radius, angle, span, coloring);
540            this.choices = choices;
541        }
542        /**
543         * Called each frame.
544         *
545         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
546         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
547         */
548        @Override
549        protected void update(float percent) {
550            int len = affected.size();
551            Coord c;
552            float f, color;
553            int idx, seed = System.identityHashCode(this), clen = choices.length;
554            final long tick = LightRNG.determine((System.currentTimeMillis() >>> 7) * seed);
555            for (int i = 0; i < len; i++) {
556                c = affected.get(i);
557                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
558                    continue;
559                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
560                        * 0.17f + percent * 1.2f;
561                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
562                    continue;
563                idx = (int) (f * colors.length);
564                if(idx >= colors.length - 1)
565                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
566                else
567                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
568                target.put(c.x, c.y, choices[(int) ((clen * (LightRNG.determine(tick + i) & 0x7FFFFFFFL)) >> 31)], color);
569            }
570        }

754         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
755         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
756         */
757        @Override
758        protected void update(float percent) {
759            int len = affected.length, index = (int)((len - 1) * percent);

769                }

774    public static Interpolation fastInSlowMidFastOut = new Interpolation() {
775        private final float value = 2, power = 3;
776        @Override
777        public float apply(float a) {
778            if (a <= 0.5f) return (1 - ((float)Math.pow(value, -power * (a * 2)) - 0.125f) * 1.1428572f) * 0.5f;
779            return (1 + (float) Math.pow(value, power * (a * 2 - 2)) - 0.25f) * 0.5714286f;
780        }
781    };

783    /**
784     * Convenience method to make a ProjectileEffect take an "arc-like" path toward the target, where it is fast at the
785     * beginning and end of its motion and is reaching the height of its arc at the center, before triggering another
786     * Action when the projectile stops (often this might be an {@link PanelEffect.ExplosionEffect}, but could be any
787     * scene2d Action).
788     * @param projectile a {@link PanelEffect.ProjectileEffect} to run as the first step
789     * @param result an {@link Action} to run after the ProjectileEffect completes
790     * @return an Action that can be added to a scene2d Actor, such as a SquidLayers or SparseLayers
791     */
792    public static Action makeGrenadeEffect(ProjectileEffect projectile, Action result)
793    {
794        projectile.useArcPathInterpolation();
795        return Actions.sequence(projectile, result);

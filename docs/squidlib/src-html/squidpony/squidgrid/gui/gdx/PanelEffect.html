014/**
015 * Various special effects that can be applied to a SquidPanel as an {@link com.badlogic.gdx.scenes.scene2d.Action}.
016 * The PanelEffect class is abstract and has implementations as static inner classes, such as {@link ExplosionEffect}.
017 * Each PanelEffect specifically affects one SquidPanel, which can be a layer in a SquidLayers object or a SquidPanel on
018 * its own. By adding the PanelEffect to any actor using {@link com.badlogic.gdx.scenes.scene2d.Actor#addAction(Action)}
019 * when the Actor has its act() method called after the normal map parts of the SquidPanel have been placed, the
020 * PanelEffect will advance and change what chars/colors are in the SquidPanel as specified in its implementation of the
021 * {@link #update(float)} method. Typically the Actor you add this to is the SquidPanel this affects, but it could also
022 * be a SquidLayers. Most PanelEffect implementations should allow most configuration to be set in their constructors.
023 * <br>
024 * Created by Tommy Ettinger on 5/24/2017.
025 */
026@Beta
027public abstract class PanelEffect extends TemporalAction{
028    public SquidPanel target;
029    public GreasedRegion validCells;
030
031    protected PanelEffect(SquidPanel targeting)
032    {
033        target = targeting;
034        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
035    }
036    protected PanelEffect(SquidPanel targeting, float duration)
037    {
038        target = targeting;
039        setDuration(duration);
040        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
041    }
042    protected PanelEffect(SquidPanel targeting, float duration, GreasedRegion valid)
043    {
044        target = targeting;
045        setDuration(duration);
046        validCells = valid;
047    }
048    @Beta
049    public static class ExplosionEffect extends PanelEffect
050    {
051        /**
052         * Normally you should set this in the constructor, and not change it later.
053         */
054        public Coord center;
055        /**
056         * Normally you should set this in the constructor, and not change it later.
057         */
058        public int radius = 2;
059        /**
060         * The default explosion colors are normal for (non-chemical, non-electrical) fire and smoke, going from orange
061         * at the start to yellow, very light yellow, and then back to a different orange before going to smoke and
062         * fading out to translucent and then transparent by the end.
063         * <br>
064         * If you want to change the colors the explosion uses, you can either pass a List of Color (or SColor or other
065         * subclasses) to the constructor or change this array directly. The float items assigned to this should be the
066         * result of calling {@link Color#toFloatBits()} or possibly the result of mixing multiple existing floats with
067         * {@link SColor#lerpFloatColors(float, float, float)}; other floats that can be directly used by libGDX, that
068         * is, packed as ABGR floats (usually the docs will mention this), can also be used.
069         */
070        public float[] colors = {
071                SColor.INTERNATIONAL_ORANGE.toFloatBits(),
072                SColor.FLORAL_LEAF.toFloatBits(),
073                SColor.LEMON.toFloatBits(),
074                SColor.LEMON_CHIFFON.toFloatBits(),
075                SColor.floatGet(0xFF6600EE),  // SColor.SAFETY_ORANGE
076                SColor.floatGet(0x595652DD),  // SColor.DB_SOOT
077                SColor.floatGet(0x59565299)}; // SColor.DB_SOOT
078        /**
079         * Used internally to determine how the explosion should spread; derived from {@link #validCells}.
080         */
081        public double[][] resMap,
082        /**
083         * The internal representation of how affected each cell is by the explosion, based on proximity to center.
084         */
085        lightMap;
086        /**
087         * The raw list of Coords that might be affected by the explosion; may include some cells that aren't going to
088         * show as exploding (it usually has some false positives), but shouldn't exclude any cells that should show as
089         * such (no false negatives). You can edit this if you need to, but it isn't recommended.
090         */
091        public List<Coord> affected;
092        /**
093         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be
094         * the full expanse of the SquidPanel. The duration will be 1 second.
095         * @param targeting the SquidPanel to affect
096         * @param center the center of the explosion
097         * @param radius the radius of the explosion, in cells
098         */
099
100        public ExplosionEffect(SquidPanel targeting, Coord center, int radius)
101        {
102            this(targeting, 1f, center, radius);
103        }
104        /**
105         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be
106         * the full expanse of the SquidPanel.
107         * @param targeting the SquidPanel to affect
108         * @param duration the duration of this PanelEffect in seconds, as a float
109         * @param center the center of the explosion
110         * @param radius the radius of the explosion, in cells
111         */
112        public ExplosionEffect(SquidPanel targeting, float duration, Coord center, int radius)
113        {
114            super(targeting, duration);
115            this.center = center;
116            this.radius = radius;
117            resMap = new double[validCells.width][validCells.height];
118            lightMap = new double[validCells.width][validCells.height];
119            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius);
120            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
121        }
122        /**
123         * Constructs an ExplosionEffect with explicit settings for most fields.
124         * @param targeting the SquidPanel to affect
125         * @param duration the duration of this PanelEffect in seconds, as a float
126         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
127         * @param center the center of the explosion
128         * @param radius the radius of the explosion, in cells
129         */
130        public ExplosionEffect(SquidPanel targeting, float duration, GreasedRegion valid, Coord center, int radius)
131        {
132            super(targeting, duration, valid);
133            this.center = center;
134            this.radius = radius;
135            resMap = ArrayTools.fill(1.0, validCells.width, validCells.height);
136            validCells.writeDoublesInto(resMap, 0.0);
137            lightMap = new double[validCells.width][validCells.height];
138            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius + 0.5);
139            validCells.not().writeDoublesInto(lightMap, 0.0);
140            validCells.not();
141            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
142        }
143
144        /**
145         * Constructs an ExplosionEffect with explicit settings for most fields but also an alternate group of Color
146         * objects that it will use to color the explosion instead of using fiery/smoke colors.
147         * @param targeting the SquidPanel to affect
148         * @param duration the duration of this PanelEffect in seconds, as a float
149         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
150         * @param center the center of the explosion
151         * @param radius the radius of the explosion, in cells
152         * @param coloring a List of Color or subclasses thereof that will replace the default fire/smoke colors here
153         */
154        public ExplosionEffect(SquidPanel targeting, float duration, GreasedRegion valid, Coord center, int radius, List<? extends Color> coloring)
155        {
156            this(targeting, duration, valid, center, radius);
157            if(colors.length != coloring.size())
158                colors = new float[coloring.size()];
159            for (int i = 0; i < colors.length; i++) {
160                colors[i] = coloring.get(i).toFloatBits();
161            }
162        }
163        /**
164         * Called each frame.
165         *
166         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
167         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
168         */
169        @Override
170        protected void update(float percent) {
171            int len = affected.size();
172            Coord c;
173            float f, color;
174            int idx, seed = System.identityHashCode(this);
175            for (int i = 0; i < len; i++) {
176                c = affected.get(i);
177                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
178                    continue;
179                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
180                        * 0.17f + percent * 1.2f;
181                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
182                    continue;
183                idx = (int) (f * colors.length);
184                if(idx >= colors.length - 1)
185                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
186                else
187                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
188                target.put(c.x, c.y, color);
189            }
190        }
191    }

310            this.choices = choices;

312        /**
313         * Called each frame.
314         *
315         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
316         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
317         */
318        @Override
319        protected void update(float percent) {
320            int len = affected.size();
321            Coord c;
322            float f, color;
323            int idx, seed = System.identityHashCode(this), clen = choices.length;
324            final long tick = LightRNG.determine((System.currentTimeMillis() >>> 7) * seed);
325            for (int i = 0; i < len; i++) {
326                c = affected.get(i);
327                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
328                    continue;
329                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
330                        * 0.17f + percent * 1.2f;
331                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
332                    continue;
333                idx = (int) (f * colors.length);
334                if(idx >= colors.length - 1)
335                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
336                else
337                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
338                target.put(c.x, c.y, choices[(int) ((clen * (LightRNG.determine(tick + i) & 0x7FFFFFFFL)) >> 31)], color);
339            }
340        }

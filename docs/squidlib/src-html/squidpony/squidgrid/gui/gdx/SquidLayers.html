306    /**
307     * Create a new SquidLayers widget with the given TextCellFactory, the given number of cells for gridWidth
308     * and gridHeight, the size in pixels for each cell given by cellWidth and cellHeight, the given
309     * SquidColorCenters for background and foreground, and the given 2D char array for an area map that may be
310     * sized differently than gridWidth by gridHeight (it is usually larger than gridWidth by gridHeight, which
311     * allows camera scrolling across the map). This requires some special work with the Camera and Viewports to
312     * get working correctly; in the squidlib module's examples,
313     * <a href="https://github.com/SquidPony/SquidLib/blob/master/squidlib/src/test/java/squidpony/gdx/examples/EverythingDemo.java">EverythingDemo</a>
314     * may be a good place to see how this can be done. You can pass null for actualMap, which will simply create
315     * a char array to use internally that is exactly gridWidth by gridHeight, in cells. Consider using predefined
316     * TextCellFactory objects from {@link DefaultResources}, which will be configured by this constructor for
317     * sizing when passed here.
318     *
319     * @param gridWidth     in grid cells
320     * @param gridHeight    in grid cells
321     * @param cellWidth     in pixels
322     * @param cellHeight    in pixels
323     * @param tcf           A TextCellFactory that will be (re-)initialized here with the given cellHeight and cellWidth.
324     * @param bgColorCenter a SquidColorCenter (possibly with a filter) to use for the background
325     * @param fgColorCenter a SquidColorCenter (possibly with a filter) to use for the foreground
326     * @param actualMap     will often be a different size than gridWidth by gridHeight, which enables camera scrolling
327     */
328    public SquidLayers(int gridWidth, int gridHeight, int cellWidth, int cellHeight, TextCellFactory tcf,
329                       SquidColorCenter bgColorCenter, SquidColorCenter fgColorCenter, char[][] actualMap) {
330
331        width = gridWidth;
332        height = gridHeight;
333
334        this.cellWidth = cellWidth;
335        this.cellHeight = cellHeight;
336
337        textFactory = tcf.width(cellWidth).height(cellHeight).initBySize();
338
339        if (actualMap == null || actualMap.length <= 0) {
340            backgroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, bgColorCenter);
341            foregroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, fgColorCenter);
342            lightnesses = ArrayTools.fill(256, width, height);
343        } else {
344            backgroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, bgColorCenter, 0, 0, ArrayTools.fill(' ', actualMap.length, actualMap[0].length));
345            foregroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, fgColorCenter, 0, 0, actualMap);
346            lightnesses = ArrayTools.fill(256, actualMap.length, actualMap[0].length);
347        }
348        animationDuration = foregroundPanel.DEFAULT_ANIMATION_DURATION;
349
350        extraPanels = new ArrayList<>();
351
352        addActorAt(0, backgroundPanel);
353        addActorAt(2, foregroundPanel);
354
355        setSize(backgroundPanel.getWidth(), backgroundPanel.getHeight());
356    }

358    /**
359     * Add an extra layer on top of the foreground layer. Use putInto methods to specify the layer when adding a char (0
360     * is background, 1 is unused, 2 is foreground, and the first call to this method creates layer 3).
361     *
362     * @return this for chaining
363     */
364    public SquidLayers addExtraLayer() {
365        SquidPanel sp;
366        if (width != foregroundPanel.getTotalWidth() || height != foregroundPanel.getTotalHeight())
367            sp = new SquidPanel(width, height, textFactory, foregroundPanel.getColorCenter(), 0, 0,
368                    ArrayTools.fill(' ', foregroundPanel.getTotalWidth(), foregroundPanel.getTotalHeight()));
369        else
370            sp = new SquidPanel(width, height, textFactory);
371        addActor(sp);
372        extraPanels.add(sp);
373        return this;
374    }

376    /**
377     * Sets the size of the text in the given layer  (but not the size of the cells) to the given width and height in
378     * pixels (which may be stretched by viewports later on, if your program uses them).
379     *
380     * @param layer the layer to affect; 0 is background, 1 is unused, 2 is foreground, 3 and higher are extra panels
381     * @param wide  the width of a glyph in pixels
382     * @param high  the height of a glyph in pixels
383     * @return this for chaining
384     */
385    public SquidLayers setTextSize(int layer, float wide, float high) {
386        SquidPanel p = backgroundPanel;
387        switch (layer) {
388            case 0:
389                break;
390            case 1:
391                break;
392            case 2:
393                p = foregroundPanel;
394                break;
395            default:
396                p = extraPanels.get(layer - 3);
397        }
398        p.setTextSize(wide, high);
399        return this;
400    }

402    /**
403     * Sets the size of the text in all layers (but not the size of the cells) to the given width and height in pixels
404     * (which may be stretched by viewports later on, if your program uses them).
405     *
406     * @param wide the width of a glyph in pixels
407     * @param high the height of a glyph in pixels
408     * @return this for chaining
409     */
410    public SquidLayers setTextSize(float wide, float high) {
411        textFactory.tweakHeight(high).tweakWidth(wide).initBySize();
412        setTextSize(0, wide, high);
413        setTextSize(2, wide, high);
414        for (int i = 0; i < extraPanels.size(); i++) {
415            setTextSize(i + 3, wide, high);
416        }
417        return this;
418    }

478    /**
479     * Place a char c into the foreground, with a foreground and background libGDX Color and a lightness variation for
480     * the background (255 will make the background equal the background panel's
481     * {@link SquidPanel#getLightingColor()}, -255 will use the background as-is, and values in between will be
482     * linearly interpolated between those two extremes).
483     *
484     * @param x                   in grid cells.
485     * @param y                   in grid cells.
486     * @param c                   a character to be drawn in the foreground
487     * @param foreground          Color for the char being drawn
488     * @param background          Color for the background
489     * @param backgroundLightness int between -255 and 255 , lower numbers are changed less, higher changed closer to the lighting color
490     * @return this for chaining
491     */
492    public SquidLayers put(int x, int y, char c, Color foreground, Color background, int backgroundLightness) {
493        foregroundPanel.put(x, y, c, foreground);
494        backgroundPanel.put(x, y, background, (lightnesses[x][y] = 256 + clamp(backgroundLightness, -255, 255)) * 0.001953125f);
495        return this;
496    }

600     * Place a 2@ char array c into the foreground, with foreground colors specified by a 2D Color array, background
601     * colors with another 2D Color array, and lightness variations for the background in a 2D int array (255 will make
602     * the background equal the background panel's {@link SquidPanel#getLightingColor()}, -255 will use the background
603     * as-is, and values in between will be linearly interpolated between those two extremes).
604     *
605     * @param x                   in grid cells.
606     * @param y                   in grid cells.
607     * @param c                   char[][] to be drawn in the foreground starting from x, y
608     * @param foregrounds         Color[][] of colors for the char being drawn
609     * @param backgrounds         Color[][] of colors for the background
610     * @param backgroundLightness int[][] with elements between -255 and 255 , lower darker, higher lighter.
611     * @return this for chaining
612     */
613    public SquidLayers put(int x, int y, char[][] c, Color[][] foregrounds, Color[][] backgrounds, int[][] backgroundLightness) {
614
615        foregroundPanel.put(x, y, c, foregrounds);
616        for (int i = x; i < getTotalWidth() && i - x < backgroundLightness.length; i++) {
617            for (int j = y; j < getTotalHeight() && j - y < backgroundLightness[i].length; j++) {
618                lightnesses[i][j] = 256 + clamp(backgroundLightness[i - x][j - y], -255, 255);
619                backgroundPanel.put(i, j, backgrounds[i - x][j - y], lightnesses[i][j] * 0.001953125f);
620            }
621        }
622        return this;
623    }

735     * @param s the string to print inside the box

738    public SquidLayers putBoxedString(int x, int y, String s) {
739        if (y > 0 && y + 1 < getTotalHeight() && x > 0 && x + 1 < getTotalWidth()) {
740            for (int j = y - 1; j < y + 2 && j < getTotalHeight(); j++) {

746                            SColor.DARK_GRAY, 0f * 0.001953125f);

749        }

751

1165     * and {@link #getTotalHeight()}, though there are good reasons why the cell sizes or text sizes might not match.
1166     * This method is probably most useful when you want a system of {@link ImageSquidPanel} layers; as long as you have
1167     * the reference to an ImageSquidPanel outside this class, you can call its additional methods there and have it
1168     * affect {@code panel} here.
1169     *
1170     * @param panel a SquidPanel, ImageSquidPanel, or other subclass of SquidPanel that should have identical size
1171     * @param index the 0-based index into the extra panels list to try to assign panel to, or to append panel to
1172     */
1173
1174    public void setExtraPanel(SquidPanel panel, int index) {
1175        if (index < 0 || panel == null || extraPanels == null)
1176            return;
1177        if (index - 3 >= extraPanels.size()) {
1178            extraPanels.add(panel);
1179            addActor(panel);
1180        } else {
1181            removeActor(extraPanels.get(index - 3));
1182            extraPanels.set(index - 3, panel);
1183            addActorAt(index, panel);
1184        }
1185    }

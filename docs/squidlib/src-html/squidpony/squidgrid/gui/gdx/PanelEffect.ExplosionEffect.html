001package squidpony.squidgrid.gui.gdx;
002
003import com.badlogic.gdx.graphics.Color;
004import com.badlogic.gdx.scenes.scene2d.Action;
005import com.badlogic.gdx.scenes.scene2d.actions.TemporalAction;
006import squidpony.ArrayTools;
007import squidpony.annotation.Beta;
008import squidpony.squidgrid.FOV;
009import squidpony.squidgrid.Radius;
010import squidpony.squidmath.Coord;
011import squidpony.squidmath.GreasedRegion;
012import squidpony.squidmath.NumberTools;
013import squidpony.squidmath.SeededNoise;
014
015import java.util.List;
016
017/**
018 * Various special effects that can be applied to a SquidPanel as an {@link com.badlogic.gdx.scenes.scene2d.Action}.
019 * The PanelEffect class is abstract and has implementations as static inner classes, such as {@link ExplosionEffect}.
020 * Each PanelEffect specifically affects one SquidPanel, which can be a layer in a SquidLayers object or a SquidPanel on
021 * its own. By adding the PanelEffect to any actor using {@link com.badlogic.gdx.scenes.scene2d.Actor#addAction(Action)}
022 * when the Actor has its act() method called after the normal map parts of the SquidPanel have been placed, the
023 * PanelEffect will advance and change what chars/colors are in the SquidPanel as specified in its implementation of the
024 * {@link #update(float)} method. Typically the Actor you add this to is the SquidPanel this affects, but it could also
025 * be a SquidLayers. Most PanelEffect implementations should allow most configuration to be set in their constructors.
026 * <br>
027 * Created by Tommy Ettinger on 5/24/2017.
028 */
029@Beta
030public abstract class PanelEffect extends TemporalAction{
031    public SquidPanel target;
032    public GreasedRegion validCells;
033
034    protected PanelEffect(SquidPanel targeting)
035    {
036        target = targeting;
037        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
038    }
039    protected PanelEffect(SquidPanel targeting, float duration)
040    {
041        target = targeting;
042        setDuration(duration);
043        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
044    }
045    protected PanelEffect(SquidPanel targeting, float duration, GreasedRegion valid)
046    {
047        target = targeting;
048        setDuration(duration);
049        validCells = valid;
050    }
051    @Beta
052    public static class ExplosionEffect extends PanelEffect
053    {
054        /**
055         * Normally you should set this in the constructor, and not change it later.
056         */
057        public Coord center;
058        /**
059         * Normally you should set this in the constructor, and not change it later.
060         */
061        public int radius = 2;
062        /**
063         * The default explosion colors are normal for (non-chemical, non-electrical) fire and smoke, going from orange
064         * at the start to yellow, very light yellow, and then back to a different orange before going to smoke and
065         * fading out to translucent and then transparent by the end.
066         * <br>
067         * If you want to change the colors the explosion uses, you can either pass a List of Color (or SColor or other
068         * subclasses) to the constructor or change this array directly. The float items assigned to this should be the
069         * result of calling {@link Color#toFloatBits()} or possibly the result of mixing multiple existing floats with
070         * {@link SColor#lerpFloatColors(float, float, float)}; other floats that can be directly used by libGDX, that
071         * is, packed as ABGR floats (usually the docs will mention this), can also be used.
072         */
073        public float[] colors = {
074                SColor.INTERNATIONAL_ORANGE.toFloatBits(),
075                SColor.FLORAL_LEAF.toFloatBits(),
076                SColor.LEMON.toFloatBits(),
077                SColor.LEMON_CHIFFON.toFloatBits(),
078                SColor.floatGet(0xFF6600EE),  // SColor.SAFETY_ORANGE
079                SColor.floatGet(0x595652DD),  // SColor.DB_SOOT
080                SColor.floatGet(0x59565299)}; // SColor.DB_SOOT
081        /**
082         * Used internally to determine how the explosion should spread; derived from {@link #validCells}.
083         */
084        public double[][] resMap,
085        /**
086         * The internal representation of how affected each cell is by the explosion, based on proximity to center.
087         */
088        lightMap;
089        /**
090         * The raw list of Coords that might be affected by the explosion; may include some cells that aren't going to
091         * show as exploding (it usually has some false positives), but shouldn't exclude any cells that should show as
092         * such (no false negatives). You can edit this if you need to, but it isn't recommended.
093         */
094        public List<Coord> affected;
095        /**
096         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be
097         * the full expanse of the SquidPanel. The duration will be 1 second.
098         * @param targeting the SquidPanel to affect
099         * @param center the center of the explosion
100         * @param radius the radius of the explosion, in cells
101         */
102
103        public ExplosionEffect(SquidPanel targeting, Coord center, int radius)
104        {
105            this(targeting, 1f, center, radius);
106        }
107        /**
108         * Constructs an ExplosionEffect with explicit settings for some fields. The valid cells this can affect will be
109         * the full expanse of the SquidPanel.
110         * @param targeting the SquidPanel to affect
111         * @param duration the duration of this PanelEffect in seconds, as a float
112         * @param center the center of the explosion
113         * @param radius the radius of the explosion, in cells
114         */
115        public ExplosionEffect(SquidPanel targeting, float duration, Coord center, int radius)
116        {
117            super(targeting, duration);
118            this.center = center;
119            this.radius = radius;
120            resMap = new double[validCells.width][validCells.height];
121            lightMap = new double[validCells.width][validCells.height];
122            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius);
123            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
124        }
125        /**
126         * Constructs an ExplosionEffect with explicit settings for most fields.
127         * @param targeting the SquidPanel to affect
128         * @param duration the duration of this PanelEffect in seconds, as a float
129         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
130         * @param center the center of the explosion
131         * @param radius the radius of the explosion, in cells
132         */
133        public ExplosionEffect(SquidPanel targeting, float duration, GreasedRegion valid, Coord center, int radius)
134        {
135            super(targeting, duration, valid);
136            this.center = center;
137            this.radius = radius;
138            resMap = ArrayTools.fill(1.0, validCells.width, validCells.height);
139            validCells.writeDoublesInto(resMap, 0.0);
140            lightMap = new double[validCells.width][validCells.height];
141            FOV.reuseFOV(resMap, lightMap, center.x, center.y, radius);
142            validCells.not().writeDoublesInto(lightMap, 0.0);
143            validCells.not();
144            affected = Radius.inCircle(center.x, center.y, radius, false, validCells.width, validCells.height);
145        }
146
147        /**
148         * Constructs an ExplosionEffect with explicit settings for most fields but also an alternate group of Color
149         * objects that it will use to color the explosion instead of using fiery/smoke colors.
150         * @param targeting the SquidPanel to affect
151         * @param duration the duration of this PanelEffect in seconds, as a float
152         * @param valid the valid cells that can be changed by this PanelEffect, as a GreasedRegion
153         * @param center the center of the explosion
154         * @param radius the radius of the explosion, in cells
155         * @param coloring a List of Color or subclasses thereof that will replace the default fire/smoke colors here
156         */
157        public ExplosionEffect(SquidPanel targeting, float duration, GreasedRegion valid, Coord center, int radius, List<? extends Color> coloring)
158        {
159            this(targeting, duration, valid, center, radius);
160            if(colors.length != coloring.size())
161                colors = new float[coloring.size()];
162            for (int i = 0; i < colors.length; i++) {
163                colors[i] = coloring.get(i).toFloatBits();
164            }
165        }
166        /**
167         * Called each frame.
168         *
169         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
170         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
171         */
172        @Override
173        protected void update(float percent) {
174            int len = affected.size();
175            Coord c;
176            float f, color;
177            int idx, seed = System.identityHashCode(this);
178            for (int i = 0; i < len; i++) {
179                c = affected.get(i);
180                if(lightMap[c.x][c.y] <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
181                    continue;
182                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
183                        * 0.17f + percent * 1.2f;
184                if(f < 0f || 0.5 * lightMap[c.x][c.y] + f < 0.4)
185                    continue;
186                idx = (int) (f * colors.length);
187                if(idx >= colors.length - 1)
188                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
189                else
190                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
191                target.put(c.x, c.y, color);
192            }
193        }
194    }
195}

124    /**
125     * Color difference metric; returns large numbers even for smallish differences.
126     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
127     * @param color1 an RGBA8888 color as an int
128     * @param color2 an RGBA8888 color as an int
129     * @return the difference between the given colors, as a positive int
130     */
131    public static int difference(final int color1, final int color2)
132    {
133        int rmean = ((color1 >>> 24) + (color2 >>> 24)) >> 1;
134        int r = (color1 >>> 24) - (color2 >>> 24);
135        int g = (color1 >>> 16 & 0xFF) - (color2 >>> 16 & 0xFF);
136        int b = (color1 >>> 8 & 0xFF) - (color2 >>> 8 & 0xFF);
137        return (((512+rmean)*r*r)>>8) + 4*g*g + (((767-rmean)*b*b)>>8);
138    }
139    /**
140     * Color difference metric; returns large numbers even for smallish differences.
141     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
142     * @param color1 an RGBA8888 color as an int
143     * @param r2 red value from 0 to 255, inclusive
144     * @param g2 green value from 0 to 255, inclusive
145     * @param b2 blue value from 0 to 255, inclusive
146     * @return the difference between the given colors, as a positive int
147     */
148    public static int difference(final int color1, int r2, int g2, int b2)
149    {
150        r2 = (r2 << 3 | r2 >>> 2);
151        g2 = (g2 << 3 | g2 >>> 2);
152        b2 = (b2 << 3 | b2 >>> 2);
153        final int rmean = ((color1 >>> 24) + r2) >> 1,
154                r = (color1 >>> 24) - r2,
155                g = (color1 >>> 16 & 0xFF) - g2 << 1,
156                b = (color1 >>> 8 & 0xFF) - b2;
157        return (((512+rmean)*r*r)>>8) + g*g + (((767-rmean)*b*b)>>8);
158    }

160    /**
161     * Color difference metric; returns large numbers even for smallish differences.
162     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
163     * @param r1 red value from 0 to 255, inclusive
164     * @param r2 red value from 0 to 255, inclusive
165     * @param g1 green value from 0 to 255, inclusive
166     * @param g2 green value from 0 to 255, inclusive
167     * @param b1 blue value from 0 to 255, inclusive
168     * @param b2 blue value from 0 to 255, inclusive
169     * @return the difference between the given colors, as a positive int
170     */
171    public static int difference(final int r1, final int r2, final int g1, final int g2, final int b1, final int b2)
172    {
173        final int rmean = (r1 + r2) >> 1,
174                r = r1 - r2,
175                g = g1 - g2 << 1,
176                b = b1 - b2;
177        return (((512+rmean)*r*r)>>8) + g*g + (((767-rmean)*b*b)>>8);
178    }
179    /**
180     * Builds the palette information this PNG8 stores from the RGBA8888 ints in {@code rgbaPalette}, up to 256 colors.
181     * Alpha is not preserved except for the first item in rgbaPalette, and only if it is {@code 0} (fully transparent
182     * black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, or only has one color, then
183     * this defaults to DawnBringer's Aurora palette with 256 hand-chosen colors (including transparent).
184     * @param rgbaPalette an array of RGBA8888 ints; all will be used up to 256 items or the length of the array
185     */
186    public void exact(int[] rgbaPalette)
187    {
188        if(rgbaPalette == null || rgbaPalette.length < 2)
189        {
190            rgbaPalette = auroraPalette;
191        }
192        Arrays.fill(paletteArray, 0);
193        Arrays.fill(paletteMapping, (byte) 0);
194        final int plen = Math.min(256, rgbaPalette.length);
195        int color, c2;
196        int dist;
197        for (int i = 0; i < plen; i++) {
198            color = rgbaPalette[i];
199            paletteArray[i] = color;
200            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
201        }
202        for (int r = 0; r < 32; r++) {
203            for (int g = 0; g < 32; g++) {
204                for (int b = 0; b < 32; b++) {
205                    c2 = r << 10 | g << 5 | b;
206                    if(paletteMapping[c2] == 0)
207                    {
208                        dist = 0x7FFFFFFF;
209                        for (int i = 1; i < 256; i++) {
210                            if(dist > (dist = Math.min(dist, difference(paletteArray[i], r, g, b))))
211                                paletteMapping[c2] = (byte)i;
212                        }
213                    }
214                }
215            }
216        }
217    }

233     * Alpha is not preserved except for the first item in colorPalette, and only if its r, g, b, and a values are all

243        {

246        }
247        Arrays.fill(paletteArray, 0);
248        Arrays.fill(paletteMapping, (byte) 0);
249        final int plen = Math.min(Math.min(256, colorPalette.length), limit);
250        int color, c2;
251        int dist;
252        for (int i = 0; i < plen; i++) {
253            color = Color.rgba8888(colorPalette[i]);
254            paletteArray[i] = color;
255            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
256        }
257        for (int r = 0; r < 32; r++) {
258            for (int g = 0; g < 32; g++) {
259                for (int b = 0; b < 32; b++) {
260                    c2 = r << 10 | g << 5 | b;
261                    if(paletteMapping[c2] == 0)
262                    {
263                        dist = 0x7FFFFFFF;
264                        for (int i = 1; i < 256; i++) {
265                            if(dist > (dist = Math.min(dist, difference(paletteArray[i], r, g, b))))
266                                paletteMapping[c2] = (byte)i;
267                        }
268                    }
269                }
270            }
271        }

274    /**
275     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
276     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
277     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
278     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
279     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
280     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
281     * uses a threshold value to determine whether it should permit a less-common color into the palette, and if the
282     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least 400, it is
283     * allowed in the palette, otherwise it is kept out for being too similar to existing colors. This doesn't return a
284     * value but instead stores the palette info in this object; a PaletteReducer can be assigned to the
285     * {@link PNG8#palette} field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
286     * @param pixmap a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
287     */
288    public void analyze(Pixmap pixmap) {
289        analyze(pixmap, 400);
290    }
291    /**
292     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
293     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
294     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
295     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
296     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
297     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
298     * takes a threshold value to determine whether it should permit a less-common color into the palette, and if the
299     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least
300     * {@code threshold}, it is allowed in the palette, otherwise it is kept out for being too similar to existing
301     * colors. The threshold is usually between 250 and 1000, and 400 is a good default. This doesn't return a value but
302     * instead stores the palette info in this object; a PaletteReducer can be assigned to the {@link PNG8#palette}
303     * field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
304     * @param pixmap a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
305     * @param threshold a minimum color difference as produced by {@link #difference(int, int)}; usually between 250 and 1000, 400 is a good default
306     */
307    public void analyze(Pixmap pixmap, int threshold) {
308        Arrays.fill(paletteArray, 0);
309        Arrays.fill(paletteMapping, (byte) 0);
310        int color;
311        final ByteBuffer pixels = pixmap.getPixels();
312        IntIntMap counts = new IntIntMap(256);
313        int hasTransparent = 0;
314        int[] reds = new int[256], greens = new int[256], blues = new int[256];
315        switch (pixmap.getFormat()) {
316            case RGBA8888: {
317                while (pixels.remaining() >= 4) {
318                    color = (pixels.getInt() & 0xF8F8F880);
319                    if ((color & 0x80) != 0) {
320                        color |= (color >>> 5 & 0x07070700) | 0xFE;
321                        counts.getAndIncrement(color, 0, 1);
322                    } else {
323                        hasTransparent = 1;
324                    }
325                }
326                if (counts.size + hasTransparent <= 256) {
327                    int i = hasTransparent;
328                    IntIntMap.Keys ks = counts.keys();
329                    while (ks.hasNext()) {
330                        color = ks.next();
331                        paletteArray[i] = color;
332                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
333                        paletteMapping[color] = (byte) i;
334                        reds[i] = color >>> 10;
335                        greens[i] = color >>> 5 & 31;
336                        blues[i] = color & 31;
337                        i++;
338                    }
339                } else // reduce color count
340                {
341                    IntIntMap.Entries es = counts.entries();
342                    SortedIntList<IntIntMap.Entry> sil = new SortedIntList<>();
343                    while (es.hasNext()) {
344                        IntIntMap.Entry ent = es.next(), ent2 = new IntIntMap.Entry();
345                        ent2.key = ent.key;
346                        ent2.value = ent.value;
347                        sil.insert(-ent.value, ent2);
348                    }
349                    Iterator<SortedIntList.Node<IntIntMap.Entry>> it = sil.iterator();
350                    PER_BEST:
351                    for (int i = 1; i < 256 && it.hasNext(); ) {
352                        color = it.next().value.key;
353                        for (int j = 1; j < i; j++) {
354                            if (difference(color, paletteArray[j]) < threshold)
355                                continue PER_BEST;
356                        }
357                        paletteArray[i] = color;
358                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
359                        paletteMapping[color] = (byte) i;
360                        reds[i] = color >>> 10;
361                        greens[i] = color >>> 5 & 31;
362                        blues[i] = color & 31;
363                        i++;
364                    }
365                }
366                int c2, dist;
367                for (int r = 0; r < 32; r++) {
368                    for (int g = 0; g < 32; g++) {
369                        for (int b = 0; b < 32; b++) {
370                            c2 = r << 10 | g << 5 | b;
371                            if (paletteMapping[c2] == 0) {
372                                dist = 0x7FFFFFFF;
373                                for (int i = 1; i < 256; i++) {
374                                    if (dist > (dist = Math.min(dist, difference(reds[i], r, greens[i], g, blues[i], b))))
375                                        paletteMapping[c2] = (byte) i;
376                                }
377                            }
378                        }
379                    }
380                }
381            }
382            break;
383            case RGB888: {
384                while (pixels.remaining() >= 6) {
385                    color = (pixels.getInt() & 0xF8F8F800);
386                    color |= (color >>> 5 & 0x07070700) | 0xFE;
387                    pixels.position(pixels.position() - 1);
388                    counts.getAndIncrement(color, 0, 1);
389                }
390                if (pixels.remaining() >= 3) {
391                    color = ((pixels.get() & 0xF8) << 24 | (pixels.get() & 0xF8) << 16 | (pixels.get() & 0xF8) << 8);
392                    color |= (color >>> 5 & 0x07070700) | 0xFE;
393                    counts.getAndIncrement(color, 0, 1);
394                }
395                if (counts.size <= 256) {
396                    int i = 0;
397                    IntIntMap.Keys ks = counts.keys();
398                    while (ks.hasNext()) {
399                        color = ks.next();
400                        paletteArray[i] = color;
401                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
402                        paletteMapping[color] = (byte) i;
403                        reds[i] = color >>> 10;
404                        greens[i] = color >>> 5 & 31;
405                        blues[i] = color & 31;
406                        i++;
407                    }
408                } else {
409                    IntIntMap.Entries es = counts.entries();
410                    SortedIntList<IntIntMap.Entry> sil = new SortedIntList<>();
411                    while (es.hasNext()) {
412                        IntIntMap.Entry ent = es.next(), ent2 = new IntIntMap.Entry();
413                        ent2.key = ent.key;
414                        ent2.value = ent.value;
415                        sil.insert(-ent.value, ent2);
416                    }
417                    Iterator<SortedIntList.Node<IntIntMap.Entry>> it = sil.iterator();
418                    PER_BEST:
419                    for (int i = 1; i < 256 && it.hasNext(); ) {
420                        color = it.next().value.key;
421                        for (int j = 1; j < i; j++) {
422                            if (difference(color, paletteArray[j]) < threshold)
423                                continue PER_BEST;
424                        }
425                        paletteArray[i] = color;
426                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
427                        paletteMapping[color] = (byte) i;
428                        reds[i] = color >>> 10;
429                        greens[i] = color >>> 5 & 31;
430                        blues[i] = color & 31;
431                        i++;
432                    }
433                }
434                int c2, dist;
435                for (int r = 0; r < 32; r++) {
436                    for (int g = 0; g < 32; g++) {
437                        for (int b = 0; b < 32; b++) {
438                            c2 = r << 10 | g << 5 | b;
439                            if (paletteMapping[c2] == 0) {
440                                dist = 0x7FFFFFFF;
441                                for (int i = 1; i < 256; i++) {
442                                    if (dist > (dist = Math.min(dist, difference(reds[i], r, greens[i], g, blues[i], b))))
443                                        paletteMapping[c2] = (byte) i;
444                                }
445                            }
446                        }
447                    }
448                }
449            }
450        }
451        pixels.rewind();
452    }

454    /**
455     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, dithering when it can.
456     * If you want to reduce the colors in a Pixmap based on what it currently contains, call
457     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
458     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;
459     * {@link #exact(int[])} is the tool for that job.
460     * <br>
461     * This method is not incredibly fast because of the extra calculations it has to do for dithering, but if you can
462     * compute the PaletteReducer once and reuse it, that will save some time.
463     * @param pixmap a Pixmap that will be modified in place
464     * @return the given Pixmap, for chaining
465     */
466    public Pixmap reduce (Pixmap pixmap) {
467        boolean hasTransparent = (paletteArray[0] == 0);
468        final int lineLen = pixmap.getWidth(), h = pixmap.getHeight();
469        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
470        if (curErrorRedBytes == null) {
471            curErrorRed = (curErrorRedBytes = new ByteArray(lineLen)).items;
472            nextErrorRed = (nextErrorRedBytes = new ByteArray(lineLen)).items;
473            curErrorGreen = (curErrorGreenBytes = new ByteArray(lineLen)).items;
474            nextErrorGreen = (nextErrorGreenBytes = new ByteArray(lineLen)).items;
475            curErrorBlue = (curErrorBlueBytes = new ByteArray(lineLen)).items;
476            nextErrorBlue = (nextErrorBlueBytes = new ByteArray(lineLen)).items;
477        } else {
478            curErrorRed = curErrorRedBytes.ensureCapacity(lineLen);
479            nextErrorRed = nextErrorRedBytes.ensureCapacity(lineLen);
480            curErrorGreen = curErrorGreenBytes.ensureCapacity(lineLen);
481            nextErrorGreen = nextErrorGreenBytes.ensureCapacity(lineLen);
482            curErrorBlue = curErrorBlueBytes.ensureCapacity(lineLen);
483            nextErrorBlue = nextErrorBlueBytes.ensureCapacity(lineLen);
484            for (int i = 0; i < lineLen; i++) {
485                nextErrorRed[i] = 0;
486                nextErrorGreen[i] = 0;
487                nextErrorBlue[i] = 0;
488            }
489
490        }
491        Pixmap.Blending blending = pixmap.getBlending();
492        pixmap.setBlending(Pixmap.Blending.None);
493        int color, used, rdiff, gdiff, bdiff;
494        byte er, eg, eb, paletteIndex;
495        for (int y = 0; y < h; y++) {
496            int ny = y + 1;
497            for (int i = 0; i < lineLen; i++) {
498                curErrorRed[i] = nextErrorRed[i];
499                curErrorGreen[i] = nextErrorGreen[i];
500                curErrorBlue[i] = nextErrorBlue[i];
501                nextErrorRed[i] = 0;
502                nextErrorGreen[i] = 0;
503                nextErrorBlue[i] = 0;
504            }
505            for (int px = 0; px < lineLen; px++) {
506                color = pixmap.getPixel(px, y) & 0xF8F8F880;
507                if ((color & 0x80) == 0 && hasTransparent)
508                    pixmap.drawPixel(px, y, 0);
509                else {
510                    er = curErrorRed[px];
511                    eg = curErrorGreen[px];
512                    eb = curErrorBlue[px];
513                    color |= (color >>> 5 & 0x07070700) | 0xFE;
514                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
515                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
516                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
517                    paletteIndex =
518                            paletteMapping[((rr << 7) & 0x7C00)
519                                    | ((gg << 2) & 0x3E0)
520                                    | ((bb >>> 3))];
521                    used = paletteArray[paletteIndex & 0xFF];
522                    pixmap.drawPixel(px, y, used);
523                    rdiff = (color>>>24)-    (used>>>24);
524                    gdiff = (color>>>16&255)-(used>>>16&255);
525                    bdiff = (color>>>8&255)- (used>>>8&255);
526                    if(px < lineLen - 1)
527                    {
528                        curErrorRed[px+1]   += rdiff >> 1;
529                        curErrorGreen[px+1] += gdiff >> 1;
530                        curErrorBlue[px+1]  += bdiff >> 1;
531                    }
532                    if(ny < h)
533                    {
534                        if(px > 0)
535                        {
536                            nextErrorRed[px-1]   += rdiff >> 2;
537                            nextErrorGreen[px-1] += gdiff >> 2;
538                            nextErrorBlue[px-1]  += bdiff >> 2;
539                        }
540                        nextErrorRed[px]   += rdiff >> 2;
541                        nextErrorGreen[px] += gdiff >> 2;
542                        nextErrorBlue[px]  += bdiff >> 2;
543                    }
544                }
545            }
546
547        }
548        pixmap.setBlending(blending);
549        return pixmap;
550    }

552    /**
553     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, without dithering. This produces
554     * blocky solid sections of color in most images where the palette isn't exact, instead of checkerboard-like
555     * dithering patterns. If you want to reduce the colors in a Pixmap based on what it currently contains, call
556     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
557     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;
558     * {@link #exact(int[])} is the tool for that job.
559     * @param pixmap a Pixmap that will be modified in place
560     * @return the given Pixmap, for chaining
561     */
562    public Pixmap reduceSolid (Pixmap pixmap) {
563        boolean hasTransparent = (paletteArray[0] == 0);
564        final int lineLen = pixmap.getWidth(), h = pixmap.getHeight();
565        Pixmap.Blending blending = pixmap.getBlending();
566        pixmap.setBlending(Pixmap.Blending.None);
567        int color;
568        for (int y = 0; y < h; y++) {
569            for (int px = 0; px < lineLen; px++) {
570                color = pixmap.getPixel(px, y);
571                if ((color & 0x80) == 0 && hasTransparent)
572                    pixmap.drawPixel(px, y, 0);
573                else {
574                    int rr = ((color >>> 24)       );
575                    int gg = ((color >>> 16) & 0xFF);
576                    int bb = ((color >>> 8)  & 0xFF);
577                    pixmap.drawPixel(px, y, paletteArray[
578                            paletteMapping[((rr << 7) & 0x7C00)
579                            | ((gg << 2) & 0x3E0)
580                            | ((bb >>> 3))] & 0xFF]);
581                }
582            }
583
584        }
585        pixmap.setBlending(blending);
586        return pixmap;
587    }

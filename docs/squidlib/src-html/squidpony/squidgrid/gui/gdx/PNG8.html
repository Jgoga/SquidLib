163    /**
164     * Writes the given Pixmap to the requested FileHandle, computing an 8-bit palette from the most common colors in
165     * pixmap. If there are 256 or less colors and none are transparent, this will use 256 colors in its palette exactly
166     * with no transparent entry, but if there are more than 256 colors or any are transparent, then one color will be
167     * used for "fully transparent" and 255 opaque colors will be used.
168     * @param file a FileHandle that must be writable, and will have the given Pixmap written as a PNG-8 image
169     * @param pixmap a Pixmap to write to the given file
170     * @throws IOException if file writing fails for any reason
171     */

173        write(file, pixmap, true);
174    }

176    /**
177     * Writes the given Pixmap to the requested FileHandle, optionally computing an 8-bit palette from the most common
178     * colors in pixmap. When computePalette is true, if there are 256 or less colors and none are transparent, this
179     * will use 256 colors in its palette exactly with no transparent entry, but if there are more than 256 colors or
180     * any are transparent, then one color will be used for "fully transparent" and 255 opaque colors will be used. When
181     * computePalette is false, this uses the last palette this had computed, or a 253-color bold palette with one
182     * fully-transparent color if no palette had been computed yet.
183     * @param file a FileHandle that must be writable, and will have the given Pixmap written as a PNG-8 image
184     * @param pixmap a Pixmap to write to the given file
185     * @param computePalette if true, this will analyze the Pixmap and use the most common colors
186     * @throws IOException if file writing fails for any reason
187     */
188    public void write (FileHandle file, Pixmap pixmap, boolean computePalette) throws IOException {
189        OutputStream output = file.write(false);
190        try {
191            write(output, pixmap, computePalette);
192        } finally {
193            StreamUtils.closeQuietly(output);
194        }
195    }

237    /**
238     * Writes the pixmap to the stream without closing the stream, optionally computing an 8-bit palette from the given
239     * Pixmap. If {@link #palette} is null (the default unless it has been assigned a PaletteReducer value), this will
240     * compute a palette from the given Pixmap regardless of computePalette.
241     * @param output an OutputStream that will not be closed
242     * @param pixmap a Pixmap to write to the given output stream
243     * @param computePalette if true, this will analyze the Pixmap and use the most common colors
244     * @param dither true if this should dither colors that can't be represented exactly
245     */
246    public void write (OutputStream output, Pixmap pixmap, boolean computePalette, boolean dither) throws IOException
247    {
248        if(dither) writeDithered(output, pixmap, computePalette);
249        else writeSolid(output, pixmap, computePalette);
250    }
251    private void writeSolid (OutputStream output, Pixmap pixmap, boolean computePalette) throws IOException{
252        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
253        if(palette == null)
254        {
255            palette = new PaletteReducer(pixmap);
256        }
257        else if(computePalette)
258        {
259            palette.analyze(pixmap);
260        }
261        final int[] paletteArray = palette.paletteArray;
262        final byte[] paletteMapping = palette.paletteMapping;
263
264
265        DataOutputStream dataOutput = new DataOutputStream(output);
266        dataOutput.write(SIGNATURE);
267
268        buffer.writeInt(IHDR);
269        buffer.writeInt(pixmap.getWidth());
270        buffer.writeInt(pixmap.getHeight());
271        buffer.writeByte(8); // 8 bits per component.
272        buffer.writeByte(COLOR_INDEXED);
273        buffer.writeByte(COMPRESSION_DEFLATE);
274        buffer.writeByte(FILTER_NONE);
275        buffer.writeByte(INTERLACE_NONE);
276        buffer.endChunk(dataOutput);
277
278        buffer.writeInt(PLTE);
279        for (int i = 0; i < paletteArray.length; i++) {
280            int p = paletteArray[i];
281            buffer.write(p>>>24);
282            buffer.write(p>>>16);
283            buffer.write(p>>>8);
284        }
285        buffer.endChunk(dataOutput);
286
287        boolean hasTransparent = false;
288        if(paletteArray[0] == 0) {
289            hasTransparent = true;
290            buffer.writeInt(TRNS);
291            buffer.write(0);
292            buffer.endChunk(dataOutput);
293        }
294        buffer.writeInt(IDAT);
295        deflater.reset();
296
297        int lineLen = pixmap.getWidth();
298        byte[] lineOut, curLine, prevLine;
299        if (lineOutBytes == null) {
300            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
301            curLine = (curLineBytes = new ByteArray(lineLen)).items;
302            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
303        } else {
304            lineOut = lineOutBytes.ensureCapacity(lineLen);
305            curLine = curLineBytes.ensureCapacity(lineLen);
306            prevLine = prevLineBytes.ensureCapacity(lineLen);
307            for (int i = 0, n = lastLineLen; i < n; i++)
308            {
309                prevLine[i] = 0;
310            }
311        }
312
313        lastLineLen = lineLen;
314
315        ByteBuffer pixels = pixmap.getPixels();
316        int oldPosition = pixels.position(), color;
317        final int w = pixmap.getWidth();
318        for (int y = 0, h = pixmap.getHeight(); y < h; y++) {
319            int py = flipY ? (h - y - 1) : y;
320            for (int px = 0; px < w; px++) {
321                color = pixmap.getPixel(px, py);
322                if ((color & 0x80) == 0 && hasTransparent)
323                    curLine[px] = 0;
324                else {
325                    int rr = ((color >>> 24)       );
326                    int gg = ((color >>> 16) & 0xFF);
327                    int bb = ((color >>> 8)  & 0xFF);
328                    curLine[px] = paletteMapping[((rr << 7) & 0x7C00)
329                                    | ((gg << 2) & 0x3E0)
330                                    | ((bb >>> 3))];
331                }
332            }
333
334            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
335
336            //Paeth
337            for (int x = 1; x < lineLen; x++) {
338                int a = curLine[x - 1] & 0xff;
339                int b = prevLine[x] & 0xff;
340                int c = prevLine[x - 1] & 0xff;
341                int p = a + b - c;
342                int pa = p - a;
343                if (pa < 0) pa = -pa;
344                int pb = p - b;
345                if (pb < 0) pb = -pb;
346                int pc = p - c;
347                if (pc < 0) pc = -pc;
348                if (pa <= pb && pa <= pc)
349                    c = a;
350                else if (pb <= pc)
351                    c = b;
352                lineOut[x] = (byte)(curLine[x] - c);
353            }
354
355            deflaterOutput.write(PAETH);
356            deflaterOutput.write(lineOut, 0, lineLen);
357
358            byte[] temp = curLine;
359            curLine = prevLine;
360            prevLine = temp;
361        }
362        pixels.position(oldPosition);
363        deflaterOutput.finish();
364        buffer.endChunk(dataOutput);
365
366        buffer.writeInt(IEND);
367        buffer.endChunk(dataOutput);
368
369        output.flush();
370    }

372    private void writeDithered (OutputStream output, Pixmap pixmap, boolean computePalette) throws IOException{
373        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
374        if(palette == null)
375        {
376            palette = new PaletteReducer(pixmap);
377        }
378        else if(computePalette)
379        {
380            palette.analyze(pixmap);
381        }
382        final int[] paletteArray = palette.paletteArray;
383        final byte[] paletteMapping = palette.paletteMapping;
384
385
386        DataOutputStream dataOutput = new DataOutputStream(output);
387        dataOutput.write(SIGNATURE);
388
389        buffer.writeInt(IHDR);
390        buffer.writeInt(pixmap.getWidth());
391        buffer.writeInt(pixmap.getHeight());
392        buffer.writeByte(8); // 8 bits per component.
393        buffer.writeByte(COLOR_INDEXED);
394        buffer.writeByte(COMPRESSION_DEFLATE);
395        buffer.writeByte(FILTER_NONE);
396        buffer.writeByte(INTERLACE_NONE);
397        buffer.endChunk(dataOutput);
398
399        buffer.writeInt(PLTE);
400        for (int i = 0; i < paletteArray.length; i++) {
401            int p = paletteArray[i];
402            buffer.write(p>>>24);
403            buffer.write(p>>>16);
404            buffer.write(p>>>8);
405        }
406        buffer.endChunk(dataOutput);
407
408        boolean hasTransparent = false;
409        if(paletteArray[0] == 0) {
410            hasTransparent = true;
411            buffer.writeInt(TRNS);
412            buffer.write(0);
413            buffer.endChunk(dataOutput);
414        }
415        buffer.writeInt(IDAT);
416        deflater.reset();
417
418        int lineLen = pixmap.getWidth();
419        byte[] lineOut, curLine, prevLine;
420        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
421        if (lineOutBytes == null) {
422            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
423            curLine = (curLineBytes = new ByteArray(lineLen)).items;
424            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
425        } else {
426            lineOut = lineOutBytes.ensureCapacity(lineLen);
427            curLine = curLineBytes.ensureCapacity(lineLen);
428            prevLine = prevLineBytes.ensureCapacity(lineLen);
429            for (int i = 0, n = lastLineLen; i < n; i++)
430            {
431                prevLine[i] = 0;
432            }
433        }
434        if(palette.curErrorRedBytes == null)
435        {
436            curErrorRed = (palette.curErrorRedBytes = new ByteArray(lineLen)).items;
437            nextErrorRed = (palette.nextErrorRedBytes = new ByteArray(lineLen)).items;
438            curErrorGreen = (palette.curErrorGreenBytes = new ByteArray(lineLen)).items;
439            nextErrorGreen = (palette.nextErrorGreenBytes = new ByteArray(lineLen)).items;
440            curErrorBlue = (palette.curErrorBlueBytes = new ByteArray(lineLen)).items;
441            nextErrorBlue = (palette.nextErrorBlueBytes = new ByteArray(lineLen)).items;
442        } else {
443            curErrorRed = palette.curErrorRedBytes.ensureCapacity(lineLen);
444            nextErrorRed = palette.nextErrorRedBytes.ensureCapacity(lineLen);
445            curErrorGreen = palette.curErrorGreenBytes.ensureCapacity(lineLen);
446            nextErrorGreen = palette.nextErrorGreenBytes.ensureCapacity(lineLen);
447            curErrorBlue = palette.curErrorBlueBytes.ensureCapacity(lineLen);
448            nextErrorBlue = palette.nextErrorBlueBytes.ensureCapacity(lineLen);
449            for (int i = 0; i < lineLen; i++) {
450                nextErrorRed[i] = 0;
451                nextErrorGreen[i] = 0;
452                nextErrorBlue[i] = 0;
453            }
454
455        }
456
457
458        lastLineLen = lineLen;
459
460        ByteBuffer pixels = pixmap.getPixels();
461        int oldPosition = pixels.position(), color, used, rdiff, gdiff, bdiff;
462        byte er, eg, eb, paletteIndex;
463        final int w = pixmap.getWidth();
464        for (int y = 0, h = pixmap.getHeight(); y < h; y++) {
465            int py = flipY ? (h - y - 1) : y;
466            int ny = flipY ? (h - y - 2) : y + 1;
467            for (int i = 0; i < lineLen; i++) {
468                curErrorRed[i] = nextErrorRed[i];
469                curErrorGreen[i] = nextErrorGreen[i];
470                curErrorBlue[i] = nextErrorBlue[i];
471                nextErrorRed[i] = 0;
472                nextErrorGreen[i] = 0;
473                nextErrorBlue[i] = 0;
474            }
475            for (int px = 0; px < w; px++) {
476                color = pixmap.getPixel(px, py) & 0xF8F8F880;
477                if ((color & 0x80) == 0 && hasTransparent)
478                    curLine[px] = 0;
479                else {
480                    er = curErrorRed[px];
481                    eg = curErrorGreen[px];
482                    eb = curErrorBlue[px];
483                    color |= (color >>> 5 & 0x07070700) | 0xFE;
484                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
485                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
486                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
487                    curLine[px] = paletteIndex =
488                            paletteMapping[((rr << 7) & 0x7C00)
489                                    | ((gg << 2) & 0x3E0)
490                                    | ((bb >>> 3))];
491                    used = paletteArray[paletteIndex & 0xFF];
492                    rdiff = (color>>>24)-    (used>>>24);
493                    gdiff = (color>>>16&255)-(used>>>16&255);
494                    bdiff = (color>>>8&255)- (used>>>8&255);
495                    if(px < w - 1)
496                    {
497                        curErrorRed[px+1]   += rdiff >> 1;
498                        curErrorGreen[px+1] += gdiff >> 1;
499                        curErrorBlue[px+1]  += bdiff >> 1;
500                    }
501                    if(ny >= 0 && ny < h)
502                    {
503                        if(px > 0)
504                        {
505                            nextErrorRed[px-1]   += rdiff >> 2;
506                            nextErrorGreen[px-1] += gdiff >> 2;
507                            nextErrorBlue[px-1]  += bdiff >> 2;
508                        }
509                        nextErrorRed[px]   += rdiff >> 2;
510                        nextErrorGreen[px] += gdiff >> 2;
511                        nextErrorBlue[px]  += bdiff >> 2;
512                    }
513                }
514            }
515
516            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
517
518            //Paeth
519            for (int x = 1; x < lineLen; x++) {
520                int a = curLine[x - 1] & 0xff;
521                int b = prevLine[x] & 0xff;
522                int c = prevLine[x - 1] & 0xff;
523                int p = a + b - c;
524                int pa = p - a;
525                if (pa < 0) pa = -pa;
526                int pb = p - b;
527                if (pb < 0) pb = -pb;
528                int pc = p - c;
529                if (pc < 0) pc = -pc;
530                if (pa <= pb && pa <= pc)
531                    c = a;
532                else if (pb <= pc)
533                    c = b;
534                lineOut[x] = (byte)(curLine[x] - c);
535            }
536
537            deflaterOutput.write(PAETH);
538            deflaterOutput.write(lineOut, 0, lineLen);
539
540            byte[] temp = curLine;
541            curLine = prevLine;
542            prevLine = temp;
543        }
544        pixels.position(oldPosition);
545        deflaterOutput.finish();
546        buffer.endChunk(dataOutput);
547
548        buffer.writeInt(IEND);
549        buffer.endChunk(dataOutput);
550
551        output.flush();
552    }

554    /** Disposal will happen automatically in {@link #finalize()} but can be done explicitly if desired. */
555    public void dispose () {
556        deflater.end();
557    }
558
559    static class ChunkBuffer extends DataOutputStream {
560        final ByteArrayOutputStream buffer;
561        final CRC32 crc;
562
563        ChunkBuffer (int initialSize) {
564            this(new ByteArrayOutputStream(initialSize), new CRC32());
565        }
566
567        private ChunkBuffer (ByteArrayOutputStream buffer, CRC32 crc) {
568            super(new CheckedOutputStream(buffer, crc));
569            this.buffer = buffer;
570            this.crc = crc;
571        }
572
573        public void endChunk (DataOutputStream target) throws IOException {
574            flush();
575            target.writeInt(buffer.size() - 4);
576            buffer.writeTo(target);
577            target.writeInt((int)crc.getValue());
578            buffer.reset();
579            crc.reset();
580        }
581    }

150    public void build253Palette()
151    {
152        Arrays.fill(paletteArray, 0);
153        int i = 0, j, rl, gl, bl, rMin, rMax=0, gMin, gMax, bMin, bMax;
154        for (int r = 0; r < 6; r++) {
155            rl = SColor.redPossibleLUT[r] & 0xFF;
156            rMin=rMax;
157            for (j = rMin; j < 32 && (SColor.redLUT[j] & 0xFF) == rl; j++) { }
158            rMax=j;
159            gMax = 0;
160            for (int g = 0; g < 7; g++) {
161                gl = SColor.greenPossibleLUT[g] & 0xFF;
162                gMin=gMax;
163                for (j = gMin; j < 32 && (SColor.greenLUT[j] >> 8 & 0xFF) == gl; j++) { }
164                gMax=j;
165                bMax = 0;
166                for (int b = 0; b < 6; b++) {
167                    bl = SColor.bluePossibleLUT[b] & 0xFF;
168                    bMin=bMax;
169                    for (j = bMin; j < 32 && (SColor.blueLUT[j] >> 16 & 0xFF) == bl; j++) { }
170                    bMax=j;
171                    paletteArray[++i] =
172                            (rl << 24
173                                    | (gl << 16 & 0xFF0000)
174                                    | (bl << 8 & 0xFF00) | 0xFE);
175                    for (int rm = rMin; rm < rMax; rm++) {
176                        for (int gm = gMin; gm < gMax; gm++) {
177                            Arrays.fill(paletteMapping, (rm << 10) + (gm << 5) + (bMin), (rm << 10) + (gm << 5) + (bMax), (byte)i);
178                        }
179                    }
180                }
181            }
182        }

193    public static int difference2(final int color1, final int color2)
194    {
195        int rmean = ((color1 >>> 24) + (color2 >>> 24)) >> 1;
196        int r = (color1 >>> 24) - (color2 >>> 24);
197        int g = (color1 >>> 16 & 0xFF) - (color2 >>> 16 & 0xFF);
198        int b = (color1 >>> 8 & 0xFF) - (color2 >>> 8 & 0xFF);
199        return (((512+rmean)*r*r)>>8) + 4*g*g + (((767-rmean)*b*b)>>8);
200    }
201    public static int difference2(final int color1, int r2, int g2, int b2)
202    {
203        r2 = (r2 << 3 | r2 >>> 2);
204        g2 = (g2 << 3 | g2 >>> 2);
205        b2 = (b2 << 3 | b2 >>> 2);
206        final int rmean = ((color1 >>> 24) + r2) >> 1, 
207                r = (color1 >>> 24) - r2,
208                g = (color1 >>> 16 & 0xFF) - g2 << 1,
209                b = (color1 >>> 8 & 0xFF) - b2;
210        return (((512+rmean)*r*r)>>8) + g*g + (((767-rmean)*b*b)>>8);
211    }
212    public static int difference2(final int r1, final int r2, final int g1, final int g2, final int b1, final int b2)
213    {
214        final int rmean = (r1 + r2) >> 1,
215                r = r1 - r2,
216                g = g1 - g2 << 1,
217                b = b1 - b2;
218        return (((512+rmean)*r*r)>>8) + g*g + (((767-rmean)*b*b)>>8);
219    }

269    public void buildKnownPalette(int[] rgbaPalette)
270    {
271        if(rgbaPalette == null || rgbaPalette.length < 2)
272        {
273            build253Palette();
274            return;
275        }
276        Arrays.fill(paletteArray, 0);
277        Arrays.fill(paletteMapping, (byte) 0);
278        final int plen = Math.min(256, rgbaPalette.length);
279        int color, c2;
280        int dist;
281        for (int i = 0; i < plen; i++) {
282            color = rgbaPalette[i];
283            paletteArray[i] = color;
284            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
285        }
286        for (int r = 0; r < 32; r++) {
287            for (int g = 0; g < 32; g++) {
288                for (int b = 0; b < 32; b++) {
289                    c2 = r << 10 | g << 5 | b;
290                    if(paletteMapping[c2] == 0)
291                    {
292                        dist = 0x7FFFFFFF;
293                        for (int i = 1; i < 256; i++) {
294                            if(dist > (dist = Math.min(dist, difference2(paletteArray[i], r, g, b))))
295                                paletteMapping[c2] = (byte)i;
296                        }
297                    }
298                }
299            }
300        }
301    }
302    public void buildComputedPalette(Pixmap pixmap) {
303        Arrays.fill(paletteArray, 0);
304        int color;
305        final ByteBuffer pixels = pixmap.getPixels();
306        IntIntMap counts = new IntIntMap(256);
307        int hasTransparent = 0;
308        switch (pixmap.getFormat()) {
309            case RGBA8888: {
310                while (pixels.remaining() >= 4) {
311                    color = (pixels.getInt() & 0xF8F8F880);
312                    if ((color & 0x80) != 0) {
313                        color |= (color >>> 5 & 0x07070700) | 0xFE;
314                        counts.getAndIncrement(color, 0, 1);
315                    }
316                    else
317                    {
318                        hasTransparent = 1;
319                    }
320                }
321                if(counts.size + hasTransparent <= 256)
322                {
323                    int idx = hasTransparent;
324                    IntIntMap.Keys ks = counts.keys();
325                    while(ks.hasNext())
326                    {
327                        color = ks.next();
328                        paletteArray[idx] = color;
329                        paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) idx++;
330                    }
331                }
332                else
333                {
334                    IntIntMap.Entries es = counts.entries();
335                    SortedIntList<IntIntMap.Entry> sil = new SortedIntList<>();
336                    while (es.hasNext())
337                    {
338                        IntIntMap.Entry ent = es.next(), ent2 = new IntIntMap.Entry();
339                        ent2.key = ent.key;
340                        ent2.value = ent.value;
341                        sil.insert(-ent.value, ent2);
342                    }
343                    Arrays.fill(paletteMapping, (byte) 0);
344                    Iterator<SortedIntList.Node<IntIntMap.Entry>> it = sil.iterator();
345                    int[] reds = new int[256], greens = new int[256], blues = new int[256]; 
346                    for (int i = 1; i < 256 && it.hasNext(); i++) {
347                        color = it.next().value.key;
348                        paletteArray[i] = color;
349                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
350                        paletteMapping[color] = (byte) i;
351                        reds[i] = color >>> 10;
352                        greens[i] = color >>> 5 & 31;
353                        blues[i] = color & 31;
354                    }
355                    int c2, dist;
356                    for (int r = 0; r < 32; r++) {
357                        for (int g = 0; g < 32; g++) {
358                            for (int b = 0; b < 32; b++) {
359                                c2 = r << 10 | g << 5 | b;
360                                if(paletteMapping[c2] == 0)
361                                {
362                                    dist = 0x7FFFFFFF;
363                                    for (int i = 1; i < 256; i++) {
364//                                        if(dist > (dist = Math.min(dist, quickDistance(reds[i] - r, greens[i] - g, blues[i] - b))))
365                                        if(dist > (dist = Math.min(dist, difference2(reds[i], r, greens[i], g, blues[i], b))))
366                                            paletteMapping[c2] = (byte)i;
367                                    }
368                                }
369                            }
370                        }
371                    }
372                }
373
374            }
375            case RGB888: {
376                while (pixels.remaining() >= 6) {
377                    color = (pixels.getInt() & 0xF8F8F800);
378                    color |= (color >>> 5 & 0x07070700) | 0xFE;
379                    pixels.position(pixels.position() - 1);
380                    counts.getAndIncrement(color, 0, 1);
381                }
382                if (pixels.remaining() >= 3) {
383                    color = ((pixels.get() & 0xF8) << 24 | (pixels.get() & 0xF8) << 16 | (pixels.get() & 0xF8) << 8);
384                    color |= (color >>> 5 & 0x07070700) | 0xFE;
385                    counts.getAndIncrement(color, 0, 1);
386                }
387                if(counts.size <= 256)
388                {
389                    int idx = 0;
390                    IntIntMap.Keys ks = counts.keys();
391                    while(ks.hasNext())
392                    {
393                        color = ks.next();
394                        paletteArray[idx] = color;
395                        paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) idx++;
396                    }
397                }
398                else
399                {
400                    IntIntMap.Entries es = counts.entries();
401                    SortedIntList<IntIntMap.Entry> sil = new SortedIntList<>();
402                    while (es.hasNext())
403                    {
404                        IntIntMap.Entry ent = es.next(), ent2 = new IntIntMap.Entry();
405                        ent2.key = ent.key;
406                        ent2.value = ent.value;
407                        sil.insert(-ent.value, ent2);
408                    }
409                    Arrays.fill(paletteMapping, (byte) 0);
410                    Iterator<SortedIntList.Node<IntIntMap.Entry>> it = sil.iterator();
411                    int[] reds = new int[256], greens = new int[256], blues = new int[256];
412                    for (int i = 1; i < 256 && it.hasNext(); i++) {
413                        color = it.next().value.key;
414                        paletteArray[i] = color;
415                        color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
416                        paletteMapping[color] = (byte) i;
417                        reds[i] = color >>> 10;
418                        greens[i] = color >>> 5 & 31;
419                        blues[i] = color & 31;
420                    }
421                    int c2, dist;
422                    for (int r = 0; r < 32; r++) {
423                        for (int g = 0; g < 32; g++) {
424                            for (int b = 0; b < 32; b++) {
425                                c2 = r << 10 | g << 5 | b;
426                                if(paletteMapping[c2] == 0)
427                                {
428                                    dist = 0x7FFFFFFF;
429                                    for (int i = 1; i < 256; i++) {
430                                        if(dist > (dist = Math.min(dist, difference2(reds[i], r, greens[i], g, blues[i], b))))
431                                            paletteMapping[c2] = (byte)i;
432                                    }
433                                }
434                            }
435                        }
436                    }
437                }
438
439            }
440        }
441        pixels.rewind();
442        //compressed = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
443    }

515    public void write (OutputStream output, Pixmap pixmap, boolean computePalette) throws IOException {
516        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
517        if(computePalette) 
518            buildComputedPalette(pixmap);
519        DataOutputStream dataOutput = new DataOutputStream(output);

522        buffer.writeInt(IHDR);
523        buffer.writeInt(pixmap.getWidth());
524        buffer.writeInt(pixmap.getHeight());
525        buffer.writeByte(8); // 8 bits per component.
526        buffer.writeByte(COLOR_INDEXED);
527        buffer.writeByte(COMPRESSION_DEFLATE);
528        buffer.writeByte(FILTER_NONE);
529        buffer.writeByte(INTERLACE_NONE);

532        buffer.writeInt(PLTE);
533        for (int i = 0; i < paletteArray.length; i++) {

537            buffer.write(p>>>8);

540        if(paletteArray[0] == 0) {
541            buffer.writeInt(TRNS);
542            buffer.write(0);
543            buffer.endChunk(dataOutput);
544        }
545        buffer.writeInt(IDAT);
546        deflater.reset();
547
548        int lineLen = pixmap.getWidth();
549        byte[] lineOut, curLine, prevLine;
550        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
551        if (lineOutBytes == null) {
552            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;

554            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;

559            curErrorBlue = (curErrorBlueBytes = new ByteArray(lineLen)).items;
560            nextErrorBlue = (nextErrorBlueBytes = new ByteArray(lineLen)).items;
561        } else {
562            lineOut = lineOutBytes.ensureCapacity(lineLen);
563            curLine = curLineBytes.ensureCapacity(lineLen);
564            prevLine = prevLineBytes.ensureCapacity(lineLen);
565            curErrorRed = curErrorRedBytes.ensureCapacity(lineLen);
566            nextErrorRed = nextErrorRedBytes.ensureCapacity(lineLen);
567            curErrorGreen = curErrorGreenBytes.ensureCapacity(lineLen);
568            nextErrorGreen = nextErrorGreenBytes.ensureCapacity(lineLen);
569            curErrorBlue = curErrorBlueBytes.ensureCapacity(lineLen);
570            nextErrorBlue = nextErrorBlueBytes.ensureCapacity(lineLen);
571            for (int i = 0, n = lastLineLen; i < n; i++)
572            {
573                prevLine[i] = 0;
574            }

576                nextErrorRed[i] = 0;
577                nextErrorGreen[i] = 0;
578                nextErrorBlue[i] = 0;
579            }

581        }

589        for (int y = 0, h = pixmap.getHeight(); y < h; y++) {
590            int py = flipY ? (h - y - 1) : y;
591            int ny = flipY ? (h - y - 2) : y + 1;
592            for (int i = 0; i < lineLen; i++) {
593                curErrorRed[i] = nextErrorRed[i];
594                curErrorGreen[i] = nextErrorGreen[i];
595                curErrorBlue[i] = nextErrorBlue[i];
596                nextErrorRed[i] = 0;
597                nextErrorGreen[i] = 0;

606//                }

608            for (int px = 0; px < w; px++) {
609                color = pixmap.getPixel(px, py);
610                if ((color & 0x80) == 0)
611                    curLine[px] = 0;
612                else {
613                    er = curErrorRed[px];
614                    eg = curErrorRed[px];
615                    eb = curErrorRed[px];
616                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er << 1), 0, 0xFF);
617                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg << 1), 0, 0xFF);

691    static class ChunkBuffer extends DataOutputStream {
692        final ByteArrayOutputStream buffer;
693        final CRC32 crc;
694
695        ChunkBuffer (int initialSize) {
696            this(new ByteArrayOutputStream(initialSize), new CRC32());
697        }
698
699        private ChunkBuffer (ByteArrayOutputStream buffer, CRC32 crc) {
700            super(new CheckedOutputStream(buffer, crc));
701            this.buffer = buffer;
702            this.crc = crc;
703        }
704
705        public void endChunk (DataOutputStream target) throws IOException {
706            flush();
707            target.writeInt(buffer.size() - 4);
708            buffer.writeTo(target);
709            target.writeInt((int)crc.getValue());
710            buffer.reset();
711            crc.reset();
712        }
713    }

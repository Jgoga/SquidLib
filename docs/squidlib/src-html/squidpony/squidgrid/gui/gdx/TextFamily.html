015/**
016 * A variant of {@link TextCellFactory} that allows switching between regular, bold, italic, and bold italic styles.
017 * While almost interchangeable with TextCellFactory, it must be set up differently, using the
018 * {@link #fontFamily(String, String...)} or {@link #fontFamilyDistance(String, String...)} methods, and (potentially
019 * importantly) it does not correctly render chars above {@code '\\u3fff'} due to using some of the bits that normally
020 * represent late-in-Unicode character codes to instead represent bold and italic modes. Two TextFamily values are
021 * present in DefaultResources, {@link DefaultResources#getLeanFamily()} and {@link DefaultResources#getSlabFamily()};
022 * using them is currently the recommended way because this class is somewhat picky about how it can be built.
023 * You may want to use {@link GDXMarkup#colorString(CharSequence)} to produce an {@link squidpony.panel.IColoredString}
024 * that contains the specially-altered chars that store bold and italic mode data.
025 * Created by Tommy Ettinger on 10/26/2017.
026 */
027public class TextFamily extends TextCellFactory {
028    
029    public static final int REGULAR = 0, BOLD = 1, ITALIC = 2, BOLD_ITALIC = 3;
030    public int currentStyle = 0;
031    public BitmapFont[] family = new BitmapFont[4];
032    protected float[] dfx = new float[4], dfy = new float[4];
033    /**

041    /**
042     * A default valued factory that uses the given {@link AssetManager} to load
043     * the font file. Use this constructor if you are likely to load the same
044     * font over and over (recall that, without an {@link AssetManager}, each
045     * instance of {@link TextCellFactory} will load its font from disk). This
046     * primarily matters if you are using fonts not bundled with SquidLib, since
047     * accessing a BitmapFont with a method (not a String) from DefaultResources
048     * caches the BitmapFont already.
049     *
050     * @param assetManager an ordinary libGDX AssetManager
051     */
052    public TextFamily(/* Nullable */ AssetManager assetManager) {
053        super(assetManager);
054    }
055    @Override
056    public TextFamily copy()
057    {
058        TextFamily next = new TextFamily(assetManager);
059        //next.bmpFont = bmpFont;
060        if(bmpFont == null)
061            bmpFont = DefaultResources.getIncludedFont();
062        if(family[0] != null)
063            next.family[0] = DefaultResources.copyFont(family[0]);
064        if(family[1] != null)
065            next.family[1] = DefaultResources.copyFont(family[1]);
066        if(family[2] != null)
067            next.family[2] = DefaultResources.copyFont(family[2]);
068        if(family[3] != null)
069            next.family[3] = DefaultResources.copyFont(family[3]);
070        System.arraycopy(dfx, 0, next.dfx, 0, 4);
071        System.arraycopy(dfy, 0, next.dfy, 0, 4);
072        next.currentStyle = currentStyle;
073        next.bmpFont = next.family[next.currentStyle];
074        next.block = block;
075        next.swap = new OrderedMap<>(swap);
076        next.distanceField = distanceField;
077        next.msdf = msdf;
078        next.distanceFieldScaleX = distanceFieldScaleX;
079        next.distanceFieldScaleY = distanceFieldScaleY;
080        next.shader = null;
081        next.fitting = fitting;
082        next.height = height;
083        next.width = width;
084        next.actualCellWidth = actualCellWidth;
085        next.actualCellHeight = actualCellHeight;
086        next.descent = descent;
087        next.lineHeight = lineHeight;
088        //next.modifiedHeight = modifiedHeight;
089        next.smoothingMultiplier = smoothingMultiplier;
090        next.scc = scc;
091        next.directionGlyph = directionGlyph;
092        if(initializedBySize)
093            next.initBySize();
094        else if(initializedByFont)
095            next.initByFont();
096        return next;
097    }

099    private BitmapFont loadFont(TextureAtlas atlas, String fontName) {
100        String fnt = fontName + ".fnt";
101        if (Gdx.files.internal(fnt).exists())
102            return new BitmapFont(Gdx.files.internal(fnt), atlas.findRegion(fontName));
103        else if (Gdx.files.classpath(fnt).exists())
104            return new BitmapFont(Gdx.files.classpath(fnt), atlas.findRegion(fontName));
105        else
106            return DefaultResources.getIncludedFont();
107    }
108    private BitmapFont loadFont(BitmapFontLoader.BitmapFontParameter atlasFinder, String fontName) {
109        assetManager.load(new AssetDescriptor<>(fontName, BitmapFont.class, atlasFinder));
110        assetManager.finishLoading();
111        return assetManager.get(fontName, BitmapFont.class);
112    }
113    public TextFamily defaultFamilyLeanDistance()
114    {

117                "Iosevka-distance-family", "Iosevka-Bold-distance-family",
118                "Iosevka-Italic-distance-family", "Iosevka-Bold-Italic-distance-family");
119    }
120    public TextFamily defaultFamilySlabDistance()
121    {
122        setSmoothingMultiplier(1.75f);
123        return fontFamilyDistance("Iosevka_Slab_Family.atlas",
124                "Iosevka-Slab-distance-family", "Iosevka-Slab-Bold-distance-family",
125                "Iosevka-Slab-Italic-distance-family", "Iosevka-Slab-Bold-Italic-distance-family");
126    }
127    public TextFamily fontFamilyDistance(String atlasName, String... fontNames)
128    {
129        fontFamily(atlasName, fontNames);
130        distanceField = true;
131        for (int i = 0; i < 4; i++) {
132            dfx[i] = family[i].getSpaceWidth() - 1f;
133            dfy[i] = family[i].getLineHeight() - 1f;
134        }
135        distanceFieldScaleX = dfx[currentStyle];
136        distanceFieldScaleY = dfy[currentStyle];
137        return this;
138    }
139    public TextFamily fontFamily(String atlasName, String... fontNames)
140    {
141        int len;
142        if(atlasName == null || fontNames == null || (len = fontNames.length) == 0)
143            return this;
144        if(assetManager != null) 
145        {
146            BitmapFontLoader.BitmapFontParameter atlasFinder = new BitmapFontLoader.BitmapFontParameter();
147            atlasFinder.atlasName = atlasName;
148            switch (len)
149            {
150                case 1:
151                    family[3] = family[2] = family[1] = family[0] = loadFont(atlasFinder, fontNames[0]);
152                    break;
153                case 2:
154                    family[2] = family[0] = loadFont(atlasFinder, fontNames[0]);
155                    family[3] = family[1] = loadFont(atlasFinder, fontNames[1]);
156                    break;
157                case 3:
158                    family[0] = loadFont(atlasFinder, fontNames[0]);
159                    family[1] = loadFont(atlasFinder, fontNames[1]);
160                    family[3] = family[2] = loadFont(atlasFinder, fontNames[2]);
161                    break;                 
162                default:
163                    family[0] = loadFont(atlasFinder, fontNames[0]);
164                    family[1] = loadFont(atlasFinder, fontNames[1]);
165                    family[2] = loadFont(atlasFinder, fontNames[2]);
166                    family[3] = loadFont(atlasFinder, fontNames[3]);
167            }
168        }
169        else
170        {
171            TextureAtlas atlas;
172            if(Gdx.files.internal(atlasName).exists())
173                atlas = new TextureAtlas(atlasName);
174            else if(Gdx.files.classpath(atlasName).exists())
175                atlas = new TextureAtlas(Gdx.files.classpath(atlasName));
176            else return this;
177            switch (len)
178            {
179                case 1:
180                    family[3] = family[2] = family[1] = family[0] = loadFont(atlas, fontNames[0]);
181                    break;
182                case 2:
183                    family[2] = family[0] = loadFont(atlas, fontNames[0]);
184                    family[3] = family[1] = loadFont(atlas, fontNames[1]);
185                    break;
186                case 3:
187                    family[0] = loadFont(atlas, fontNames[0]);
188                    family[1] = loadFont(atlas, fontNames[1]);
189                    family[3] = family[2] = loadFont(atlas, fontNames[2]);
190                    break;
191                default:
192                    family[0] = loadFont(atlas, fontNames[0]);
193                    family[1] = loadFont(atlas, fontNames[1]);
194                    family[2] = loadFont(atlas, fontNames[2]);
195                    family[3] = loadFont(atlas, fontNames[3]);
196            }
197        }
198        bmpFont = family[currentStyle];
199        return this;
200    }

202    @Override
203    public TextCellFactory initBySize() {
204        if(bmpFont == null)
205            bmpFont = DefaultResources.getIncludedFont();
206
207        //bmpFont.setFixedWidthGlyphs(fitting);
208        Pixmap temp = new Pixmap(1, 1, Pixmap.Format.RGBA8888);
209        temp.setColor(Color.WHITE);
210        temp.fill();
211        block = new Texture(1, 1, Pixmap.Format.RGBA8888);
212        block.draw(temp, 0, 0);
213        temp.dispose();
214        if(msdf)
215        {
216            for (int i = 0; i < 4; i++) {
217                family[i].getData().setScale(width / dfx[i], height / dfy[i]);
218            }
219            shader = new ShaderProgram(DefaultResources.vertexShader, DefaultResources.msdfFragmentShader);
220            if (!shader.isCompiled()) {
221                Gdx.app.error("shader", "Distance Field font shader compilation failed:\n" + shader.getLog());
222            }
223            //lineTweak = lineHeight / 20f;
224            //distanceFieldScaleX *= (((float)width) / height) / (distanceFieldScaleX / distanceFieldScaleY);
225        }
226        else if(distanceField)
227        {
228            for (int i = 0; i < 4; i++) {
229                family[i].getData().setScale(width / dfx[i], height / dfy[i]);
230            }
231            //bmpFont.getData().setScale(width / distanceFieldScaleX, height / distanceFieldScaleY);
232
233            shader = new ShaderProgram(DefaultResources.vertexShader, DefaultResources.fragmentShader);
234            if (!shader.isCompiled()) {
235                Gdx.app.error("shader", "Distance Field font shader compilation failed:\n" + shader.getLog());
236            }
237            //lineTweak = lineHeight / 20f;
238            //distanceFieldScaleX *= (((float)width) / height) / (distanceFieldScaleX / distanceFieldScaleY);
239        }
240        else {
241            shader = SpriteBatch.createDefaultShader();
242            //lineTweak = lineHeight * 0.0625f;
243        }
244        lineHeight = bmpFont.getLineHeight();
245        descent = bmpFont.getDescent();
246        //(msdf ? -8 * bmpFont.getDescent(): bmpFont.getDescent());
247        style = new Label.LabelStyle(bmpFont, null);
248        BitmapFont.Glyph g = bmpFont.getData().getGlyph(directionGlyph);
249        dirMarker = new TextureRegion(bmpFont.getRegion(g.page), g.srcX, g.srcY, g.width, g.height);
250        initialized = true;
251        initializedBySize = true;
252        return this;
253    }

255    /**
256     * Use the specified Batch to draw a String (often just one char long) in the specified LibGDX Color, with x and y
257     * determining the world-space coordinates for the upper-left corner.
258     *
259     * @param batch the LibGDX Batch to do the drawing
260     * @param c the char to draw, often but not necessarily one char. Can be null to draw a solid block instead.
261     * @param encodedColor the LibGDX Color to use, converted to float as by {@link Color#toFloatBits()}
262     * @param x x of the upper-left corner of the region of text in world coordinates.
263     * @param y y of the upper-left corner of the region of text in world coordinates.
264     */
265    public void draw(Batch batch, char c, float encodedColor, float x, float y) {
266        setStyle(c >>> 14);
267        c &= '\u3fff';
268        super.draw(batch, c, encodedColor, x, y);
269    }

271    /**
272     * Changes the currently-used font style, such as from regular to italic or bold; values for style can range from 0
273     * to 3 (inclusive), and normally you would use one of the constants {@link #REGULAR}, {@link #BOLD},
274     * {@link #ITALIC}, or {@link #BOLD_ITALIC} as that parameter.
275     * @param style an int (can be 0, 1, 2, or 3) that determines which style to use; typically a constant in this class
276     */
277    @Override
278    public void setStyle(int style)
279    {
280        currentStyle = style & 3;
281        bmpFont = family[currentStyle];
282        if(distanceField)
283        {
284            distanceFieldScaleX = dfx[currentStyle];
285            distanceFieldScaleY = dfy[currentStyle];
286            lineHeight = bmpFont.getLineHeight();
287            descent = bmpFont.getDescent();
288
289        }
290    }

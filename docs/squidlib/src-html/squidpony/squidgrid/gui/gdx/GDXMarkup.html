012/**
013 * GDXMarkup implements IMarkup for libGDX Color objects, and can start blocks of markup that libGDX understands that
014 * display text in a given Color. Typically, the singleton {@link #instance} would be passed to a class that uses
015 * IMarkup of Color, and then anything else will be handled internally as Colors are given to the using class. This does
016 * extend GDX's markup to handle bold and italic options for text; this only works if you are using a {@link TextFamily}
017 * as your {@link TextCellFactory}, such as {@link DefaultResources#getSlabFamily()}, and only if you use
018 * {@link #colorString(CharSequence)} or {@link #styleString(CharSequence)} to generate a value that can be drawn later.
019 * <br>
020 * The notation for colors is the same as in the rest of libGDX, but if you make an IColoredString with colorString(),
021 * it doesn't need any flag to be changed on your BitmapFont (like it does for GDX markup normally). This notation looks
022 * like {@code [#FF00FF]Magenta text here[]}, which starts a tag for a hex color, uses the hex value for bright magenta,
023 * then contains the text "Magenta text here" which will be shown in bright magenta, followed by "[]" to change the
024 * color (and style, see next) back to the default white (and normal style). You can also use the names of colors, as
025 * defined in the documentation for every {@link SColor}, and some libGDX Color values as well; this looks like
026 * {@code [Inside Of A Bottle]Gray text[]} to produce the words "Gray text" with the color
027 * {@link SColor#INSIDE_OF_A_BOTTLE} (note that the docs for that SColor say what the precise name is, and case needs to
028 * match; you can also look up the {@link SColor#name} field). You can use {@code [[} to escape an opening bracket, and
029 * {@code []} to reset formatting. As an addition to GDX color markup, if using a TextFamily you can toggle the font
030 * style as bold with {@code [*]} and as italic with {@code [/]}. If bold is on when this encounters another bold tag,
031 * it will turn bold off; the same is true for italic. These formatting styles can overlap and do not need to be nested
032 * as in HTML; this notation is valid: {@code [*]bold, [/]bold and italic, [*] just italic,[] plain}. While {@code []}
033 * resets both color and style to white color and regular style, there is also {@code [,]} to reset only style, or
034 * {@code [WHITE]} to reset only color (to white).
035 * <br>
036 * Created by Tommy Ettinger on 1/23/2016.
037 */
038public class GDXMarkup implements IMarkup<Color>{
039    public static GDXMarkup instance = new GDXMarkup();
040    public GDXMarkup()
041    {
042
043    }
044    private static String floatToHex(float f)
045    {
046        //String s = Integer.toHexString(MathUtils.round(f * 255));
047        //if(s.length() < 2) return "0" + s;
048        //else return s;
049        int r = MathUtils.round(f * 255);
050        return String.valueOf(StringKit.hexDigits[r >>> 8 & 15]) + StringKit.hexDigits[r & 15];
051    }
052    @Override
053    public String getMarkup(Color value) {
054        return "[#" + floatToHex(value.r) + floatToHex(value.g) + floatToHex(value.b) + floatToHex(value.a) + "]";
055    }
056
057    @Override
058    public String closeMarkup() {
059        return "[]";
060    }
061
062    private final Matcher markupMatcher = Pattern.compile("({=p}[^\\[]+)|(?:\\[({=e}\\[))|(?:\\[#({=h}[0-9A-Fa-f]{6,8})\\])|(?:\\[({=b}\\*)\\])|(?:\\[({=i}/)\\])|(?:\\[({=u},)\\])|(?:\\[({=n}[^\\]]+?)\\])|(?:\\[({=r}\\]))").matcher();
063    private static final char BOLD = '\u4000', ITALIC = '\u8000', REGULAR = '\0';
064    private final StringBuilder sb = new StringBuilder(128);
065
066    /**
067     * Takes a CharSequence (such as a String or StringBuilder) that contains the markup this class understands, and
068     * produces an IColoredString (of Color) with the color markup tags used to mark text in colors in the resulting
069     * IColoredString, and any style markup tags used to mark sections of text as bold or italic for a TextFamily to
070     * render (normal TextCellFactory rendering may show bold/italic text as gibberish).
071     * @param markupString a String or other CharSequence containing color and/or style markup tags
072     * @return an IColoredString (of Color) with all the markup applied and removed from the text after applying
073     */
074    public IColoredString<Color> colorString(final CharSequence markupString)
075    {
076        markupMatcher.setTarget(markupString);
077        IColoredString<Color> cs = new IColoredString.Impl<>();
078        Color current = Color.WHITE;
079        char mod = REGULAR;
080        while (markupMatcher.find())
081        {
082            if(markupMatcher.getGroup("p", sb))
083            {
084                for (int i = 0; i < sb.length(); i++) {
085                    sb.setCharAt(i, (char)(sb.charAt(i) | mod));
086                }
087                cs.append(sb.toString(), current);
088            }
089            else if(markupMatcher.getGroup("e", sb))
090            {
091                cs.append((char) ('[' | mod), current);
092            }
093            else if(markupMatcher.getGroup("r", sb))
094            {
095                current = Color.WHITE;
096                mod = REGULAR;
097            }
098            else if(markupMatcher.getGroup("u", sb))
099            {
100                mod = REGULAR;
101            }
102            else if(markupMatcher.getGroup("h", sb))
103            {
104                current.set(StringKit.intFromHex(sb));
105            }
106            else if(markupMatcher.getGroup("n", sb))
107            {
108                current = Colors.get(sb.toString());
109            }
110            else if(markupMatcher.getGroup("b", sb))
111            {
112                mod ^= BOLD;
113            }
114            else if(markupMatcher.getGroup("i", sb))
115            {
116                mod ^= ITALIC;
117            }
118            sb.setLength(0);
119
120        }
121        return cs;
122    }
123
124    /**
125     * Similar to {@link #colorString(CharSequence)}, but leaves color tags as they are and only uses style tags, like
126     * {@code [*]} for bold and {@code [/]} for italic. The StringBuilder this returns can be converted to a String or
127     * used directly for further modification, but styles will probably only render correctly if using a TextFamily.
128     * You should be aware that if {@code []} is used to reset both color and style in the given markupString, then only
129     * the style will be reset here but the {@code []} will be removed, which may affect colors if the result is later
130     * given to something that expects color markup to also have been closed. A solution for this is to use {@code [,]}
131     * to reset only styles, and to avoid using {@code []} to change color by explicitly using {@code [WHITE]} to set
132     * the text color back to pure white. This way is only necessary if you have color markup in the markupString you
133     * pass to this method.
134     * @param markupString a String containing color markup (which is left as-is) and/or style markup (which is used)
135     * @return a StringBuilder based on markupString that has the style markup applied and other markup left there.
136     */
137    public StringBuilder styleString(final CharSequence markupString)
138    {
139        markupMatcher.setTarget(markupString);
140        char mod = REGULAR;
141        StringBuilder fsb = new StringBuilder(markupString.length());
142        while (markupMatcher.find())
143        {
144            if(markupMatcher.getGroup("p", sb))
145            {
146                for (int i = 0; i < sb.length(); i++) {
147                    sb.setCharAt(i, (char)(sb.charAt(i) | mod));
148                }
149                fsb.append(sb);
150            }
151            else if(markupMatcher.getGroup("e", sb))
152            {
153                fsb.append((char) ('[' | mod));
154            }
155            else if(markupMatcher.isCaptured("r") || markupMatcher.isCaptured("u"))
156            {
157                mod = REGULAR;
158            }
159            else if(markupMatcher.getGroup("h", sb))
160            {
161                markupMatcher.getGroup(0, fsb);
162            }
163            else if(markupMatcher.getGroup("n", sb))
164            {
165                markupMatcher.getGroup(0, fsb);
166            }
167            else if(markupMatcher.getGroup("b", sb))
168            {
169                mod ^= BOLD;
170            }
171            else if(markupMatcher.getGroup("i", sb))
172            {
173                mod ^= ITALIC;
174            }
175            sb.setLength(0);
176
177        }
178        return fsb;
179    }

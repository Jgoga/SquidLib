001package squidpony.squidgrid.gui.gdx;
002import com.badlogic.gdx.graphics.Color;
003import com.badlogic.gdx.math.MathUtils;
004import squidpony.squidmath.NumberTools;
005
006import static squidpony.squidgrid.gui.gdx.SColor.floatGet;
007
008/**
009 * Pre-made FloatFilter classes that you can use to filter colors without producing extra Color objects.
010 * <br>
011 * Created by Tommy Ettinger on 7/22/2018.
012 */
013public final class FloatFilters {
014    private FloatFilters()
015    {
016        // don't build me!
017    }
018
019    /**
020     * Wraps the functionality of {@link SColor#toEditedFloat(float, float, float, float, float)} so it can be called as
021     * a FloatFilter, adding values to hue, saturation, and value (clamping saturation and value and wrapping hue).
022     */
023    public static class HSVFilter extends FloatFilter
024    {
025        public float hueAddend, saturationAddend, valueAddend;
026        
027        public HSVFilter(float saturation, float value)
028        {
029            this(0f, saturation, value);
030        }
031        
032        public HSVFilter(float hueAdd, float saturationAdd, float valueAdd)
033        {
034            hueAddend = hueAdd;
035            saturationAddend = saturationAdd;
036            valueAddend = valueAdd;
037        }
038        /**
039         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
040         *
041         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
042         * @return a packed float color, as produced by {@link Color#toFloatBits()}
043         */
044        @Override
045        public float alter(float color) {
046            return SColor.toEditedFloat(color, hueAddend, saturationAddend, valueAddend, 0f);
047        }
048    }
049
050    /**
051     * Changes all colors this alters to have the same hue as, and a closer saturation to, a target color given in the
052     * constructor, as well as optionally multiplying saturation of the result and/or adding lightness/value. A good
053     * example usage of this is to make a sepia-tone effect with
054     * {@code new FloatFilters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
055     */
056    public static class ColorizeFilter extends FloatFilter
057    {
058        public float targetHue, targetSaturation, saturationMultiplier, valueAddend;
059
060        public ColorizeFilter(float color)
061        {
062            this(color, 1f, 0f);
063        }
064        public ColorizeFilter(Color color)
065        {
066            this(color.toFloatBits(), 1f, 0f);
067        }
068        public ColorizeFilter(Color color, float saturationMul, float valueAdd)
069        {
070            this(color.toFloatBits(), saturationMul, valueAdd);
071        }
072        public ColorizeFilter(float color, float saturationMul, float valueAdd)
073        {
074            targetHue = SColor.hueOfFloat(color);
075            targetSaturation = SColor.saturationOfFloat(color);
076            saturationMultiplier = saturationMul;
077            valueAddend = valueAdd;
078        }
079        /**
080         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
081         *
082         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
083         * @return a packed float color, as produced by {@link Color#toFloatBits()}
084         */
085        @Override
086        public float alter(float color) {
087            float saturation = saturationMultiplier, value = valueAddend;
088            final int bits = NumberTools.floatToIntBits(color);
089            final float s,
090                    r = (bits & 0x000000ff) * 0x1.010102p-8f,
091                    g = (bits & 0x0000ff00) * 0x1.010102p-16f,
092                    b = (bits & 0x00ff0000) * 0x1.010102p-24f;
093            final float min = Math.min(Math.min(r, g), b);   //Min. value of RGB
094            final float max = Math.max(Math.max(r, g), b);   //Max value of RGB, equivalent to value
095            final float delta = max - min;                   //Delta RGB value
096            if ( delta < 0.0039f )                           //This is a gray, no chroma...
097            {
098                s = targetSaturation * 0.5f;
099            }
100            else                                             //Chromatic data...
101            {
102                s = ((delta / max) + targetSaturation) * 0.5f;
103            }
104            saturation = MathUtils.clamp(s * saturation, 0f, 1f);
105            value = MathUtils.clamp(max + value, 0f, 1f);
106            float opacity = MathUtils.clamp(((bits & 0xfe000000) >>> 24) * 0x1.020408p-8f, 0f, 1f);
107
108            if (saturation <= 0.0039f) {
109                return floatGet(value, value, value, opacity);
110            } else if (value <= 0.0039f) {
111                return NumberTools.intBitsToFloat((int) (opacity * 254f) << 24 & 0xFE000000);
112            } else {
113                final float hu = targetHue * 6f;
114                final int i = (int) hu;
115                final float x = value * (1 - saturation);
116                final float y = value * (1 - saturation * (hu - i));
117                final float z = value * (1 - saturation * (1 - (hu - i)));
118
119                switch (i) {
120                    case 0:
121                        return floatGet(value, z, x, opacity);
122                    case 1:
123                        return floatGet(y, value, x, opacity);
124                    case 2:
125                        return floatGet(x, value, z, opacity);
126                    case 3:
127                        return floatGet(x, y, value, opacity);
128                    case 4:
129                        return floatGet(z, x, value, opacity);
130                    default:
131                        return floatGet(value, x, y, opacity);
132                }
133            }
134        }
135    }
136    /**
137     * A FloatFilter that alters primarily-red and primarily-green colors so they can be more easily be distinguished by
138     * people with at least some forms of red-green color-blindness (deuteranopia should be handled well, protanopia
139     * very well, and tritanopia may not benefit at all). Causes reds to be darkened and greens to be lightened if the
140     * other of the pair is not present in similar quantities (which is the case for yellows and blues).
141     */
142    public static class DistinctRedGreenFilter extends FloatFilter {
143        /**
144         * Constructs a DistinctRedGreenFilter. This class is a simple wrapper around a function that doesn't need
145         * member variables, so there should be little overhead with this filter.
146         */
147        public DistinctRedGreenFilter() {
148        }
149
150        /**
151         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
152         *
153         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
154         * @return a packed float color, as produced by {@link Color#toFloatBits()}
155         */
156        @Override
157        public float alter(float color) {
158            final int bits = NumberTools.floatToIntBits(color),
159                    r = bits & 0xFF, g = bits >>> 8 & 0xFF, b = bits >>> 16 & 0xFF, a = bits >>> 24,
160                    diff = g - r;
161            if(diff > 101)
162                return floatGet(Math.min(1f, 0x1.010102p-16f * r * (203 + (diff >> 1))),
163                        Math.min(1f, 0x1.010102p-16f * g * (228 + (diff >> 1))),
164                        Math.min(1f, 0x1.010102p-16f * b * (203 + (diff >> 1))),
165                        0x1.010102p-8f * a);
166            else if(diff < -75)
167                return floatGet(Math.min(1f, 0x1.010102p-16f * r * (152 - diff)),
168                        Math.min(1f, 0x1.010102p-16f * g * (177 - diff)),
169                        Math.min(1f, 0x1.010102p-16f * b * (177 - diff)),
170                        0x1.010102p-8f * a);
171            else
172                return color;
173        }
174    }

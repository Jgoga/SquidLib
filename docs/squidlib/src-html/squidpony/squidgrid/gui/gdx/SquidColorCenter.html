041    @Override
042    protected Color create(int red, int green, int blue, int opacity) {
043        if (filter == null)
044                        /* No filtering */
045            return new Color(red / 255f, green / 255f, blue / 255f, opacity / 255f);
046        else
047                        /* Some filtering */
048            return filter.alter(red / 255f, green / 255f, blue / 255f, opacity / 255f);
049    }
050    @Override
051    public Color filter(Color c)
052    {
053        if(c == null)
054            return Color.CLEAR;
055        else
056            return super.filter(c);
057    }

059    public Color get(long c)
060    {
061        return get((int)((c >> 24) & 0xff), (int)((c >> 16) & 0xff), (int)((c >> 8) & 0xff), (int)(c & 0xff));
062    }
063    public Color get(float r, float g, float b, float a)
064    {
065        return get(Math.round(255 * r), Math.round(255 * g), Math.round(255 * b), Math.round(255 * a));

068    /**
069     * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
070     * @param start the initial Color
071     * @param end the "target" color
072     * @param change the degree to change closer to end; a change of 0.0f produces start, 1.0f produces end
073     * @return a new Color
074     */
075    @Override
076    public Color lerp(Color start, Color end, float change)
077    {
078        if(start == null || end == null)
079            return Color.CLEAR;
080        return get(
081                start.r + change * (end.r - start.r),
082                start.g + change * (end.g - start.g),
083                start.b + change * (end.b - start.b),
084                start.a + change * (end.a - start.a)
085        );
086    }
087    /**
088     * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
089     * @param start the initial Color
090     * @param end the "target" color
091     * @param change the degree to change closer to end; a change of 0.0 produces start, 1.0 produces end
092     * @return a new Color
093     */
094    public Color lerp(Color start, Color end, double change)
095    {
096        return lerp(start, end, (float)change);
097    }

118    public static int encode (Color color) {
119        if (color == null)
120            return 0;
121        return (Math.round(color.r * 255.0f) << 24)
122                | (Math.round(color.g * 255.0f) << 16)
123                | (Math.round(color.b * 255.0f) << 8)
124                | Math.round(color.a * 255.0f);
125    }

408    /**
409     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
410     * gradient. The interpolation argument can be used to make the color stay close to fromColor and/or toColor longer
411     * than it would normally, or shorter if the middle colors are desirable.
412     * @param fromColor the color to start with, included in the gradient
413     * @param toColor the color to end on, included in the gradient
414     * @param steps the number of elements to use in the gradient
415     * @param interpolation a libGDX Interpolation that defines how quickly the color changes during the transition
416     * @return an ArrayList composed of the blending steps from fromColor to toColor, with length equal to steps
417     */
418    public ArrayList<Color> gradient(Color fromColor, Color toColor, int steps, Interpolation interpolation)
419    {
420        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
421        colors.add(filter(fromColor));
422        if(steps < 2)
423            return colors;
424        for (float i = 1; i < steps; i++) {
425            colors.add(lerp(fromColor, toColor, interpolation.apply(i / (steps - 1))));
426        }
427        return colors;
428    }

430    /**
431     * Finds a gradient with the specified number of steps going from fromColor to midColor, then midColor to (possibly)
432     * fromColor, with both included in the gradient but fromColor only repeated at the end if the number of steps is
433     * odd. The interpolation argument can be used to make the color linger for a while with colors close to fromColor
434     * or midColor, or to do the opposite and quickly change from one and spend more time in the middle.
435     * @param fromColor the color to start with (and end with, if steps is an odd number), included in the gradient
436     * @param midColor the color to use in the middle of the loop, included in the gradient
437     * @param steps the number of elements to use in the gradient, will be at least 3
438     * @param interpolation a libGDX Interpolation that defines how quickly the color changes at the start and end of
439     *                      each transition, both from fromColor to midColor as well as back to fromColor
440     * @return an ArrayList composed of the blending steps from fromColor to midColor to fromColor again, with length equal to steps
441     */
442    public ArrayList<Color> loopingGradient(Color fromColor, Color midColor, int steps, Interpolation interpolation)
443    {
444        ArrayList<Color> colors = new ArrayList<>((steps > 3) ? steps : 3);
445        colors.add(filter(fromColor));
446        for (float i = 1; i < steps / 2; i++) {
447            colors.add(lerp(fromColor, midColor, interpolation.apply(i / (steps / 2))));
448        }
449        for (float i = 0, c = steps / 2; c < steps; i++, c++) {
450            colors.add(lerp(midColor, fromColor, interpolation.apply(i / (steps / 2))));
451        }
452        return colors;
453    }

469    /**
470     * Generates a hue-shifted rainbow of colors, starting at red and going through orange, yellow, green, blue, and
471     * purple before getting close to red at the end again. If the given number of steps is less than 6 or so, you should
472     * expect to see only some of those colors; if steps is larger (36 may be reasonable for gradients), you are more
473     * likely to see colors that appear for shorter spans on the color wheel, like orange.
474     * Uses the given saturation and value for all colors in the rainbow, and only changes hue.
475     * @param saturation the saturation of the rainbow's colors; 1.0 is boldest and 0.0 is grayscale
476     * @param value the brightness of the rainbow's colors; 1.0 is brightest
477     * @param steps the number of different Color elements to generate in the returned ArrayList
478     * @return an ArrayList of Color where each element goes around the color wheel, starting at red, then orange, etc.
479     */
480    public ArrayList<Color> rainbow(float saturation, float value, int steps)
481    {
482        steps = (steps > 1) ? steps : 1;
483        ArrayList<Color> colors = new ArrayList<>(steps);
484        for (float i = 0; i < 1f - 0.5f / steps; i+= 1.0f / steps) {
485            colors.add(filter(getHSV(i, saturation, value)));
486        }
487        return colors;
488    }

505    /**
506     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
507     * gradient. This does not typically take a direct path on its way between fromColor and toColor, and is useful to
508     * generate a wide variety of colors that can be confined to a rough amount of maximum difference by choosing values
509     * for fromColor and toColor that are more similar.
510     * <br>
511     * Try using colors for fromColor and toColor that have different r, g, and b values, such as gray and white, then
512     * compare to colors that don't differ on, for example, r, such as bright red and pink. In the first case, red,
513     * green, blue, and many other colors will be generated if there are enough steps; in the second case, red will be
514     * at the same level in all generated colors (very high, so no pure blue or pure green, but purple and yellow are
515     * possible). This should help illustrate how this chooses how far to "zig-zag" off the straight-line path.
516     * @param fromColor the color to start with, included in the gradient
517     * @param toColor the color to end on, included in the gradient
518     * @param steps the number of elements to use in the gradient; ideally no greater than 345 to avoid duplicates
519     * @return an ArrayList composed of the zig-zag steps from fromColor to toColor, with length equal to steps
520     */
521    public ArrayList<Color> zigzagGradient(Color fromColor, Color toColor, int steps)
522    {
523        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
524        colors.add(filter(fromColor));
525        if(steps < 2)
526            return colors;
527        float dr = toColor.r - fromColor.r, dg = toColor.g - fromColor.g, db = toColor.b - fromColor.b,
528                a = fromColor.a, cr, cg, cb;
529        int decoded;
530        for (float i = 1; i < steps; i++) {
531            // 345 happens to be the distance on our 3D Hilbert curve that corresponds to (7,7,7).
532            decoded = Math.round(345 * (i / (steps - 1)));
533            cr = (CoordPacker.hilbert3X[decoded] / 7f) * dr + fromColor.r;
534            cg = (CoordPacker.hilbert3Y[decoded] / 7f) * dg + fromColor.g;
535            cb = (CoordPacker.hilbert3Z[decoded] / 7f) * db + fromColor.b;
536            colors.add(get(cr, cg, cb, a));
537        }
538        return colors;
539    }

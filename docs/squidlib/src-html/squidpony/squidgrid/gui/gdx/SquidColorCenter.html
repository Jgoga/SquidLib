013/**
014 * A concrete implementation of {@link IColorCenter} for libgdx's {@link com.badlogic.gdx.graphics.Color}.
015 * Supports filtering any colors that this creates using a {@link Filter}, such as one from {@link Filters}.
016 * This class largely supersedes the earlier {@link SColorFactory} class, and supports similar operations
017 * while also allowing filters to modify the returned colors. Some things use different terms between the
018 * two classes; {@link SColorFactory#blend(SColor, SColor, double)} is {@link #lerp(Color, Color, double)}
019 * here, and {@link SColorFactory#setFloor(int)} is {@link #setGranularity(int)} (with different behavior).
020 * @author smelC
021 * @author Tommy Ettinger
022 * @see SColor Another way to obtain colors by using pre-allocated (and named) instances.
023 */
024public class SquidColorCenter extends IColorCenter.Skeleton<Color> {

026    /**
027     * How different requested colors need to be to make a new color; should range from 0 to at most 6.
028     * If this is 0, all requested colors will be looked up (using a cached version if the exact request had been made
029     * before), but if this is greater than 0, then exponentially less colors will be used, using the cache for twice as
030     * many requests at granularity 1 (2 raised to the granularity), four times as many at granularity 2, and so on.
031     * Defaults to 1, which seems to help ensure visually-identical colors are not created more than once.
032     */
033    public int granularity = 1;
034
035    /**
036     * Gets the granularity, which is how different requested colors need to be to make a new color; can be from 0 to 6.
037     * If this is 0, all requested colors will be looked up (using a cached version if the exact request had been made
038     * before), but if this is greater than 0, then exponentially less colors will be used, using the cache for twice as
039     * many requests at granularity 1 (2 raised to the granularity), four times as many at granularity 2, and so on.
040     * If no granularity was set, the default is 1.
041     * @return the current granularity, as an int
042     */
043    public int getGranularity() {
044        return granularity;
045    }

050     * before), but if this is greater than 0, then exponentially less colors will be used, using the cache for twice as
051     * many requests at granularity 1 (2 raised to the granularity), four times as many at granularity 2, and so on.

053     * @param granularity the granularity to use; will be clamped between 0 and 6
054     */
055    public void setGranularity(int granularity) {
056        this.granularity = MathUtils.clamp(granularity, 0, 6);

114    /**
115     * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
116     * @param start the initial Color
117     * @param end the "target" color
118     * @param change the degree to change closer to end; a change of 0.0f produces start, 1.0f produces end
119     * @return a new Color
120     */
121    @Override
122    public Color lerp(Color start, Color end, float change)
123    {
124        if(start == null || end == null)
125            return Color.CLEAR;
126        return get(
127                start.r + change * (end.r - start.r),
128                start.g + change * (end.g - start.g),
129                start.b + change * (end.b - start.b),
130                start.a + change * (end.a - start.a)
131        );
132    }
133    /**
134     * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
135     * @param start the initial Color
136     * @param end the "target" color
137     * @param change the degree to change closer to end; a change of 0.0 produces start, 1.0 produces end
138     * @return a new Color
139     */
140    public Color lerp(Color start, Color end, double change)
141    {
142        return lerp(start, end, (float)change);
143    }

164    public static int encode (Color color) {
165        if (color == null)
166            return 0;
167        return (Math.round(color.r * 255.0f) << 24)
168                | (Math.round(color.g * 255.0f) << 16)
169                | (Math.round(color.b * 255.0f) << 8)
170                | Math.round(color.a * 255.0f);
171    }

454    /**
455     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
456     * gradient. The interpolation argument can be used to make the color stay close to fromColor and/or toColor longer
457     * than it would normally, or shorter if the middle colors are desirable.
458     * @param fromColor the color to start with, included in the gradient
459     * @param toColor the color to end on, included in the gradient
460     * @param steps the number of elements to use in the gradient
461     * @param interpolation a libGDX Interpolation that defines how quickly the color changes during the transition
462     * @return an ArrayList composed of the blending steps from fromColor to toColor, with length equal to steps
463     */
464    public ArrayList<Color> gradient(Color fromColor, Color toColor, int steps, Interpolation interpolation)
465    {
466        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
467        colors.add(filter(fromColor));
468        if(steps < 2)
469            return colors;
470        for (float i = 1; i < steps; i++) {
471            colors.add(lerp(fromColor, toColor, interpolation.apply(i / (steps - 1))));
472        }
473        return colors;
474    }

476    /**
477     * Finds a gradient with the specified number of steps going from fromColor to midColor, then midColor to (possibly)
478     * fromColor, with both included in the gradient but fromColor only repeated at the end if the number of steps is
479     * odd. The interpolation argument can be used to make the color linger for a while with colors close to fromColor
480     * or midColor, or to do the opposite and quickly change from one and spend more time in the middle.
481     * @param fromColor the color to start with (and end with, if steps is an odd number), included in the gradient
482     * @param midColor the color to use in the middle of the loop, included in the gradient
483     * @param steps the number of elements to use in the gradient, will be at least 3
484     * @param interpolation a libGDX Interpolation that defines how quickly the color changes at the start and end of
485     *                      each transition, both from fromColor to midColor as well as back to fromColor
486     * @return an ArrayList composed of the blending steps from fromColor to midColor to fromColor again, with length equal to steps
487     */
488    public ArrayList<Color> loopingGradient(Color fromColor, Color midColor, int steps, Interpolation interpolation)
489    {
490        ArrayList<Color> colors = new ArrayList<>((steps > 3) ? steps : 3);
491        colors.add(filter(fromColor));
492        for (float i = 1; i < steps / 2; i++) {
493            colors.add(lerp(fromColor, midColor, interpolation.apply(i / (steps / 2))));
494        }
495        for (float i = 0, c = steps / 2; c < steps; i++, c++) {
496            colors.add(lerp(midColor, fromColor, interpolation.apply(i / (steps / 2))));
497        }
498        return colors;
499    }

533     * purple before getting close to red at the end again. If the given number of steps is less than 6 or so, you should
534     * expect to see only some of those colors; if steps is larger (36 may be reasonable for gradients), you are more
535     * likely to see colors that appear for shorter spans on the color wheel, like orange.
536     * Uses the given saturation, value, and opacity for all colors in the rainbow, and only changes hue.
537     * @param saturation the saturation of the rainbow's colors; 1.0 is boldest and 0.0 is grayscale
538     * @param value the brightness of the rainbow's colors; 1.0 is brightest
539     * @param opacity the alpha value of all colors in the rainbow; 0.0 is fully transparent and 1.0 is opaque
540     * @param steps the number of different Color elements to generate in the returned ArrayList
541     * @return an ArrayList of Color where each element goes around the color wheel, starting at red, then orange, etc.
542     */
543    public ArrayList<Color> rainbow(float saturation, float value, float opacity, int steps)
544    {
545        steps = (steps > 1) ? steps : 1;
546        ArrayList<Color> colors = new ArrayList<>(steps);
547        for (float i = 0; i < 1f - 0.5f / steps; i+= 1.0f / steps) {
548            colors.add(filter(getHSV(i, saturation, value, opacity)));
549        }
550        return colors;
551    }

578     * @param steps the number of different Color elements to generate in the returned ArrayList
579     * @return an ArrayList of Color where each element goes around the color wheel, starting at red, then orange, etc.
580     */
581    public ArrayList<Color> rainbow(double saturation, double value, double opacity, int steps)
582    {
583        return rainbow((float)saturation, (float)value, (float)opacity, steps);
584    }
585    /**
586     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
587     * gradient. This does not typically take a direct path on its way between fromColor and toColor, and is useful to
588     * generate a wide variety of colors that can be confined to a rough amount of maximum difference by choosing values
589     * for fromColor and toColor that are more similar.
590     * <br>
591     * Try using colors for fromColor and toColor that have different r, g, and b values, such as gray and white, then
592     * compare to colors that don't differ on, for example, r, such as bright red and pink. In the first case, red,
593     * green, blue, and many other colors will be generated if there are enough steps; in the second case, red will be
594     * at the same level in all generated colors (very high, so no pure blue or pure green, but purple and yellow are
595     * possible). This should help illustrate how this chooses how far to "zig-zag" off the straight-line path.
596     * @param fromColor the color to start with, included in the gradient
597     * @param toColor the color to end on, included in the gradient
598     * @param steps the number of elements to use in the gradient; ideally no greater than 345 to avoid duplicates
599     * @return an ArrayList composed of the zig-zag steps from fromColor to toColor, with length equal to steps
600     */
601    public ArrayList<Color> zigzagGradient(Color fromColor, Color toColor, int steps)
602    {
603        CoordPacker.init();
604        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
605        colors.add(filter(fromColor));
606        if(steps < 2)
607            return colors;
608        float dr = toColor.r - fromColor.r, dg = toColor.g - fromColor.g, db = toColor.b - fromColor.b,
609                a = fromColor.a, cr, cg, cb;
610        int decoded;
611        for (float i = 1; i < steps; i++) {
612            // 345 happens to be the distance on our 3D Hilbert curve that corresponds to (7,7,7).
613            decoded = Math.round(345 * (i / (steps - 1)));
614            cr = (CoordPacker.hilbert3X[decoded] / 7f) * dr + fromColor.r;
615            cg = (CoordPacker.hilbert3Y[decoded] / 7f) * dg + fromColor.g;
616            cb = (CoordPacker.hilbert3Z[decoded] / 7f) * db + fromColor.b;
617            colors.add(get(cr, cg, cb, a));
618        }
619        return colors;
620    }

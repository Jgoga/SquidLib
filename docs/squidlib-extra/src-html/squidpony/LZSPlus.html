034/**
035 * LZ-String compression, taking Strings and compressing them to other Strings, optionally with some encryption.
036 * Created by Tommy Ettinger on 7/13/2017.

038public final class LZSPlus {

040    /**
041     * Compresses the given text using LZ-String compression; does not encrypt the result.
042     * @param uncompressedStr text to compress
043     * @return a compressed version of the given text
044     */
045    public static String compress(String uncompressedStr) {
046        return compress(uncompressedStr, null);
047    }
048
049    /**
050     * Compresses the given text using LZ-String compression and encrypts (somewhat) the compressed result so it can't
051     * be read back without the same keys as an int array. Shorter int arrays give less security to encryption, though
052     * there isn't much security to begin with. You can produce a decent-quality array for this purpose with
053     * {@link Garbler#makeKeyArray(int, String)}; the size parameter could reasonably be anywhere from 2 to 32. If the
054     * keys array is null or empty, this only compresses and does not perform an additional encryption step.
055     * @param uncompressedStr text to compress and optionally encrypt
056     * @param keys the int array that will be used to encrypt the output, and will be required to decrypt the result; may be null
057     * @return a compressed version of the given text
058     */
059    public static String compress(String uncompressedStr, long[] keys) {
060        if (uncompressedStr == null) return null;
061        if (uncompressedStr.isEmpty()) return "";
062        int i, value;
063        ObjectIntMap<String> context_dictionary = new ObjectIntMap<String>();
064        ObjectSet<String> context_dictionaryToCreate = new ObjectSet<String>();
065        String context_c;
066        String context_wc;
067        String context_w = "";
068        int context_enlargeIn = 2; // Compensate for the first entry which should not count
069        int context_dictSize = 3;
070        int context_numBits = 2;
071        int len = uncompressedStr.length();
072        StringBuilder context_data = new StringBuilder(len >>> 1);
073        int context_data_val = 0;
074        int context_data_position = 0;
075        int ii;
076
077        for (ii = 0; ii < len; ii++) {
078            context_c = uncompressedStr.substring(ii, ii+1);
079            if (!context_dictionary.containsKey(context_c)) {
080                context_dictionary.put(context_c, context_dictSize++);
081                context_dictionaryToCreate.add(context_c);
082            }
083
084            context_wc = context_w + context_c;
085            if (context_dictionary.containsKey(context_wc)) {
086                context_w = context_wc;
087            } else {
088                if (context_dictionaryToCreate.contains(context_w)) {
089                    if ((value = context_w.charAt(0)) < 256) {
090                        for (i = 0; i < context_numBits; i++) {
091                            context_data_val = (context_data_val << 1);
092                            if (context_data_position == 14) {
093                                context_data_position = 0;
094                                context_data.append((char) (context_data_val + 32));
095                                context_data_val = 0;
096                            } else {
097                                context_data_position++;
098                            }
099                        }
100                        for (i = 0; i < 8; i++) {
101                            context_data_val = (context_data_val << 1) | (value & 1);
102                            if (context_data_position == 14) {
103                                context_data_position = 0;
104                                context_data.append((char) (context_data_val + 32));
105                                context_data_val = 0;
106                            } else {
107                                context_data_position++;
108                            }
109                            value >>= 1;
110                        }
111                    } else {
112                        value = 1;
113                        for (i = 0; i < context_numBits; i++) {
114                            context_data_val = (context_data_val << 1) | value;
115                            if (context_data_position == 14) {
116                                context_data_position = 0;
117                                context_data.append((char) (context_data_val + 32));
118                                context_data_val = 0;
119                            } else {
120                                context_data_position++;
121                            }
122                            value = 0;
123                        }
124                        value = context_w.charAt(0);
125                        for (i = 0; i < 16; i++) {
126                            context_data_val = (context_data_val << 1) | (value & 1);
127                            if (context_data_position == 14) {
128                                context_data_position = 0;
129                                context_data.append((char) (context_data_val + 32));
130                                context_data_val = 0;
131                            } else {
132                                context_data_position++;
133                            }
134                            value >>= 1;
135                        }
136                    }
137                    context_enlargeIn--;
138                    if (context_enlargeIn == 0) {
139                        context_enlargeIn = 1 << context_numBits++;
140                    }
141                    context_dictionaryToCreate.remove(context_w);
142                } else {
143                    value = context_dictionary.get(context_w, -1);
144                    for (i = 0; i < context_numBits; i++) {
145                        context_data_val = (context_data_val << 1) | (value & 1);
146                        if (context_data_position == 14) {
147                            context_data_position = 0;
148                            context_data.append((char) (context_data_val + 32));
149                            context_data_val = 0;
150                        } else {
151                            context_data_position++;
152                        }
153                        value >>= 1;
154                    }
155
156                }
157                context_enlargeIn--;
158                if (context_enlargeIn == 0) {
159                    context_enlargeIn = 1 << context_numBits++;
160                }
161                // Add wc to the dictionary.
162                context_dictionary.put(context_wc, context_dictSize++);
163                context_w = context_c;
164            }
165        }
166
167        // Output the code for w.
168        if (!context_w.isEmpty()) {
169            if (context_dictionaryToCreate.contains(context_w)) {
170                if (context_w.charAt(0) < 256) {
171                    for (i = 0; i < context_numBits; i++) {
172                        context_data_val = (context_data_val << 1);
173                        if (context_data_position == 14) {
174                            context_data_position = 0;
175                            context_data.append((char) (context_data_val + 32));
176                            context_data_val = 0;
177                        } else {
178                            context_data_position++;
179                        }
180                    }
181                    value = context_w.charAt(0);
182                    for (i = 0; i < 8; i++) {
183                        context_data_val = (context_data_val << 1) | (value & 1);
184                        if (context_data_position == 14) {
185                            context_data_position = 0;
186                            context_data.append((char) (context_data_val + 32));
187                            context_data_val = 0;
188                        } else {
189                            context_data_position++;
190                        }
191                        value >>= 1;
192                    }
193                } else {
194                    value = 1;
195                    for (i = 0; i < context_numBits; i++) {
196                        context_data_val = (context_data_val << 1) | value;
197                        if (context_data_position == 14) {
198                            context_data_position = 0;
199                            context_data.append((char) (context_data_val + 32));
200                            context_data_val = 0;
201                        } else {
202                            context_data_position++;
203                        }
204                        value = 0;
205                    }
206                    value = context_w.charAt(0);
207                    for (i = 0; i < 16; i++) {
208                        context_data_val = (context_data_val << 1) | (value & 1);
209                        if (context_data_position == 14) {
210                            context_data_position = 0;
211                            context_data.append((char) (context_data_val + 32));
212                            context_data_val = 0;
213                        } else {
214                            context_data_position++;
215                        }
216                        value >>= 1;
217                    }
218                }
219
220                context_dictionaryToCreate.remove(context_w);
221            } else {
222                value = context_dictionary.get(context_w, -1);
223                for (i = 0; i < context_numBits; i++) {
224                    context_data_val = (context_data_val << 1) | (value & 1);
225                    if (context_data_position == 14) {
226                        context_data_position = 0;
227                        context_data.append((char) (context_data_val + 32));
228                        context_data_val = 0;
229                    } else {
230                        context_data_position++;
231                    }
232                    value >>= 1;
233                }
234
235            }
236        }
237
238        // Mark the end of the stream
239        value = 2;
240        for (i = 0; i < context_numBits; i++) {
241            context_data_val = (context_data_val << 1) | (value & 1);
242            if (context_data_position == 14) {
243                context_data_position = 0;
244                context_data.append((char) (context_data_val + 32));
245                context_data_val = 0;
246            } else {
247                context_data_position++;
248            }
249            value >>= 1;
250        }
251
252        // Flush the last char
253        while (true) {
254            context_data_val = (context_data_val << 1);
255            if (context_data_position == 14) {
256                context_data.append((char) (context_data_val + 32));
257                break;
258            } else
259                context_data_position++;
260        }
261        if(keys != null && keys.length > 0)
262        {
263            int mainLength = context_data.length(), smallLength = mainLength >> 2, rem = mainLength & 3;
264            char[] raw = new char[mainLength];
265            context_data.getChars(0, mainLength, raw, 0);
266            for (int j = 0; j < keys.length; j++) {
267                long z = keys[j], inc = 0x9E3779BAL * z + 0x632BE5ABL, r;
268                for (int k = 0, n = 0; k < smallLength; k++) {
269                    r = splitMix64(z += inc);
270                    raw[n] ^= (r & 0x7fff);
271                    raw[n++] += 32;
272                    raw[n] ^= (r >>> 16 & 0x7fff);
273                    raw[n++] += 32;
274                    raw[n] ^= (r >>> 32 & 0x7fff);
275                    raw[n++] += 32;
276                    raw[n] ^= (r >>> 49);
277                    raw[n++] += 32;
278                }
279                z += inc;
280                switch(rem)
281                {
282                    case 3:
283                        raw[mainLength-3] ^= splitMix64(z) >>> 49;
284                    case 2:
285                        raw[mainLength-2] ^= splitMix64(z) >>> 16 & 0x7fff;
286                    case 1:
287                        raw[mainLength-1] ^= splitMix64(z) & 0x7fff;
288                }
289            }
290            return String.valueOf(raw);
291        }
292        else
293            return context_data.toString();
294    }
295    /**
296     * Decompresses text that was compressed with LZ-String compression; does not reverse decryption so it can only
297     * decompress Strings produced by {@link #compress(String)}, or {@link #compress(String, long[])} with an empty or
298     * null keys parameter.
299     * @param compressed text that was compressed by {@link #compress(String)}
300     * @return the original text, decompressed from the given text
301     */
302
303    public static String decompress(String compressed) {
304        return decompress(compressed, null);
305    }

307    /**
308     * Decompresses text that was compressed with LZ-String compression, reversing any encryption if the keys int array
309     * matches the int array passed to {@link #compress(String, long[])} (keys can be null if no array was passed).
310     * @param compressed text that was compressed by {@link #compress(String, long[])}
311     * @param keys the int array that was used to encrypt the output, and must match to decrypt the result; may be null
312     * @return the original text, decompressed and decrypted from compressed
313     */
314    public static String decompress(String compressed, long[] keys) {
315        if (compressed == null)
316            return null;
317        if (compressed.isEmpty())
318            return "";
319        final char[] comp = compressed.toCharArray();
320        int length = comp.length;
321        if(keys != null && keys.length > 0)
322        {
323            int smallLength = length >> 2, rem = length & 3;
324            for (int j = keys.length - 1; j >= 0; j--) {
325                long z = keys[j], inc = 0x9E3779BAL * z + 0x632BE5ABL, r;
326                z += (1+smallLength) * inc;
327                switch (rem)
328                {
329                    case 3:
330                        comp[length-3] ^= splitMix64(z) >>> 49;
331                    case 2:
332                        comp[length-2] ^= splitMix64(z) >>> 16 & 0x7fff;
333                    case 1:
334                        comp[length-1] ^= splitMix64(z) & 0x7fff;
335                }
336                for (int k = 0, n = (length & -4) - 1; k < smallLength ; k++) {
337                    r = splitMix64(z -= inc);
338                    comp[n] -= 32;
339                    comp[n--] ^= (r >>> 49);
340                    comp[n] -= 32;
341                    comp[n--] ^= (r >>> 32 & 0x7fff);
342                    comp[n] -= 32;
343                    comp[n--] ^= (r >>> 16 & 0x7fff);
344                    comp[n] -= 32;
345                    comp[n--] ^= (r & 0x7fff);
346
347                }
348            }
349        }
350
351
352        ArrayList<String> dictionary = new ArrayList<String>();
353        int enlargeIn = 4, dictSize = 4, numBits = 3, position = 16384, index = 1, resb, maxpower, power;
354        String entry, w, c;
355        ArrayList<String> result = new ArrayList<String>();
356        char bits, val = (char) (comp[0] - 32);
357
358        for (char i = 0; i < 3; i++) {
359            dictionary.add(i, String.valueOf(i));
360        }
361
362        bits = 0;
363        maxpower = 2;
364        power = 0;
365        while (power != maxpower) {
366            resb = val & position;
367            position >>= 1;
368            if (position == 0) {
369                position = 16384;
370                val = (char) (comp[index++] - 32);
371            }
372            bits |= (resb > 0 ? 1 : 0) << power++;
373        }
374
375        switch (bits) {
376            case 0:
377                bits = 0;
378                maxpower = 8;
379                power = 0;
380                while (power != maxpower) {
381                    resb = val & position;
382                    position >>= 1;
383                    if (position == 0) {
384                        position = 16384;
385                        val = (char) (comp[index++] - 32);
386                    }
387                    bits |= (resb > 0 ? 1 : 0) << power++;
388                }
389                c = String.valueOf(bits);
390                break;
391            case 1:
392                bits = 0;
393                maxpower = 16;
394                power = 0;
395                while (power != maxpower) {
396                    resb = val & position;
397                    position >>= 1;
398                    if (position == 0) {
399                        position = 16384;
400                        val = (char) (comp[index++] - 32);
401                    }
402                    bits |= (resb > 0 ? 1 : 0) << power++;
403                }
404                c = String.valueOf(bits);
405                break;
406            default:
407                return "";
408        }
409        dictionary.add(c);
410        w = c;
411        result.add(w);
412        while (true) {
413            if (index > length) {
414                return "";
415            }
416
417            bits = 0;
418            maxpower = numBits;
419            power = 0;
420            while (power != maxpower) {
421                resb = val & position;
422                position >>= 1;
423                if (position == 0) {
424                    position = 16384;
425                    val = (char) (comp[index++] - 32);
426                }
427                bits |= (resb > 0 ? 1 : 0) << power++;
428            }
429            int cc;
430            switch (cc = bits) {
431                case 0:
432                    bits = 0;
433                    maxpower = 8;
434                    power = 0;
435                    while (power != maxpower) {
436                        resb = val & position;
437                        position >>= 1;
438                        if (position == 0) {
439                            position = 16384;
440                            val = (char) (comp[index++] - 32);
441                        }
442                        bits |= (resb > 0 ? 1 : 0) << power++;
443                    }
444
445                    dictionary.add(String.valueOf(bits));
446                    cc = dictSize++;
447                    enlargeIn--;
448                    break;
449                case 1:
450                    bits = 0;
451                    maxpower = 16;
452                    power = 0;
453                    while (power != maxpower) {
454                        resb = val & position;
455                        position >>= 1;
456                        if (position == 0) {
457                            position = 16384;
458                            val = (char) (comp[index++] - 32);
459                        }
460                        bits |= (resb > 0 ? 1 : 0) << power++;
461                    }
462                    dictionary.add(String.valueOf(bits));
463                    cc = dictSize++;
464                    enlargeIn--;
465                    break;
466                case 2:
467                    StringBuilder sb = new StringBuilder(result.size());
468                    for (String s : result)
469                        sb.append(s);
470                    return sb.toString();
471            }
472
473            if (enlargeIn == 0) {
474                enlargeIn = 1 << numBits;
475                numBits++;
476            }
477
478            if (cc < dictionary.size() && dictionary.get(cc) != null) {
479                entry = dictionary.get(cc);
480            } else {
481                if (cc == dictSize) {
482                    entry = w + w.charAt(0);
483                } else {
484                    return "";
485                }
486            }
487            result.add(entry);
488
489            // Add w+entry[0] to the dictionary.
490            dictionary.add(w + entry.charAt(0));
491            dictSize++;
492            enlargeIn--;
493
494            w = entry;
495
496            if (enlargeIn == 0) {
497                enlargeIn = 1 << numBits;
498                numBits++;
499            }
500
501        }
502
503    }

505    /**
506     * Get a long from this with {@code splitMix64(z += 0x9E3779B97F4A7C15L)}, where z is a long to use as state.
507     * 0x9E3779B97F4A7C15L can be changed for any odd long if the same number is used across calls.
508     * @param state long, must be changed with each call; {@code splitMix64(z += 0x9E3779B97F4A7C15L)} is recommended
509     * @return a pseudo-random long
510     */
511    private static long splitMix64(long state) {
512        state = (((state *= 0x9E3779B97F4A7C15L) >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
513        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
514        return state ^ (state >>> 31);
515    }
